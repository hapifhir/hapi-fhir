28  import org.springframework.context.annotation.Bean;

32  import org.springframework.transaction.annotation.EnableTransactionManagement;
33  
34  import ca.uhn.fhir.context.FhirContext;
35  import ca.uhn.fhir.jpa.config.BaseConfig;
36  import ca.uhn.fhir.jpa.dao.FulltextSearchSvcImpl;
37  import ca.uhn.fhir.jpa.dao.IFhirSystemDao;
38  import ca.uhn.fhir.jpa.dao.IFulltextSearchSvc;
39  import ca.uhn.fhir.jpa.dao.dstu3.SearchParamExtractorDstu3;
40  import ca.uhn.fhir.jpa.provider.dstu3.TerminologyUploaderProviderDstu3;
41  import ca.uhn.fhir.jpa.term.HapiTerminologySvcDstu3;
42  import ca.uhn.fhir.jpa.term.IHapiTerminologyLoaderSvc;
43  import ca.uhn.fhir.jpa.term.IHapiTerminologySvcDstu3;
44  import ca.uhn.fhir.jpa.term.TerminologyLoaderSvc;
45  import ca.uhn.fhir.jpa.validation.JpaValidationSupportChainDstu3;
46  import ca.uhn.fhir.validation.IValidatorModule;
47  
48  @Configuration
49  @EnableTransactionManagement
50  public class BaseDstu3Config extends BaseConfig {
51  
52  	@Bean(autowire = Autowire.BY_TYPE)
53  	public IHapiTerminologySvcDstu3 terminologyService() {
54  		return new HapiTerminologySvcDstu3();
55  	}
56  	
57  	@Bean(autowire = Autowire.BY_TYPE)
58  	public IHapiTerminologyLoaderSvc terminologyLoaderService() {
59  		return new TerminologyLoaderSvc();
60  	}
61  
62  	@Bean
63  	@Primary
64  	public FhirContext defaultFhirContext() {
65  		return fhirContextDstu3();
66  	}
67  
68  	@Bean(name = "myInstanceValidatorDstu3")
69  	@Lazy
70  	public IValidatorModule instanceValidatorDstu3() {
71  		FhirInstanceValidator val = new FhirInstanceValidator();
72  		val.setBestPracticeWarningLevel(BestPracticeWarningLevel.Warning);
73  		val.setValidationSupport(validationSupportChainDstu3());
74  		return val;
75  	}
76  
77  	@Bean(name = "myJpaValidationSupportDstu3", autowire = Autowire.BY_NAME)
78  	public ca.uhn.fhir.jpa.dao.dstu3.IJpaValidationSupportDstu3 jpaValidationSupportDstu3() {
79  		ca.uhn.fhir.jpa.dao.dstu3.JpaValidationSupportDstu3 retVal = new ca.uhn.fhir.jpa.dao.dstu3.JpaValidationSupportDstu3();
80  		return retVal;
81  	}
82  
83  	@Bean(autowire = Autowire.BY_TYPE)
84  	public IFulltextSearchSvc searchDaoDstu3() {
85  		FulltextSearchSvcImpl searchDao = new FulltextSearchSvcImpl();
86  		return searchDao;
87  	}
88  
89  	@Bean(name = "mySystemDaoDstu3", autowire = Autowire.BY_NAME)
90  	public IFhirSystemDao<org.hl7.fhir.dstu3.model.Bundle, org.hl7.fhir.dstu3.model.Meta> systemDaoDstu3() {
91  		ca.uhn.fhir.jpa.dao.dstu3.FhirSystemDaoDstu3 retVal = new ca.uhn.fhir.jpa.dao.dstu3.FhirSystemDaoDstu3();
92  		return retVal;
93  	}
94  
95  	@Bean(name = "mySystemProviderDstu3")
96  	public ca.uhn.fhir.jpa.provider.dstu3.JpaSystemProviderDstu3 systemProviderDstu3() {
97  		ca.uhn.fhir.jpa.provider.dstu3.JpaSystemProviderDstu3 retVal = new ca.uhn.fhir.jpa.provider.dstu3.JpaSystemProviderDstu3();
98  		retVal.setContext(defaultFhirContext());
99  		retVal.setDao(systemDaoDstu3());
100 		return retVal;
101 	}
102 
103 	@Primary
104 	@Bean(autowire = Autowire.BY_NAME, name = "myJpaValidationSupportChainDstu3")
105 	public IValidationSupport validationSupportChainDstu3() {
106 		return new JpaValidationSupportChainDstu3();
107 	}
108 
109 	@Bean(autowire = Autowire.BY_TYPE)
110 	public SearchParamExtractorDstu3 searchParamExtractor() {
111 		return new SearchParamExtractorDstu3();
112 	}
113 
114 	@Bean(autowire=Autowire.BY_TYPE)
115 	public TerminologyUploaderProviderDstu3 terminologyUploaderProvider() {
116 		TerminologyUploaderProviderDstu3 retVal = new TerminologyUploaderProviderDstu3();
117 		retVal.setContext(defaultFhirContext());
118 		return retVal;
119 	}
120 	
121 
122 }

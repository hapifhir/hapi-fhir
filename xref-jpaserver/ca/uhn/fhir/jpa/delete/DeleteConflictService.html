48  @Service
49  public class DeleteConflictService {
50  	private static final Logger ourLog = LoggerFactory.getLogger(DeleteConflictService.class);
51  	public static final int FIRST_QUERY_RESULT_COUNT = 1;
52  	public static final int RETRY_QUERY_RESULT_COUNT = 60;
53  	public static final int MAX_RETRY_ATTEMPTS = 10;
54  
55  	@Autowired
56  	DeleteConflictFinderService myDeleteConflictFinderService;
57  	@Autowired
58  	DaoConfig myDaoConfig;
59  	@Autowired
60  	protected IResourceLinkDao myResourceLinkDao;
61  	@Autowired
62  	private FhirContext myFhirContext;
63  	@Autowired
64  	protected IInterceptorBroadcaster myInterceptorBroadcaster;
65  
66  	public int validateOkToDelete(DeleteConflictList theDeleteConflicts, ResourceTable theEntity, boolean theForValidate, RequestDetails theRequest) {
67  
68  		// We want the list of resources that are marked to be the same list even as we
69  		// drill into conflict resolution stacks.. this allows us to not get caught by
70  		// circular references
71  		DeleteConflictList newConflicts = new DeleteConflictList(theDeleteConflicts);
72  
73  		// In most cases, there will be no hooks, and so we only need to check if there is at least FIRST_QUERY_RESULT_COUNT conflict and populate that.
74  		// Only in the case where there is a hook do we need to go back and collect larger batches of conflicts for processing.
75  
76  		DeleteConflictOutcome outcome = findAndHandleConflicts(theRequest, newConflicts, theEntity, theForValidate, FIRST_QUERY_RESULT_COUNT);
77  
78  		int retryCount = 0;
79  		while (outcome != null) {
80  			int shouldRetryCount = Math.min(outcome.getShouldRetryCount(), MAX_RETRY_ATTEMPTS);
81  			if (!(retryCount < shouldRetryCount)) break;
82  			newConflicts = new DeleteConflictList();
83  			outcome = findAndHandleConflicts(theRequest, newConflicts, theEntity, theForValidate, RETRY_QUERY_RESULT_COUNT);
84  			++retryCount;
85  		}
86  		theDeleteConflicts.addAll(newConflicts);
87  		return retryCount;
88  	}
89  
90  	private DeleteConflictOutcome findAndHandleConflicts(RequestDetails theRequest, DeleteConflictList theDeleteConflicts, ResourceTable theEntity, boolean theForValidate, int theMinQueryResultCount) {
91  		List<ResourceLink> resultList = myDeleteConflictFinderService.findConflicts(theEntity, theMinQueryResultCount);
92  		if (resultList.isEmpty()) {
93  			return null;
94  		}
95  
96  		return handleConflicts(theRequest, theDeleteConflicts, theEntity, theForValidate, resultList);
97  	}
98  
99  	private DeleteConflictOutcome handleConflicts(RequestDetails theRequest, DeleteConflictList theDeleteConflicts, ResourceTable theEntity, boolean theForValidate, List<ResourceLink> theResultList) {
100 		if (!myDaoConfig.isEnforceReferentialIntegrityOnDelete() && !theForValidate) {
101 			ourLog.debug("Deleting {} resource dependencies which can no longer be satisfied", theResultList.size());
102 			myResourceLinkDao.deleteAll(theResultList);
103 			return null;
104 		}
105 
106 		addConflictsToList(theDeleteConflicts, theEntity, theResultList);
107 
108 		if (theDeleteConflicts.isEmpty()) {
109 			return new DeleteConflictOutcome();
110 		}
111 
112 		// Notify Interceptors about pre-action call
113 		HookParams hooks = new HookParams()
114 			.add(DeleteConflictList.class, theDeleteConflicts)
115 			.add(RequestDetails.class, theRequest)
116 			.addIfMatchesType(ServletRequestDetails.class, theRequest);
117 		return (DeleteConflictOutcome)JpaInterceptorBroadcaster.doCallHooksAndReturnObject(myInterceptorBroadcaster, theRequest, Pointcut.STORAGE_PRESTORAGE_DELETE_CONFLICTS, hooks);
118 	}
119 
120 	private void addConflictsToList(DeleteConflictList theDeleteConflicts, ResourceTable theEntity, List<ResourceLink> theResultList) {
121 		for (ResourceLink link : theResultList) {
122 			IdDt targetId = theEntity.getIdDt();
123 			IdDt sourceId = link.getSourceResource().getIdDt();
124 			String sourcePath = link.getSourcePath();
125 			if (theDeleteConflicts.isResourceIdMarkedForDeletion(sourceId)) {
126 				if (theDeleteConflicts.isResourceIdMarkedForDeletion(targetId)) {
127 					continue;
128 				}
129 			}
130 
131 			theDeleteConflicts.add(new DeleteConflict(sourceId, sourcePath, targetId));
132 		}
133 	}
134 
135 	public void validateDeleteConflictsEmptyOrThrowException(DeleteConflictList theDeleteConflicts) {
136 		if (theDeleteConflicts.isEmpty()) {
137 			return;
138 		}
139 
140 		IBaseOperationOutcome oo = OperationOutcomeUtil.newInstance(myFhirContext);
141 		String firstMsg = null;
142 
143 		Iterator<DeleteConflict> iterator = theDeleteConflicts.iterator();
144 		while (iterator.hasNext()) {
145 			DeleteConflict next = iterator.next();
146 			StringBuilder b = new StringBuilder();
147 			b.append("Unable to delete ");
148 			b.append(next.getTargetId().toUnqualifiedVersionless().getValue());
149 			b.append(" because at least one resource has a reference to this resource. First reference found was resource ");
150 			b.append(next.getSourceId().toUnqualifiedVersionless().getValue());
151 			b.append(" in path ");
152 			b.append(next.getSourcePath());
153 			String msg = b.toString();
154 			if (firstMsg == null) {
155 				firstMsg = msg;
156 			}
157 			OperationOutcomeUtil.addIssue(myFhirContext, oo, BaseHapiFhirDao.OO_SEVERITY_ERROR, msg, null, "processing");
158 		}
159 
160 		throw new ResourceVersionConflictException(firstMsg, oo);
161 	}

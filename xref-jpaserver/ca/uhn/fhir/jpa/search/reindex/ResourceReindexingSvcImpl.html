138 	private void initExecutor() {
139 		// Create the threadpool executor used for reindex jobs
140 		int reindexThreadCount = myDaoConfig.getReindexThreadCount();
141 		RejectedExecutionHandler rejectHandler = new Executors.BlockPolicy();
142 		myTaskExecutor = new ThreadPoolExecutor(0, reindexThreadCount,
143 			0L, TimeUnit.MILLISECONDS,
144 			new LinkedBlockingQueue<>(100),
145 			myReindexingThreadFactory,
146 			rejectHandler
147 		);
148 	}

156 	@Override
157 	@Transactional(Transactional.TxType.REQUIRED)
158 	public void markAllResourcesForReindexing(String theType) {
159 		String typeDesc;
160 		if (isNotBlank(theType)) {
161 			myReindexJobDao.markAllOfTypeAsDeleted(theType);
162 			typeDesc = theType;
163 		} else {
164 			myReindexJobDao.markAllOfTypeAsDeleted();
165 			typeDesc = "(any)";
166 		}
167 
168 		ResourceReindexJobEntity job = new ResourceReindexJobEntity();
169 		job.setResourceType(theType);
170 		job.setThresholdHigh(DateUtils.addMinutes(new Date(), 5));
171 		job = myReindexJobDao.saveAndFlush(job);
172 
173 		ourLog.info("Marking all resources of type {} for reindexing - Got job ID[{}]", typeDesc, job.getId());
174 	}

176 	@Override
177 	@Transactional(Transactional.TxType.NEVER)
178 	@Scheduled(fixedDelay = 10 * DateUtils.MILLIS_PER_SECOND)
179 	public Integer runReindexingPass() {
180 		if (myDaoConfig.isSchedulingDisabled()) {
181 			return null;
182 		}
183 		if (myIndexingLock.tryLock()) {
184 			try {
185 				return doReindexingPassInsideLock();
186 			} finally {
187 				myIndexingLock.unlock();
188 			}
189 		}
190 		return null;
191 	}
192 
193 	private Integer doReindexingPassInsideLock() {
194 		expungeJobsMarkedAsDeleted();
195 		return runReindexJobs();
196 	}
197 
198 	@Override
199 	public int forceReindexingPass() {
200 		myIndexingLock.lock();
201 		try {
202 			return doReindexingPassInsideLock();
203 		} finally {
204 			myIndexingLock.unlock();
205 		}
206 	}

208 	@Override
209 	public void cancelAndPurgeAllJobs() {
210 		ourLog.info("Cancelling and purging all resource reindexing jobs");
211 		myTxTemplate.execute(t -> {
212 			myReindexJobDao.markAllOfTypeAsDeleted();
213 			return null;
214 		});
215 
216 		myTaskExecutor.shutdown();
217 		initExecutor();
218 
219 		expungeJobsMarkedAsDeleted();
220 	}

222 	private int runReindexJobs() {
223 		Collection<ResourceReindexJobEntity> jobs = myTxTemplate.execute(t -> myReindexJobDao.findAll(PageRequest.of(0, 10), false));
224 		assert jobs != null;
225 
226 		int count = 0;
227 		for (ResourceReindexJobEntity next : jobs) {
228 
229 			if (next.getThresholdHigh().getTime() < System.currentTimeMillis()) {
230 				markJobAsDeleted(next);
231 				continue;
232 			}
233 
234 			count += runReindexJob(next);
235 		}
236 		return count;
237 	}

239 	private void markJobAsDeleted(ResourceReindexJobEntity next) {
240 		myTxTemplate.execute(t -> {
241 			myReindexJobDao.markAsDeletedById(next.getId());
242 			return null;
243 		});
244 	}

246 	private int runReindexJob(ResourceReindexJobEntity theJob) {
247 		if (theJob.getSuspendedUntil() != null) {
248 			if (theJob.getSuspendedUntil().getTime() > System.currentTimeMillis()) {
249 				return 0;
250 			}
251 		}
252 
253 		ourLog.info("Performing reindex pass for JOB[{}]", theJob.getId());
254 		StopWatch sw = new StopWatch();
255 		AtomicInteger counter = new AtomicInteger();
256 
257 		// Calculate range
258 		Date low = theJob.getThresholdLow() != null ? theJob.getThresholdLow() : BEGINNING_OF_TIME;
259 		Date high = theJob.getThresholdHigh();
260 
261 		// Query for resources within threshold
262 		Slice<Long> range = myTxTemplate.execute(t -> {
263 			PageRequest page = PageRequest.of(0, 10000);
264 			if (isNotBlank(theJob.getResourceType())) {
265 				return myResourceTableDao.findIdsOfResourcesWithinUpdatedRangeOrderedFromOldest(page, theJob.getResourceType(), low, high);
266 			} else {
267 				return myResourceTableDao.findIdsOfResourcesWithinUpdatedRangeOrderedFromOldest(page, low, high);
268 			}
269 		});
270 		Validate.notNull(range);
271 		int count = range.getNumberOfElements();
272 
273 		// Submit each resource requiring reindexing
274 		List<Future<Date>> futures = range
275 			.stream()
276 			.map(t -> myTaskExecutor.submit(new ResourceReindexingTask(t, counter)))
277 			.collect(Collectors.toList());
278 
279 		Date latestDate = null;
280 		boolean haveMultipleDates = false;
281 		for (Future<Date> next : futures) {
282 			Date nextDate;
283 			try {
284 				nextDate = next.get();
285 			} catch (Exception e) {
286 				ourLog.error("Failure reindexing", e);
287 				Date suspendedUntil = DateUtils.addMinutes(new Date(), 1);
288 				myTxTemplate.execute(t -> {
289 					myReindexJobDao.setSuspendedUntil(suspendedUntil);
290 					return null;
291 				});
292 				return counter.get();
293 			}
294 
295 			if (nextDate != null) {
296 				if (latestDate != null) {
297 					if (latestDate.getTime() != nextDate.getTime()) {
298 						haveMultipleDates = true;
299 					}
300 				}
301 				if (latestDate == null || latestDate.getTime() < nextDate.getTime()) {
302 					latestDate = new Date(nextDate.getTime());
303 				}
304 			}
305 		}
306 
307 		// Just in case we end up in some sort of infinite loop. This shouldn't happen, and couldn't really
308 		// happen unless there were 10000 resources with the exact same update time down to the
309 		// millisecond.
310 		Date newLow;
311 		if (latestDate == null) {
312 			markJobAsDeleted(theJob);
313 			return 0;
314 		}
315 		if (latestDate.getTime() == low.getTime()) {
316 			ourLog.error("Final pass time for reindex JOB[{}] has same ending low value: {}", theJob.getId(), latestDate);
317 			newLow = new Date(latestDate.getTime() + 1);
318 		} else if (!haveMultipleDates) {
319 			newLow = new Date(latestDate.getTime() + 1);
320 		} else {
321 			newLow = latestDate;
322 		}
323 
324 		myTxTemplate.execute(t -> {
325 			myReindexJobDao.setThresholdLow(theJob.getId(), newLow);
326 			return null;
327 		});
328 
329 		ourLog.info("Completed pass of reindex JOB[{}] - Indexed {} resources in {} ({} / sec) - Have indexed until: {}", theJob.getId(), count, sw.toString(), sw.formatThroughput(count, TimeUnit.SECONDS), theJob.getThresholdLow());
330 		return counter.get();
331 	}

333 	private void expungeJobsMarkedAsDeleted() {
334 		myTxTemplate.execute(t -> {
335 			Collection<ResourceReindexJobEntity> toDelete = myReindexJobDao.findAll(PageRequest.of(0, 10), true);
336 			toDelete.forEach(job -> {
337 				ourLog.info("Purging deleted job[{}]", job.getId());
338 				myReindexJobDao.deleteById(job.getId());
339 			});
340 			return null;
341 		});
342 	}

344 	@SuppressWarnings("JpaQlInspection")
345 	private void markResourceAsIndexingFailed(final long theId) {
346 		TransactionTemplate txTemplate = new TransactionTemplate(myTxManager);
347 		txTemplate.setPropagationBehavior(TransactionDefinition.PROPAGATION_REQUIRES_NEW);
348 		txTemplate.execute((TransactionCallback<Void>) theStatus -> {
349 			ourLog.info("Marking resource with PID {} as indexing_failed", new Object[]{theId});
350 
351 			myResourceTableDao.updateIndexStatus(theId, BaseHapiFhirDao.INDEX_STATUS_INDEXING_FAILED);
352 
353 			Query q = myEntityManager.createQuery("DELETE FROM ResourceTag t WHERE t.myResourceId = :id");
354 			q.setParameter("id", theId);
355 			q.executeUpdate();
356 
357 			q = myEntityManager.createQuery("DELETE FROM ResourceIndexedSearchParamCoords t WHERE t.myResourcePid = :id");
358 			q.setParameter("id", theId);
359 			q.executeUpdate();
360 
361 			q = myEntityManager.createQuery("DELETE FROM ResourceIndexedSearchParamDate t WHERE t.myResourcePid = :id");
362 			q.setParameter("id", theId);
363 			q.executeUpdate();
364 
365 			q = myEntityManager.createQuery("DELETE FROM ResourceIndexedSearchParamNumber t WHERE t.myResourcePid = :id");
366 			q.setParameter("id", theId);
367 			q.executeUpdate();
368 
369 			q = myEntityManager.createQuery("DELETE FROM ResourceIndexedSearchParamQuantity t WHERE t.myResourcePid = :id");
370 			q.setParameter("id", theId);
371 			q.executeUpdate();
372 
373 			q = myEntityManager.createQuery("DELETE FROM ResourceIndexedSearchParamString t WHERE t.myResourcePid = :id");
374 			q.setParameter("id", theId);
375 			q.executeUpdate();
376 
377 			q = myEntityManager.createQuery("DELETE FROM ResourceIndexedSearchParamToken t WHERE t.myResourcePid = :id");
378 			q.setParameter("id", theId);
379 			q.executeUpdate();
380 
381 			q = myEntityManager.createQuery("DELETE FROM ResourceIndexedSearchParamUri t WHERE t.myResourcePid = :id");
382 			q.setParameter("id", theId);
383 			q.executeUpdate();
384 
385 			q = myEntityManager.createQuery("DELETE FROM ResourceLink t WHERE t.mySourceResourcePid = :id");
386 			q.setParameter("id", theId);
387 			q.executeUpdate();
388 
389 			q = myEntityManager.createQuery("DELETE FROM ResourceLink t WHERE t.myTargetResourcePid = :id");
390 			q.setParameter("id", theId);
391 			q.executeUpdate();
392 
393 			return null;
394 		});

397 	private class ResourceReindexingTask implements Callable<Date> {
398 		private final Long myNextId;
399 		private final AtomicInteger myCounter;
400 		private Date myUpdated;
401 
402 		ResourceReindexingTask(Long theNextId, AtomicInteger theCounter) {
403 			myNextId = theNextId;
404 			myCounter = theCounter;
405 		}
406 
407 
408 		@SuppressWarnings("unchecked")
409 		private <T extends IBaseResource> void doReindex(ResourceTable theResourceTable, T theResource) {
410 			RuntimeResourceDefinition resourceDefinition = myContext.getResourceDefinition(theResource.getClass());
411 			Class<T> resourceClass = (Class<T>) resourceDefinition.getImplementingClass();
412 			final IFhirResourceDao<T> dao = myDaoRegistry.getResourceDao(resourceClass);
413 			dao.reindex(theResource, theResourceTable);
414 
415 			myCounter.incrementAndGet();
416 		}
417 
418 		@Override
419 		public Date call() {
420 			Throwable reindexFailure;
421 			try {
422 				reindexFailure = myTxTemplate.execute(t -> {
423 					ResourceTable resourceTable = myResourceTableDao.findById(myNextId).orElseThrow(IllegalStateException::new);
424 					myUpdated = resourceTable.getUpdatedDate();
425 
426 					try {
427 						/*
428 						 * This part is because from HAPI 1.5 - 1.6 we changed the format of forced ID to be "type/id" instead of just "id"
429 						 */
430 						ForcedId forcedId = resourceTable.getForcedId();
431 						if (forcedId != null) {
432 							if (isBlank(forcedId.getResourceType())) {
433 								ourLog.info("Updating resource {} forcedId type to {}", forcedId.getForcedId(), resourceTable.getResourceType());
434 								forcedId.setResourceType(resourceTable.getResourceType());
435 								myForcedIdDao.save(forcedId);
436 							}
437 						}
438 
439 						IFhirResourceDao<?> dao = myDaoRegistry.getResourceDao(resourceTable.getResourceType());
440 						IBaseResource resource = dao.toResource(resourceTable, false);
441 						if (resource == null) {
442 							throw new InternalErrorException("Could not find resource version " + resourceTable.getIdDt().toUnqualified().getValue() + " in database");
443 						}
444 						doReindex(resourceTable, resource);
445 						return null;
446 
447 					} catch (Exception e) {
448 						ourLog.error("Failed to index resource {}: {}", resourceTable.getIdDt(), e.toString(), e);
449 						t.setRollbackOnly();
450 						return e;
451 					}
452 				});
453 			} catch (ResourceVersionConflictException e) {
454 				/*
455 				 * We reindex in multiple threads, so it's technically possible that two threads try
456 				 * to index resources that cause a constraint error now (i.e. because a unique index has been
457 				 * added that didn't previously exist). In this case, one of the threads would succeed and
458 				 * not get this error, so we'll let the other one fail and try
459 				 * again later.
460 				 */
461 				ourLog.info("Failed to reindex {} because of a version conflict. Leaving in unindexed state: {}", e.getMessage());
462 				reindexFailure = null;
463 			}
464 
465 			if (reindexFailure != null) {
466 				ourLog.info("Setting resource PID[{}] status to ERRORED", myNextId);
467 				markResourceAsIndexingFailed(myNextId);
468 			}
469 
470 			return myUpdated;
471 		}

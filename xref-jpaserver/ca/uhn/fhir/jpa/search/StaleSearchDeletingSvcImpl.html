28  import com.google.common.annotations.VisibleForTesting;
29  import org.apache.commons.lang3.time.DateUtils;
30  import org.hl7.fhir.dstu3.model.InstantType;
31  import org.springframework.beans.factory.annotation.Autowired;
32  import org.springframework.data.domain.PageRequest;
33  import org.springframework.data.domain.Slice;
34  import org.springframework.scheduling.annotation.Scheduled;
35  import org.springframework.transaction.PlatformTransactionManager;
36  import org.springframework.transaction.TransactionStatus;
37  import org.springframework.transaction.annotation.Propagation;
38  import org.springframework.transaction.annotation.Transactional;
39  import org.springframework.transaction.support.TransactionCallback;
40  import org.springframework.transaction.support.TransactionTemplate;
41  
42  import java.util.Date;
43  
44  /**
45   * Deletes old searches
46   */
47  public class StaleSearchDeletingSvcImpl implements IStaleSearchDeletingSvc {
48  	public static final long DEFAULT_CUTOFF_SLACK = 10 * DateUtils.MILLIS_PER_SECOND;
49  	private static final org.slf4j.Logger ourLog = org.slf4j.LoggerFactory.getLogger(StaleSearchDeletingSvcImpl.class);
50  	private static Long ourNowForUnitTests;
51  	/*
52  	 * We give a bit of extra leeway just to avoid race conditions where a query result
53  	 * is being reused (because a new client request came in with the same params) right before
54  	 * the result is to be deleted
55  	 */
56  	private long myCutoffSlack = DEFAULT_CUTOFF_SLACK;
57  	@Autowired
58  	private DaoConfig myDaoConfig;
59  	@Autowired
60  	private ISearchDao mySearchDao;
61  	@Autowired
62  	private ISearchIncludeDao mySearchIncludeDao;
63  	@Autowired
64  	private ISearchResultDao mySearchResultDao;
65  	@Autowired
66  	private PlatformTransactionManager myTransactionManager;
67  
68  	private void deleteSearch(final Long theSearchPid) {
69  		Search searchToDelete = mySearchDao.findOne(theSearchPid);
70  		if (searchToDelete != null) {
71  			ourLog.info("Deleting search {}/{} - Created[{}] -- Last returned[{}]", searchToDelete.getId(), searchToDelete.getUuid(), new InstantType(searchToDelete.getCreated()), new InstantType(searchToDelete.getSearchLastReturned()));
72  			mySearchIncludeDao.deleteForSearch(searchToDelete.getId());
73  			mySearchResultDao.deleteForSearch(searchToDelete.getId());
74  			mySearchDao.delete(searchToDelete);
75  		}
76  	}
77  
78  	@Override
79  	@Transactional(propagation = Propagation.NOT_SUPPORTED)
80  	public void pollForStaleSearchesAndDeleteThem() {
81  		if (!myDaoConfig.isExpireSearchResults()) {
82  			return;
83  		}
84  
85  		long cutoffMillis = myDaoConfig.getExpireSearchResultsAfterMillis();
86  		if (myDaoConfig.getReuseCachedSearchResultsForMillis() != null) {
87  			cutoffMillis = Math.max(cutoffMillis, myDaoConfig.getReuseCachedSearchResultsForMillis());
88  		}
89  		final Date cutoff = new Date((now() - cutoffMillis) - myCutoffSlack);
90  
91  		if (ourNowForUnitTests != null) {
92  			ourLog.info("Searching for searches which are before {} - now is {}", new InstantType(cutoff), new InstantType(new Date(now())));
93  		}
94  
95  		ourLog.debug("Searching for searches which are before {}", cutoff);
96  
97  		TransactionTemplate tt = new TransactionTemplate(myTransactionManager);
98  		int count = tt.execute(new TransactionCallback<Integer>() {
99  			@Override
100 			public Integer doInTransaction(TransactionStatus theStatus) {
101 				Slice<Long> toDelete = mySearchDao.findWhereLastReturnedBefore(cutoff, new PageRequest(0, 1000));
102 				for (final Long next : toDelete) {
103 					deleteSearch(next);
104 				}
105 				return toDelete.getContent().size();
106 			}
107 		});
108 
109 		if (count > 0) {
110 			long total = tt.execute(new TransactionCallback<Long>() {
111 				@Override
112 				public Long doInTransaction(TransactionStatus theStatus) {
113 					return mySearchDao.count();
114 				}
115 			});
116 			ourLog.info("Deleted {} searches, {} remaining", count, total);
117 		}
118 
119 	}
120 
121 	@Scheduled(fixedDelay = DEFAULT_CUTOFF_SLACK)
122 	@Transactional(propagation = Propagation.NOT_SUPPORTED)
123 	@Override
124 	public synchronized void schedulePollForStaleSearches() {
125 		if (!myDaoConfig.isSchedulingDisabled()) {
126 			pollForStaleSearchesAndDeleteThem();
127 		}
128 	}
129 
130 	@VisibleForTesting
131 	public void setCutoffSlackForUnitTest(long theCutoffSlack) {
132 		myCutoffSlack = theCutoffSlack;
133 	}
134 
135 	private static long now() {
136 		if (ourNowForUnitTests != null) {
137 			return ourNowForUnitTests;
138 		}
139 		return System.currentTimeMillis();
140 	}
141 
142 	/**
143 	 * This is for unit tests only, do not call otherwise
144 	 */
145 	@VisibleForTesting
146 	public static void setNowForUnitTests(Long theNowForUnitTests) {
147 		ourNowForUnitTests = theNowForUnitTests;
148 	}

150 }

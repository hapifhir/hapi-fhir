23  import ca.uhn.fhir.jpa.entity.*;
24  import ca.uhn.fhir.jpa.model.entity.*;
25  import ca.uhn.fhir.util.StopWatch;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  import org.springframework.beans.factory.annotation.Autowired;
29  import org.springframework.stereotype.Service;
30  import org.springframework.transaction.PlatformTransactionManager;
31  import org.springframework.transaction.TransactionDefinition;
32  import org.springframework.transaction.support.TransactionTemplate;
33  
34  import javax.persistence.EntityManager;
35  import javax.persistence.PersistenceContext;
36  import javax.persistence.PersistenceContextType;
37  import java.util.concurrent.atomic.AtomicInteger;
38  
39  @Service
40  public class ExpungeEverythingService {
41  	private static final Logger ourLog = LoggerFactory.getLogger(ExpungeEverythingService.class);
42  	@Autowired
43  	private PlatformTransactionManager myPlatformTransactionManager;
44  	@PersistenceContext(type = PersistenceContextType.TRANSACTION)
45  	protected EntityManager myEntityManager;
46  
47  	void expungeEverything() {
48  
49  		final AtomicInteger counter = new AtomicInteger();
50  
51  		ourLog.info("BEGINNING GLOBAL $expunge");
52  		TransactionTemplate txTemplate = new TransactionTemplate(myPlatformTransactionManager);
53  		txTemplate.setPropagationBehavior(TransactionDefinition.PROPAGATION_REQUIRES_NEW);
54  		txTemplate.execute(t -> {
55  			counter.addAndGet(doExpungeEverythingQuery("UPDATE " + ResourceHistoryTable.class.getSimpleName() + " d SET d.myForcedId = null"));
56  			counter.addAndGet(doExpungeEverythingQuery("UPDATE " + ResourceTable.class.getSimpleName() + " d SET d.myForcedId = null"));
57  			counter.addAndGet(doExpungeEverythingQuery("UPDATE " + TermCodeSystem.class.getSimpleName() + " d SET d.myCurrentVersion = null"));
58  			return null;
59  		});
60  		txTemplate.execute(t -> {
61  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + SearchParamPresent.class.getSimpleName() + " d"));
62  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + ForcedId.class.getSimpleName() + " d"));
63  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + ResourceIndexedSearchParamDate.class.getSimpleName() + " d"));
64  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + ResourceIndexedSearchParamNumber.class.getSimpleName() + " d"));
65  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + ResourceIndexedSearchParamQuantity.class.getSimpleName() + " d"));
66  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + ResourceIndexedSearchParamString.class.getSimpleName() + " d"));
67  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + ResourceIndexedSearchParamToken.class.getSimpleName() + " d"));
68  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + ResourceIndexedSearchParamUri.class.getSimpleName() + " d"));
69  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + ResourceIndexedSearchParamCoords.class.getSimpleName() + " d"));
70  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + ResourceIndexedCompositeStringUnique.class.getSimpleName() + " d"));
71  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + ResourceLink.class.getSimpleName() + " d"));
72  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + SearchResult.class.getSimpleName() + " d"));
73  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + SearchInclude.class.getSimpleName() + " d"));
74  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + TermConceptParentChildLink.class.getSimpleName() + " d"));
75  			return null;
76  		});
77  		txTemplate.execute(t -> {
78  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + TermConceptMapGroupElementTarget.class.getSimpleName() + " d"));
79  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + TermConceptMapGroupElement.class.getSimpleName() + " d"));
80  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + TermConceptMapGroup.class.getSimpleName() + " d"));
81  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + TermConceptMap.class.getSimpleName() + " d"));
82  			return null;
83  		});
84  		txTemplate.execute(t -> {
85  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + TermConceptProperty.class.getSimpleName() + " d"));
86  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + TermConceptDesignation.class.getSimpleName() + " d"));
87  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + TermConcept.class.getSimpleName() + " d"));
88  			for (TermCodeSystem next : myEntityManager.createQuery("SELECT c FROM " + TermCodeSystem.class.getName() + " c", TermCodeSystem.class).getResultList()) {
89  				next.setCurrentVersion(null);
90  				myEntityManager.merge(next);
91  			}
92  			return null;
93  		});
94  		txTemplate.execute(t -> {
95  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + TermCodeSystemVersion.class.getSimpleName() + " d"));
96  			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + TermCodeSystem.class.getSimpleName() + " d"));
97  			return null;
98  		});
99  		txTemplate.execute(t -> {
100 			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + SubscriptionTable.class.getSimpleName() + " d"));
101 			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + ResourceHistoryTag.class.getSimpleName() + " d"));
102 			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + ResourceTag.class.getSimpleName() + " d"));
103 			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + TagDefinition.class.getSimpleName() + " d"));
104 			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + ResourceHistoryTable.class.getSimpleName() + " d"));
105 			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + ResourceTable.class.getSimpleName() + " d"));
106 			counter.addAndGet(doExpungeEverythingQuery("DELETE from " + org.hibernate.search.jpa.Search.class.getSimpleName() + " d"));
107 			return null;
108 		});
109 
110 		ourLog.info("COMPLETED GLOBAL $expunge - Deleted {} rows", counter.get());
111 	}
112 
113 	private int doExpungeEverythingQuery(String theQuery) {
114 		StopWatch sw = new StopWatch();
115 		int outcome = myEntityManager.createQuery(theQuery).executeUpdate();
116 		ourLog.debug("Query affected {} rows in {}: {}", outcome, sw.toString(), theQuery);
117 		return outcome;
118 	}
119 }

102 	private void handleTransactionCreateOrUpdateOutcome(Map<IIdType, IIdType> idSubstitutions, Map<IIdType, DaoMethodOutcome> idToPersistedOutcome, IIdType nextResourceId, DaoMethodOutcome outcome,
103 																		 BUNDLEENTRY newEntry, String theResourceType, IBaseResource theRes, ServletRequestDetails theRequestDetails) {
104 		IIdType newId = outcome.getId().toUnqualifiedVersionless();
105 		IIdType resourceId = isPlaceholder(nextResourceId) ? nextResourceId : nextResourceId.toUnqualifiedVersionless();
106 		if (newId.equals(resourceId) == false) {
107 			idSubstitutions.put(resourceId, newId);
108 			if (isPlaceholder(resourceId)) {
109 				/*
110 				 * The correct way for substitution IDs to be is to be with no resource type, but we'll accept the qualified kind too just to be lenient.
111 				 */
112 				IIdType id = myContext.getVersion().newIdType();
113 				id.setValue(theResourceType + '/' + resourceId.getValue());
114 				idSubstitutions.put(id, newId);
115 			}
116 		}
117 		idToPersistedOutcome.put(newId, outcome);
118 		if (outcome.getCreated().booleanValue()) {
119 			myVersionAdapter.setResponseStatus(newEntry, toStatusString(Constants.STATUS_HTTP_201_CREATED));
120 		} else {
121 			myVersionAdapter.setResponseStatus(newEntry, toStatusString(Constants.STATUS_HTTP_200_OK));
122 		}
123 		Date lastModifier = getLastModified(theRes);
124 		myVersionAdapter.setResponseLastModified(newEntry, lastModifier);
125 
126 		if (theRequestDetails != null) {
127 			if (outcome.getResource() != null) {
128 				String prefer = theRequestDetails.getHeader(Constants.HEADER_PREFER);
129 				PreferReturnEnum preferReturn = RestfulServerUtils.parsePreferHeader(prefer);
130 				if (preferReturn != null) {
131 					if (preferReturn == PreferReturnEnum.REPRESENTATION) {
132 						myVersionAdapter.setResource(newEntry, outcome.getResource());
133 					}
134 				}
135 			}
136 		}
137 
138 	}

144 	private String performIdSubstitutionsInMatchUrl(Map<IIdType, IIdType> theIdSubstitutions, String theMatchUrl) {
145 		String matchUrl = theMatchUrl;
146 		if (isNotBlank(matchUrl)) {
147 			for (Map.Entry<IIdType, IIdType> nextSubstitutionEntry : theIdSubstitutions.entrySet()) {
148 				IIdType nextTemporaryId = nextSubstitutionEntry.getKey();
149 				IIdType nextReplacementId = nextSubstitutionEntry.getValue();
150 				String nextTemporaryIdPart = nextTemporaryId.getIdPart();
151 				String nextReplacementIdPart = nextReplacementId.getValueAsString();
152 				if (isUrn(nextTemporaryId) && nextTemporaryIdPart.length() > URN_PREFIX.length()) {
153 					matchUrl = matchUrl.replace(nextTemporaryIdPart, nextReplacementIdPart);
154 					matchUrl = matchUrl.replace(UrlUtil.escapeUrlParam(nextTemporaryIdPart), nextReplacementIdPart);
155 				}
156 			}
157 		}
158 		return matchUrl;
159 	}

177 		return processTransactionAsSubRequest((ServletRequestDetails) theRequestDetails, theRequest, actionName);

180 	private BUNDLE processTransactionAsSubRequest(ServletRequestDetails theRequestDetails, BUNDLE theRequest, String theActionName) {
181 		BaseHapiFhirDao.markRequestAsProcessingSubRequest(theRequestDetails);
182 		try {
183 			return processTransaction(theRequestDetails, theRequest, theActionName);
184 		} finally {
185 			BaseHapiFhirDao.clearRequestAsProcessingSubRequest(theRequestDetails);
186 		}
187 	}

189 	private BUNDLE batch(final RequestDetails theRequestDetails, BUNDLE theRequest) {
190 		ourLog.info("Beginning batch with {} resources", myVersionAdapter.getEntries(theRequest).size());
191 		long start = System.currentTimeMillis();
192 
193 		TransactionTemplate txTemplate = new TransactionTemplate(myTxManager);
194 		txTemplate.setPropagationBehavior(TransactionDefinition.PROPAGATION_REQUIRES_NEW);
195 
196 		BUNDLE resp = myVersionAdapter.createBundle(org.hl7.fhir.r4.model.Bundle.BundleType.BATCHRESPONSE.toCode());
197 
198 		/*
199 		 * For batch, we handle each entry as a mini-transaction in its own database transaction so that if one fails, it doesn't prevent others
200 		 */
201 
202 		for (final BUNDLEENTRY nextRequestEntry : myVersionAdapter.getEntries(theRequest)) {
203 
204 			BaseServerResponseExceptionHolder caughtEx = new BaseServerResponseExceptionHolder();
205 
206 			TransactionCallback<BUNDLE> callback = theStatus -> {
207 				BUNDLE subRequestBundle = myVersionAdapter.createBundle(org.hl7.fhir.r4.model.Bundle.BundleType.TRANSACTION.toCode());
208 				myVersionAdapter.addEntry(subRequestBundle, nextRequestEntry);
209 
210 				BUNDLE subResponseBundle = processTransactionAsSubRequest((ServletRequestDetails) theRequestDetails, subRequestBundle, "Batch sub-request");
211 				return subResponseBundle;
212 			};
213 
214 			try {
215 				// FIXME: this doesn't need to be a callback
216 				BUNDLE nextResponseBundle = callback.doInTransaction(null);
217 
218 				BUNDLEENTRY subResponseEntry = myVersionAdapter.getEntries(nextResponseBundle).get(0);
219 				myVersionAdapter.addEntry(resp, subResponseEntry);
220 
221 				/*
222 				 * If the individual entry didn't have a resource in its response, bring the sub-transaction's OperationOutcome across so the client can see it
223 				 */
224 				if (myVersionAdapter.getResource(subResponseEntry) == null) {
225 					BUNDLEENTRY nextResponseBundleFirstEntry = myVersionAdapter.getEntries(nextResponseBundle).get(0);
226 					myVersionAdapter.setResource(subResponseEntry, myVersionAdapter.getResource(nextResponseBundleFirstEntry));
227 				}
228 
229 			} catch (BaseServerResponseException e) {
230 				caughtEx.setException(e);
231 			} catch (Throwable t) {
232 				ourLog.error("Failure during BATCH sub transaction processing", t);
233 				caughtEx.setException(new InternalErrorException(t));
234 			}
235 
236 			if (caughtEx.getException() != null) {
237 				BUNDLEENTRY nextEntry = myVersionAdapter.addEntry(resp);
238 
239 				populateEntryWithOperationOutcome(caughtEx.getException(), nextEntry);
240 
241 				myVersionAdapter.setResponseStatus(nextEntry, toStatusString(caughtEx.getException().getStatusCode()));
242 			}
243 
244 		}
245 
246 		long delay = System.currentTimeMillis() - start;
247 		ourLog.info("Batch completed in {}ms", new Object[]{delay});
248 
249 		return resp;
250 	}

252 	private BUNDLE processTransaction(final ServletRequestDetails theRequestDetails, final BUNDLE theRequest, final String theActionName) {
253 		validateDependencies();
254 
255 		String transactionType = myVersionAdapter.getBundleType(theRequest);
256 		if (org.hl7.fhir.r4.model.Bundle.BundleType.BATCH.toCode().equals(transactionType)) {
257 			return batch(theRequestDetails, theRequest);
258 		}
259 
260 		if (transactionType == null) {
261 			String message = "Transaction Bundle did not specify valid Bundle.type, assuming " + Bundle.BundleType.TRANSACTION.toCode();
262 			ourLog.warn(message);
263 			transactionType = org.hl7.fhir.r4.model.Bundle.BundleType.TRANSACTION.toCode();
264 		}
265 		if (!org.hl7.fhir.r4.model.Bundle.BundleType.TRANSACTION.toCode().equals(transactionType)) {
266 			throw new InvalidRequestException("Unable to process transaction where incoming Bundle.type = " + transactionType);
267 		}
268 
269 		ourLog.debug("Beginning {} with {} resources", theActionName, myVersionAdapter.getEntries(theRequest).size());
270 
271 		final Date updateTime = new Date();
272 		final StopWatch transactionStopWatch = new StopWatch();
273 
274 		final Set<IIdType> allIds = new LinkedHashSet<>();
275 		final Map<IIdType, IIdType> idSubstitutions = new HashMap<>();
276 		final Map<IIdType, DaoMethodOutcome> idToPersistedOutcome = new HashMap<>();
277 		List<BUNDLEENTRY> requestEntries = myVersionAdapter.getEntries(theRequest);
278 
279 		// Do all entries have a verb?
280 		for (int i = 0; i < myVersionAdapter.getEntries(theRequest).size(); i++) {
281 			BUNDLEENTRY nextReqEntry = requestEntries.get(i);
282 			String verb = myVersionAdapter.getEntryRequestVerb(nextReqEntry);
283 			if (verb == null || !isValidVerb(verb)) {
284 				throw new InvalidRequestException(myContext.getLocalizer().getMessage(BaseHapiFhirSystemDao.class, "transactionEntryHasInvalidVerb", verb, i));
285 			}
286 		}
287 
288 		/*
289 		 * We want to execute the transaction request bundle elements in the order
290 		 * specified by the FHIR specification (see TransactionSorter) so we save the
291 		 * original order in the request, then sort it.
292 		 *
293 		 * Entries with a type of GET are removed from the bundle so that they
294 		 * can be processed at the very end. We do this because the incoming resources
295 		 * are saved in a two-phase way in order to deal with interdependencies, and
296 		 * we want the GET processing to use the final indexing state
297 		 */
298 		final BUNDLE response = myVersionAdapter.createBundle(org.hl7.fhir.r4.model.Bundle.BundleType.TRANSACTIONRESPONSE.toCode());
299 		List<BUNDLEENTRY> getEntries = new ArrayList<>();
300 		final IdentityHashMap<BUNDLEENTRY, Integer> originalRequestOrder = new IdentityHashMap<>();
301 		for (int i = 0; i < requestEntries.size(); i++) {
302 			originalRequestOrder.put(requestEntries.get(i), i);
303 			myVersionAdapter.addEntry(response);
304 			if (myVersionAdapter.getEntryRequestVerb(requestEntries.get(i)).equals("GET")) {
305 				getEntries.add(requestEntries.get(i));
306 			}
307 		}
308 
309 		/*
310 		 * See FhirSystemDaoDstu3Test#testTransactionWithPlaceholderIdInMatchUrl
311 		 * Basically if the resource has a match URL that references a placeholder,
312 		 * we try to handle the resource with the placeholder first.
313 		 */
314 		Set<String> placeholderIds = new HashSet<>();
315 		final List<BUNDLEENTRY> entries = requestEntries;
316 		for (BUNDLEENTRY nextEntry : entries) {
317 			String fullUrl = myVersionAdapter.getFullUrl(nextEntry);
318 			if (isNotBlank(fullUrl) && fullUrl.startsWith(URN_PREFIX)) {
319 				placeholderIds.add(fullUrl);
320 			}
321 		}
322 		Collections.sort(entries, new TransactionSorter(placeholderIds));
323 
324 		/*
325 		 * All of the write operations in the transaction (PUT, POST, etc.. basically anything
326 		 * except GET) are performed in their own database transaction before we do the reads.
327 		 * We do this because the reads (specifically the searches) often spawn their own
328 		 * secondary database transaction and if we allow that within the primary
329 		 * database transaction we can end up with deadlocks if the server is under
330 		 * heavy load with lots of concurrent transactions using all available
331 		 * database connections.
332 		 */
333 		TransactionTemplate txManager = new TransactionTemplate(myTxManager);
334 		Map<BUNDLEENTRY, ResourceTable> entriesToProcess = txManager.execute(status -> {
335 			Map<BUNDLEENTRY, ResourceTable> retVal = doTransactionWriteOperations(theRequestDetails, theActionName, updateTime, allIds, idSubstitutions, idToPersistedOutcome, response, originalRequestOrder, entries, transactionStopWatch);
336 
337 			transactionStopWatch.startTask("Commit writes to database");
338 			return retVal;
339 		});
340 		transactionStopWatch.endCurrentTask();
341 
342 		for (Map.Entry<BUNDLEENTRY, ResourceTable> nextEntry : entriesToProcess.entrySet()) {
343 			String responseLocation = nextEntry.getValue().getIdDt().toUnqualified().getValue();
344 			String responseEtag = nextEntry.getValue().getIdDt().getVersionIdPart();
345 			myVersionAdapter.setResponseLocation(nextEntry.getKey(), responseLocation);
346 			myVersionAdapter.setResponseETag(nextEntry.getKey(), responseEtag);
347 		}
348 
349 		/*
350 		 * Loop through the request and process any entries of type GET
351 		 */
352 		if (getEntries.size() > 0) {
353 			transactionStopWatch.startTask("Process " + getEntries.size() + " GET entries");
354 		}
355 		for (BUNDLEENTRY nextReqEntry : getEntries) {
356 			Integer originalOrder = originalRequestOrder.get(nextReqEntry);
357 			BUNDLEENTRY nextRespEntry = myVersionAdapter.getEntries(response).get(originalOrder);
358 
359 			ServletSubRequestDetails requestDetails = new ServletSubRequestDetails();
360 			requestDetails.setServletRequest(theRequestDetails.getServletRequest());
361 			requestDetails.setRequestType(RequestTypeEnum.GET);
362 			requestDetails.setServer(theRequestDetails.getServer());
363 
364 			String url = extractTransactionUrlOrThrowException(nextReqEntry, "GET");
365 
366 			int qIndex = url.indexOf('?');
367 			ArrayListMultimap<String, String> paramValues = ArrayListMultimap.create();
368 			requestDetails.setParameters(new HashMap<>());
369 			if (qIndex != -1) {
370 				String params = url.substring(qIndex);
371 				List<NameValuePair> parameters = BaseHapiFhirDao.translateMatchUrl(params);
372 				for (NameValuePair next : parameters) {
373 					paramValues.put(next.getName(), next.getValue());
374 				}
375 				for (Map.Entry<String, Collection<String>> nextParamEntry : paramValues.asMap().entrySet()) {
376 					String[] nextValue = nextParamEntry.getValue().toArray(new String[nextParamEntry.getValue().size()]);
377 					requestDetails.addParameter(nextParamEntry.getKey(), nextValue);
378 				}
379 				url = url.substring(0, qIndex);
380 			}
381 
382 			requestDetails.setRequestPath(url);
383 			requestDetails.setFhirServerBase(theRequestDetails.getFhirServerBase());
384 
385 			theRequestDetails.getServer().populateRequestDetailsFromRequestPath(requestDetails, url);
386 			BaseMethodBinding<?> method = theRequestDetails.getServer().determineResourceMethod(requestDetails, url);
387 			if (method == null) {
388 				throw new IllegalArgumentException("Unable to handle GET " + url);
389 			}
390 
391 			if (isNotBlank(myVersionAdapter.getEntryRequestIfMatch(nextReqEntry))) {
392 				requestDetails.addHeader(Constants.HEADER_IF_MATCH, myVersionAdapter.getEntryRequestIfMatch(nextReqEntry));
393 			}
394 			if (isNotBlank(myVersionAdapter.getEntryRequestIfNoneExist(nextReqEntry))) {
395 				requestDetails.addHeader(Constants.HEADER_IF_NONE_EXIST, myVersionAdapter.getEntryRequestIfNoneExist(nextReqEntry));
396 			}
397 			if (isNotBlank(myVersionAdapter.getEntryRequestIfNoneMatch(nextReqEntry))) {
398 				requestDetails.addHeader(Constants.HEADER_IF_NONE_MATCH, myVersionAdapter.getEntryRequestIfNoneMatch(nextReqEntry));
399 			}
400 
401 			Validate.isTrue(method instanceof BaseResourceReturningMethodBinding, "Unable to handle GET {}", url);
402 			try {
403 				IBaseResource resource = ((BaseResourceReturningMethodBinding) method).doInvokeServer(theRequestDetails.getServer(), requestDetails);
404 				if (paramValues.containsKey(Constants.PARAM_SUMMARY) || paramValues.containsKey(Constants.PARAM_CONTENT)) {
405 					resource = filterNestedBundle(requestDetails, resource);
406 				}
407 				myVersionAdapter.setResource(nextRespEntry, resource);
408 				myVersionAdapter.setResponseStatus(nextRespEntry, toStatusString(Constants.STATUS_HTTP_200_OK));
409 			} catch (NotModifiedException e) {
410 				myVersionAdapter.setResponseStatus(nextRespEntry, toStatusString(Constants.STATUS_HTTP_304_NOT_MODIFIED));
411 			} catch (BaseServerResponseException e) {
412 				ourLog.info("Failure processing transaction GET {}: {}", url, e.toString());
413 				myVersionAdapter.setResponseStatus(nextRespEntry, toStatusString(e.getStatusCode()));
414 				populateEntryWithOperationOutcome(e, nextRespEntry);
415 			}
416 
417 		}
418 		transactionStopWatch.endCurrentTask();
419 
420 		ourLog.info("Transaction timing:\n{}", transactionStopWatch.formatTaskDurations());
421 
422 		return response;
423 	}

426 		try {
427 			return org.hl7.fhir.r4.model.Bundle.HTTPVerb.fromCode(theVerb) != null;
428 		} catch (FHIRException theE) {
429 			return false;
430 		}

434 	 * This method is called for nested bundles (e.g. if we received a transaction with an entry that
435 	 * was a GET search, this method is called on the bundle for the search result, that will be placed in the
436 	 * outer bundle). This method applies the _summary and _content parameters to the output of
437 	 * that bundle.
438 	 * <p>
439 	 * TODO: This isn't the most efficient way of doing this.. hopefully we can come up with something better in the future.

442 		IParser p = myContext.newJsonParser();
443 		RestfulServerUtils.configureResponseParser(theRequestDetails, p);
444 		return p.parseResource(theResource.getClass(), p.encodeResourceToString(theResource));
445 	}

462 	private Map<BUNDLEENTRY, ResourceTable> doTransactionWriteOperations(ServletRequestDetails theRequestDetails, String theActionName, Date theUpdateTime, Set<IIdType> theAllIds,
463 																								Map<IIdType, IIdType> theIdSubstitutions, Map<IIdType, DaoMethodOutcome> theIdToPersistedOutcome, BUNDLE theResponse, IdentityHashMap<BUNDLEENTRY, Integer> theOriginalRequestOrder, List<BUNDLEENTRY> theEntries, StopWatch theTransactionStopWatch) {
464 		Set<String> deletedResources = new HashSet<>();
465 		List<DeleteConflict> deleteConflicts = new ArrayList<>();
466 		Map<BUNDLEENTRY, ResourceTable> entriesToProcess = new IdentityHashMap<>();
467 		Set<ResourceTable> nonUpdatedEntities = new HashSet<>();
468 		Set<ResourceTable> updatedEntities = new HashSet<>();
469 		Map<String, Class<? extends IBaseResource>> conditionalRequestUrls = new HashMap<>();
470 
471 		/*
472 		 * Loop through the request and process any entries of type
473 		 * PUT, POST or DELETE
474 		 */
475 		for (int i = 0; i < theEntries.size(); i++) {
476 
477 			if (i % 100 == 0) {
478 				ourLog.debug("Processed {} non-GET entries out of {}", i, theEntries.size());
479 			}
480 
481 			BUNDLEENTRY nextReqEntry = theEntries.get(i);
482 			IBaseResource res = myVersionAdapter.getResource(nextReqEntry);
483 			IIdType nextResourceId = null;
484 			if (res != null) {
485 
486 				nextResourceId = res.getIdElement();
487 
488 				if (!nextResourceId.hasIdPart()) {
489 					if (isNotBlank(myVersionAdapter.getFullUrl(nextReqEntry))) {
490 						nextResourceId = newIdType(myVersionAdapter.getFullUrl(nextReqEntry));
491 					}
492 				}
493 
494 				if (nextResourceId.hasIdPart() && nextResourceId.getIdPart().matches("[a-zA-Z]+\\:.*") && !isPlaceholder(nextResourceId)) {
495 					throw new InvalidRequestException("Invalid placeholder ID found: " + nextResourceId.getIdPart() + " - Must be of the form 'urn:uuid:[uuid]' or 'urn:oid:[oid]'");
496 				}
497 
498 				if (nextResourceId.hasIdPart() && !nextResourceId.hasResourceType() && !isPlaceholder(nextResourceId)) {
499 					nextResourceId = newIdType(toResourceName(res.getClass()), nextResourceId.getIdPart());
500 					res.setId(nextResourceId);
501 				}
502 
503 				/*
504 				 * Ensure that the bundle doesn't have any duplicates, since this causes all kinds of weirdness
505 				 */
506 				if (isPlaceholder(nextResourceId)) {
507 					if (!theAllIds.add(nextResourceId)) {
508 						throw new InvalidRequestException(myContext.getLocalizer().getMessage(BaseHapiFhirSystemDao.class, "transactionContainsMultipleWithDuplicateId", nextResourceId));
509 					}
510 				} else if (nextResourceId.hasResourceType() && nextResourceId.hasIdPart()) {
511 					IIdType nextId = nextResourceId.toUnqualifiedVersionless();
512 					if (!theAllIds.add(nextId)) {
513 						throw new InvalidRequestException(myContext.getLocalizer().getMessage(BaseHapiFhirSystemDao.class, "transactionContainsMultipleWithDuplicateId", nextId));
514 					}
515 				}
516 
517 			}
518 
519 			String verb = myVersionAdapter.getEntryRequestVerb(nextReqEntry);
520 			String resourceType = res != null ? myContext.getResourceDefinition(res).getName() : null;
521 			Integer order = theOriginalRequestOrder.get(nextReqEntry);
522 			BUNDLEENTRY nextRespEntry = myVersionAdapter.getEntries(theResponse).get(order);
523 
524 			theTransactionStopWatch.startTask("Bundle.entry[" + i + "]: " + verb + " " + defaultString(resourceType));
525 
526 			switch (verb) {
527 				case "POST": {
528 					// CREATE
529 					@SuppressWarnings("rawtypes")
530 					IFhirResourceDao resourceDao = getDaoOrThrowException(res.getClass());
531 					res.setId((String) null);
532 					DaoMethodOutcome outcome;
533 					String matchUrl = myVersionAdapter.getEntryRequestIfNoneExist(nextReqEntry);
534 					matchUrl = performIdSubstitutionsInMatchUrl(theIdSubstitutions, matchUrl);
535 					outcome = resourceDao.create(res, matchUrl, false, theRequestDetails);
536 					if (nextResourceId != null) {
537 						handleTransactionCreateOrUpdateOutcome(theIdSubstitutions, theIdToPersistedOutcome, nextResourceId, outcome, nextRespEntry, resourceType, res, theRequestDetails);
538 					}
539 					entriesToProcess.put(nextRespEntry, outcome.getEntity());
540 					if (outcome.getCreated() == false) {
541 						nonUpdatedEntities.add(outcome.getEntity());
542 					} else {
543 						if (isNotBlank(matchUrl)) {
544 							conditionalRequestUrls.put(matchUrl, res.getClass());
545 						}
546 					}
547 
548 					break;
549 				}
550 				case "DELETE": {
551 					// DELETE
552 					String url = extractTransactionUrlOrThrowException(nextReqEntry, verb);
553 					UrlUtil.UrlParts parts = UrlUtil.parseUrl(url);
554 					ca.uhn.fhir.jpa.dao.IFhirResourceDao<? extends IBaseResource> dao = toDao(parts, verb, url);
555 					int status = Constants.STATUS_HTTP_204_NO_CONTENT;
556 					if (parts.getResourceId() != null) {
557 						IIdType deleteId = newIdType(parts.getResourceType(), parts.getResourceId());
558 						if (!deletedResources.contains(deleteId.getValueAsString())) {
559 							DaoMethodOutcome outcome = dao.delete(deleteId, deleteConflicts, theRequestDetails);
560 							if (outcome.getEntity() != null) {
561 								deletedResources.add(deleteId.getValueAsString());
562 								entriesToProcess.put(nextRespEntry, outcome.getEntity());
563 							}
564 						}
565 					} else {
566 						String matchUrl = parts.getResourceType() + '?' + parts.getParams();
567 						matchUrl = performIdSubstitutionsInMatchUrl(theIdSubstitutions, matchUrl);
568 						DeleteMethodOutcome deleteOutcome = dao.deleteByUrl(matchUrl, deleteConflicts, theRequestDetails);
569 						List<ResourceTable> allDeleted = deleteOutcome.getDeletedEntities();
570 						for (ResourceTable deleted : allDeleted) {
571 							deletedResources.add(deleted.getIdDt().toUnqualifiedVersionless().getValueAsString());
572 						}
573 						if (allDeleted.isEmpty()) {
574 							status = Constants.STATUS_HTTP_204_NO_CONTENT;
575 						}
576 
577 						myVersionAdapter.setResponseOutcome(nextRespEntry, deleteOutcome.getOperationOutcome());
578 					}
579 
580 					myVersionAdapter.setResponseStatus(nextRespEntry, toStatusString(status));
581 
582 					break;
583 				}
584 				case "PUT": {
585 					// UPDATE
586 					@SuppressWarnings("rawtypes")
587 					IFhirResourceDao resourceDao = getDaoOrThrowException(res.getClass());
588 
589 					String url = extractTransactionUrlOrThrowException(nextReqEntry, verb);
590 
591 					DaoMethodOutcome outcome;
592 					UrlUtil.UrlParts parts = UrlUtil.parseUrl(url);
593 					if (isNotBlank(parts.getResourceId())) {
594 						String version = null;
595 						if (isNotBlank(myVersionAdapter.getEntryRequestIfMatch(nextReqEntry))) {
596 							version = ParameterUtil.parseETagValue(myVersionAdapter.getEntryRequestIfMatch(nextReqEntry));
597 						}
598 						res.setId(newIdType(parts.getResourceType(), parts.getResourceId(), version));
599 						outcome = resourceDao.update(res, null, false, theRequestDetails);
600 					} else {
601 						res.setId((String) null);
602 						String matchUrl;
603 						if (isNotBlank(parts.getParams())) {
604 							matchUrl = parts.getResourceType() + '?' + parts.getParams();
605 						} else {
606 							matchUrl = parts.getResourceType();
607 						}
608 						matchUrl = performIdSubstitutionsInMatchUrl(theIdSubstitutions, matchUrl);
609 						outcome = resourceDao.update(res, matchUrl, false, theRequestDetails);
610 						if (Boolean.TRUE.equals(outcome.getCreated())) {
611 							conditionalRequestUrls.put(matchUrl, res.getClass());
612 						}
613 					}
614 
615 					if (outcome.getCreated() == Boolean.FALSE) {
616 						updatedEntities.add(outcome.getEntity());
617 					}
618 
619 					handleTransactionCreateOrUpdateOutcome(theIdSubstitutions, theIdToPersistedOutcome, nextResourceId, outcome, nextRespEntry, resourceType, res, theRequestDetails);
620 					entriesToProcess.put(nextRespEntry, outcome.getEntity());
621 					break;
622 				}
623 				case "GET":
624 				default:
625 					break;
626 
627 			}
628 
629 			theTransactionStopWatch.endCurrentTask();
630 		}
631 
632 
633 		/*
634 		 * Make sure that there are no conflicts from deletions. E.g. we can't delete something
635 		 * if something else has a reference to it.. Unless the thing that has a reference to it
636 		 * was also deleted as a part of this transaction, which is why we check this now at the
637 		 * end.
638 		 */
639 
640 		deleteConflicts.removeIf(next ->
641 			deletedResources.contains(next.getTargetId().toUnqualifiedVersionless().getValue()));
642 		myDao.validateDeleteConflictsEmptyOrThrowException(deleteConflicts);
643 
644 		/*
645 		 * Perform ID substitutions and then index each resource we have saved
646 		 */
647 
648 		FhirTerser terser = myContext.newTerser();
649 		theTransactionStopWatch.startTask("Index " + theIdToPersistedOutcome.size() + " resources");
650 		for (DaoMethodOutcome nextOutcome : theIdToPersistedOutcome.values()) {
651 			IBaseResource nextResource = nextOutcome.getResource();
652 			if (nextResource == null) {
653 				continue;
654 			}
655 
656 			// References
657 			List<ResourceReferenceInfo> allRefs = terser.getAllResourceReferences(nextResource);
658 			for (ResourceReferenceInfo nextRef : allRefs) {
659 				IIdType nextId = nextRef.getResourceReference().getReferenceElement();
660 				if (!nextId.hasIdPart()) {
661 					continue;
662 				}
663 				if (theIdSubstitutions.containsKey(nextId)) {
664 					IIdType newId = theIdSubstitutions.get(nextId);
665 					ourLog.debug(" * Replacing resource ref {} with {}", nextId, newId);
666 					nextRef.getResourceReference().setReference(newId.getValue());
667 				} else if (nextId.getValue().startsWith("urn:")) {
668 					throw new InvalidRequestException("Unable to satisfy placeholder ID " + nextId.getValue() + " found in element named '" + nextRef.getName() + "' within resource of type: " + nextResource.getIdElement().getResourceType());
669 				} else {
670 					ourLog.debug(" * Reference [{}] does not exist in bundle", nextId);
671 				}
672 			}
673 
674 			// URIs
675 			Class<? extends IPrimitiveType<?>> uriType = (Class<? extends IPrimitiveType<?>>) myContext.getElementDefinition("uri").getImplementingClass();
676 			List<? extends IPrimitiveType<?>> allUris = terser.getAllPopulatedChildElementsOfType(nextResource, uriType);
677 			for (IPrimitiveType<?> nextRef : allUris) {
678 				if (nextRef instanceof IIdType) {
679 					continue; // No substitution on the resource ID itself!
680 				}
681 				IIdType nextUriString = newIdType(nextRef.getValueAsString());
682 				if (theIdSubstitutions.containsKey(nextUriString)) {
683 					IIdType newId = theIdSubstitutions.get(nextUriString);
684 					ourLog.debug(" * Replacing resource ref {} with {}", nextUriString, newId);
685 					nextRef.setValueAsString(newId.getValue());
686 				} else {
687 					ourLog.debug(" * Reference [{}] does not exist in bundle", nextUriString);
688 				}
689 			}
690 
691 			IPrimitiveType<Date> deletedInstantOrNull = ResourceMetadataKeyEnum.DELETED_AT.get((IAnyResource) nextResource);
692 			Date deletedTimestampOrNull = deletedInstantOrNull != null ? deletedInstantOrNull.getValue() : null;
693 
694 			if (updatedEntities.contains(nextOutcome.getEntity())) {
695 				myDao.updateInternal(theRequestDetails, nextResource, true, false, theRequestDetails, nextOutcome.getEntity(), nextResource.getIdElement(), nextOutcome.getPreviousResource());
696 			} else if (!nonUpdatedEntities.contains(nextOutcome.getEntity())) {
697 				myDao.updateEntity(theRequestDetails, nextResource, nextOutcome.getEntity(), deletedTimestampOrNull, true, false, theUpdateTime, false, true);
698 			}
699 		}
700 
701 		theTransactionStopWatch.endCurrentTask();
702 		theTransactionStopWatch.startTask("Flush writes to database");
703 
704 		flushJpaSession();
705 
706 		theTransactionStopWatch.endCurrentTask();
707 		if (conditionalRequestUrls.size() > 0) {
708 			theTransactionStopWatch.startTask("Check for conflicts in conditional resources");
709 		}
710 
711 		/*
712 		 * Double check we didn't allow any duplicates we shouldn't have
713 		 */
714 		for (Map.Entry<String, Class<? extends IBaseResource>> nextEntry : conditionalRequestUrls.entrySet()) {
715 			String matchUrl = nextEntry.getKey();
716 			Class<? extends IBaseResource> resType = nextEntry.getValue();
717 			if (isNotBlank(matchUrl)) {
718 				IFhirResourceDao<?> resourceDao = myDao.getDao(resType);
719 				Set<Long> val = resourceDao.processMatchUrl(matchUrl);
720 				if (val.size() > 1) {
721 					throw new InvalidRequestException(
722 						"Unable to process " + theActionName + " - Request would cause multiple resources to match URL: \"" + matchUrl + "\". Does transaction request contain duplicates?");
723 				}
724 			}
725 		}
726 
727 		theTransactionStopWatch.endCurrentTask();
728 
729 		for (IIdType next : theAllIds) {
730 			IIdType replacement = theIdSubstitutions.get(next);
731 			if (replacement == null) {
732 				continue;
733 			}
734 			if (replacement.equals(next)) {
735 				continue;
736 			}
737 			ourLog.debug("Placeholder resource ID \"{}\" was replaced with permanent ID \"{}\"", next, replacement);
738 		}
739 		return entriesToProcess;
740 	}

861 	/**
862 	 * Transaction Order, per the spec:
863 	 * <p>
864 	 * Process any DELETE interactions
865 	 * Process any POST interactions
866 	 * Process any PUT interactions
867 	 * Process any GET interactions
868 	 */
869 	//@formatter:off
870 	public class TransactionSorter implements Comparator<BUNDLEENTRY> {
871 
872 		private Set<String> myPlaceholderIds;
873 
874 		public TransactionSorter(Set<String> thePlaceholderIds) {
875 			myPlaceholderIds = thePlaceholderIds;
876 		}
877 
878 		@Override
879 		public int compare(BUNDLEENTRY theO1, BUNDLEENTRY theO2) {
880 			int o1 = toOrder(theO1);
881 			int o2 = toOrder(theO2);
882 
883 			if (o1 == o2) {
884 				String matchUrl1 = toMatchUrl(theO1);
885 				String matchUrl2 = toMatchUrl(theO2);
886 				if (isBlank(matchUrl1) && isBlank(matchUrl2)) {
887 					return 0;
888 				}
889 				if (isBlank(matchUrl1)) {
890 					return -1;
891 				}
892 				if (isBlank(matchUrl2)) {
893 					return 1;
894 				}
895 
896 				boolean match1containsSubstitutions = false;
897 				boolean match2containsSubstitutions = false;
898 				for (String nextPlaceholder : myPlaceholderIds) {
899 					if (matchUrl1.contains(nextPlaceholder)) {
900 						match1containsSubstitutions = true;
901 					}
902 					if (matchUrl2.contains(nextPlaceholder)) {
903 						match2containsSubstitutions = true;
904 					}
905 				}
906 
907 				if (match1containsSubstitutions && match2containsSubstitutions) {
908 					return 0;
909 				}
910 				if (!match1containsSubstitutions && !match2containsSubstitutions) {
911 					return 0;
912 				}
913 				if (match1containsSubstitutions) {
914 					return 1;
915 				} else {
916 					return -1;
917 				}
918 			}
919 
920 			return o1 - o2;
921 		}
922 
923 		private String toMatchUrl(BUNDLEENTRY theEntry) {
924 			String verb = myVersionAdapter.getEntryRequestVerb(theEntry);
925 			if (verb.equals("POST")) {
926 				return myVersionAdapter.getEntryIfNoneExist(theEntry);
927 			}
928 			if (verb.equals("PUT") || verb.equals("DELETE")) {
929 				String url = extractTransactionUrlOrThrowException(theEntry, verb);
930 				UrlUtil.UrlParts parts = UrlUtil.parseUrl(url);
931 				if (isBlank(parts.getResourceId())) {
932 					return parts.getResourceType() + '?' + parts.getParams();
933 				}
934 			}
935 			return null;
936 		}
937 
938 		private int toOrder(BUNDLEENTRY theO1) {
939 			int o1 = 0;
940 			if (myVersionAdapter.getEntryRequestVerb(theO1) != null) {
941 				switch (myVersionAdapter.getEntryRequestVerb(theO1)) {
942 					case "DELETE":
943 						o1 = 1;
944 						break;
945 					case "POST":
946 						o1 = 2;
947 						break;
948 					case "PUT":
949 						o1 = 3;
950 						break;
951 					case "GET":
952 						o1 = 4;
953 						break;
954 					default:
955 						o1 = 0;
956 						break;
957 				}
958 			}
959 			return o1;
960 		}
961 
962 	}

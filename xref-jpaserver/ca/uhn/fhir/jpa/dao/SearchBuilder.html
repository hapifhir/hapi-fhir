164 	private void addPredicateHas(List<List<? extends IQueryParameterType>> theHasParameters) {
165 
166 		for (List<? extends IQueryParameterType> nextOrList : theHasParameters) {
167 
168 			StringBuilder valueBuilder = new StringBuilder();
169 			String targetResourceType = null;
170 			String owningParameter = null;
171 			String parameterName = null;
172 			for (IQueryParameterType nextParam : nextOrList) {
173 				HasParam next = (HasParam) nextParam;
174 				if (valueBuilder.length() > 0) {
175 					valueBuilder.append(',');
176 				}
177 				valueBuilder.append(UrlUtil.escapeUrlParam(next.getValueAsQueryToken(myContext)));
178 				targetResourceType = next.getTargetResourceType();
179 				owningParameter = next.getOwningFieldName();
180 				parameterName = next.getParameterName();
181 			}
182 
183 			if (valueBuilder.length() == 0) {
184 				continue;
185 			}
186 
187 			String matchUrl = targetResourceType + '?' + UrlUtil.escapeUrlParam(parameterName) + '=' + valueBuilder.toString();
188 			RuntimeResourceDefinition targetResourceDefinition;
189 			try {
190 				targetResourceDefinition = myContext.getResourceDefinition(targetResourceType);
191 			} catch (DataFormatException e) {
192 				throw new InvalidRequestException("Invalid resource type: " + targetResourceType);
193 			}
194 
195 			String paramName = parameterName.replaceAll("\\..*", "");
196 			RuntimeSearchParam owningParameterDef = myCallingDao.getSearchParamByName(targetResourceDefinition, paramName);
197 			if (owningParameterDef == null) {
198 				throw new InvalidRequestException("Unknown parameter name: " + targetResourceType + ':' + parameterName);
199 			}
200 
201 			owningParameterDef = myCallingDao.getSearchParamByName(targetResourceDefinition, owningParameter);
202 			if (owningParameterDef == null) {
203 				throw new InvalidRequestException("Unknown parameter name: " + targetResourceType + ':' + owningParameter);
204 			}
205 
206 			Class<? extends IBaseResource> resourceType = targetResourceDefinition.getImplementingClass();
207 			Set<Long> match = myCallingDao.processMatchUrl(matchUrl, resourceType);
208 			if (match.isEmpty()) {
209 				// Pick a PID that can never match
210 				match = Collections.singleton(-1L);
211 			}
212 
213 			Join<ResourceTable, ResourceLink> join = myResourceTableRoot.join("myIncomingResourceLinks", JoinType.LEFT);
214 
215 			Predicate predicate = join.get("mySourceResourcePid").in(match);
216 			myPredicates.add(predicate);
217 		}
218 	}

221 		for (List<? extends IQueryParameterType> nextList : theList) {

248 
249 		Join<ResourceTable, ResourceIndexedSearchParamNumber> join = createOrReuseJoin(JoinEnum.NUMBER, theParamName);
250 
251 		if (theList.get(0).getMissing() != null) {
252 			addPredicateParamMissing(theResourceName, theParamName, theList.get(0).getMissing(), join);
253 			return;
254 		}
255 
256 		List<Predicate> codePredicates = new ArrayList<Predicate>();
257 		for (IQueryParameterType nextOr : theList) {
258 			IQueryParameterType params = nextOr;
259 
260 			if (params instanceof NumberParam) {
261 				NumberParam param = (NumberParam) params;
262 
263 				BigDecimal value = param.getValue();
264 				if (value == null) {
265 					continue;
266 				}
267 
268 				final Expression<BigDecimal> fromObj = join.get("myValue");
269 				ParamPrefixEnum prefix = ObjectUtils.defaultIfNull(param.getPrefix(), ParamPrefixEnum.EQUAL);
270 				String invalidMessageName = "invalidNumberPrefix";
271 
272 				Predicate num = createPredicateNumeric(theResourceName, theParamName, join, myBuilder, params, prefix, value, fromObj, invalidMessageName);
273 				codePredicates.add(num);
274 
275 			} else {
276 				throw new IllegalArgumentException("Invalid token type: " + params.getClass());
277 			}
278 
279 		}
280 
281 		myPredicates.add(myBuilder.or(toArray(codePredicates)));
282 	}

319 	 * Add reference predicate to the current search
320 	 */
321 	private void addPredicateReference(String theResourceName, String theParamName, List<? extends IQueryParameterType> theList) {
322 		assert theParamName.contains(".") == false;
323 
324 		if (theList.get(0).getMissing() != null) {
325 			addPredicateParamMissing(theResourceName, theParamName, theList.get(0).getMissing());
326 			return;
327 		}
328 
329 		Join<ResourceTable, ResourceLink> join = createOrReuseJoin(JoinEnum.REFERENCE, theParamName);
330 
331 		List<Predicate> codePredicates = new ArrayList<Predicate>();
332 
333 		for (IQueryParameterType nextOr : theList) {
334 			IQueryParameterType params = nextOr;
335 
336 			if (params instanceof ReferenceParam) {
337 				ReferenceParam ref = (ReferenceParam) params;
338 
339 				if (isBlank(ref.getChain())) {
340 					IIdType dt = new IdDt(ref.getBaseUrl(), ref.getResourceType(), ref.getIdPart(), null);
341 
342 					if (dt.hasBaseUrl()) {
343 						if (myCallingDao.getConfig().getTreatBaseUrlsAsLocal().contains(dt.getBaseUrl())) {
344 							dt = dt.toUnqualified();
345 						} else {
346 							ourLog.debug("Searching for resource link with target URL: {}", dt.getValue());
347 							Predicate eq = myBuilder.equal(join.get("myTargetResourceUrl"), dt.getValue());
348 							codePredicates.add(eq);
349 							continue;
350 						}
351 					}
352 
353 					List<Long> targetPid;
354 					try {
355 						targetPid = myCallingDao.translateForcedIdToPids(dt);
356 					} catch (ResourceNotFoundException e) {
357 						// Use a PID that will never exist
358 						targetPid = Collections.singletonList(-1L);
359 					}
360 					for (Long next : targetPid) {
361 						ourLog.debug("Searching for resource link with target PID: {}", next);
362 
363 						Predicate pathPredicate = createResourceLinkPathPredicate(theResourceName, theParamName, join);
364 						Predicate pidPredicate = myBuilder.equal(join.get("myTargetResourcePid"), next);
365 						codePredicates.add(myBuilder.and(pathPredicate, pidPredicate));
366 					}
367 
368 				} else {
369 
370 					final List<Class<? extends IBaseResource>> resourceTypes;
371 					String resourceId;
372 					if (!ref.getValue().matches("[a-zA-Z]+\\/.*")) {
373 
374 						RuntimeSearchParam param = mySearchParamRegistry.getActiveSearchParam(theResourceName, theParamName);
375 						resourceTypes = new ArrayList<>();
376 
377 						Set<String> targetTypes = param.getTargets();
378 
379 						if (targetTypes != null && !targetTypes.isEmpty()) {
380 							for (String next : targetTypes) {
381 								resourceTypes.add(myContext.getResourceDefinition(next).getImplementingClass());
382 							}
383 						}
384 
385 						if (resourceTypes.isEmpty()) {
386 							RuntimeResourceDefinition resourceDef = myContext.getResourceDefinition(theResourceName);
387 							RuntimeSearchParam searchParamByName = myCallingDao.getSearchParamByName(resourceDef, theParamName);
388 							if (searchParamByName == null) {
389 								throw new InternalErrorException("Could not find parameter " + theParamName);
390 							}
391 							String paramPath = searchParamByName.getPath();
392 							if (paramPath.endsWith(".as(Reference)")) {
393 								paramPath = paramPath.substring(0, paramPath.length() - ".as(Reference)".length()) + "Reference";
394 							}
395 
396 							if (paramPath.contains(".extension(")) {
397 								int startIdx = paramPath.indexOf(".extension(");
398 								int endIdx = paramPath.indexOf(')', startIdx);
399 								if (startIdx != -1 && endIdx != -1) {
400 									paramPath = paramPath.substring(0, startIdx + 10) + paramPath.substring(endIdx + 1);
401 								}
402 							}
403 
404 							BaseRuntimeChildDefinition def = myContext.newTerser().getDefinition(myResourceType, paramPath);
405 							if (def instanceof RuntimeChildChoiceDefinition) {
406 								RuntimeChildChoiceDefinition choiceDef = (RuntimeChildChoiceDefinition) def;
407 								resourceTypes.addAll(choiceDef.getResourceTypes());
408 							} else if (def instanceof RuntimeChildResourceDefinition) {
409 								RuntimeChildResourceDefinition resDef = (RuntimeChildResourceDefinition) def;
410 								resourceTypes.addAll(resDef.getResourceTypes());
411 							} else {
412 								throw new ConfigurationException("Property " + paramPath + " of type " + myResourceName + " is not a resource: " + def.getClass());
413 							}
414 						}
415 
416 						if (resourceTypes.isEmpty()) {
417 							for (BaseRuntimeElementDefinition<?> next : myContext.getElementDefinitions()) {
418 								if (next instanceof RuntimeResourceDefinition) {
419 									RuntimeResourceDefinition nextResDef = (RuntimeResourceDefinition) next;
420 									resourceTypes.add(nextResDef.getImplementingClass());
421 								}
422 							}
423 						}
424 
425 						resourceId = ref.getValue();
426 
427 					} else {
428 						RuntimeResourceDefinition resDef = myContext.getResourceDefinition(ref.getResourceType());
429 						resourceTypes = new ArrayList<Class<? extends IBaseResource>>(1);
430 						resourceTypes.add(resDef.getImplementingClass());
431 						resourceId = ref.getIdPart();
432 					}
433 
434 					boolean foundChainMatch = false;
435 
436 					String chain = ref.getChain();
437 					String remainingChain = null;
438 					int chainDotIndex = chain.indexOf('.');
439 					if (chainDotIndex != -1) {
440 						remainingChain = chain.substring(chainDotIndex + 1);
441 						chain = chain.substring(0, chainDotIndex);
442 					}
443 
444 					for (Class<? extends IBaseResource> nextType : resourceTypes) {
445 						RuntimeResourceDefinition typeDef = myContext.getResourceDefinition(nextType);
446 						String subResourceName = typeDef.getName();
447 
448 						IFhirResourceDao<?> dao = myCallingDao.getDao(nextType);
449 						if (dao == null) {
450 							ourLog.debug("Don't have a DAO for type {}", nextType.getSimpleName());
451 							continue;
452 						}
453 
454 						int qualifierIndex = chain.indexOf(':');
455 						String qualifier = null;
456 						if (qualifierIndex != -1) {
457 							qualifier = chain.substring(qualifierIndex);
458 							chain = chain.substring(0, qualifierIndex);
459 						}
460 
461 						boolean isMeta = BaseHapiFhirDao.RESOURCE_META_PARAMS.containsKey(chain);
462 						RuntimeSearchParam param = null;
463 						if (!isMeta) {
464 							param = myCallingDao.getSearchParamByName(typeDef, chain);
465 							if (param == null) {
466 								ourLog.debug("Type {} doesn't have search param {}", nextType.getSimpleName(), param);
467 								continue;
468 							}
469 						}
470 
471 						IQueryParameterType chainValue;
472 						if (remainingChain != null) {
473 							if (param == null || param.getParamType() != RestSearchParameterTypeEnum.REFERENCE) {
474 								ourLog.debug("Type {} parameter {} is not a reference, can not chain {}", new Object[]{nextType.getSimpleName(), chain, remainingChain});
475 								continue;
476 							}
477 
478 							chainValue = new ReferenceParam();
479 							chainValue.setValueAsQueryToken(myContext, theParamName, qualifier, resourceId);
480 							((ReferenceParam) chainValue).setChain(remainingChain);
481 						} else if (isMeta) {
482 							IQueryParameterType type = BaseHapiFhirDao.newInstanceType(chain);
483 							type.setValueAsQueryToken(myContext, theParamName, qualifier, resourceId);
484 							chainValue = type;
485 						} else {
486 							chainValue = toParameterType(param, qualifier, resourceId);
487 						}
488 
489 						foundChainMatch = true;
490 
491 						Subquery<Long> subQ = myResourceTableQuery.subquery(Long.class);
492 						Root<ResourceTable> subQfrom = subQ.from(ResourceTable.class);
493 						subQ.select(subQfrom.get("myId").as(Long.class));
494 
495 						List<List<? extends IQueryParameterType>> andOrParams = new ArrayList<List<? extends IQueryParameterType>>();
496 						andOrParams.add(Collections.singletonList(chainValue));
497 
498 						/*
499 						 * We're doing a chain call, so push the current query root
500 						 * and predicate list down and put new ones at the top of the
501 						 * stack and run a subuery
502 						 */
503 						Root<ResourceTable> stackRoot = myResourceTableRoot;
504 						ArrayList<Predicate> stackPredicates = myPredicates;
505 						Map<JoinKey, Join<?, ?>> stackIndexJoins = myIndexJoins;
506 						myResourceTableRoot = subQfrom;
507 						myPredicates = Lists.newArrayList();
508 						myIndexJoins = Maps.newHashMap();
509 
510 						// Create the subquery predicates
511 						myPredicates.add(myBuilder.equal(myResourceTableRoot.get("myResourceType"), subResourceName));
512 						myPredicates.add(myBuilder.isNull(myResourceTableRoot.get("myDeleted")));
513 						searchForIdsWithAndOr(subResourceName, chain, andOrParams);
514 
515 						subQ.where(toArray(myPredicates));
516 
517 						/*
518 						 * Pop the old query root and predicate list back
519 						 */
520 						myResourceTableRoot = stackRoot;
521 						myPredicates = stackPredicates;
522 						myIndexJoins = stackIndexJoins;
523 
524 						Predicate pathPredicate = createResourceLinkPathPredicate(theResourceName, theParamName, join);
525 						Predicate pidPredicate = join.get("myTargetResourcePid").in(subQ);
526 						codePredicates.add(myBuilder.and(pathPredicate, pidPredicate));
527 
528 					}
529 
530 					if (!foundChainMatch) {
531 						throw new InvalidRequestException(myContext.getLocalizer().getMessage(BaseHapiFhirResourceDao.class, "invalidParameterChain", theParamName + '.' + ref.getChain()));
532 					}
533 				}
534 
535 			} else {
536 				throw new IllegalArgumentException("Invalid token type (expecting ReferenceParam): " + params.getClass());

766 
767 		Join<ResourceTable, ResourceIndexedSearchParamUri> join = createOrReuseJoin(JoinEnum.URI, theParamName);
768 
769 		if (theList.get(0).getMissing() != null) {
770 			addPredicateParamMissing(theResourceName, theParamName, theList.get(0).getMissing(), join);
771 			return;
772 		}
773 
774 		List<Predicate> codePredicates = new ArrayList<Predicate>();
775 		for (IQueryParameterType nextOr : theList) {
776 			IQueryParameterType params = nextOr;
777 
778 			if (params instanceof UriParam) {
779 				UriParam param = (UriParam) params;
780 
781 				String value = param.getValue();
782 				if (value == null) {
783 					continue;
784 				}
785 
786 				Predicate predicate;
787 				if (param.getQualifier() == UriParamQualifierEnum.ABOVE) {
788 
789 					/*
790 					 * :above is an inefficient query- It means that the user is supplying a more specific URL (say
791 					 * http://example.com/foo/bar/baz) and that we should match on any URLs that are less
792 					 * specific but otherwise the same. For example http://example.com and http://example.com/foo would both
793 					 * match.
794 					 *
795 					 * We do this by querying the DB for all candidate URIs and then manually checking each one. This isn't
796 					 * very efficient, but this is also probably not a very common type of query to do.
797 					 *
798 					 * If we ever need to make this more efficient, lucene could certainly be used as an optimization.
799 					 */
800 					ourLog.info("Searching for candidate URI:above parameters for Resource[{}] param[{}]", myResourceName, theParamName);
801 					Collection<String> candidates = myResourceIndexedSearchParamUriDao.findAllByResourceTypeAndParamName(myResourceName, theParamName);
802 					List<String> toFind = new ArrayList<>();
803 					for (String next : candidates) {
804 						if (value.length() >= next.length()) {
805 							if (value.substring(0, next.length()).equals(next)) {
806 								toFind.add(next);
807 							}
808 						}
809 					}
810 
811 					if (toFind.isEmpty()) {
812 						continue;
813 					}
814 
815 					predicate = join.<Object>get("myUri").as(String.class).in(toFind);
816 
817 				} else if (param.getQualifier() == UriParamQualifierEnum.BELOW) {
818 					predicate = myBuilder.like(join.get("myUri").as(String.class), createLeftMatchLikeExpression(value));
819 				} else {
820 					predicate = myBuilder.equal(join.get("myUri").as(String.class), value);
821 				}
822 				codePredicates.add(predicate);
823 			} else {
824 				throw new IllegalArgumentException("Invalid URI type: " + params.getClass());
825 			}
826 
827 		}
828 
829 		/*
830 		 * If we haven't found any of the requested URIs in the candidates, then we'll
831 		 * just add a predicate that can never match
832 		 */
833 		if (codePredicates.isEmpty()) {
834 			Predicate predicate = myBuilder.isNull(join.get("myMissing").as(String.class));
835 			myPredicates.add(predicate);
836 			return;
837 		}
838 
839 		Predicate orPredicate = myBuilder.or(toArray(codePredicates));
840 
841 		Predicate paramNamePredicate = myBuilder.equal(join.get("myParamName"), theParamName);
842 		Predicate outerPredicate = myBuilder.and(paramNamePredicate, orPredicate);
843 
844 		myPredicates.add(outerPredicate);
845 	}

985 														  IQueryParameterType theParam, ParamPrefixEnum thePrefix, BigDecimal theValue, final Expression<BigDecimal> thePath,
986 														  String invalidMessageName) {
987 		Predicate num;
988 		switch (thePrefix) {
989 			case GREATERTHAN:
990 				num = builder.gt(thePath, theValue);
991 				break;
992 			case GREATERTHAN_OR_EQUALS:
993 				num = builder.ge(thePath, theValue);
994 				break;
995 			case LESSTHAN:
996 				num = builder.lt(thePath, theValue);
997 				break;
998 			case LESSTHAN_OR_EQUALS:
999 				num = builder.le(thePath, theValue);
1000 				break;
1001 			case APPROXIMATE:
1002 			case EQUAL:
1003 			case NOT_EQUAL:
1004 				BigDecimal mul = calculateFuzzAmount(thePrefix, theValue);
1005 				BigDecimal low = theValue.subtract(mul, MathContext.DECIMAL64);
1006 				BigDecimal high = theValue.add(mul, MathContext.DECIMAL64);
1007 				Predicate lowPred;
1008 				Predicate highPred;
1009 				if (thePrefix != ParamPrefixEnum.NOT_EQUAL) {
1010 					lowPred = builder.ge(thePath.as(BigDecimal.class), low);
1011 					highPred = builder.le(thePath.as(BigDecimal.class), high);
1012 					num = builder.and(lowPred, highPred);
1013 					ourLog.trace("Searching for {} <= val <= {}", low, high);
1014 				} else {
1015 					// Prefix was "ne", so reverse it!
1016 					lowPred = builder.lt(thePath.as(BigDecimal.class), low);
1017 					highPred = builder.gt(thePath.as(BigDecimal.class), high);
1018 					num = builder.or(lowPred, highPred);
1019 				}
1020 				break;
1021 			default:
1022 				String msg = myContext.getLocalizer().getMessage(SearchBuilder.class, invalidMessageName, thePrefix.getValue(), theParam.getValueAsQueryToken(myContext));
1023 				throw new InvalidRequestException(msg);
1024 		}
1025 
1026 		if (theParamName == null) {
1027 			return num;
1028 		}
1029 		return combineParamIndexPredicateWithParamNamePredicate(theResourceName, theParamName, theFrom, num);
1030 	}

1958 	/**
1959 	 * Figures out the tolerance for a search. For example, if the user is searching for <code>4.00</code>, this method
1960 	 * returns <code>0.005</code> because we shold actually match values which are
1961 	 * <code>4 (+/-) 0.005</code> according to the FHIR specs.
1962 	 */
1963 	static BigDecimal calculateFuzzAmount(ParamPrefixEnum cmpValue, BigDecimal theValue) {
1964 		if (cmpValue == ParamPrefixEnum.APPROXIMATE) {
1965 			return theValue.multiply(new BigDecimal(0.1));
1966 		} else {
1967 			String plainString = theValue.toPlainString();
1968 			int dotIdx = plainString.indexOf('.');
1969 			if (dotIdx == -1) {
1970 				return new BigDecimal(0.5);
1971 			}
1972 
1973 			int precision = plainString.length() - (dotIdx);
1974 			double mul = Math.pow(10, -precision);
1975 			double val = mul * 5.0d;
1976 			return new BigDecimal(val);
1977 		}

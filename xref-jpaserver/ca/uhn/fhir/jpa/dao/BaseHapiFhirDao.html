237 	protected Set<ResourceLink> extractResourceLinks(ResourceTable theEntity, IBaseResource theResource) {
238 		Set<ResourceLink> retVal = new HashSet<ResourceLink>();
239 
240 		RuntimeResourceDefinition def = getContext().getResourceDefinition(theResource);
241 		for (RuntimeSearchParam nextSpDef : def.getSearchParams()) {
242 
243 			if (nextSpDef.getParamType() != RestSearchParameterTypeEnum.REFERENCE) {
244 				continue;
245 			}
246 
247 			String nextPathsUnsplit = nextSpDef.getPath();
248 			if (isBlank(nextPathsUnsplit)) {
249 				continue;
250 			}
251 
252 			boolean multiType = false;
253 			if (nextPathsUnsplit.endsWith("[x]")) {
254 				multiType = true;
255 			}
256 
257 			String[] nextPathsSplit = nextPathsUnsplit.split("\\|");
258 			for (String nextPath : nextPathsSplit) {
259 				nextPath = nextPath.trim();
260 
261 				List<Class<? extends IBaseResource>> allowedTypesInField = null;
262 				for (Object nextObject : extractValues(nextPath, theResource)) {
263 					if (nextObject == null) {
264 						continue;
265 					}
266 
267 					ResourceLink nextEntity;
268 					if (nextObject instanceof IBaseReference) {
269 						IBaseReference nextValue = (IBaseReference) nextObject;
270 						if (nextValue.isEmpty()) {
271 							continue;
272 						}
273 						if (nextValue.getReferenceElement().isEmpty() || nextValue.getReferenceElement().getValue().startsWith("#")) {
274 							// This is a blank or contained resource reference
275 							continue;
276 						}
277 
278 						String typeString = nextValue.getReferenceElement().getResourceType();
279 						if (isBlank(typeString)) {
280 							throw new InvalidRequestException(
281 									"Invalid resource reference found at path[" + nextPathsUnsplit + "] - Does not contain resource type - " + nextValue.getReferenceElement().getValue());
282 						}
283 						RuntimeResourceDefinition resourceDefinition;
284 						try {
285 							resourceDefinition = getContext().getResourceDefinition(typeString);
286 						} catch (DataFormatException e) {
287 							throw new InvalidRequestException("Invalid resource reference found at path[" + nextPathsUnsplit + "] - Resource type is unknown or not supported on this server - "
288 									+ nextValue.getReferenceElement().getValue());
289 						}
290 
291 						Class<? extends IBaseResource> type = resourceDefinition.getImplementingClass();
292 						String id = nextValue.getReferenceElement().getIdPart();
293 						if (StringUtils.isBlank(id)) {
294 							throw new InvalidRequestException(
295 									"Invalid resource reference found at path[" + nextPathsUnsplit + "] - Does not contain resource ID - " + nextValue.getReferenceElement().getValue());
296 						}
297 
298 						IFhirResourceDao<?> dao = getDao(type);
299 						if (dao == null) {
300 							StringBuilder b = new StringBuilder();
301 							b.append("This server (version ");
302 							b.append(myContext.getVersion().getVersion());
303 							b.append(") is not able to handle resources of type[");
304 							b.append(nextValue.getReferenceElement().getResourceType());
305 							b.append("] - Valid resource types for this server: ");
306 							b.append(myResourceTypeToDao.keySet().toString());
307 
308 							throw new InvalidRequestException(b.toString());
309 						}
310 						Long valueOf;
311 						try {
312 							valueOf = translateForcedIdToPid(nextValue.getReferenceElement());
313 						} catch (ResourceNotFoundException e) {
314 							String resName = getContext().getResourceDefinition(type).getName();
315 							throw new InvalidRequestException("Resource " + resName + "/" + id + " not found, specified in path: " + nextPathsUnsplit);
316 						}
317 						ResourceTable target = myEntityManager.find(ResourceTable.class, valueOf);
318 						RuntimeResourceDefinition targetResourceDef = getContext().getResourceDefinition(type);
319 						if (target == null) {
320 							String resName = targetResourceDef.getName();
321 							throw new InvalidRequestException("Resource " + resName + "/" + id + " not found, specified in path: " + nextPathsUnsplit);
322 						}
323 
324 						if (!typeString.equals(target.getResourceType())) {
325 							throw new UnprocessableEntityException("Resource contains reference to " + nextValue.getReferenceElement().getValue() + " but resource with ID "
326 									+ nextValue.getReferenceElement().getIdPart() + " is actually of type " + target.getResourceType());
327 						}
328 
329 						/*
330 						 * Is the target type an allowable type of resource for the path where it is referenced?
331 						 */
332 
333 						if (allowedTypesInField == null) {
334 							BaseRuntimeChildDefinition childDef = getContext().newTerser().getDefinition(theResource.getClass(), nextPath);
335 							if (childDef instanceof RuntimeChildResourceDefinition) {
336 								RuntimeChildResourceDefinition resRefDef = (RuntimeChildResourceDefinition) childDef;
337 								allowedTypesInField = resRefDef.getResourceTypes();
338 							} else {
339 								allowedTypesInField = new ArrayList<Class<? extends IBaseResource>>();
340 								allowedTypesInField.add(IBaseResource.class);
341 							}
342 						}
343 
344 						boolean acceptableLink = false;
345 						for (Class<? extends IBaseResource> next : allowedTypesInField) {
346 							if (next.isAssignableFrom(targetResourceDef.getImplementingClass())) {
347 								acceptableLink = true;
348 								break;
349 							}
350 						}
351 
352 						if (!acceptableLink) {
353 							throw new UnprocessableEntityException("Invalid reference found at path '" + nextPath + "'. Resource type '" + targetResourceDef.getName() + "' is not valid for this path");
354 						}
355 
356 						nextEntity = new ResourceLink(nextPath, theEntity, target);
357 					} else {
358 						if (!multiType) {
359 							if (nextSpDef.getName().equals("sourceuri")) {
360 								continue; // TODO: disable this eventually - ConceptMap:sourceuri is of type reference but points to a URI
361 							}
362 							throw new ConfigurationException("Search param " + nextSpDef.getName() + " is of unexpected datatype: " + nextObject.getClass());
363 						} else {
364 							continue;
365 						}
366 					}
367 					if (nextEntity != null) {
368 						retVal.add(nextEntity);
369 					}
370 				}
371 			}
372 		}
373 
374 		theEntity.setHasLinks(retVal.size() > 0);
375 
376 		return retVal;
377 	}

434 		if (profiles != null) {
435 			for (IIdType next : profiles) {
436 				TagDefinition tag = getTag(TagTypeEnum.PROFILE, NS_JPA_PROFILE, next.getValue(), null);

489 	private void findMatchingTagIds(String theResourceName, IIdType theResourceId, Set<Long> tagIds, Class<? extends BaseTag> entityClass) {
490 		{
491 			CriteriaBuilder builder = myEntityManager.getCriteriaBuilder();
492 			CriteriaQuery<Tuple> cq = builder.createTupleQuery();
493 			Root<? extends BaseTag> from = cq.from(entityClass);
494 			cq.multiselect(from.get("myTagId").as(Long.class)).distinct(true);
495 
496 			if (theResourceName != null) {
497 				Predicate typePredicate = builder.equal(from.get("myResourceType"), theResourceName);
498 				if (theResourceId != null) {
499 					cq.where(typePredicate, builder.equal(from.get("myResourceId"), translateForcedIdToPid(theResourceId)));
500 				} else {
501 					cq.where(typePredicate);
502 				}

1603 
1604 			String paramName = next.getName();
1605 			String qualifier = null;
1606 			for (int i = 0; i < paramName.length(); i++) {
1607 				switch (paramName.charAt(i)) {
1608 				case '.':
1609 				case ':':
1610 					qualifier = paramName.substring(i);
1611 					paramName = paramName.substring(0, i);
1612 					i = Integer.MAX_VALUE - 1;
1613 					break;
1614 				}
1615 			}
1616 
1617 			QualifiedParamList paramList = QualifiedParamList.splitQueryStringByCommasIgnoreEscape(qualifier, next.getValue());
1618 			nameToParamLists.put(paramName, paramList);
1619 		}
1620 
1621 		for (String nextParamName : nameToParamLists.keySet()) {
1622 			List<QualifiedParamList> paramList = nameToParamLists.get(nextParamName);
1623 			if (Constants.PARAM_LASTUPDATED.equals(nextParamName)) {
1624 				if (paramList != null && paramList.size() > 0) {
1625 					if (paramList.size() > 2) {
1626 						throw new InvalidRequestException("Failed to parse match URL[" + theMatchUrl + "] - Can not have more than 2 " + Constants.PARAM_LASTUPDATED + " parameter repetitions");
1627 					} else {
1628 						DateRangeParam p1 = new DateRangeParam();
1629 						p1.setValuesAsQueryTokens(paramList);
1630 						paramMap.setLastUpdated(p1);
1631 					}
1632 				}
1633 				continue;
1634 			}
1635 
1636 			if (Constants.PARAM_COUNT.equals(nextParamName)) {
1637 				if (paramList.size() > 0 && paramList.get(0).size() > 0) {
1638 					String intString = paramList.get(0).get(0);
1639 					try {
1640 						paramMap.setCount(Integer.parseInt(intString));
1641 					} catch (NumberFormatException e) {
1642 						throw new InvalidRequestException("Invalid " + Constants.PARAM_COUNT + " value: " + intString);
1643 					}
1644 				}
1645 				continue;
1646 			}
1647 
1648 			if (RESOURCE_META_PARAMS.containsKey(nextParamName)) {
1649 				if (isNotBlank(paramList.get(0).getQualifier()) && paramList.get(0).getQualifier().startsWith(".")) {
1650 					throw new InvalidRequestException("Invalid parameter chain: " + nextParamName + paramList.get(0).getQualifier());
1651 				}
1652 				IQueryParameterAnd<?> type = newInstanceAnd(nextParamName);
1653 				type.setValuesAsQueryTokens((paramList));
1654 				paramMap.add(nextParamName, type);
1655 			} else if (nextParamName.startsWith("_")) {
1656 				// ignore these since they aren't search params (e.g. _sort)
1657 			} else {
1658 				RuntimeSearchParam paramDef = resourceDef.getSearchParam(nextParamName);
1659 				if (paramDef == null) {
1660 					throw new InvalidRequestException("Failed to parse match URL[" + theMatchUrl + "] - Resource type " + resourceDef.getName() + " does not have a parameter with name: " + nextParamName);
1661 				}
1662 
1663 				IQueryParameterAnd<?> param = MethodUtil.parseQueryParams(paramDef, nextParamName, paramList);
1664 				paramMap.add(nextParamName, param);
1665 			}
1666 		}
1667 		return paramMap;
1668 	}

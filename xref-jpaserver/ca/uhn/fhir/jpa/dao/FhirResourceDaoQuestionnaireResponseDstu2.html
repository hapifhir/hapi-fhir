44  public class FhirResourceDaoQuestionnaireResponseDstu2 extends FhirResourceDaoDstu2<QuestionnaireResponse> {
45  
46  	@Autowired
47  	@Qualifier("myFhirContextDstu2Hl7Org")
48  	private FhirContext myRefImplCtx;
49  	
50  	private Boolean myValidateResponses;
51  
52  	/**
53  	 * Initialize the bean
54  	 */
55  	@PostConstruct
56  	public void initialize() {
57  		try {
58  			Class.forName("org.hl7.fhir.instance.model.QuestionnaireResponse");
59  			myValidateResponses = true;
60  		} catch (ClassNotFoundException e) {
61  			myValidateResponses = Boolean.FALSE;
62  		}
63  	}
64  	
65  	@Override
66  	protected void validateResourceForStorage(QuestionnaireResponse theResource, ResourceTable theEntityToSave) {
67  		super.validateResourceForStorage(theResource, theEntityToSave);
68  		if (!myValidateResponses) {
69  			return;
70  		}
71  		
72  		QuestionnaireResponse qa = (QuestionnaireResponse) theResource;
73  		if (qa == null || qa.getQuestionnaire() == null || qa.getQuestionnaire().getReference() == null || qa.getQuestionnaire().getReference().isEmpty()) {
74  			return;
75  		}
76  
77  		FhirValidator val = myRefImplCtx.newValidator();
78  		val.setValidateAgainstStandardSchema(false);
79  		val.setValidateAgainstStandardSchematron(false);
80  
81  		FhirQuestionnaireResponseValidator module = new FhirQuestionnaireResponseValidator();
82  		module.setResourceLoader(new JpaResourceLoader());
83  		val.registerValidatorModule(module);
84  
85  		ValidationResult result = val.validateWithResult(myRefImplCtx.newJsonParser().parseResource(getContext().newJsonParser().encodeResourceToString(qa)));
86  		if (!result.isSuccessful()) {
87  			IBaseOperationOutcome oo = getContext().newJsonParser().parseResource(OperationOutcome.class, myRefImplCtx.newJsonParser().encodeResourceToString(result.toOperationOutcome()));
88  			throw new UnprocessableEntityException(getContext(), oo);
89  		}
90  	}
91  
92  	public class JpaResourceLoader implements IResourceLoader {
93  
94  		public JpaResourceLoader() {
95  			super();
96  		}
97  
98  		@Override
99  		public <T extends IBaseResource> T load(Class<T> theType, IIdType theId) throws ResourceNotFoundException {
100 
101 			/*
102 			 * The QuestionnaireResponse validator uses RI structures, so for now we need to convert between that and HAPI
103 			 * structures. This is a bit hackish, but hopefully it will go away at some point.
104 			 */
105 			if ("ValueSet".equals(theType.getSimpleName())) {
106 				IFhirResourceDao<ValueSet> dao = getDao(ValueSet.class);
107 				ValueSet in = dao.read(theId, null);
108 				String encoded = getContext().newJsonParser().encodeResourceToString(in);
109 
110 				// TODO: this is temporary until structures-dstu2 catches up to structures-hl7org.dstu2
111 				encoded = encoded.replace("\"define\"", "\"codeSystem\"");
112 
113 				return myRefImplCtx.newJsonParser().parseResource(theType, encoded);
114 			} else if ("Questionnaire".equals(theType.getSimpleName())) {
115 				IFhirResourceDao<Questionnaire> dao = getDao(Questionnaire.class);
116 				Questionnaire vs = dao.read(theId, null);
117 				return myRefImplCtx.newJsonParser().parseResource(theType, getContext().newJsonParser().encodeResourceToString(vs));
118 			} else {
119 				// Should not happen, validator will only ask for these two
120 				throw new IllegalStateException("Unexpected request to load resource of type " + theType);
121 			}
122 
123 		}
124 
125 	}

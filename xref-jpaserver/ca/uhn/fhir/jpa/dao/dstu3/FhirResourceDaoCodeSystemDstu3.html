182 	private List<TermConcept> toPersistedConcepts(List<ConceptDefinitionComponent> theConcept, TermCodeSystemVersion theCodeSystemVersion) {
183 		ArrayList<TermConcept> retVal = new ArrayList<TermConcept>();
184 
185 		for (ConceptDefinitionComponent next : theConcept) {
186 			if (isNotBlank(next.getCode())) {
187 				TermConcept termConcept = new TermConcept();
188 				termConcept.setCode(next.getCode());
189 				termConcept.setCodeSystem(theCodeSystemVersion);
190 				termConcept.setDisplay(next.getDisplay());
191 				termConcept.addChildren(toPersistedConcepts(next.getConcept(), theCodeSystemVersion), RelationshipTypeEnum.ISA);
192 				retVal.add(termConcept);
193 			}
194 		}
195 
196 		return retVal;
197 	}

199 	@Override
200 	protected ResourceTable updateEntity(IBaseResource theResource, ResourceTable theEntity, Date theDeletedTimestampOrNull, boolean thePerformIndexing,
201 			boolean theUpdateVersion, Date theUpdateTime) {
202 		ResourceTable retVal = super.updateEntity(theResource, theEntity, theDeletedTimestampOrNull, thePerformIndexing, theUpdateVersion, theUpdateTime);
203 
204 		CodeSystem cs = (CodeSystem) theResource;
205 
206 		if (cs != null && isNotBlank(cs.getUrl())) {
207 			String codeSystemUrl = cs.getUrl();
208 			if (cs.getContent() == CodeSystemContentMode.COMPLETE || cs.getContent() == null) {
209 				ourLog.info("CodeSystem {} has a status of {}, going to store concepts in terminology tables", retVal.getIdDt().getValue(), cs.getContentElement().getValueAsString());
210 				TermCodeSystemVersion persCs = new TermCodeSystemVersion();
211 				persCs.setResource(retVal);
212 				persCs.setResourceVersionId(retVal.getVersion());
213 				persCs.getConcepts().addAll(toPersistedConcepts(cs.getConcept(), persCs));
214 
215 				myTerminologySvc.storeNewCodeSystemVersion(retVal.getId(), codeSystemUrl, persCs);
216 			}
217 		}
218 
219 		return retVal;
220 	}

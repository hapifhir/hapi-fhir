60  	private UrlParts parseUrl(String theAction, String theUrl) {
61  		UrlParts retVal = new UrlParts();
62  
63  		//@formatter:off
64  		/* 
65  		 * We assume that the URL passed in is in one of the following forms:
66  		 * [Resource Type]?[Search Params]
67  		 * [Resource Type]/[Resource ID]
68  		 * [Resource Type]/[Resource ID]/_history/[Version ID]
69  		 */
70  		//@formatter:on
71  		int nextStart = 0;
72  		boolean nextIsHistory = false;
73  
74  		for (int idx = 0; idx < theUrl.length(); idx++) {
75  			char nextChar = theUrl.charAt(idx);
76  			boolean atEnd = (idx + 1) == theUrl.length();
77  			if (nextChar == '?' || nextChar == '/' || atEnd) {
78  				int endIdx = atEnd ? idx + 1 : idx;
79  				String nextSubstring = theUrl.substring(nextStart, endIdx);
80  				if (retVal.getResourceType() == null) {
81  					retVal.setResourceType(nextSubstring);
82  				} else if (retVal.getResourceId() == null) {
83  					retVal.setResourceId(nextSubstring);
84  				} else if (nextIsHistory) {
85  					retVal.setVersionId(nextSubstring);
86  				} else {
87  					if (nextSubstring.equals(Constants.URL_TOKEN_HISTORY)) {
88  						nextIsHistory = true;
89  					} else {
90  						String msg = getContext().getLocalizer().getMessage(BaseFhirSystemDao.class, "transactionInvalidUrl", theAction, theUrl);
91  						throw new InvalidRequestException(msg);
92  					}
93  				}
94  				if (nextChar == '?') {
95  					if (theUrl.length() > idx + 1) {
96  						retVal.setParams(theUrl.substring(idx + 1, theUrl.length()));
97  					}
98  					break;
99  				}
100 				nextStart = idx + 1;
101 			}
102 		}
103 
104 		RuntimeResourceDefinition resType = getContext().getResourceDefinition(retVal.getResourceType());
105 		IFhirResourceDao<? extends IResource> dao = null;
106 		if (resType != null) {
107 			dao = getDao(resType.getImplementingClass());
108 		}
109 		if (dao == null) {
110 			String msg = getContext().getLocalizer().getMessage(BaseFhirSystemDao.class, "transactionInvalidUrl", theAction, theUrl);
111 			throw new InvalidRequestException(msg);
112 		}
113 		retVal.setDao(dao);
114 
115 		if (retVal.getResourceId() == null && retVal.getParams() == null) {
116 			String msg = getContext().getLocalizer().getMessage(BaseFhirSystemDao.class, "transactionInvalidUrl", theAction, theUrl);
117 			throw new InvalidRequestException(msg);
118 		}
119 
120 		return retVal;
121 	}

484 	private String extractTransactionUrlOrThrowException(Entry nextEntry, HTTPVerbEnum verb) {
485 		String url = nextEntry.getTransaction().getUrl();
486 		if (isBlank(url)) {
487 			throw new InvalidRequestException(getContext().getLocalizer().getMessage(BaseFhirSystemDao.class, "transactionMissingUrl", verb.name()));
488 		}
489 		return url;
490 	}

492 	private static void handleTransactionCreateOrUpdateOutcome(Map<IdDt, IdDt> idSubstitutions, Map<IdDt, DaoMethodOutcome> idToPersistedOutcome, IdDt nextResourceId, DaoMethodOutcome outcome, Entry newEntry) {
493 		IdDt newId = outcome.getId().toUnqualifiedVersionless();
494 		IdDt resourceId = nextResourceId.toUnqualifiedVersionless();
495 		if (newId.equals(resourceId) == false) {
496 			idSubstitutions.put(resourceId, newId);
497 		}
498 		idToPersistedOutcome.put(newId, outcome);
499 		if (outcome.getCreated().booleanValue()) {
500 			newEntry.getTransactionResponse().setStatus(Long.toString(Constants.STATUS_HTTP_201_CREATED));
501 		} else {
502 			newEntry.getTransactionResponse().setStatus(Long.toString(Constants.STATUS_HTTP_200_OK));
503 		}
504 		newEntry.getTransactionResponse().setLocation(outcome.getId().toUnqualified().getValue());
505 		newEntry.getTransactionResponse().setEtag(outcome.getId().getVersionIdPart());
506 	}

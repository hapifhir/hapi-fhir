64  	private String extractTransactionUrlOrThrowException(Entry nextEntry, HTTPVerbEnum verb) {
65  		String url = nextEntry.getTransaction().getUrl();
66  		if (isBlank(url)) {
67  			throw new InvalidRequestException(getContext().getLocalizer().getMessage(BaseHapiFhirSystemDao.class, "transactionMissingUrl", verb.name()));
68  		}
69  		return url;
70  	}

84  	private UrlParts parseUrl(String theAction, String theUrl) {
85  		UrlParts retVal = new UrlParts();
86  
87  		//@formatter:off
88  		/* 
89  		 * We assume that the URL passed in is in one of the following forms:
90  		 * [Resource Type]?[Search Params]
91  		 * [Resource Type]/[Resource ID]
92  		 * [Resource Type]/[Resource ID]/_history/[Version ID]
93  		 */
94  		//@formatter:on
95  		int nextStart = 0;
96  		boolean nextIsHistory = false;
97  
98  		for (int idx = 0; idx < theUrl.length(); idx++) {
99  			char nextChar = theUrl.charAt(idx);
100 			boolean atEnd = (idx + 1) == theUrl.length();
101 			if (nextChar == '?' || nextChar == '/' || atEnd) {
102 				int endIdx = atEnd ? idx + 1 : idx;
103 				String nextSubstring = theUrl.substring(nextStart, endIdx);
104 				if (retVal.getResourceType() == null) {
105 					retVal.setResourceType(nextSubstring);
106 				} else if (retVal.getResourceId() == null) {
107 					retVal.setResourceId(nextSubstring);
108 				} else if (nextIsHistory) {
109 					retVal.setVersionId(nextSubstring);
110 				} else {
111 					if (nextSubstring.equals(Constants.URL_TOKEN_HISTORY)) {
112 						nextIsHistory = true;
113 					} else {
114 						String msg = getContext().getLocalizer().getMessage(BaseHapiFhirSystemDao.class, "transactionInvalidUrl", theAction, theUrl);
115 						throw new InvalidRequestException(msg);
116 					}
117 				}
118 				if (nextChar == '?') {
119 					if (theUrl.length() > idx + 1) {
120 						retVal.setParams(theUrl.substring(idx + 1, theUrl.length()));
121 					}
122 					break;
123 				}
124 				nextStart = idx + 1;
125 			}
126 		}
127 
128 		RuntimeResourceDefinition resType;
129 		try {
130 			resType = getContext().getResourceDefinition(retVal.getResourceType());
131 		} catch (DataFormatException e) {
132 			String msg = getContext().getLocalizer().getMessage(BaseHapiFhirSystemDao.class, "transactionInvalidUrl", theAction, theUrl);
133 			throw new InvalidRequestException(msg);
134 		}
135 		IFhirResourceDao<? extends IResource> dao = null;
136 		if (resType != null) {
137 			dao = getDao(resType.getImplementingClass());
138 		}
139 		if (dao == null) {
140 			String msg = getContext().getLocalizer().getMessage(BaseHapiFhirSystemDao.class, "transactionInvalidUrl", theAction, theUrl);
141 			throw new InvalidRequestException(msg);
142 		}
143 		retVal.setDao(dao);
144 
145 		if (retVal.getResourceId() == null && retVal.getParams() == null) {
146 			String msg = getContext().getLocalizer().getMessage(BaseHapiFhirSystemDao.class, "transactionInvalidUrl", theAction, theUrl);
147 			throw new InvalidRequestException(msg);
148 		}
149 
150 		return retVal;
151 	}

153 	@SuppressWarnings("unchecked")
154 	@Transactional(propagation = Propagation.REQUIRED)
155 	@Override
156 	public Bundle transaction(Bundle theResources) {
157 		ourLog.info("Beginning transaction with {} resources", theResources.getEntry().size());
158 		long start = System.currentTimeMillis();
159 
160 		Set<IdDt> allIds = new LinkedHashSet<IdDt>();
161 		Map<IdDt, IdDt> idSubstitutions = new HashMap<IdDt, IdDt>();
162 		Map<IdDt, DaoMethodOutcome> idToPersistedOutcome = new HashMap<IdDt, DaoMethodOutcome>();
163 
164 		Bundle response = new Bundle();
165 		OperationOutcome oo = new OperationOutcome();
166 		response.addEntry().setResource(oo);
167 
168 		for (int i = 0; i < theResources.getEntry().size(); i++) {
169 			Entry nextEntry = theResources.getEntry().get(i);
170 			IResource res = nextEntry.getResource();
171 			IdDt nextResourceId = null;
172 			if (res != null) {
173 
174 				nextResourceId = res.getId();
175 				if (nextResourceId.hasIdPart() && !nextResourceId.hasResourceType() && !isPlaceholder(nextResourceId)) {
176 					nextResourceId = new IdDt(toResourceName(res.getClass()), nextResourceId.getIdPart());
177 					res.setId(nextResourceId);
178 				}
179 
180 				/*
181 				 * Ensure that the bundle doesn't have any duplicates, since this causes all kinds of weirdness
182 				 */
183 				if (isPlaceholder(nextResourceId)) {
184 					if (!allIds.add(nextResourceId)) {
185 						throw new InvalidRequestException(getContext().getLocalizer().getMessage(BaseHapiFhirSystemDao.class, "transactionContainsMultipleWithDuplicateId", nextResourceId));
186 					}
187 				} else if (nextResourceId.hasResourceType() && nextResourceId.hasIdPart()) {
188 					IdDt nextId = nextResourceId.toUnqualifiedVersionless();
189 					if (!allIds.add(nextId)) {
190 						throw new InvalidRequestException(getContext().getLocalizer().getMessage(BaseHapiFhirSystemDao.class, "transactionContainsMultipleWithDuplicateId", nextId));
191 					}
192 				}
193 
194 			}
195 
196 			HTTPVerbEnum verb = nextEntry.getTransaction().getMethodElement().getValueAsEnum();
197 			if (verb == null) {
198 				throw new InvalidRequestException(getContext().getLocalizer().getMessage(BaseHapiFhirSystemDao.class, "transactionEntryHasInvalidVerb", nextEntry.getTransaction().getMethod()));
199 			}
200 
201 			String resourceType = res != null ? getContext().getResourceDefinition(res).getName() : null;
202 
203 			switch (verb) {
204 			case POST: {
205 				// CREATE
206 				@SuppressWarnings("rawtypes")
207 				IFhirResourceDao resourceDao = getDao(res.getClass());
208 				res.setId((String) null);
209 				DaoMethodOutcome outcome;
210 				Entry newEntry = response.addEntry();
211 				outcome = resourceDao.create(res, nextEntry.getTransaction().getIfNoneExist(), false);
212 				handleTransactionCreateOrUpdateOutcome(idSubstitutions, idToPersistedOutcome, nextResourceId, outcome, newEntry, resourceType);
213 				break;
214 			}
215 			case DELETE: {
216 				// DELETE
217 				Entry newEntry = response.addEntry();
218 				String url = extractTransactionUrlOrThrowException(nextEntry, verb);
219 				UrlParts parts = parseUrl(verb.getCode(), url);
220 				if (parts.getResourceId() != null) {
221 					parts.getDao().delete(new IdDt(parts.getResourceType(), parts.getResourceId()));
222 				} else {
223 					parts.getDao().deleteByUrl(parts.getResourceType() + '?' + parts.getParams());
224 				}
225 
226 				newEntry.getTransactionResponse().setStatus(Integer.toString(Constants.STATUS_HTTP_204_NO_CONTENT));
227 				break;
228 			}
229 			case PUT: {
230 				// UPDATE
231 				@SuppressWarnings("rawtypes")
232 				IFhirResourceDao resourceDao = getDao(res.getClass());
233 
234 				DaoMethodOutcome outcome;
235 				Entry newEntry = response.addEntry();
236 
237 				String url = extractTransactionUrlOrThrowException(nextEntry, verb);
238 
239 				UrlParts parts = parseUrl(verb.getCode(), url);
240 				if (isNotBlank(parts.getResourceId())) {
241 					res.setId(new IdDt(parts.getResourceType(), parts.getResourceId()));
242 					outcome = resourceDao.update(res, null, false);
243 				} else {
244 					res.setId((String) null);
245 					outcome = resourceDao.update(res, parts.getResourceType() + '?' + parts.getParams(), false);
246 				}
247 
248 				handleTransactionCreateOrUpdateOutcome(idSubstitutions, idToPersistedOutcome, nextResourceId, outcome, newEntry, resourceType);
249 				break;
250 			}
251 			case GET: {
252 				// SEARCH/READ/VREAD
253 				String url = extractTransactionUrlOrThrowException(nextEntry, verb);
254 				UrlParts parts = parseUrl(verb.getCode(), url);
255 
256 				@SuppressWarnings("rawtypes")
257 				IFhirResourceDao resourceDao = parts.getDao();
258 
259 				String ifNoneMatch = nextEntry.getTransaction().getIfNoneMatch();
260 				if (isNotBlank(ifNoneMatch)) {
261 					ifNoneMatch = MethodUtil.parseETagValue(ifNoneMatch);
262 				}
263 				
264 				if (parts.getResourceId() != null && parts.getParams() == null) {
265 					IResource found;
266 					boolean notChanged = false;
267 					if (parts.getVersionId() != null) {
268 						if (isNotBlank(ifNoneMatch)) {
269 							throw new InvalidRequestException("Unable to perform vread on '" + url + "' with ifNoneMatch also set. Do not include a version in the URL to perform a conditional read.");
270 						}
271 						found = resourceDao.read(new IdDt(parts.getResourceType(), parts.getResourceId(), parts.getVersionId()));
272 					} else {
273 						found = resourceDao.read(new IdDt(parts.getResourceType(), parts.getResourceId()));
274 						if (isNotBlank(ifNoneMatch) && ifNoneMatch.equals(found.getId().getVersionIdPart())) {
275 							notChanged = true;
276 						}
277 					}
278 					Entry entry = response.addEntry();
279 					if (notChanged == false) {
280 						entry.setResource(found);
281 					}
282 					EntryTransactionResponse resp = entry.getTransactionResponse();
283 					resp.setLocation(found.getId().toUnqualified().getValue());
284 					resp.setEtag(found.getId().getVersionIdPart());
285 					if (!notChanged) {
286 						resp.setStatus(Integer.toString(Constants.STATUS_HTTP_200_OK));
287 					} else {
288 						resp.setStatus(Integer.toString(Constants.STATUS_HTTP_304_NOT_MODIFIED));
289 					}
290 				} else if (parts.getParams() != null) {
291 					RuntimeResourceDefinition def = getContext().getResourceDefinition(parts.getDao().getResourceType());
292 					SearchParameterMap params = translateMatchUrl(url, def);
293 					IBundleProvider bundle = parts.getDao().search(params);
294 
295 					Bundle searchBundle = new Bundle();
296 					searchBundle.setTotal(bundle.size());
297 
298 					int configuredMax = 100; // this should probably be configurable or something
299 					if (bundle.size() > configuredMax) {
300 						oo.addIssue().setSeverity(IssueSeverityEnum.WARNING)
301 								.setDetails("Search nested within transaction found more than " + configuredMax + " matches, but paging is not supported in nested transactions");
302 					}
303 					List<IBaseResource> resourcesToAdd = bundle.getResources(0, Math.min(bundle.size(), configuredMax));
304 					for (IBaseResource next : resourcesToAdd) {
305 						searchBundle.addEntry().setResource((IResource) next);
306 					}
307 
308 					Entry newEntry = response.addEntry();
309 					newEntry.setResource(searchBundle);
310 					newEntry.getTransactionResponse().setStatus(Integer.toString(Constants.STATUS_HTTP_200_OK));
311 				}
312 			}
313 			}
314 
315 		}
316 
317 		FhirTerser terser = getContext().newTerser();
318 
319 		for (DaoMethodOutcome nextOutcome : idToPersistedOutcome.values()) {
320 			IResource nextResource = (IResource) nextOutcome.getResource();
321 			if (nextResource == null) {
322 				continue;
323 			}
324 
325 			List<BaseResourceReferenceDt> allRefs = terser.getAllPopulatedChildElementsOfType(nextResource, BaseResourceReferenceDt.class);
326 			for (BaseResourceReferenceDt nextRef : allRefs) {
327 				IdDt nextId = nextRef.getReference();
328 				if (idSubstitutions.containsKey(nextId)) {
329 					IdDt newId = idSubstitutions.get(nextId);
330 					ourLog.info(" * Replacing resource ref {} with {}", nextId, newId);
331 					nextRef.setReference(newId);
332 				} else {
333 					ourLog.debug(" * Reference [{}] does not exist in bundle", nextId);
334 				}
335 			}
336 
337 			InstantDt deletedInstantOrNull = ResourceMetadataKeyEnum.DELETED_AT.get(nextResource);
338 			Date deletedTimestampOrNull = deletedInstantOrNull != null ? deletedInstantOrNull.getValue() : null;
339 			updateEntity(nextResource, nextOutcome.getEntity(), false, deletedTimestampOrNull, true, false);
340 		}
341 
342 		long delay = System.currentTimeMillis() - start;
343 		ourLog.info("Transaction completed in {}ms", new Object[] { delay });
344 
345 		oo.addIssue().setSeverity(IssueSeverityEnum.INFORMATION).setDetails("Transaction completed in " + delay + "ms");
346 
347 		for (IdDt next : allIds) {
348 			IdDt replacement = idSubstitutions.get(next);
349 			if (replacement == null) {
350 				continue;
351 			}
352 			if (replacement.equals(next)) {
353 				continue;
354 			}
355 			oo.addIssue().setSeverity(IssueSeverityEnum.INFORMATION).setDetails("Placeholder resource ID \"" + next + "\" was replaced with permanent ID \"" + replacement + "\"");
356 		}
357 
358 		notifyWriteCompleted();
359 
360 		response.setType(BundleTypeEnum.TRANSACTION_RESPONSE);
361 		return response;
362 	}

364 	private static void handleTransactionCreateOrUpdateOutcome(Map<IdDt, IdDt> idSubstitutions, Map<IdDt, DaoMethodOutcome> idToPersistedOutcome, IdDt nextResourceId, DaoMethodOutcome outcome,
365 			Entry newEntry, String theResourceType) {
366 		IdDt newId = (IdDt) outcome.getId().toUnqualifiedVersionless();
367 		IdDt resourceId = isPlaceholder(nextResourceId) ? nextResourceId : nextResourceId.toUnqualifiedVersionless();
368 		if (newId.equals(resourceId) == false) {
369 			idSubstitutions.put(resourceId, newId);
370 			if (isPlaceholder(resourceId)) {
371 				/*
372 				 * The correct way for substitution IDs to be is to be with no resource type, but we'll accept the qualified kind too just to be lenient.
373 				 */
374 				idSubstitutions.put(new IdDt(theResourceType + '/' + resourceId.getValue()), newId);
375 			}
376 		}
377 		idToPersistedOutcome.put(newId, outcome);
378 		if (outcome.getCreated().booleanValue()) {
379 			newEntry.getTransactionResponse().setStatus(Long.toString(Constants.STATUS_HTTP_201_CREATED));
380 		} else {
381 			newEntry.getTransactionResponse().setStatus(Long.toString(Constants.STATUS_HTTP_200_OK));
382 		}
383 		newEntry.getTransactionResponse().setLocation(outcome.getId().toUnqualified().getValue());
384 		newEntry.getTransactionResponse().setEtag(outcome.getId().getVersionIdPart());
385 	}

387 	private static boolean isPlaceholder(IdDt theId) {
388 		if ("urn:oid:".equals(theId.getBaseUrl()) || "urn:uuid:".equals(theId.getBaseUrl())) {
389 			return true;
390 		}
391 		return false;
392 	}

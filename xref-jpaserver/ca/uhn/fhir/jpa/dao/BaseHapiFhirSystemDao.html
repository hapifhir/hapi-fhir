96  	private int doPerformReindexingPassForResources(final Integer theCount, TransactionTemplate txTemplate) {
97  		return txTemplate.execute(new TransactionCallback<Integer>() {
98  			@SuppressWarnings("unchecked")
99  			@Override
100 			public Integer doInTransaction(TransactionStatus theStatus) {
101 				TypedQuery<ResourceTable> q = myEntityManager.createQuery("SELECT t FROM " + ResourceTable.class.getSimpleName() + " t WHERE t.myIndexStatus IS null", ResourceTable.class);
102 
103 				int maxResult = 500;
104 				if (theCount != null) {
105 					maxResult = Math.min(theCount, 2000);
106 				}
107 
108 				q.setMaxResults(maxResult);
109 				List<ResourceTable> resources = q.getResultList();
110 				if (resources.isEmpty()) {
111 					return 0;
112 				}
113 
114 				ourLog.info("Indexing {} resources", resources.size());
115 
116 				int count = 0;
117 				long start = System.currentTimeMillis();
118 
119 				for (ResourceTable resourceTable : resources) {
120 					try {
121 						/*
122 						 * This part is because from HAPI 1.5 - 1.6 we changed the format of forced ID to be "type/id" instead of just "id"
123 						 */
124 						ForcedId forcedId = resourceTable.getForcedId();
125 						if (forcedId != null) {
126 							if (isBlank(forcedId.getResourceType())) {
127 								ourLog.info("Updating resource {} forcedId type to {}", forcedId.getForcedId(), resourceTable.getResourceType());
128 								forcedId.setResourceType(resourceTable.getResourceType());
129 								myForcedIdDao.save(forcedId);
130 							}
131 						}
132 
133 						final IBaseResource resource = toResource(resourceTable, false);
134 
135 						@SuppressWarnings("rawtypes")
136 						final IFhirResourceDao dao = getDao(resource.getClass());
137 
138 						dao.reindex(resource, resourceTable);
139 					} catch (Exception e) {
140 						ourLog.error("Failed to index resource {}: {}", new Object[] { resourceTable.getIdDt(), e.toString(), e });
141 						throw new ReindexFailureException(resourceTable.getId());
142 					}
143 					count++;
144 				}
145 
146 				long delay = System.currentTimeMillis() - start;
147 				long avg = (delay / resources.size());
148 				ourLog.info("Indexed {} / {} resources in {}ms - Avg {}ms / resource", new Object[] { count, resources.size(), delay, avg });
149 
150 				return resources.size();
151 			}
152 		});
153 	}

167 	@Override
168 	public Map<String, Long> getResourceCounts() {
169 		CriteriaBuilder builder = myEntityManager.getCriteriaBuilder();
170 		CriteriaQuery<Tuple> cq = builder.createTupleQuery();
171 		Root<?> from = cq.from(ResourceTable.class);
172 		cq.multiselect(from.get("myResourceType").as(String.class), builder.count(from.get("myResourceType")).as(Long.class));
173 		cq.groupBy(from.get("myResourceType"));
174 
175 		TypedQuery<Tuple> q = myEntityManager.createQuery(cq);
176 
177 		Map<String, Long> retVal = new HashMap<String, Long>();
178 		for (Tuple next : q.getResultList()) {
179 			String resourceName = next.get(0, String.class);
180 			Long count = next.get(1, Long.class);
181 			retVal.put(resourceName, count);
182 		}
183 		return retVal;
184 	}

222 	private void markResourceAsIndexingFailed(final long theId) {
223 		TransactionTemplate txTemplate = new TransactionTemplate(myTxManager);
224 		txTemplate.setPropagationBehavior(TransactionTemplate.PROPAGATION_REQUIRES_NEW);
225 		txTemplate.execute(new TransactionCallback<Void>() {
226 			@Override
227 			public Void doInTransaction(TransactionStatus theStatus) {
228 				ourLog.info("Marking resource with PID {} as indexing_failed", new Object[] { theId });
229 				Query q = myEntityManager.createQuery("UPDATE ResourceTable t SET t.myIndexStatus = :status WHERE t.myId = :id");
230 				q.setParameter("status", INDEX_STATUS_INDEXING_FAILED);
231 				q.setParameter("id", theId);
232 				q.executeUpdate();
233 				
234 				q = myEntityManager.createQuery("DELETE FROM ResourceTag t WHERE t.myResourceId = :id");
235 				q.setParameter("id", theId);
236 				q.executeUpdate();
237 
238 				q = myEntityManager.createQuery("DELETE FROM ResourceIndexedSearchParamCoords t WHERE t.myResourcePid = :id");
239 				q.setParameter("id", theId);
240 				q.executeUpdate();
241 
242 				q = myEntityManager.createQuery("DELETE FROM ResourceIndexedSearchParamDate t WHERE t.myResourcePid = :id");
243 				q.setParameter("id", theId);
244 				q.executeUpdate();
245 
246 				q = myEntityManager.createQuery("DELETE FROM ResourceIndexedSearchParamNumber t WHERE t.myResourcePid = :id");
247 				q.setParameter("id", theId);
248 				q.executeUpdate();
249 
250 				q = myEntityManager.createQuery("DELETE FROM ResourceIndexedSearchParamQuantity t WHERE t.myResourcePid = :id");
251 				q.setParameter("id", theId);
252 				q.executeUpdate();
253 
254 				q = myEntityManager.createQuery("DELETE FROM ResourceIndexedSearchParamString t WHERE t.myResourcePid = :id");
255 				q.setParameter("id", theId);
256 				q.executeUpdate();
257 
258 				q = myEntityManager.createQuery("DELETE FROM ResourceIndexedSearchParamToken t WHERE t.myResourcePid = :id");
259 				q.setParameter("id", theId);
260 				q.executeUpdate();
261 
262 				q = myEntityManager.createQuery("DELETE FROM ResourceIndexedSearchParamUri t WHERE t.myResourcePid = :id");
263 				q.setParameter("id", theId);
264 				q.executeUpdate();
265 
266 				q = myEntityManager.createQuery("DELETE FROM ResourceLink t WHERE t.mySourceResourcePid = :id");
267 				q.setParameter("id", theId);
268 				q.executeUpdate();
269 
270 				q = myEntityManager.createQuery("DELETE FROM ResourceLink t WHERE t.myTargetResourcePid = :id");
271 				q.setParameter("id", theId);
272 				q.executeUpdate();
273 
274 				return null;
275 			}
276 		});
277 	}

279 	@Override
280 	@Transactional(propagation = Propagation.NOT_SUPPORTED)
281 	public int performReindexingPass(final Integer theCount) {
282 		if (!myReindexLock.tryLock()) {
283 			return 0;
284 		}
285 		try {
286 			return doPerformReindexingPass(theCount);
287 		} catch (ReindexFailureException e) {
288 			ourLog.warn("Reindexing failed for resource {}", e.getResourceId());
289 			markResourceAsIndexingFailed(e.getResourceId());
290 			return -1;
291 		} finally {
292 			myReindexLock.unlock();
293 		}
294 	}

300 	protected ResourceTable tryToLoadEntity(IdDt nextId) {
301 		ResourceTable entity;
302 		try {
303 			Long pid = translateForcedIdToPid(nextId.getResourceType(), nextId.getIdPart());
304 			entity = myEntityManager.find(ResourceTable.class, pid);
305 		} catch (ResourceNotFoundException e) {
306 			entity = null;
307 		}
308 		return entity;
309 	}

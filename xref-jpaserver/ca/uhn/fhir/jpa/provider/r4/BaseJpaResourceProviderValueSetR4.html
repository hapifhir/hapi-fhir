38  	@Operation(name = "$expand", idempotent = true)
39  	public ValueSet expand(
40  		HttpServletRequest theServletRequest,
41  		@IdParam(optional = true) IdType theId,
42  		@OperationParam(name = "valueSet", min = 0, max = 1) ValueSet theValueSet,
43  		@OperationParam(name = "url", min = 0, max = 1) UriType theUrl,
44  		@OperationParam(name = "filter", min = 0, max = 1) StringType theFilter,
45  		RequestDetails theRequestDetails) {
46  
47  		boolean haveId = theId != null && theId.hasIdPart();
48  		boolean haveIdentifier = theUrl != null && isNotBlank(theUrl.getValue());
49  		boolean haveValueSet = theValueSet != null && theValueSet.isEmpty() == false;
50  
51  		if (!haveId && !haveIdentifier && !haveValueSet) {
52  			throw new InvalidRequestException("$expand operation at the type level (no ID specified) requires a url or a valueSet as a part of the request");
53  		}
54  
55  		if (moreThanOneTrue(haveId, haveIdentifier, haveValueSet)) {
56  			throw new InvalidRequestException("$expand must EITHER be invoked at the instance level, or have a url specified, or have a ValueSet specified. Can not combine these options.");
57  		}
58  
59  		startRequest(theServletRequest);
60  		try {
61  			IFhirResourceDaoValueSet<ValueSet, Coding, CodeableConcept> dao = (IFhirResourceDaoValueSet<ValueSet, Coding, CodeableConcept>) getDao();
62  			if (haveId) {
63  				return dao.expand(theId, toFilterString(theFilter), theRequestDetails);
64  			} else if (haveIdentifier) {
65  				return dao.expandByIdentifier(theUrl.getValue(), toFilterString(theFilter));
66  			} else {
67  				return dao.expand(theValueSet, toFilterString(theFilter));
68  			}
69  
70  		} finally {
71  			endRequest(theServletRequest);
72  		}
73  	}
74  
75  
76  	private String toFilterString(StringType theFilter) {
77  		return theFilter != null ? theFilter.getValue() : null;
78  	}

81  	@SuppressWarnings("unchecked")
82  	@Operation(name = "$validate-code", idempotent = true, returnParameters = {
83  		@OperationParam(name = "result", type = BooleanType.class, min = 1),
84  		@OperationParam(name = "message", type = StringType.class),
85  		@OperationParam(name = "display", type = StringType.class)
86  	})
87  	public Parameters validateCode(
88  		HttpServletRequest theServletRequest,
89  		@IdParam(optional = true) IdType theId,
90  		@OperationParam(name = "url", min = 0, max = 1) UriType theValueSetUrl,
91  		@OperationParam(name = "code", min = 0, max = 1) CodeType theCode,
92  		@OperationParam(name = "system", min = 0, max = 1) UriType theSystem,
93  		@OperationParam(name = "display", min = 0, max = 1) StringType theDisplay,
94  		@OperationParam(name = "coding", min = 0, max = 1) Coding theCoding,
95  		@OperationParam(name = "codeableConcept", min = 0, max = 1) CodeableConcept theCodeableConcept,
96  		RequestDetails theRequestDetails
97  	) {
98  
99  		startRequest(theServletRequest);
100 		try {
101 			IFhirResourceDaoValueSet<ValueSet, Coding, CodeableConcept> dao = (IFhirResourceDaoValueSet<ValueSet, Coding, CodeableConcept>) getDao();
102 			ValidateCodeResult result = dao.validateCode(theValueSetUrl, theId, theCode, theSystem, theDisplay, theCoding, theCodeableConcept, theRequestDetails);
103 			Parameters retVal = new Parameters();
104 			retVal.addParameter().setName("result").setValue(new BooleanType(result.isResult()));
105 			if (isNotBlank(result.getMessage())) {
106 				retVal.addParameter().setName("message").setValue(new StringType(result.getMessage()));
107 			}
108 			if (isNotBlank(result.getDisplay())) {
109 				retVal.addParameter().setName("display").setValue(new StringType(result.getDisplay()));
110 			}
111 			return retVal;
112 		} finally {
113 			endRequest(theServletRequest);
114 		}
115 	}

118 	private static boolean moreThanOneTrue(boolean... theBooleans) {
119 		boolean haveOne = false;
120 		for (boolean next : theBooleans) {
121 			if (next) {
122 				if (haveOne) {
123 					return true;
124 				} else {
125 					haveOne = true;
126 				}
127 			}
128 		}
129 		return false;
130 	}

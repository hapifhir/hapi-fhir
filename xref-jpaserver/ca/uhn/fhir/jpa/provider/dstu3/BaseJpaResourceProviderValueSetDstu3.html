49  	@Operation(name = "$expand", idempotent = true)
50  	public ValueSet expand(
51  			HttpServletRequest theServletRequest,
52  			@IdParam(optional=true) IdType theId,
53  			@OperationParam(name="valueSet", min=0, max=1) ValueSet theValueSet,
54  			@OperationParam(name="identifier", min=0, max=1) UriType theIdentifier,
55  			@OperationParam(name = "filter", min=0, max=1) StringType theFilter) {
56  		//@formatter:on
57  		
58  		boolean haveId = theId != null && theId.hasIdPart();
59  		boolean haveIdentifier = theIdentifier != null && isNotBlank(theIdentifier.getValue());
60  		boolean haveValueSet = theValueSet != null && theValueSet.isEmpty() == false;
61  		
62  		if (!haveId && !haveIdentifier && !haveValueSet) {
63  			throw new InvalidRequestException("$expand operation at the type level (no ID specified) requires an identifier or a valueSet as a part of the request");
64  		}
65  
66  		if (moreThanOneTrue(haveId, haveIdentifier, haveValueSet)) {
67  			throw new InvalidRequestException("$expand must EITHER be invoked at the type level, or have an identifier specified, or have a ValueSet specified. Can not combine these options.");
68  		}
69  		
70  		startRequest(theServletRequest);
71  		try {
72  			IFhirResourceDaoValueSet<ValueSet, Coding, CodeableConcept> dao = (IFhirResourceDaoValueSet<ValueSet, Coding, CodeableConcept>) getDao();
73  			if (haveId) {
74  				return dao.expand(theId, toFilterString(theFilter));
75  			} else if (haveIdentifier) {
76  				return dao.expandByIdentifier(theIdentifier.getValue(), toFilterString(theFilter));
77  			} else {
78  				return dao.expand(theValueSet, toFilterString(theFilter));
79  			}
80  			
81  		} finally {
82  			endRequest(theServletRequest);
83  		}
84  	}
85  
86  
87  	private static boolean moreThanOneTrue(boolean... theBooleans) {
88  		boolean haveOne = false;
89  		for (boolean next : theBooleans) {
90  			if (next) {
91  				if (haveOne) {
92  					return true;
93  				} else {
94  					haveOne = true;
95  				}
96  			}
97  		}
98  		return false;
99  	}

102 	private String toFilterString(StringType theFilter) {
103 		return theFilter != null ? theFilter.getValue() : null;
104 	}

107 	@Operation(name = "$lookup", idempotent = true, returnParameters= {
108 		@OperationParam(name="name", type=StringType.class, min=1),
109 		@OperationParam(name="version", type=StringType.class, min=0),
110 		@OperationParam(name="display", type=StringType.class, min=1),
111 		@OperationParam(name="abstract", type=BooleanType.class, min=1),
112 	})
113 	public Parameters lookup(
114 			HttpServletRequest theServletRequest,
115 			@OperationParam(name="code", min=0, max=1) CodeType theCode, 
116 			@OperationParam(name="system", min=0, max=1) UriType theSystem,
117 			@OperationParam(name="coding", min=0, max=1) Coding theCoding 
118 			) {
119 		//@formatter:on
120 		
121 		startRequest(theServletRequest);
122 		try {
123 			IFhirResourceDaoValueSet<ValueSet, Coding, CodeableConcept> dao = (IFhirResourceDaoValueSet<ValueSet, Coding, CodeableConcept>) getDao();
124 			LookupCodeResult result = dao.lookupCode(theCode, theSystem, theCoding);
125 			if (result.isFound()==false) {
126 				throw new ResourceNotFoundException("Unable to find code[" + result.getSearchedForCode() + "] in system[" + result.getSearchedForSystem() + "]");
127 			}
128 			Parameters retVal = new Parameters();
129 			retVal.addParameter().setName("name").setValue(new StringType(result.getCodeSystemDisplayName()));
130 			if (isNotBlank(result.getCodeSystemVersion())) {
131 				retVal.addParameter().setName("version").setValue(new StringType(result.getCodeSystemVersion()));
132 			}
133 			retVal.addParameter().setName("display").setValue(new StringType(result.getCodeDisplay()));
134 			retVal.addParameter().setName("abstract").setValue(new BooleanType(result.isCodeIsAbstract()));			
135 			return retVal;
136 		} finally {
137 			endRequest(theServletRequest);
138 		}
139 	}

143 	@Operation(name = "$validate-code", idempotent = true, returnParameters= {
144 		@OperationParam(name="result", type=BooleanType.class, min=1),
145 		@OperationParam(name="message", type=StringType.class),
146 		@OperationParam(name="display", type=StringType.class)
147 	})
148 	public Parameters validateCode(
149 			HttpServletRequest theServletRequest,
150 			@IdParam(optional=true) IdType theId, 
151 			@OperationParam(name="identifier", min=0, max=1) UriType theValueSetIdentifier, 
152 			@OperationParam(name="code", min=0, max=1) CodeType theCode, 
153 			@OperationParam(name="system", min=0, max=1) UriType theSystem,
154 			@OperationParam(name="display", min=0, max=1) StringType theDisplay,
155 			@OperationParam(name="coding", min=0, max=1) Coding theCoding,
156 			@OperationParam(name="codeableConcept", min=0, max=1) CodeableConcept theCodeableConcept
157 			) {
158 		//@formatter:on
159 		
160 		startRequest(theServletRequest);
161 		try {
162 			IFhirResourceDaoValueSet<ValueSet, Coding, CodeableConcept> dao = (IFhirResourceDaoValueSet<ValueSet, Coding, CodeableConcept>) getDao();
163 			ValidateCodeResult result = dao.validateCode(theValueSetIdentifier, theId, theCode, theSystem, theDisplay, theCoding, theCodeableConcept);
164 			Parameters retVal = new Parameters();
165 			retVal.addParameter().setName("result").setValue(new BooleanType(result.isResult()));
166 			if (isNotBlank(result.getMessage())) {
167 				retVal.addParameter().setName("message").setValue(new StringType(result.getMessage()));
168 			}
169 			if (isNotBlank(result.getDisplay())) {
170 				retVal.addParameter().setName("display").setValue(new StringType(result.getDisplay()));
171 			}
172 			return retVal;
173 		} finally {
174 			endRequest(theServletRequest);
175 		}

23  import static org.apache.commons.lang3.StringUtils.defaultString;
24  import static org.apache.commons.lang3.StringUtils.isNotBlank;

26  import java.io.FileInputStream;
27  import java.io.IOException;
28  import java.util.ArrayList;
29  import java.util.List;
30  
31  import javax.servlet.http.HttpServletRequest;
32  
33  import org.apache.commons.io.IOUtils;
34  import org.hl7.fhir.dstu3.model.Attachment;
35  import org.hl7.fhir.dstu3.model.IntegerType;
36  import org.hl7.fhir.dstu3.model.Parameters;
37  import org.hl7.fhir.dstu3.model.StringType;
38  import org.hl7.fhir.dstu3.model.UriType;
39  import org.springframework.beans.factory.annotation.Autowired;

41  import ca.uhn.fhir.jpa.provider.BaseJpaProvider;
42  import ca.uhn.fhir.jpa.term.IHapiTerminologyLoaderSvc;
43  import ca.uhn.fhir.jpa.term.IHapiTerminologyLoaderSvc.UploadStatistics;
44  import ca.uhn.fhir.rest.annotation.Operation;
45  import ca.uhn.fhir.rest.annotation.OperationParam;
46  import ca.uhn.fhir.rest.method.RequestDetails;
47  import ca.uhn.fhir.rest.server.exceptions.InternalErrorException;
48  import ca.uhn.fhir.rest.server.exceptions.InvalidRequestException;
49  
50  public class TerminologyUploaderProviderDstu3 extends BaseJpaProvider {
51  	public static final String UPLOAD_EXTERNAL_CODE_SYSTEM = "$upload-external-code-system";
52  
53  	private static final org.slf4j.Logger ourLog = org.slf4j.LoggerFactory.getLogger(TerminologyUploaderProviderDstu3.class);
54  
55  	@Autowired
56  	private IHapiTerminologyLoaderSvc myTerminologyLoaderSvc;
57  	
58  	//@formatter:off
59  	@Operation(name = UPLOAD_EXTERNAL_CODE_SYSTEM, idempotent = false, returnParameters= {
60  		@OperationParam(name="conceptCount", type=IntegerType.class, min=1)
61  	})
62  	public Parameters uploadExternalCodeSystem(
63  			HttpServletRequest theServletRequest,
64  			@OperationParam(name="url", min=1) UriType theUrl,
65  			@OperationParam(name="package", min=0) Attachment thePackage,
66  			@OperationParam(name="localfile", min=0, max=OperationParam.MAX_UNLIMITED) List<StringType> theLocalFile,
67  			RequestDetails theRequestDetails 
68  			) {
69  		//@formatter:on
70  		
71  		startRequest(theServletRequest);
72  		try {
73  			List<byte[]> data = new ArrayList<byte[]>();
74  			if (theLocalFile != null && theLocalFile.size() > 0) {
75  				for (StringType nextLocalFile : theLocalFile) {
76  					if (isNotBlank(nextLocalFile.getValue())) {
77  						ourLog.info("Reading in local file: {}", nextLocalFile.getValue());
78  						try {
79  							byte[] nextData = IOUtils.toByteArray(new FileInputStream(nextLocalFile.getValue()));
80  							data.add(nextData);
81  						} catch (IOException e) {
82  							throw new InternalErrorException(e);
83  						}
84  					}
85  				}
86  			} else if (thePackage == null || thePackage.getData() == null || thePackage.getData().length == 0) {
87  				throw new InvalidRequestException("No 'localfile' or 'package' parameter, or package had no data");
88  			} else {
89  				data = new ArrayList<byte[]>();
90  				data.add(thePackage.getData());
91  				thePackage.setData(null);
92  			}
93  			
94  			String url = theUrl != null ? theUrl.getValueAsString() : null;
95  			url = defaultString(url);
96  
97  			UploadStatistics stats;
98  			if (IHapiTerminologyLoaderSvc.SCT_URL.equals(url)) {
99  				stats = myTerminologyLoaderSvc.loadSnomedCt((data), theRequestDetails);
100 			} else if (IHapiTerminologyLoaderSvc.LOINC_URL.equals(url)) {
101 					stats = myTerminologyLoaderSvc.loadLoinc((data), theRequestDetails);
102 			} else {
103 				throw new InvalidRequestException("Unknown URL: " + url);
104 			}
105 			
106 			Parameters retVal = new Parameters();
107 			retVal.addParameter().setName("conceptCount").setValue(new IntegerType(stats.getConceptCount()));
108 			return retVal;
109 		} finally {
110 			endRequest(theServletRequest);
111 		}
112 	}

196 				theValueSetCodeAccumulator.excludeConcept(theCodeSystem, theCode);

344 			optionalTermValueSet = myValueSetDao.findByUrl(theValueSetToExpand.getUrl());
345 		} else {
346 			throw new UnprocessableEntityException("ValueSet to be expanded must provide either ValueSet.id or ValueSet.url");
347 		}
348 
349 		if (!optionalTermValueSet.isPresent()) {
350 			ourLog.warn("ValueSet is not present in terminology tables. Will perform in-memory expansion without parameters. {}", getValueSetInfo(theValueSetToExpand));
351 			return expandValueSetInMemory(theValueSetToExpand, null); // In-memory expansion.
352 		}
353 
354 		TermValueSet termValueSet = optionalTermValueSet.get();
355 
356 		if (termValueSet.getExpansionStatus() != TermValueSetPreExpansionStatusEnum.EXPANDED) {
357 			ourLog.warn("{} is present in terminology tables but not ready for persistence-backed invocation of operation $expand. Will perform in-memory expansion without parameters. Current status: {} | {}",
358 				getValueSetInfo(theValueSetToExpand), termValueSet.getExpansionStatus().name(), termValueSet.getExpansionStatus().getDescription());
359 			return expandValueSetInMemory(theValueSetToExpand, null); // In-memory expansion.
360 		}
361 
362 		ValueSet.ValueSetExpansionComponent expansionComponent = new ValueSet.ValueSetExpansionComponent();
363 		expansionComponent.setIdentifier(UUID.randomUUID().toString());
364 		expansionComponent.setTimestamp(new Date());
365 
366 		populateExpansionComponent(expansionComponent, termValueSet, theOffset, theCount);
367 
368 		ValueSet valueSet = new ValueSet();
369 		valueSet.setStatus(Enumerations.PublicationStatus.ACTIVE);
370 		valueSet.setCompose(theValueSetToExpand.getCompose());
371 		valueSet.setExpansion(expansionComponent);
372 		return valueSet;
373 	}

389 	private void expandConcepts(ValueSet.ValueSetExpansionComponent theExpansionComponent, TermValueSet theTermValueSet, int theOffset, int theCount) {
390 		int conceptsExpanded = 0;
391 		int designationsExpanded = 0;
392 		int toIndex = theOffset + theCount;
393 		Collection<TermValueSetConceptView> conceptViews = myTermValueSetConceptViewDao.findByTermValueSetId(theOffset, toIndex, theTermValueSet.getId());
394 
395 		if (conceptViews.isEmpty()) {
396 			logConceptsExpanded("No concepts to expand. ", theTermValueSet, conceptsExpanded);
397 			return;
398 		}
399 
400 		Map<Long, ValueSet.ValueSetExpansionContainsComponent> pidToConcept = new HashMap<>();
401 		for (TermValueSetConceptView conceptView : conceptViews) {
402 
403 			Long conceptPid = conceptView.getConceptPid();
404 			ValueSet.ValueSetExpansionContainsComponent containsComponent;
405 
406 			if (!pidToConcept.containsKey(conceptPid)) {
407 				containsComponent = theExpansionComponent.addContains();
408 				containsComponent.setSystem(conceptView.getConceptSystemUrl());
409 				containsComponent.setCode(conceptView.getConceptCode());
410 				containsComponent.setDisplay(conceptView.getConceptDisplay());
411 				pidToConcept.put(conceptPid, containsComponent);
412 			} else {
413 				containsComponent = pidToConcept.get(conceptPid);
414 			}
415 
416 			// TODO: DM 2019-08-17 - Implement includeDesignations parameter for $expand operation to designations optional.
417 			if (conceptView.getDesignationPid() != null) {
418 				ValueSet.ConceptReferenceDesignationComponent designationComponent = containsComponent.addDesignation();
419 				designationComponent.setLanguage(conceptView.getDesignationLang());
420 				designationComponent.setUse(new Coding(
421 					conceptView.getDesignationUseSystem(),
422 					conceptView.getDesignationUseCode(),
423 					conceptView.getDesignationUseDisplay()));
424 				designationComponent.setValue(conceptView.getDesignationVal());
425 
426 				if (++designationsExpanded % 250 == 0) {
427 					logDesignationsExpanded("Expansion of designations in progress. ", theTermValueSet, designationsExpanded);
428 				}
429 			}
430 
431 			if (++conceptsExpanded % 250 == 0) {
432 				logConceptsExpanded("Expansion of concepts in progress. ", theTermValueSet, conceptsExpanded);
433 			}
434 		}
435 
436 		logDesignationsExpanded("Finished expanding designations. ", theTermValueSet, designationsExpanded);
437 		logConceptsExpanded("Finished expanding concepts. ", theTermValueSet, conceptsExpanded);
438 	}

458 	@SuppressWarnings("ConstantConditions")
459 	private void expandValueSet(ValueSet theValueSetToExpand, IValueSetConceptAccumulator theValueSetCodeAccumulator, AtomicInteger theCodeCounter, VersionIndependentConcept theWantConceptOrNull) {
460 		Set<String> addedCodes = new HashSet<>();
461 
462 		StopWatch sw = new StopWatch();
463 		String valueSetInfo = getValueSetInfo(theValueSetToExpand);
464 		ourLog.debug("Working with {}", valueSetInfo);
465 
466 		// Handle includes
467 		ourLog.debug("Handling includes");
468 		for (ValueSet.ConceptSetComponent include : theValueSetToExpand.getCompose().getInclude()) {
469 			for (int i = 0; ; i++) {
470 				int queryIndex = i;
471 				Boolean shouldContinue = myTxTemplate.execute(t -> {
472 					boolean add = true;
473 					return expandValueSetHandleIncludeOrExclude(theValueSetCodeAccumulator, addedCodes, include, add, theCodeCounter, queryIndex, theWantConceptOrNull);
474 				});
475 				if (!shouldContinue) {
476 					break;
477 				}
478 			}
479 		}
480 
481 		// If the accumulator filled up, abort
482 		if (theValueSetCodeAccumulator.getCapacityRemaining() != null && theValueSetCodeAccumulator.getCapacityRemaining() <= 0) {
483 			String msg = myContext.getLocalizer().getMessage(BaseTermReadSvcImpl.class, "expansionTooLarge", myDaoConfig.getMaximumExpansionSize());
484 			throw new ExpansionTooCostlyException(msg);
485 		}
486 
487 		// Handle excludes
488 		ourLog.debug("Handling excludes");
489 		for (ValueSet.ConceptSetComponent exclude : theValueSetToExpand.getCompose().getExclude()) {
490 			for (int i = 0; ; i++) {
491 				int queryIndex = i;
492 				Boolean shouldContinue = myTxTemplate.execute(t -> {
493 					boolean add = false;
494 					return expandValueSetHandleIncludeOrExclude(theValueSetCodeAccumulator, addedCodes, exclude, add, theCodeCounter, queryIndex, null);
495 				});

560 
561 		String system = theIncludeOrExclude.getSystem();
562 		boolean hasSystem = isNotBlank(system);
563 		boolean hasValueSet = theIncludeOrExclude.getValueSet().size() > 0;
564 
565 		if (hasSystem) {
566 
567 			if (theWantConceptOrNull != null && theWantConceptOrNull.getSystem() != null && !Constants.codeSystemNotNeeded(theWantConceptOrNull.getSystem()) && !system.equals(theWantConceptOrNull.getSystem())) {
568 				return false;
569 			}
570 
571 			ourLog.debug("Starting {} expansion around CodeSystem: {}", (theAdd ? "inclusion" : "exclusion"), system);
572 
573 			TermCodeSystem cs = myCodeSystemDao.findByCodeSystemUri(system);
574 			if (cs != null) {
575 
576 				TermCodeSystemVersion csv = cs.getCurrentVersion();
577 				FullTextEntityManager em = org.hibernate.search.jpa.Search.getFullTextEntityManager(myEntityManager);
578 
579 				/*
580 				 * If FullText searching is not enabled, we can handle only basic expansions
581 				 * since we're going to do it without the database.
582 				 */
583 				if (myFulltextSearchSvc == null) {
584 					expandWithoutHibernateSearch(theValueSetCodeAccumulator, csv, theAddedCodes, theIncludeOrExclude, system, theAdd, theCodeCounter);
585 					return false;
586 				}
587 
588 				/*
589 				 * Ok, let's use hibernate search to build the expansion
590 				 */
591 				QueryBuilder qb = em.getSearchFactory().buildQueryBuilder().forEntity(TermConcept.class).get();
592 				BooleanJunction<?> bool = qb.bool();
593 
594 				bool.must(qb.keyword().onField("myCodeSystemVersionPid").matching(csv.getPid()).createQuery());
595 
596 				if (theWantConceptOrNull != null) {
597 					bool.must(qb.keyword().onField("myCode").matching(theWantConceptOrNull.getCode()).createQuery());
598 				}
599 
600 				/*
601 				 * Filters
602 				 */
603 				handleFilters(bool, system, qb, theIncludeOrExclude);
604 
605 				Query luceneQuery = bool.createQuery();
606 
607 				/*
608 				 * Include/Exclude Concepts
609 				 */
610 				List<Term> codes = theIncludeOrExclude
611 					.getConcept()
612 					.stream()
613 					.filter(Objects::nonNull)
614 					.map(ValueSet.ConceptReferenceComponent::getCode)
615 					.filter(StringUtils::isNotBlank)
616 					.map(t -> new Term("myCode", t))
617 					.collect(Collectors.toList());
618 				if (codes.size() > 0) {
619 					MultiPhraseQuery query = new MultiPhraseQuery();
620 					query.add(codes.toArray(new Term[0]));
621 					luceneQuery = new BooleanQuery.Builder()
622 						.add(luceneQuery, BooleanClause.Occur.MUST)
623 						.add(query, BooleanClause.Occur.MUST)
624 						.build();
625 				}
626 
627 				/*
628 				 * Execute the query
629 				 */
630 				FullTextQuery jpaQuery = em.createFullTextQuery(luceneQuery, TermConcept.class);
631 
632 				/*
633 				 * DM 2019-08-21 - Processing slows after any ValueSets with many codes explicitly identified. This might
634 				 * be due to the dark arts that is memory management. Will monitor but not do anything about this right now.
635 				 */
636 				BooleanQuery.setMaxClauseCount(10000);
637 
638 				StopWatch sw = new StopWatch();
639 				AtomicInteger count = new AtomicInteger(0);
640 
641 				int maxResultsPerBatch = 10000;
642 
643 				/*
644 				 * If the accumulator is bounded, we may reduce the size of the query to
645 				 * Lucene in order to be more efficient.
646 				 */
647 				if (theAdd) {
648 					Integer accumulatorCapacityRemaining = theValueSetCodeAccumulator.getCapacityRemaining();
649 					if (accumulatorCapacityRemaining != null) {
650 						maxResultsPerBatch = Math.min(maxResultsPerBatch, accumulatorCapacityRemaining + 1);
651 					}
652 					if (maxResultsPerBatch <= 0) {
653 						return false;
654 					}
655 				}
656 
657 				jpaQuery.setMaxResults(maxResultsPerBatch);
658 				jpaQuery.setFirstResult(theQueryIndex * maxResultsPerBatch);
659 
660 				ourLog.debug("Beginning batch expansion for {} with max results per batch: {}", (theAdd ? "inclusion" : "exclusion"), maxResultsPerBatch);
661 
662 				StopWatch swForBatch = new StopWatch();
663 				AtomicInteger countForBatch = new AtomicInteger(0);
664 
665 				List resultList = jpaQuery.getResultList();
666 				int resultsInBatch = resultList.size();
667 				int firstResult = jpaQuery.getFirstResult();
668 				for (Object next : resultList) {
669 					count.incrementAndGet();
670 					countForBatch.incrementAndGet();
671 					TermConcept concept = (TermConcept) next;
672 					try {
673 						addCodeIfNotAlreadyAdded(theValueSetCodeAccumulator, theAddedCodes, concept, theAdd, theCodeCounter);
674 					} catch (ExpansionTooCostlyException e) {
675 						return false;
676 					}
677 				}
678 
679 				ourLog.debug("Batch expansion for {} with starting index of {} produced {} results in {}ms", (theAdd ? "inclusion" : "exclusion"), firstResult, countForBatch, swForBatch.getMillis());
680 
681 				if (resultsInBatch < maxResultsPerBatch) {
682 					ourLog.debug("Expansion for {} produced {} results in {}ms", (theAdd ? "inclusion" : "exclusion"), count, sw.getMillis());
683 					return false;
684 				} else {
685 					return true;
686 				}
687 
688 			} else {
689 				// No CodeSystem matching the URL found in the database.
690 
691 				CodeSystem codeSystemFromContext = getCodeSystemFromContext(system);
692 
693 				if (!theIncludeOrExclude.getConcept().isEmpty()) {
694 					for (ValueSet.ConceptReferenceComponent next : theIncludeOrExclude.getConcept()) {
695 						String nextCode = next.getCode();
696 						if (theWantConceptOrNull == null || theWantConceptOrNull.getCode().equals(nextCode)) {
697 							if (isNoneBlank(system, nextCode) && !theAddedCodes.contains(system + "|" + nextCode)) {
698 
699 								if (codeSystemFromContext != null) {
700 									CodeSystem.ConceptDefinitionComponent code = findCode(codeSystemFromContext.getConcept(), nextCode);
701 									if (code != null) {
702 										String display = code.getDisplay();
703 										addOrRemoveCode(theValueSetCodeAccumulator, theAddedCodes, theAdd, system, nextCode, display);
704 									}
705 								} else {
706 
707 									// This code just plain doesn't exist in any known codesystem, but the valueset
708 									// explicitly includes it. We'll trust the valueset in this case. This happens for
709 									// codesytems such a USPS. There is probably a better way to handle this...
710 									addOrRemoveCode(theValueSetCodeAccumulator, theAddedCodes, theAdd, system, nextCode, null);
711 								}
712 							}
713 						}
714 					}
715 				} else {
716 					if (codeSystemFromContext == null) {
717 						String msg = myContext.getLocalizer().getMessage(BaseTermReadSvcImpl.class, "expansionRefersToUnknownCs", system);
718 						ourLog.warn(msg);
719 						theValueSetCodeAccumulator.addMessage(msg);
720 						return false;
721 					}
722 
723 					List<CodeSystem.ConceptDefinitionComponent> concept = codeSystemFromContext.getConcept();
724 					addConceptsToList(theValueSetCodeAccumulator, theAddedCodes, system, concept, theAdd, theWantConceptOrNull);
725 				}
726 
727 				return false;
728 			}
729 		} else if (hasValueSet) {
730 
731 			for (CanonicalType nextValueSet : theIncludeOrExclude.getValueSet()) {
732 				ourLog.debug("Starting {} expansion around ValueSet: {}", (theAdd ? "inclusion" : "exclusion"), nextValueSet.getValueAsString());
733 
734 				List<VersionIndependentConcept> expanded = expandValueSet(nextValueSet.getValueAsString());
735 				Map<String, TermCodeSystem> uriToCodeSystem = new HashMap<>();
736 
737 				for (VersionIndependentConcept nextConcept : expanded) {
738 					if (theAdd) {
739 
740 						if (!uriToCodeSystem.containsKey(nextConcept.getSystem())) {
741 							TermCodeSystem codeSystem = myCodeSystemDao.findByCodeSystemUri(nextConcept.getSystem());
742 							uriToCodeSystem.put(nextConcept.getSystem(), codeSystem);
743 						}
744 
745 						TermCodeSystem codeSystem = uriToCodeSystem.get(nextConcept.getSystem());
746 						if (codeSystem != null) {
747 							myConceptDao
748 								.findByCodeSystemAndCode(codeSystem.getCurrentVersion(), nextConcept.getCode())
749 								.ifPresent(concept ->
750 									addCodeIfNotAlreadyAdded(theValueSetCodeAccumulator, theAddedCodes, concept, theAdd, theCodeCounter)
751 								);
752 						} else {
753 							// This will happen if we're expanding against a built-in (part of FHIR) ValueSet that
754 							// isn't actually in the database anywhere
755 							Collection<TermConceptDesignation> emptyCollection = Collections.emptyList();
756 							addCodeIfNotAlreadyAdded(theValueSetCodeAccumulator, theAddedCodes, emptyCollection, theAdd, theCodeCounter, nextConcept.getSystem(), nextConcept.getCode(), null);
757 						}
758 					}
759 					if (isNoneBlank(nextConcept.getSystem(), nextConcept.getCode()) && !theAdd && theAddedCodes.remove(nextConcept.getSystem() + "|" + nextConcept.getCode())) {
760 						theValueSetCodeAccumulator.excludeConcept(nextConcept.getSystem(), nextConcept.getCode());
761 					}
762 				}
763 
764 			}
765 
766 			return false;
767 
768 		} else {
769 			throw new InvalidRequestException("ValueSet contains " + (theAdd ? "include" : "exclude") + " criteria with no system defined");
770 		}

795 		}
796

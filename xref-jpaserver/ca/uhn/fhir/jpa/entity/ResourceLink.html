23  import java.io.Serializable;
24  
25  import javax.persistence.Column;
26  import javax.persistence.Entity;
27  import javax.persistence.GeneratedValue;
28  import javax.persistence.GenerationType;
29  import javax.persistence.Id;
30  import javax.persistence.JoinColumn;
31  import javax.persistence.ManyToOne;
32  import javax.persistence.Table;
33  
34  import org.apache.commons.lang3.Validate;
35  
36  @Entity
37  @Table(name = "HFJ_RES_LINK"/*, indexes= {@Index(name="IDX_RL_TPATHRES", columnList= "SRC_PATH,TARGET_RESOURCE_ID")}*/)
38  @org.hibernate.annotations.Table(appliesTo="HFJ_RES_LINK",indexes= {
39  		@org.hibernate.annotations.Index(name="IDX_RL_TPATHRES", columnNames= {"SRC_PATH", "TARGET_RESOURCE_ID"})})
40  public class ResourceLink implements Serializable {
41  
42  	private static final long serialVersionUID = 1L;
43  
44  	@GeneratedValue(strategy = GenerationType.AUTO)
45  	@Id
46  	@Column(name = "PID")
47  	private Long myId;
48  
49  	@Column(name = "SRC_PATH", length = 100, nullable = false)
50  	private String mySourcePath;
51  
52  	@ManyToOne(optional = false)
53  	@JoinColumn(name = "SRC_RESOURCE_ID", referencedColumnName="RES_ID", nullable=false)
54  	private ResourceTable mySourceResource;
55  
56  	@Column(name = "SRC_RESOURCE_ID", insertable = false, updatable = false, nullable=false)
57  	private Long mySourceResourcePid;
58  
59  	@ManyToOne(optional = false)
60  	@JoinColumn(name = "TARGET_RESOURCE_ID", referencedColumnName="RES_ID", nullable=false)
61  	private ResourceTable myTargetResource;
62  
63  	@Column(name = "TARGET_RESOURCE_ID", insertable = false, updatable = false,nullable=false)
64  	private Long myTargetResourcePid;
65  
66  	public ResourceLink() {
67  		//nothing
68  	}
69  	
70  	@Override
71  	public String toString() {
72  		StringBuilder b = new StringBuilder();
73  		b.append("ResourceLink[");
74  		b.append("path=").append(mySourcePath);
75  		b.append(", src=").append(mySourceResource.getId());
76  		b.append(", target=").append(myTargetResource.getId());
77  		
78  		b.append("]");
79  		return b.toString();
80  	}
81  
82  	public ResourceLink(String theSourcePath, ResourceTable theSourceResource, ResourceTable theTargetResource) {
83  		super();
84  		mySourcePath = theSourcePath;
85  		mySourceResource = theSourceResource;
86  		myTargetResource = theTargetResource;
87  	}
88  
89  	public String getSourcePath() {
90  		return mySourcePath;
91  	}
92  
93  	public ResourceTable getSourceResource() {
94  		return mySourceResource;
95  	}
96  
97  	public Long getSourceResourcePid() {
98  		return mySourceResourcePid;
99  	}
100 
101 	public ResourceTable getTargetResource() {
102 		return myTargetResource;
103 	}
104 
105 	public Long getTargetResourcePid() {
106 		return myTargetResourcePid;
107 	}
108 
109 	public void setSourcePath(String theSourcePath) {
110 		mySourcePath = theSourcePath;
111 	}
112 
113 	public void setSourceResource(ResourceTable theSourceResource) {
114 		mySourceResource = theSourceResource;
115 	}
116 
117 	public void setSourceResourcePid(Long theSourceResourcePid) {
118 		mySourceResourcePid = theSourceResourcePid;
119 	}
120 
121 	public void setTargetResource(ResourceTable theTargetResource) {
122 		Validate.notNull(theTargetResource);
123 		myTargetResource = theTargetResource;
124 	}
125 
126 	public void setTargetResourcePid(Long theTargetResourcePid) {
127 		myTargetResourcePid = theTargetResourcePid;
128 	}
129 
130 }

23  import javax.persistence.*;
24  import java.io.Serializable;
25  
26  //@formatter:off
27  @Table(name = "TRM_CODESYSTEM", uniqueConstraints = {
28  	@UniqueConstraint(name = "IDX_CS_CODESYSTEM", columnNames = {"CODE_SYSTEM_URI"})
29  })
30  @Entity()
31  //@formatter:on
32  public class TermCodeSystem implements Serializable {
33  	private static final long serialVersionUID = 1L;
34  
35  	@Column(name = "CODE_SYSTEM_URI", nullable = false)
36  	private String myCodeSystemUri;
37  
38  	@OneToOne()
39  	@JoinColumn(name = "CURRENT_VERSION_PID", referencedColumnName = "PID", nullable = true, foreignKey = @ForeignKey(name = "FK_TRMCODESYSTEM_CURVER"))
40  	private TermCodeSystemVersion myCurrentVersion;
41  	@Id()
42  	@SequenceGenerator(name = "SEQ_CODESYSTEM_PID", sequenceName = "SEQ_CODESYSTEM_PID")
43  	@GeneratedValue(strategy = GenerationType.AUTO, generator = "SEQ_CODESYSTEM_PID")
44  	@Column(name = "PID")
45  	private Long myPid;
46  	@OneToOne()
47  	@JoinColumn(name = "RES_ID", referencedColumnName = "RES_ID", nullable = false, updatable = false, foreignKey = @ForeignKey(name = "FK_TRMCODESYSTEM_RES"))
48  	private ResourceTable myResource;
49  	@Column(name = "RES_ID", insertable = false, updatable = false)
50  	private Long myResourcePid;
51  
52  	public String getCodeSystemUri() {
53  		return myCodeSystemUri;
54  	}
55  
56  	public void setCodeSystemUri(String theCodeSystemUri) {
57  		myCodeSystemUri = theCodeSystemUri;
58  	}
59  
60  	public TermCodeSystemVersion getCurrentVersion() {
61  		return myCurrentVersion;
62  	}
63  
64  	public void setCurrentVersion(TermCodeSystemVersion theCurrentVersion) {
65  		myCurrentVersion = theCurrentVersion;
66  	}
67  
68  	public Long getPid() {
69  		return myPid;
70  	}
71  
72  	public ResourceTable getResource() {
73  		return myResource;
74  	}
75  
76  	public void setResource(ResourceTable theResource) {
77  		myResource = theResource;
78  	}
79  }

52  /**
53   * Interceptor that allows for cascading deletes (deletes that resolve constraint issues).
54   * <p>
55   * For example, if <code>DiagnosticReport/A</code> has a reference to <code>Observation/B</code>
56   * it is not normally possible to delete <code>Observation/B</code> without first deleting
57   * <code>DiagnosticReport/A</code>. With this interceptor in place, it is.
58   * </p>
59   * <p>
60   * When using this interceptor, client requests must include the parameter
61   * <code>_cascade=true</code> on the DELETE URL in order to activate
62   * cascading delete, or include the request header <code>X-Cascade-Delete: true</code>
63   * </p>
64   */
65  @Interceptor
66  public class CascadingDeleteInterceptor {
67  
68  	private static final Logger ourLog = LoggerFactory.getLogger(CascadingDeleteInterceptor.class);
69  	private static final String CASCADED_DELETES_KEY = CascadingDeleteInterceptor.class.getName() + "_CASCADED_DELETES_KEY";
70  	private static final String CASCADED_DELETES_FAILED_KEY = CascadingDeleteInterceptor.class.getName() + "_CASCADED_DELETES_FAILED_KEY";
71  
72  	private final DaoRegistry myDaoRegistry;
73  	private final IInterceptorBroadcaster myInterceptorBroadcaster;
74  
75  	/**
76  	 * Constructor
77  	 *
78  	 * @param theDaoRegistry The DAO registry (must not be null)
79  	 */
80  	public CascadingDeleteInterceptor(DaoRegistry theDaoRegistry, IInterceptorBroadcaster theInterceptorBroadcaster) {
81  		Validate.notNull(theDaoRegistry, "theDaoRegistry must not be null");
82  		Validate.notNull(theInterceptorBroadcaster, "theInterceptorBroadcaster must not be null");
83  		myDaoRegistry = theDaoRegistry;
84  		myInterceptorBroadcaster = theInterceptorBroadcaster;
85  	}
86  
87  	@Hook(Pointcut.STORAGE_PRESTORAGE_DELETE_CONFLICTS)
88  	public DeleteConflictOutcome handleDeleteConflicts(DeleteConflictList theConflictList, RequestDetails theRequest) {
89  		ourLog.debug("Have delete conflicts: {}", theConflictList);
90  
91  		if (!shouldCascade(theRequest)) {
92  			return null;
93  		}
94  
95  		List<String> cascadedDeletes = getCascadedDeletesMap(theRequest, true);
96  		for (Iterator<DeleteConflict> iter = theConflictList.iterator(); iter.hasNext(); ) {
97  			DeleteConflict next = iter.next();
98  			IdDt nextSource = next.getSourceId();
99  			String nextSourceId = nextSource.toUnqualifiedVersionless().getValue();
100 
101 			if (!cascadedDeletes.contains(nextSourceId)) {
102 
103 				IFhirResourceDao dao = myDaoRegistry.getResourceDao(nextSource.getResourceType());
104 
105 				// Interceptor call: STORAGE_CASCADE_DELETE
106 				IBaseResource resource = dao.read(nextSource);
107 				HookParams params = new HookParams()
108 					.add(RequestDetails.class, theRequest)
109 					.addIfMatchesType(ServletRequestDetails.class, theRequest)
110 					.add(DeleteConflictList.class, theConflictList)
111 					.add(IBaseResource.class, resource);
112 				JpaInterceptorBroadcaster.doCallHooks(myInterceptorBroadcaster, theRequest, Pointcut.STORAGE_CASCADE_DELETE, params);
113 
114 				// Actually perform the delete
115 				ourLog.info("Have delete conflict {} - Cascading delete", next);
116 				dao.delete(nextSource, theRequest);
117 
118 				cascadedDeletes.add(nextSourceId);
119 
120 			}
121 		}
122 
123 		return new DeleteConflictOutcome().setShouldRetryCount(MAX_RETRY_ATTEMPTS);
124 	}
125 
126 	@SuppressWarnings("unchecked")
127 	private List<String> getCascadedDeletesMap(RequestDetails theRequest, boolean theCreate) {
128 		List<String> retVal = (List<String>) theRequest.getUserData().get(CASCADED_DELETES_KEY);
129 		if (retVal == null && theCreate) {
130 			retVal = new ArrayList<>();
131 			theRequest.getUserData().put(CASCADED_DELETES_KEY, retVal);
132 		}
133 		return retVal;
134 	}
135 
136 	@Hook(Pointcut.SERVER_OUTGOING_FAILURE_OPERATIONOUTCOME)
137 	public void outgoingFailureOperationOutcome(RequestDetails theRequestDetails, IBaseOperationOutcome theResponse) {
138 		if (theRequestDetails != null) {
139 
140 			String failedDeleteMessage = (String) theRequestDetails.getUserData().get(CASCADED_DELETES_FAILED_KEY);
141 			if (isNotBlank(failedDeleteMessage)) {
142 				FhirContext ctx = theRequestDetails.getFhirContext();
143 				String severity = OperationOutcome.IssueSeverity.INFORMATION.toCode();
144 				String code = OperationOutcome.IssueType.INFORMATIONAL.toCode();
145 				String details = failedDeleteMessage;
146 				OperationOutcomeUtil.addIssue(ctx, theResponse, severity, details, null, code);
147 			}
148 
149 		}
150 	}
151 
152 
153 	@Hook(Pointcut.SERVER_OUTGOING_RESPONSE)
154 	public void outgoingResponse(RequestDetails theRequestDetails, ResponseDetails theResponseDetails, IBaseResource theResponse) {
155 		if (theRequestDetails != null) {
156 
157 			// Successful delete list
158 			List<String> deleteList = getCascadedDeletesMap(theRequestDetails, false);
159 			if (deleteList != null) {
160 				if (theResponseDetails.getResponseCode() == 200) {
161 					if (theResponse instanceof IBaseOperationOutcome) {
162 						FhirContext ctx = theRequestDetails.getFhirContext();
163 						IBaseOperationOutcome oo = (IBaseOperationOutcome) theResponse;
164 						String severity = OperationOutcome.IssueSeverity.INFORMATION.toCode();
165 						String code = OperationOutcome.IssueType.INFORMATIONAL.toCode();
166 						String details = ctx.getLocalizer().getMessage(CascadingDeleteInterceptor.class, "successMsg", deleteList.size(), deleteList);
167 						OperationOutcomeUtil.addIssue(ctx, oo, severity, details, null, code);
168 					}
169 				}
170 			}
171 
172 		}
173 	}

176 	/**
177 	 * Subclasses may override
178 	 *
179 	 * @param theRequest The REST request
180 	 * @return Returns true if cascading delete should be allowed
181 	 */
182 	@SuppressWarnings("WeakerAccess")
183 	protected boolean shouldCascade(RequestDetails theRequest) {
184 		if (theRequest != null) {
185 			String[] cascadeParameters = theRequest.getParameters().get(Constants.PARAMETER_CASCADE_DELETE);
186 			if (cascadeParameters != null && Arrays.asList(cascadeParameters).contains(Constants.CASCADE_DELETE)) {
187 				return true;
188 			}
189 
190 			String cascadeHeader = theRequest.getHeader(Constants.HEADER_CASCADE);
191 			if (Constants.CASCADE_DELETE.equals(cascadeHeader)) {
192 				return true;
193 			}
194 
195 			// Add a message to the response
196 			String message = theRequest.getFhirContext().getLocalizer().getMessage(CascadingDeleteInterceptor.class, "noParam");
197 			theRequest.getUserData().put(CASCADED_DELETES_FAILED_KEY, message);
198 		}
199 
200 		return false;
201 	}

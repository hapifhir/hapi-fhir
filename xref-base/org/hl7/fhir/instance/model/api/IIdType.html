25  /**
26   * Base interface for ID datatype. 
27   * 
28   * <p>
29   * <b>Concrete Implementations:</b> This interface is often returned and/or accepted by methods in HAPI's API
30   * where either {@link ca.uhn.fhir.model.primitive.IdDt} (the HAPI structure ID type) or 
31   * <code>org.hl7.fhir.instance.model.IdType</code> (the RI structure ID type) will be used, depending on 
32   * which version of the strctures your application is using.   
33   * </p>
34   */
35  public interface IIdType extends IPrimitiveType<String> {
36  
37  	void applyTo(IBaseResource theResource);
38  
39  	/**
40  	 * Returns the server base URL if this ID contains one. For example, the base URL is
41  	 * the 'http://example.com/fhir' in the following ID: <code>http://example.com/fhir/Patient/123/_history/55</code>
42  	 */
43  	String getBaseUrl();
44  
45  	/**
46  	 * Returns only the logical ID part of this ID. For example, given the ID
47  	 * "http://example,.com/fhir/Patient/123/_history/456", this method would
48  	 * return "123".
49  	 */
50  	String getIdPart();
51  
52  	/**
53  	 * Returns the ID part of this ID (e.g. in the ID http://example.com/Patient/123/_history/456 this would be the
54  	 * part "123") parsed as a {@link Long}.
55  	 * 
56  	 * @throws NumberFormatException If the value can't be parsed as a long
57  	 */
58  	Long getIdPartAsLong();
59  
60  	String getResourceType();
61  
62  	/**
63  	 * Returns the value of this ID. Note that this value may be a fully qualified URL, a relative/partial URL, or a simple ID. Use {@link #getIdPart()} to get just the ID portion.
64  	 * 
65  	 * @see #getIdPart()
66  	 */
67  	String getValue();
68  
69  	String getVersionIdPart();
70  
71  	/**
72  	 * Returns the version ID part of this ID (e.g. in the ID http://example.com/Patient/123/_history/456 this would be the
73  	 * part "456") parsed as a {@link Long}.
74  	 * 
75  	 * @throws NumberFormatException If the value can't be parsed as a long
76  	 */
77  	Long getVersionIdPartAsLong();
78  
79  	boolean hasBaseUrl();
80  
81  	/**
82  	 * Returns <code>true</code> if this ID contains an actual ID part. For example, the ID part is
83  	 * the '123' in the following ID: <code>http://example.com/fhir/Patient/123/_history/55</code>;
84  	 */
85  	boolean hasIdPart();
86  
87  	boolean hasResourceType();
88  
89  	boolean hasVersionIdPart();
90  
91  	/**
92  	 * Returns <code>true</code> if this ID contains an absolute URL (in other words, a URL starting with "http://" or "https://"
93  	 */
94  	boolean isAbsolute();
95  
96  	boolean isEmpty();
97  
98  	/**
99  	 * Returns <code>true</code> if the {@link #getIdPart() ID part of this object} is valid according to the FHIR rules for valid IDs. 
100 	 * <p>
101 	 * The FHIR specification states:
102 	 * <code>Any combination of upper or lower case ASCII letters ('A'..'Z', and 'a'..'z', numerals ('0'..'9'), '-' and '.', with a length limit of 64 characters. (This might be an integer, an un-prefixed OID, UUID or any other identifier pattern that meets these constraints.) regex: [A-Za-z0-9\-\.]{1,64}</code>
103 	 * </p>
104 	 */
105 	boolean isIdPartValid();
106 
107 	/**
108 	 * Returns <code>true</code> if the {@link #getIdPart() ID part of this object} contains
109 	 * only numbers 
110 	 */
111 	boolean isIdPartValidLong();
112 
113 	/**
114 	 * Returns <code>true</code> if the ID is a local reference (in other words, it begins with the '#' character)
115 	 */
116 	boolean isLocal();
117 
118 	/**
119 	 * Returns <code>true</code> if the {@link #getVersionIdPart() version ID part of this object} contains
120 	 * only numbers 
121 	 */
122 	boolean isVersionIdPartValidLong();
123 
124 	@Override
125 	IIdType setValue(String theString);
126 
127 	IIdType toUnqualified();
128 
129 	IIdType toUnqualifiedVersionless();
130 
131 	IIdType toVersionless();
132 
133 	/**
134 	 * Returns a copy of this object, but with a different {@link #getResourceType() resource type} 
135 	 * (or if this object does not have a resource type currently, returns a copy of this object with
136 	 * the given resource type).
137 	 * <p>
138 	 * Note that if this object represents a local reference (e.g. <code>#foo</code>) or
139 	 * a URN (e.g. <code>urn:oid:1.2.3.4</code>) this method will simply return a copy
140 	 * of this object with no modifications.
141 	 * </p>
142 	 */
143 	IIdType withResourceType(String theResName);
144 	
145 	/**
146 	 * Returns a copy of this object, but with a different {@link #getResourceType() resource type} 
147 	 * and {@link #getBaseUrl() base URL}
148 	 * (or if this object does not have a resource type currently, returns a copy of this object with
149 	 * the given server base and resource type).
150 	 * <p>
151 	 * Note that if this object represents a local reference (e.g. <code>#foo</code>) or
152 	 * a URN (e.g. <code>urn:oid:1.2.3.4</code>) this method will simply return a copy
153 	 * of this object with no modifications.
154 	 * </p>
155 	 */
156 	IIdType withServerBase(String theServerBase, String theResourceName);
157 
158 	/**
159 	 * Returns a copy of this object, but with a different {@link #getVersionIdPart() version ID} 
160 	 * (or if this object does not have a resource type currently, returns a copy of this object with
161 	 * the given version).
162 	 * <p>
163 	 * Note that if this object represents a local reference (e.g. <code>#foo</code>) or
164 	 * a URN (e.g. <code>urn:oid:1.2.3.4</code>) this method will simply return a copy
165 	 * of this object with no modifications.
166 	 * </p>
167 	 */
168 	IIdType withVersion(String theVersion);
169 
170 	/**
171 	 * Sets the value of this ID by combining all of the individual parts.
172 	 * <p>
173 	 * <b>Required parameters:</b> The following rules apply to the parameters of this method (in this case, populated means
174 	 * a non-empty string and not populated means <code>null</code> or an empty string)
175 	 * </p>
176 	 * <ul>
177 	 * <li>All values may be not populated</li>
178 	 * <li>If <b>theVersionIdPart</b> is populated, <b>theResourceType</b> and <b>theIdPart</b> must be populated</li>
179 	 * <li>If <b>theBaseUrl</b> is populated and <b>theIdPart</b> is populated, <b>theResourceType</b> must be populated</li>
180 	 * </ul>
181 	 * 
182 	 * @return Returns a reference to <code>this</code> for easy method chaining
183 	 */
184 	IIdType setParts(String theBaseUrl, String theResourceType, String theIdPart, String theVersionIdPart);
185 
186 }

125 	public static String escapeXml(String doco) {
126 		if (doco == null)
127 			return "";
128 		
129 		StringBuilder b = new StringBuilder();
130 		for (char c : doco.toCharArray()) {
131 		  if (c == '<')
132 			  b.append("&lt;");
133 		  else if (c == '>')
134 			  b.append("&gt;");
135 		  else if (c == '&')
136 			  b.append("&amp;");
137       else if (c == '"')
138         b.append("&quot;");
139 		  else 
140 			  b.append(c);
141 		}		
142 		return b.toString();
143 	}

457   public static String path(String... args) throws IOException {
458     StringBuilder s = new StringBuilder();
459     boolean d = false;
460     for(String arg: args) {
461       if (!d)
462         d = !noString(arg);
463       else if (!s.toString().endsWith(File.separator))
464         s.append(File.separator);
465       String a = arg;
466       a = a.replace("\\", File.separator);
467       if (s.length() > 0 && a.startsWith(File.separator))
468         a = a.substring(File.separator.length());
469       
470       while (a.startsWith(".."+File.separator)) {
471         String p = s.toString().substring(0, s.length()-1);
472         if (!p.contains(File.separator))
473           throw new IOException("illegal path underrun "+Arrays.asList(args));
474         s = new StringBuilder(p.substring(0,  p.lastIndexOf(File.separator))+File.separator);
475         a = a.substring(3);
476       }
477       if ("..".equals(a)) {
478         int i = s.substring(0, s.length()-1).lastIndexOf(File.separator);
479         s = new StringBuilder(s.substring(0, i+1));
480       } else
481         s.append(a);
482     }
483     return s.toString();
484   }

626 	public static String normalize(String s) {
627 		if (noString(s))
628 			return null;
629 	  StringBuilder b = new StringBuilder();
630 	  boolean isWhitespace = false;
631 	  for (int i = 0; i < s.length(); i++) {
632 	  	char c = s.charAt(i);
633 	  	if (!Character.isWhitespace(c)) { 
634 	  		b.append(Character.toLowerCase(c));
635 	  		isWhitespace = false;
636 	  	} else if (!isWhitespace) {
637 	  		b.append(' ');
638 	  		isWhitespace = true;	  		
639 	  	} 
640 	  }
641 	  return b.toString().trim();
642   }

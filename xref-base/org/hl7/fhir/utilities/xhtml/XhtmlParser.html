216 	private void addTextNode(XhtmlNode node, StringBuilder s) {
217 		String t = isTrimWhitespace() ? s.toString().trim() : s.toString();
218 		if (t.length() > 0) {
219 			lastText = t;
220 			// System.out.println(t);
221 			node.addText(t);
222 			s.setLength(0);
223 		}
224 	}

226 	private boolean attributeIsOk(String elem, String attr, String value) throws FHIRFormatError {
227 		if (validatorMode)
228 			return true;
229 		boolean ok = attributes.contains(attr) || attributes.contains(elem + "." + attr);
230 		if (ok)
231 			return true;
232 		else
233 			switch (policy) {
234 			case Accept:
235 				return true;
236 			case Drop:
237 				return false;
238 			case Reject:
239 				throw new FHIRFormatError("Illegal HTML attribute " + elem + "." + attr);
240 			}
241 
242 		if ((elem + "." + attr).equals("img.src") && !(value.startsWith("#") || value.startsWith("http:") || value.startsWith("https:"))) {
243 			switch (policy) {
244 			case Accept:
245 				return true;
246 			case Drop:
247 				return false;
248 			case Reject:
249 				throw new FHIRFormatError("Illegal Image Reference " + value);
250 			}
251 		}
252 		return false;
253 	}

255 	private NSMap checkNamespaces(QName n, XhtmlNode node, NSMap nsm, boolean root) {
256 		// what we do here is strip out any stated namespace attributes, putting them in the namesapce map
257 		// then we figure out what the namespace of this element is, and state it explicitly if it's not the default
258 
259 		// but we don't bother with any of this if we're not validating
260 		if (!validatorMode)
261 			return null;
262 		NSMap result = new NSMap(nsm);
263 		List<String> nsattrs = new ArrayList<String>();
264 		for (String an : node.getAttributes().keySet()) {
265 			if (an.equals("xmlns")) {
266 				result.def(node.getAttribute(an));
267 				nsattrs.add(an);
268 			}
269 			if (an.startsWith("xmlns:")) {
270 				result.ns(an.substring(6), node.getAttribute(an));
271 				nsattrs.add(an);
272 			}
273 		}
274 		for (String s : nsattrs)
275 			node.getAttributes().remove(s);
276 		if (n.hasNs()) {
277 			String nns = result.get(n.getNs());
278 			if (!nns.equals(result.def())) {
279 				node.getAttributes().put("xmlns", nns);
280 				result.def(nns);
281 			}
282 		} else if (root && result.hasDef()) {
283 			node.getAttributes().put("xmlns", result.def());
284 		}
285 		return result;
286 	}

288 	private String checkNS(XhtmlNode res, Element node, String defaultNS) {
289 		if (!validatorMode)
290 			return null;
291 		String ns = node.getNamespaceURI();
292 		if (ns == null)
293 			return null;
294 		if (!ns.equals(defaultNS)) {
295 			res.getAttributes().put("xmlns", ns);
296 			return ns;
297 		}
298 		return defaultNS;
299 	}

357 	private XhtmlDocument parse(String entryName) throws FHIRFormatError, IOException {
358 		XhtmlDocument result = new XhtmlDocument();
359 		skipWhiteSpaceAndComments(result);
360 		if (peekChar() != '<')
361 			throw new FHIRFormatError("Unable to Parse HTML - does not start with tag. Found " + peekChar() + descLoc());
362 		readChar();
363 		QName n = new QName(readName().toLowerCase());
364 		if ((entryName != null) && !n.getName().equals(entryName))
365 			throw new FHIRFormatError("Unable to Parse HTML - starts with '" + n + "' not '" + entryName + "'" + descLoc());
366 		XhtmlNode root = result.addTag(n.getName());
367 		parseAttributes(root);
368 		NSMap nsm = checkNamespaces(n, root, null, true);
369 		if (readChar() == '/') {
370 			if (peekChar() != '>')
371 				throw new FHIRFormatError("unexpected non-end of element " + n + " " + descLoc());
372 			readChar();
373 		} else {
374 			unwindPoint = null;
375 			List<XhtmlNode> p = new ArrayList<XhtmlNode>();
376 			parseElementInner(root, p, nsm);
377 		}
378 		return result;
379 	}

386 	private void parseAttributes(XhtmlNode node) throws FHIRFormatError, IOException {
387 		while (Character.isWhitespace(peekChar()))
388 			readChar();
389 		while (peekChar() != '>' && peekChar() != '/' && peekChar() != '\0') {
390 			String name = readName();
391 			if (name.length() == 0) {
392 				throw new FHIRFormatError("Unable to read attribute on <" + node.getName() + ">" + descLoc());
393 			}
394 			while (Character.isWhitespace(peekChar()))
395 				readChar();
396 
397 			if (isNameChar(peekChar()) || peekChar() == '>' || peekChar() == '/')
398 				node.getAttributes().put(name, null);
399 			else if (peekChar() != '=') {
400 				throw new FHIRFormatError("Unable to read attribute '" + name + "' value on <" + node.getName() + ">" + descLoc());
401 			} else {
402 				readChar();
403 				while (Character.isWhitespace(peekChar()))
404 					readChar();
405 				if (peekChar() == '"' || peekChar() == '\'')
406 					node.getAttributes().put(name, parseAttributeValue(readChar()));
407 				else
408 					node.getAttributes().put(name, parseAttributeValue('\0'));
409 			}
410 			while (Character.isWhitespace(peekChar()))
411 				readChar();
412 		}
413 	}

415 	private String parseAttributeValue(char term) throws IOException, FHIRFormatError {
416 		StringBuilder b = new StringBuilder();
417 		while (peekChar() != '\0' && peekChar() != '>' && (term != '\0' || peekChar() != '/') && peekChar() != term) {
418 			if (peekChar() == '&') {
419 				parseLiteral(b);
420 			} else
421 				b.append(readChar());
422 		}
423 		if (peekChar() == term)
424 			readChar();
425 		return b.toString();
426 	}

428 	private void parseElement(XhtmlNode parent, List<XhtmlNode> parents, NSMap nsm) throws IOException, FHIRFormatError {
429 		QName name = new QName(readName());
430 		XhtmlNode node = parent.addTag(name.getName());
431 		List<XhtmlNode> newParents = new ArrayList<XhtmlNode>();
432 		newParents.addAll(parents);
433 		newParents.add(parent);
434 		parseAttributes(node);
435 		nsm = checkNamespaces(name, node, nsm, false);
436 		if (readChar() == '/') {
437 			if (peekChar() != '>')
438 				throw new FHIRFormatError("unexpected non-end of element " + name + " " + descLoc());
439 			readChar();
440 		} else {
441 			parseElementInner(node, newParents, nsm);
442 		}
443 	}

445 	private void parseElementInner(XhtmlNode node, List<XhtmlNode> parents, NSMap nsm) throws FHIRFormatError, IOException {
446 		StringBuilder s = new StringBuilder();
447 		while (peekChar() != '\0' && !parents.contains(unwindPoint) && !(node == unwindPoint)) {
448 			if (peekChar() == '<') {
449 				addTextNode(node, s);
450 				readChar();
451 				if (peekChar() == '!') {
452 					String sc = readToCommentEnd();
453 					if (sc.startsWith("DOCTYPE"))
454 						throw new FHIRFormatError("Malformed XHTML: Found a DocType declaration, and these are not allowed (XXE security vulnerability protection)");
455 					node.addComment(sc);
456 				} else if (peekChar() == '?')
457 					node.addComment(readToTagEnd());
458 				else if (peekChar() == '/') {
459 					readChar();
460 					QName n = new QName(readToTagEnd());
461 					if (node.getName().equals(n.getName()))
462 						return;
463 					else {
464 						if (mustBeWellFormed)
465 							throw new FHIRFormatError("Malformed XHTML: Found \"</" + n.getName() + ">\" expecting \"</" + node.getName() + ">\"" + descLoc());
466 						for (int i = parents.size() - 1; i >= 0; i--) {
467 							if (parents.get(i).getName().equals(n))
468 								unwindPoint = parents.get(i);
469 						}
470 						if (unwindPoint != null) {
471 							for (int i = parents.size(); i > 0; i--) {
472 								if (i < parents.size() && parents.get(i) == unwindPoint)
473 									return;
474 								if (i == parents.size()) {
475 									parents.get(i - 1).getChildNodes().addAll(node.getChildNodes());
476 									node.getChildNodes().clear();
477 								} else {
478 									parents.get(i - 1).getChildNodes().addAll(parents.get(i).getChildNodes());
479 									parents.get(i).getChildNodes().clear();
480 								}
481 							}
482 						}
483 					}
484 				} else if (Character.isLetterOrDigit(peekChar())) {
485 					parseElement(node, parents, nsm);
486 				} else
487 					throw new FHIRFormatError("Unable to Parse HTML - node '" + node.getName() + "' has unexpected content '" + peekChar() + "' (last text = '" + lastText + "'" + descLoc());
488 			} else if (peekChar() == '&') {
489 				parseLiteral(s);
490 			} else
491 				s.append(readChar());
492 		}
493 		addTextNode(node, s);
494 	}

496 	private XhtmlNode parseFragment() throws IOException, FHIRException {
497 		skipWhiteSpace();
498 		if (peekChar() != '<')
499 			throw new FHIRException("Unable to Parse HTML - does not start with tag. Found " + peekChar() + descLoc());
500 		readChar();
501 		if (peekChar() == '?') {
502 			readToTagEnd();
503 			skipWhiteSpace();
504 			if (peekChar() != '<')
505 				throw new FHIRException("Unable to Parse HTML - does not start with tag after processing instruction. Found " + peekChar() + descLoc());
506 			readChar();
507 		}
508 		String n = readName().toLowerCase();
509 		readToTagEnd();
510 		XhtmlNode result = new XhtmlNode(NodeType.Element);
511 
512 		int colonIndex = n.indexOf(':');
513 		if (colonIndex != -1) {
514 			n = n.substring(colonIndex + 1);
515 		}
516 
517 		result.setName(n);
518 		unwindPoint = null;
519 		List<XhtmlNode> p = new ArrayList<XhtmlNode>();
520 		parseElementInner(result, p, null);
521 
522 		return result;
523 	}

554 	private void parseLiteral(StringBuilder s) throws IOException, FHIRFormatError {
555 		// UInt16 w;
556 		readChar();
557 		String c = readUntil(';');
558 		if (c.equals("apos"))
559 			s.append('\'');
560 		else if (c.equals("quot"))
561 			s.append('"');
562 		else if (c.equals("nbsp"))
563 			s.append(XhtmlNode.NBSP);
564 		else if (c.equals("amp"))
565 			s.append('&');
566 		else if (c.equals("rsquo"))
567 			s.append('');
568 		else if (c.equals("gt"))
569 			s.append('>');
570 		else if (c.equals("lt"))
571 			s.append('<');
572 		else if (c.equals("copy"))
573 			s.append((char) 169);
574 		else if (c.equals("reg"))
575 			s.append((char) 174);
576 		else if (c.equals("sect"))
577 			s.append((char) 0xA7);
578 		else if (c.charAt(0) == '#') {
579 			if (isInteger(c.substring(1), 10))
580 				s.append((char) Integer.parseInt(c.substring(1)));
581 			else if (isInteger(c.substring(1), 16))
582 				s.append((char) Integer.parseInt(c.substring(1), 16));
583 		} else if (c.equals("fnof"))
584 			s.append((char) 402); // latin small f with hook = function = florin, U+0192 ISOtech -->
585 		else if (c.equals("Alpha"))
586 			s.append((char) 913); // greek capital letter alpha, U+0391
587 		else if (c.equals("Beta"))
588 			s.append((char) 914); // greek capital letter beta, U+0392
589 		else if (c.equals("Gamma"))
590 			s.append((char) 915); // greek capital letter gamma, U+0393 ISOgrk3
591 		else if (c.equals("Delta"))
592 			s.append((char) 916); // greek capital letter delta, U+0394 ISOgrk3
593 		else if (c.equals("Epsilon"))
594 			s.append((char) 917); // greek capital letter epsilon, U+0395
595 		else if (c.equals("Zeta"))
596 			s.append((char) 918); // greek capital letter zeta, U+0396
597 		else if (c.equals("Eta"))
598 			s.append((char) 919); // greek capital letter eta, U+0397
599 		else if (c.equals("Theta"))
600 			s.append((char) 920); // greek capital letter theta, U+0398 ISOgrk3
601 		else if (c.equals("Iota"))
602 			s.append((char) 921); // greek capital letter iota, U+0399
603 		else if (c.equals("Kappa"))
604 			s.append((char) 922); // greek capital letter kappa, U+039A
605 		else if (c.equals("Lambda"))
606 			s.append((char) 923); // greek capital letter lambda, U+039B ISOgrk3
607 		else if (c.equals("Mu"))
608 			s.append((char) 924); // greek capital letter mu, U+039C
609 		else if (c.equals("Nu"))

835 	private XhtmlNode parseNode(Element node, String defaultNS) throws FHIRFormatError {
836 		XhtmlNode res = new XhtmlNode(NodeType.Element);
837 		res.setName(node.getLocalName());
838 		defaultNS = checkNS(res, node, defaultNS);
839 		for (int i = 0; i < node.getAttributes().getLength(); i++) {
840 			Attr attr = (Attr) node.getAttributes().item(i);
841 			if (attributeIsOk(res.getName(), attr.getName(), attr.getValue()) && !attr.getLocalName().startsWith("xmlns"))
842 				res.getAttributes().put(attr.getName(), attr.getValue());
843 		}
844 		Node child = node.getFirstChild();
845 		while (child != null) {
846 			if (child.getNodeType() == Node.TEXT_NODE) {
847 				res.addText(child.getTextContent());
848 			} else if (child.getNodeType() == Node.COMMENT_NODE) {
849 				res.addComment(child.getTextContent());
850 			} else if (child.getNodeType() == Node.ELEMENT_NODE) {
851 				if (elementIsOk(child.getLocalName()))
852 					res.getChildNodes().add(parseNode((Element) child, defaultNS));
853 			} else
854 				throw new FHIRFormatError("Unhandled XHTML feature: " + Integer.toString(child.getNodeType()) + descLoc());
855 			child = child.getNextSibling();
856 		}
857 		return res;
858 	}

860 	private XhtmlNode parseNode(XMLEventReader xpp) throws IOException, FHIRFormatError, XMLStreamException {
861 		XhtmlNode res = new XhtmlNode(NodeType.Element);
862 
863 		if (!xpp.hasNext()) {
864 			return res;
865 		}
866 
867 		StartElement firstEvent = (StartElement) xpp.nextEvent();
868 		res.setName(firstEvent.getSchemaType().getLocalPart());
869 
870 		for (Iterator<?> attrIter = firstEvent.getAttributes(); attrIter.hasNext();) {
871 			Attribute nextAttr = (Attribute) attrIter.next();
872 			if (attributeIsOk(firstEvent.getName().getLocalPart(), nextAttr.getName().getLocalPart(), nextAttr.getValue()))
873 				res.getAttributes().put(nextAttr.getName().getLocalPart(), nextAttr.getValue());
874 		}
875 
876 		while (xpp.hasNext()) {
877 			XMLEvent nextEvent = xpp.nextEvent();
878 			int eventType = nextEvent.getEventType();
879 			if (eventType != XMLEvent.END_ELEMENT) {
880 				break;
881 			}
882 			if (eventType == XMLEvent.CHARACTERS) {
883 				res.addText(((Characters) xpp).getData());
884 			} else if (eventType == XMLEvent.COMMENT) {
885 				res.addComment(((Comment) xpp).getText());
886 			} else if (eventType == XMLEvent.START_ELEMENT) {
887 				StartElement nextStart = (StartElement) nextEvent;
888 				if (elementIsOk(nextStart.getName().getLocalPart())) {
889 					res.getChildNodes().add(parseNode(xpp));
890 				}
891 			} else {
892 				throw new FHIRFormatError("Unhandled XHTML feature: " + Integer.toString(eventType) + descLoc());
893 			}
894 		}
895 		xpp.next();
896 		return res;
897 	}

899 	private char peekChar() throws IOException {
900 		if (cache.length() > 0)
901 			return cache.charAt(0);
902 		else if (!rdr.ready())
903 			return '\0';
904 		else {
905 			char c = (char) rdr.read();
906 			if (c == (char) -1) {
907 				cache = "";
908 				return '\0';
909 			}
910 			cache = Character.toString(c);
911 			return c;
912 		}
913 	}

919 	private char readChar() throws IOException {
920 		char c;
921 		if (cache.length() > 0) {
922 			c = cache.charAt(0);
923 			cache = cache.length() == 1 ? "" : cache.substring(1);
924 		} else if (!rdr.ready())
925 			c = '\0';
926 		else
927 			c = (char) rdr.read();
928 		if (c == '\r' || c == '\n') {
929 			if (c == '\r' || lastChar != '\r') {
930 				line++;
931 				col = 0;
932 			}
933 			lastChar = c;
934 		}
935 		col++;
936 		return c;
937 	}

946 	private String readToCommentEnd() throws IOException, FHIRFormatError {
947 		if (peekChar() == '!')
948 			readChar();
949 		StringBuilder s = new StringBuilder();
950 
951 		boolean simple = true;
952 		if (peekChar() == '-') {
953 			readChar();
954 			simple = peekChar() != '-';
955 			if (simple)
956 				s.append('-');
957 			else
958 				readChar();
959 		}
960 
961 		boolean done = false;
962 		while (!done) {
963 			char c = peekChar();
964 			if (c == '-') {
965 				readChar();
966 				if (peekChar() == '-') {
967 					readChar();
968 					if (peekChar() == '>') {
969 						done = true;
970 					} else
971 						s.append("--");
972 				} else
973 					s.append('-');
974 			} else if (simple && peekChar() == '>') {
975 				done = true;
976 			} else if (c != '\0')
977 				s.append(readChar());
978 			else if (mustBeWellFormed)
979 				throw new FHIRFormatError("Unexpected termination of html source" + descLoc());
980 		}
981 		if (peekChar() != '\0') {
982 			readChar();
983 			skipWhiteSpace();
984 		}
985 		return s.toString();
986 	}

988 	private String readToTagEnd() throws IOException, FHIRFormatError {
989 		StringBuilder s = new StringBuilder();
990 		while (peekChar() != '>' && peekChar() != '\0')
991 			s.append(readChar());
992 		if (peekChar() != '\0') {
993 			readChar();
994 			skipWhiteSpace();
995 		} else if (mustBeWellFormed)
996 			throw new FHIRFormatError("Unexpected termination of html source" + descLoc());
997 		return s.toString();
998 	}

1031 	private void skipWhiteSpaceAndComments(XhtmlNode focus) throws IOException, FHIRFormatError {
1032 		while (Character.isWhitespace(peekChar()) || (peekChar() == 0xfeff))
1033 			readChar();
1034 		if (peekChar() == '<') {
1035 			char ch = readChar();
1036 			if (peekChar() == '!') {
1037 				readChar();
1038 				if (peekChar() == '-') {
1039 					readChar();
1040 					if (peekChar() == '-') {
1041 						readChar();
1042 						if (peekChar() == ' ')
1043 							readChar();
1044 						focus.addComment(readToCommentEnd());
1045 					} else
1046 						throw new FHIRFormatError("unrecognised element type <!" + peekChar() + descLoc());
1047 				} else
1048 					focus.addDocType(readToCommentEnd());
1049 				skipWhiteSpaceAndComments(focus);
1050 			} else if (peekChar() == '?') {
1051 				String r = readToTagEnd();
1052 				focus.addInstruction(r.substring(1, r.length() - 1));
1053 				skipWhiteSpaceAndComments(focus);
1054 			} else
1055 				pushChar(ch);
1056 		}
1057 	}

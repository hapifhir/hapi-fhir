1   package ca.uhn.fhir.validation;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import org.thymeleaf.util.Validate;
7   
8   import ca.uhn.fhir.context.FhirContext;
9   
10  abstract class BaseValidationContext<T> implements IValidationContext<T> {
11  
12  	protected final FhirContext myFhirContext;
13  	private List<SingleValidationMessage> myMessages = new ArrayList<SingleValidationMessage>();
14  
15  	BaseValidationContext(FhirContext theFhirContext) {
16  		myFhirContext = theFhirContext;
17  	}
18  
19  	@Override
20  	public void addValidationMessage(SingleValidationMessage theMessage) {
21  		Validate.notNull(theMessage, "theMessage must not be null");
22  		myMessages.add(theMessage);
23  	}
24  
25  	@Override
26  	public FhirContext getFhirContext() {
27  		return myFhirContext;
28  	}
29  
30  	@Override
31  	public ValidationResult toResult() {
32  		return new ValidationResult(myFhirContext, myMessages);
33  	}
34  
35  }

126 	private Class<?> determineElementType(Field next) {
127 		Class<?> nextElementType = next.getType();
128 		if (List.class.equals(nextElementType)) {
129 			nextElementType = ReflectionUtil.getGenericCollectionTypeOfField(next);
130 		} else if (Collection.class.isAssignableFrom(nextElementType)) {
131 			throw new ConfigurationException("Field '" + next.getName() + "' in type '" + next.getClass().getCanonicalName() + "' is a Collection - Only java.util.List curently supported");
132 		}
133 		return nextElementType;
134 	}

137 	private IValueSetEnumBinder<Enum<?>> getBoundCodeBinder(Field theNext) {
138 		Class<?> bound = getGenericCollectionTypeOfCodedField(theNext);
139 		if (bound == null) {
140 			throw new ConfigurationException("Field '" + theNext + "' has no parameter for " + BoundCodeDt.class.getSimpleName() + " to determine enum type");
141 		}
142 
143 		try {
144 			Field bindingField = bound.getField("VALUESET_BINDER");
145 			return (IValueSetEnumBinder<Enum<?>>) bindingField.get(null);
146 		} catch (IllegalArgumentException e) {
147 			throw new ConfigurationException("Field '" + theNext + "' has type parameter " + bound.getCanonicalName() + " but this class has no valueset binding field", e);
148 		} catch (IllegalAccessException e) {
149 			throw new ConfigurationException("Field '" + theNext + "' has type parameter " + bound.getCanonicalName() + " but this class has no valueset binding field", e);
150 		} catch (NoSuchFieldException e) {
151 			throw new ConfigurationException("Field '" + theNext + "' has type parameter " + bound.getCanonicalName() + " but this class has no valueset binding field", e);
152 		} catch (SecurityException e) {
153 			throw new ConfigurationException("Field '" + theNext + "' has type parameter " + bound.getCanonicalName() + " but this class has no valueset binding field", e);
154 		}

185 	private void init(Map<Class<? extends IBase>, BaseRuntimeElementDefinition<?>> theExistingDefinitions, Set<Class<? extends IBase>> theDatatypes) {
186 		if (theExistingDefinitions != null) {
187 			myClassToElementDefinitions.putAll(theExistingDefinitions);
188 		}
189 
190 		int startSize = myClassToElementDefinitions.size();
191 		long start = System.currentTimeMillis();
192 		Map<String, Class<? extends IBaseResource>> resourceTypes = myNameToResourceType;
193 
194 		myVersionTypes = scanVersionPropertyFile(theDatatypes, resourceTypes, myVersion);
195 
196 		// toScan.add(DateDt.class);
197 		// toScan.add(CodeDt.class);
198 		// toScan.add(DecimalDt.class);
199 		// toScan.add(AttachmentDt.class);
200 		// toScan.add(ResourceReferenceDt.class);
201 		// toScan.add(QuantityDt.class);
202 
203 		do {
204 			for (Class<? extends IBase> nextClass : theDatatypes) {
205 				scan(nextClass);
206 			}
207 			for (Iterator<Class<? extends IBase>> iter = myScanAlso.iterator(); iter.hasNext();) {
208 				if (myClassToElementDefinitions.containsKey(iter.next())) {
209 					iter.remove();
210 				}
211 			}
212 			theDatatypes.clear();
213 			theDatatypes.addAll(myScanAlso);
214 			myScanAlso.clear();
215 		} while (!theDatatypes.isEmpty());
216 
217 		for (Entry<Class<? extends IBase>, BaseRuntimeElementDefinition<?>> nextEntry : myClassToElementDefinitions.entrySet()) {
218 			if (theExistingDefinitions != null && theExistingDefinitions.containsKey(nextEntry.getKey())) {
219 				continue;
220 			}
221 			BaseRuntimeElementDefinition<?> next = nextEntry.getValue();
222 			next.sealAndInitialize(myContext, myClassToElementDefinitions);
223 		}
224 
225 		myRuntimeChildUndeclaredExtensionDefinition = new RuntimeChildUndeclaredExtensionDefinition();
226 		myRuntimeChildUndeclaredExtensionDefinition.sealAndInitialize(myContext, myClassToElementDefinitions);
227 
228 		long time = System.currentTimeMillis() - start;
229 		int size = myClassToElementDefinitions.size() - startSize;
230 		ourLog.debug("Done scanning FHIR library, found {} model entries in {}ms", size, time);
231 	}

237 	/**
238 	 * There are two implementations of all of the annotations (e.g. {@link Child} and
239 	 * {@link org.hl7.fhir.instance.model.annotations.Child}) since the HL7.org ones will eventually replace the HAPI
240 	 * ones. Annotations can't extend each other or implement interfaces or anything like that, so rather than duplicate
241 	 * all of the annotation processing code this method just creates an interface Proxy to simulate the HAPI
242 	 * annotations if the HL7.org ones are found instead.
243 	 */
244 	@SuppressWarnings("unchecked")
245 	private <T extends Annotation> T pullAnnotation(Class<?> theContainer, AnnotatedElement theTarget, Class<T> theAnnotationType) {
246 
247 		T retVal = theTarget.getAnnotation(theAnnotationType);
248 		if (myContext.getVersion().getVersion() != FhirVersionEnum.DSTU2_HL7ORG) {
249 			return retVal;
250 		}
251 
252 		if (retVal == null) {
253 			final Class<? extends Annotation> altAnnotationClass;
254 			/*
255 			 * Use a cache to minimize Class.forName calls, since they are slow and expensive..
256 			 */
257 			if (myAnnotationForwards.containsKey(theAnnotationType) == false) {
258 				String sourceClassName = theAnnotationType.getName();
259 				String candidateAltClassName = sourceClassName.replace("ca.uhn.fhir.model.api.annotation", "org.hl7.fhir.instance.model.annotations");
260 				if (!sourceClassName.equals(candidateAltClassName)) {
261 					Class<?> forName;
262 					try {
263 						forName = Class.forName(candidateAltClassName);
264 						ourLog.debug("Forwarding annotation request for [{}] to class [{}]", theAnnotationType, forName);
265 					} catch (ClassNotFoundException e) {
266 						forName = null;
267 					}
268 					altAnnotationClass = (Class<? extends Annotation>) forName;
269 				} else {
270 					altAnnotationClass = null;
271 				}
272 				myAnnotationForwards.put(theAnnotationType, altAnnotationClass);
273 			} else {
274 				altAnnotationClass = myAnnotationForwards.get(theAnnotationType);
275 			}
276 
277 			if (altAnnotationClass == null) {
278 				return null;
279 			}
280 
281 			final Annotation altAnnotation;
282 			altAnnotation = theTarget.getAnnotation(altAnnotationClass);
283 			if (altAnnotation == null) {
284 				return null;
285 			}
286 
287 			InvocationHandler h = new InvocationHandler() {
288 
289 				@Override
290 				public Object invoke(Object theProxy, Method theMethod, Object[] theArgs) throws Throwable {
291 					Method altMethod = altAnnotationClass.getMethod(theMethod.getName(), theMethod.getParameterTypes());
292 					return altMethod.invoke(altAnnotation, theArgs);
293 				}
294 			};
295 			retVal = (T) Proxy.newProxyInstance(theAnnotationType.getClassLoader(), new Class<?>[] { theAnnotationType }, h);
296 
297 		}
298 
299 		return retVal;
300 	}

302 	private void scan(Class<? extends IBase> theClass) throws ConfigurationException {
303 		BaseRuntimeElementDefinition<?> existingDef = myClassToElementDefinitions.get(theClass);
304 		if (existingDef != null) {
305 			return;
306 		}
307 
308 		ResourceDef resourceDefinition = pullAnnotation(theClass, theClass, ResourceDef.class);
309 		if (resourceDefinition != null) {
310 			if (!IBaseResource.class.isAssignableFrom(theClass)) {
311 				throw new ConfigurationException("Resource type contains a @" + ResourceDef.class.getSimpleName() + " annotation but does not implement " + IResource.class.getCanonicalName() + ": " + theClass.getCanonicalName());
312 			}
313 			@SuppressWarnings("unchecked")
314 			Class<? extends IBaseResource> resClass = (Class<? extends IBaseResource>) theClass;
315 			scanResource(resClass, resourceDefinition);
316 		}
317 
318 		DatatypeDef datatypeDefinition = pullAnnotation(theClass, theClass, DatatypeDef.class);
319 		if (datatypeDefinition != null) {
320 			if (ICompositeType.class.isAssignableFrom(theClass)) {
321 				@SuppressWarnings("unchecked")
322 				Class<? extends ICompositeType> resClass = (Class<? extends ICompositeType>) theClass;
323 				scanCompositeDatatype(resClass, datatypeDefinition);
324 			} else if (IPrimitiveType.class.isAssignableFrom(theClass)) {
325 				@SuppressWarnings({ "unchecked" })
326 				Class<? extends IPrimitiveType<?>> resClass = (Class<? extends IPrimitiveType<?>>) theClass;
327 				scanPrimitiveDatatype(resClass, datatypeDefinition);
328 			} else {
329 				return;
330 //				throw new ConfigurationException("Resource type contains a @" + DatatypeDef.class.getSimpleName() + " annotation but does not implement " + IDatatype.class.getCanonicalName() + ": " + theClass.getCanonicalName());
331 			}
332 		}
333 
334 		Block blockDefinition = pullAnnotation(theClass, theClass, Block.class);
335 
336 		if (blockDefinition != null) {
337 			if (IResourceBlock.class.isAssignableFrom(theClass) || IBaseBackboneElement.class.isAssignableFrom(theClass) || IBaseDatatypeElement.class.isAssignableFrom(theClass)) {
338 				scanBlock(theClass);
339 			} else {
340 				throw new ConfigurationException("Type contains a @" + Block.class.getSimpleName() + " annotation but does not implement " + IResourceBlock.class.getCanonicalName() + ": " + theClass.getCanonicalName());
341 			}
342 		}
343 
344 		if (blockDefinition == null && datatypeDefinition == null && resourceDefinition == null) {
345 			throw new ConfigurationException("Resource class[" + theClass.getName() + "] does not contain any valid HAPI-FHIR annotations");
346 		}
347 	}

349 	private void scanBlock(Class<? extends IBase> theClass) {
350 		ourLog.debug("Scanning resource block class: {}", theClass.getName());
351 
352 		String resourceName = theClass.getCanonicalName();
353 		if (isBlank(resourceName)) {
354 			throw new ConfigurationException("Block type @" + Block.class.getSimpleName() + " annotation contains no name: " + theClass.getCanonicalName());
355 		}
356 
357 		RuntimeResourceBlockDefinition resourceDef = new RuntimeResourceBlockDefinition(resourceName, theClass, isStandardType(theClass));
358 		myClassToElementDefinitions.put(theClass, resourceDef);
359 
360 		scanCompositeElementForChildren(theClass, resourceDef);
361 	}

363 	private void scanCompositeDatatype(Class<? extends ICompositeType> theClass, DatatypeDef theDatatypeDefinition) {
364 		ourLog.debug("Scanning datatype class: {}", theClass.getName());
365 
366 		RuntimeCompositeDatatypeDefinition resourceDef;
367 		if (theClass.equals(ExtensionDt.class)) {
368 			resourceDef = new RuntimeExtensionDtDefinition(theDatatypeDefinition, theClass, true);
369 		} else {
370 			resourceDef = new RuntimeCompositeDatatypeDefinition(theDatatypeDefinition, theClass, isStandardType(theClass));
371 		}
372 		myClassToElementDefinitions.put(theClass, resourceDef);
373 		myNameToElementDefinitions.put(resourceDef.getName(), resourceDef);
374 		scanCompositeElementForChildren(theClass, resourceDef);
375 	}

377 	@SuppressWarnings("unchecked")
378 	private void scanCompositeElementForChildren(Class<? extends IBase> theClass, BaseRuntimeElementCompositeDefinition<?> theDefinition) {
379 		Set<String> elementNames = new HashSet<String>();
380 		TreeMap<Integer, BaseRuntimeDeclaredChildDefinition> orderToElementDef = new TreeMap<Integer, BaseRuntimeDeclaredChildDefinition>();
381 		TreeMap<Integer, BaseRuntimeDeclaredChildDefinition> orderToExtensionDef = new TreeMap<Integer, BaseRuntimeDeclaredChildDefinition>();
382 
383 		LinkedList<Class<? extends IBase>> classes = new LinkedList<Class<? extends IBase>>();
384 
385 		/*
386 		 * We scan classes for annotated fields in the class but also all of its superclasses
387 		 */
388 		Class<? extends IBase> current = theClass;
389 		do {
390 			classes.push(current);
391 			if (IBase.class.isAssignableFrom(current.getSuperclass())) {
392 				current = (Class<? extends IBase>) current.getSuperclass();
393 			} else {
394 				current = null;
395 			}
396 		} while (current != null);
397 
398 		for (Class<? extends IBase> next : classes) {
399 			scanCompositeElementForChildren(next, elementNames, orderToElementDef, orderToExtensionDef);
400 		}
401 
402 		// while (orderToElementDef.size() > 0 && orderToElementDef.firstKey() <
403 		// 0) {
404 		// BaseRuntimeDeclaredChildDefinition elementDef =
405 		// orderToElementDef.remove(orderToElementDef.firstKey());
406 		// if (elementDef.getElementName().equals("identifier")) {
407 		// orderToElementDef.put(theIdentifierOrder, elementDef);
408 		// } else {
409 		// throw new ConfigurationException("Don't know how to handle element: "
410 		// + elementDef.getElementName());
411 		// }
412 		// }
413 
414 		TreeSet<Integer> orders = new TreeSet<Integer>();
415 		orders.addAll(orderToElementDef.keySet());
416 		orders.addAll(orderToExtensionDef.keySet());
417 
418 		for (Integer i : orders) {
419 			BaseRuntimeChildDefinition nextChild = orderToElementDef.get(i);
420 			if (nextChild != null) {
421 				theDefinition.addChild(nextChild);
422 			}
423 			BaseRuntimeDeclaredChildDefinition nextExt = orderToExtensionDef.get(i);
424 			if (nextExt != null) {
425 				theDefinition.addExtension((RuntimeChildDeclaredExtensionDefinition) nextExt);
426 			}
427 		}
428 
429 	}

431 	@SuppressWarnings("unchecked")
432 	private void scanCompositeElementForChildren(Class<? extends IBase> theClass, Set<String> elementNames, TreeMap<Integer, BaseRuntimeDeclaredChildDefinition> theOrderToElementDef, TreeMap<Integer, BaseRuntimeDeclaredChildDefinition> theOrderToExtensionDef) {
433 		int baseElementOrder = theOrderToElementDef.isEmpty() ? 0 : theOrderToElementDef.lastEntry().getKey() + 1;
434 
435 		for (Field next : theClass.getDeclaredFields()) {
436 
437 			if (Modifier.isFinal(next.getModifiers())) {
438 				ourLog.trace("Ignoring constant {} on target type {}", next.getName(), theClass);
439 				continue;
440 			}
441 
442 			Child childAnnotation = pullAnnotation(theClass, next, Child.class);
443 			if (childAnnotation == null) {
444 				ourLog.trace("Ignoring non @Child field {} on target type {}", next.getName(), theClass);
445 				continue;
446 			}
447 
448 			Description descriptionAnnotation = pullAnnotation(theClass, next, Description.class);
449 
450 			TreeMap<Integer, BaseRuntimeDeclaredChildDefinition> orderMap = theOrderToElementDef;
451 			Extension extensionAttr = pullAnnotation(theClass, next, Extension.class);
452 			if (extensionAttr != null) {
453 				orderMap = theOrderToExtensionDef;
454 			}
455 
456 			String elementName = childAnnotation.name();
457 			int order = childAnnotation.order();
458 			if (order == Child.REPLACE_PARENT) {
459 
460 				if (extensionAttr != null) {
461 
462 					for (Entry<Integer, BaseRuntimeDeclaredChildDefinition> nextEntry : orderMap.entrySet()) {
463 						BaseRuntimeDeclaredChildDefinition nextDef = nextEntry.getValue();
464 						if (nextDef instanceof RuntimeChildDeclaredExtensionDefinition) {
465 							if (nextDef.getExtensionUrl().equals(extensionAttr.url())) {
466 								order = nextEntry.getKey();
467 								orderMap.remove(nextEntry.getKey());
468 								elementNames.remove(elementName);
469 								break;
470 							}
471 						}
472 					}
473 					if (order == Child.REPLACE_PARENT) {
474 						throw new ConfigurationException("Field " + next.getName() + "' on target type " + theClass.getSimpleName() + " has order() of REPLACE_PARENT (" + Child.REPLACE_PARENT + ") but no parent element with extension URL " + extensionAttr.url() + " could be found on type "
475 								+ next.getDeclaringClass().getSimpleName());
476 					}
477 
478 				} else {
479 
480 					for (Entry<Integer, BaseRuntimeDeclaredChildDefinition> nextEntry : orderMap.entrySet()) {
481 						BaseRuntimeDeclaredChildDefinition nextDef = nextEntry.getValue();
482 						if (elementName.equals(nextDef.getElementName())) {
483 							order = nextEntry.getKey();
484 							orderMap.remove(nextEntry.getKey());
485 							elementNames.remove(elementName);
486 							break;
487 						}
488 					}
489 					if (order == Child.REPLACE_PARENT) {
490 						throw new ConfigurationException("Field " + next.getName() + "' on target type " + theClass.getSimpleName() + " has order() of REPLACE_PARENT (" + Child.REPLACE_PARENT + ") but no parent element with name " + elementName + " could be found on type "
491 								+ next.getDeclaringClass().getSimpleName());
492 					}
493 
494 				}
495 
496 			}
497 			if (order < 0 && order != Child.ORDER_UNKNOWN) {
498 				throw new ConfigurationException("Invalid order '" + order + "' on @Child for field '" + next.getName() + "' on target type: " + theClass);
499 			}
500 			if (order != Child.ORDER_UNKNOWN) {
501 				order = order + baseElementOrder;
502 			}
503 			// int min = childAnnotation.min();
504 			// int max = childAnnotation.max();
505 
506 			/*
507 			 * Anything that's marked as unknown is given a new ID that is <0 so that it doesn't conflict with any given
508 			 * IDs and can be figured out later
509 			 */
510 			if (order == Child.ORDER_UNKNOWN) {
511 				order = Integer.MIN_VALUE;
512 				while (orderMap.containsKey(order)) {
513 					order++;
514 				}
515 			}
516 
517 			List<Class<? extends IBase>> choiceTypes = new ArrayList<Class<? extends IBase>>();
518 			for (Class<? extends IBase> nextChoiceType : childAnnotation.type()) {
519 				choiceTypes.add(nextChoiceType);
520 			}
521 
522 			if (orderMap.containsKey(order)) {
523 				throw new ConfigurationException("Detected duplicate field order '" + childAnnotation.order() + "' for element named '" + elementName + "' in type '" + theClass.getCanonicalName() + "'");
524 			}
525 
526 			if (elementNames.contains(elementName)) {
527 				throw new ConfigurationException("Detected duplicate field name '" + elementName + "' in type '" + theClass.getCanonicalName() + "'");
528 			}
529 
530 			Class<?> nextElementType = determineElementType(next);
531 
532 			if (BaseContainedDt.class.isAssignableFrom(nextElementType) || (childAnnotation.name().equals("contained") && IBaseResource.class.isAssignableFrom(nextElementType))) {
533 				/*
534 				 * Child is contained resources
535 				 */
536 				RuntimeChildContainedResources def = new RuntimeChildContainedResources(next, childAnnotation, descriptionAnnotation, elementName);
537 				orderMap.put(order, def);
538 
539 			} else if (IAnyResource.class.isAssignableFrom(nextElementType) || IResource.class.equals(nextElementType)) {
540 				/*
541 				 * Child is a resource as a direct child, as in Bundle.entry.resource
542 				 */
543 				RuntimeChildDirectResource def = new RuntimeChildDirectResource(next, childAnnotation, descriptionAnnotation, elementName);
544 				orderMap.put(order, def);
545 
546 			} else if (choiceTypes.size() > 1 && !BaseResourceReferenceDt.class.isAssignableFrom(nextElementType) && !IBaseReference.class.isAssignableFrom(nextElementType)) {
547 				/*
548 				 * Child is a choice element
549 				 */
550 				for (Class<? extends IBase> nextType : choiceTypes) {
551 					addScanAlso(nextType);
552 				}
553 				RuntimeChildChoiceDefinition def = new RuntimeChildChoiceDefinition(next, elementName, childAnnotation, descriptionAnnotation, choiceTypes);
554 				orderMap.put(order, def);
555 
556 			} else if (next.getType().equals(ExtensionDt.class)) {
557 
558 				RuntimeChildExtensionDt def = new RuntimeChildExtensionDt(next, elementName, childAnnotation, descriptionAnnotation);
559 				orderMap.put(order, def);
560 				if (IElement.class.isAssignableFrom(nextElementType)) {
561 					addScanAlso((Class<? extends IElement>) nextElementType);
562 				}
563 
564 			} else if (extensionAttr != null) {
565 				/*
566 				 * Child is an extension
567 				 */
568 				Class<? extends IBase> et = (Class<? extends IBase>) nextElementType;
569 
570 				Object binder = null;
571 				if (BoundCodeDt.class.isAssignableFrom(nextElementType) || IBoundCodeableConcept.class.isAssignableFrom(nextElementType)) {
572 					binder = getBoundCodeBinder(next);
573 				}
574 
575 				RuntimeChildDeclaredExtensionDefinition def = new RuntimeChildDeclaredExtensionDefinition(next, childAnnotation, descriptionAnnotation, extensionAttr, elementName, extensionAttr.url(), et, binder);
576 
577 				if (IBaseEnumeration.class.isAssignableFrom(nextElementType)) {
578 					def.setEnumerationType(ReflectionUtil.getGenericCollectionTypeOfFieldWithSecondOrderForList(next));
579 				}
580 
581 				orderMap.put(order, def);
582 				if (IBase.class.isAssignableFrom(nextElementType)) {
583 					addScanAlso((Class<? extends IBase>) nextElementType);
584 				}
585 			} else if (BaseResourceReferenceDt.class.isAssignableFrom(nextElementType) || IBaseReference.class.isAssignableFrom(nextElementType)) {
586 				/*
587 				 * Child is a resource reference
588 				 */
589 				List<Class<? extends IBaseResource>> refTypesList = new ArrayList<Class<? extends IBaseResource>>();
590 				for (Class<? extends IElement> nextType : childAnnotation.type()) {
591 					if (IBaseResource.class.isAssignableFrom(nextType) == false) {
592 						throw new ConfigurationException("Field '" + next.getName() + "' in class '" + next.getDeclaringClass().getCanonicalName() + "' is of type " + BaseResourceReferenceDt.class + " but contains a non-resource type: " + nextType.getCanonicalName());
593 					}
594 					refTypesList.add((Class<? extends IBaseResource>) nextType);
595 					addScanAlso(nextType);
596 				}
597 				RuntimeChildResourceDefinition def = new RuntimeChildResourceDefinition(next, elementName, childAnnotation, descriptionAnnotation, refTypesList);
598 				orderMap.put(order, def);
599 
600 			} else if (IResourceBlock.class.isAssignableFrom(nextElementType) || IBaseBackboneElement.class.isAssignableFrom(nextElementType) || IBaseDatatypeElement.class.isAssignableFrom(nextElementType)) {
601 				/*
602 				 * Child is a resource block (i.e. a sub-tag within a resource) TODO: do these have a better name
603 				 * according to HL7?
604 				 */
605 
606 				Class<? extends IBase> blockDef = (Class<? extends IBase>) nextElementType;
607 				addScanAlso(blockDef);
608 				RuntimeChildResourceBlockDefinition def = new RuntimeChildResourceBlockDefinition(next, childAnnotation, descriptionAnnotation, elementName, blockDef);
609 				orderMap.put(order, def);
610 
611 			} else if (IDatatype.class.equals(nextElementType) || IElement.class.equals(nextElementType) || "org.hl7.fhir.instance.model.Type".equals(nextElementType.getName()) || IBaseDatatype.class.equals(nextElementType)) {
612 
613 				RuntimeChildAny def = new RuntimeChildAny(next, elementName, childAnnotation, descriptionAnnotation);
614 				orderMap.put(order, def);
615 
616 			} else if (IDatatype.class.isAssignableFrom(nextElementType) || IPrimitiveType.class.isAssignableFrom(nextElementType) || ICompositeType.class.isAssignableFrom(nextElementType) || IBaseDatatype.class.isAssignableFrom(nextElementType)
617 					|| IBaseExtension.class.isAssignableFrom(nextElementType)) {
618 				Class<? extends IBase> nextDatatype = (Class<? extends IBase>) nextElementType;
619 
620 				addScanAlso(nextDatatype);
621 				BaseRuntimeChildDatatypeDefinition def;
622 				if (IPrimitiveType.class.isAssignableFrom(nextElementType)) {
623 					if (nextElementType.equals(BoundCodeDt.class)) {
624 						IValueSetEnumBinder<Enum<?>> binder = getBoundCodeBinder(next);
625 						def = new RuntimeChildPrimitiveBoundCodeDatatypeDefinition(next, elementName, childAnnotation, descriptionAnnotation, nextDatatype, binder);
626 					} else if (IBaseEnumeration.class.isAssignableFrom(nextElementType)) {
627 						Class<?> binderType = ReflectionUtil.getGenericCollectionTypeOfFieldWithSecondOrderForList(next);
628 						def = new RuntimeChildPrimitiveEnumerationDatatypeDefinition(next, elementName, childAnnotation, descriptionAnnotation, nextDatatype, binderType);
629 					} else {
630 						def = new RuntimeChildPrimitiveDatatypeDefinition(next, elementName, descriptionAnnotation, childAnnotation, nextDatatype);
631 					}
632 				} else if (IBaseXhtml.class.isAssignableFrom(nextElementType)) {
633 					def = new RuntimeChildXhtmlDatatypeDefinition(next, elementName, descriptionAnnotation, childAnnotation, nextDatatype);
634 				} else {
635 					if (IBoundCodeableConcept.class.isAssignableFrom(nextElementType)) {
636 						IValueSetEnumBinder<Enum<?>> binder = getBoundCodeBinder(next);
637 						def = new RuntimeChildCompositeBoundDatatypeDefinition(next, elementName, childAnnotation, descriptionAnnotation, nextDatatype, binder);
638 					} else if (BaseNarrativeDt.class.isAssignableFrom(nextElementType) || INarrative.class.isAssignableFrom(nextElementType)) {
639 						def = new RuntimeChildNarrativeDefinition(next, elementName, childAnnotation, descriptionAnnotation, nextDatatype);
640 					} else {
641 						def = new RuntimeChildCompositeDatatypeDefinition(next, elementName, childAnnotation, descriptionAnnotation, nextDatatype);
642 					}
643 				}
644 
645 				CodeableConceptElement concept = pullAnnotation(theClass, next, CodeableConceptElement.class);
646 				if (concept != null) {
647 					if (!ICodedDatatype.class.isAssignableFrom(nextDatatype)) {
648 						throw new ConfigurationException("Field '" + elementName + "' in type '" + theClass.getCanonicalName() + "' is marked as @" + CodeableConceptElement.class.getCanonicalName() + " but type is not a subtype of " + ICodedDatatype.class.getName());
649 					} else {
650 						Class<? extends ICodeEnum> type = concept.type();
651 						myScanAlsoCodeTable.add(type);
652 						def.setCodeType(type);
653 					}
654 				}
655 
656 				orderMap.put(order, def);
657 
658 			} else {
659 				throw new ConfigurationException("Field '" + elementName + "' in type '" + theClass.getCanonicalName() + "' is not a valid child type: " + nextElementType);
660 			}
661 
662 			elementNames.add(elementName);
663 		}
664 	}

666 	private String scanPrimitiveDatatype(Class<? extends IPrimitiveType<?>> theClass, DatatypeDef theDatatypeDefinition) {
667 		ourLog.debug("Scanning resource class: {}", theClass.getName());
668 
669 		String resourceName = theDatatypeDefinition.name();
670 		if (isBlank(resourceName)) {
671 			throw new ConfigurationException("Resource type @" + ResourceDef.class.getSimpleName() + " annotation contains no resource name: " + theClass.getCanonicalName());
672 		}
673 
674 		BaseRuntimeElementDefinition<?> resourceDef;
675 		if (theClass.equals(XhtmlDt.class)) {
676 			@SuppressWarnings("unchecked")
677 			Class<XhtmlDt> clazz = (Class<XhtmlDt>) theClass;
678 			resourceDef = new RuntimePrimitiveDatatypeNarrativeDefinition(resourceName, clazz, isStandardType(clazz));
679 		} else if (IBaseXhtml.class.isAssignableFrom(theClass)) {
680 			@SuppressWarnings("unchecked")
681 			Class<? extends IBaseXhtml> clazz = (Class<? extends IBaseXhtml>) theClass;
682 			resourceDef = new RuntimePrimitiveDatatypeXhtmlHl7OrgDefinition(resourceName, clazz, isStandardType(clazz));
683 		} else if (IIdType.class.isAssignableFrom(theClass)) {
684 			resourceDef = new RuntimeIdDatatypeDefinition(theDatatypeDefinition, theClass, isStandardType(theClass));
685 		} else {
686 			resourceDef = new RuntimePrimitiveDatatypeDefinition(theDatatypeDefinition, theClass, isStandardType(theClass));
687 		}
688 		myClassToElementDefinitions.put(theClass, resourceDef);
689 		if (!theDatatypeDefinition.isSpecialization()) {
690 			myNameToElementDefinitions.put(resourceName, resourceDef);
691 		}
692 
693 		return resourceName;
694 	}

696 	private String scanResource(Class<? extends IBaseResource> theClass, ResourceDef resourceDefinition) {
697 		ourLog.debug("Scanning resource class: {}", theClass.getName());
698 
699 		boolean primaryNameProvider = true;
700 		String resourceName = resourceDefinition.name();
701 		if (isBlank(resourceName)) {
702 			Class<?> parent = theClass.getSuperclass();
703 			primaryNameProvider = false;
704 			while (parent.equals(Object.class) == false && isBlank(resourceName)) {
705 				ResourceDef nextDef = pullAnnotation(theClass, parent, ResourceDef.class);
706 				if (nextDef != null) {
707 					resourceName = nextDef.name();
708 				}
709 				parent = parent.getSuperclass();
710 			}
711 			if (isBlank(resourceName)) {
712 				throw new ConfigurationException("Resource type @" + ResourceDef.class.getSimpleName() + " annotation contains no resource name(): " + theClass.getCanonicalName() + " - This is only allowed for types that extend other resource types ");
713 			}
714 		}
715 
716 		String resourceId = resourceDefinition.id();
717 		if (!isBlank(resourceId)) {
718 			if (myIdToResourceDefinition.containsKey(resourceId)) {
719 				throw new ConfigurationException("The following resource types have the same ID of '" + resourceId + "' - " + theClass.getCanonicalName() + " and " + myIdToResourceDefinition.get(resourceId).getImplementingClass().getCanonicalName());
720 			}
721 		}
722 
723 		RuntimeResourceDefinition resourceDef = new RuntimeResourceDefinition(myContext, resourceName, theClass, resourceDefinition, isStandardType(theClass));
724 		myClassToElementDefinitions.put(theClass, resourceDef);
725 		if (primaryNameProvider) {
726 			if (resourceDef.getStructureVersion() == myVersion) {
727 				myNameToResourceDefinitions.put(resourceName, resourceDef);
728 			}
729 		}
730 		scanCompositeElementForChildren(theClass, resourceDef);
731 
732 		myIdToResourceDefinition.put(resourceId, resourceDef);
733 
734 		scanResourceForSearchParams(theClass, resourceDef);
735 
736 		return resourceName;
737 	}

739 	private void scanResourceForSearchParams(Class<? extends IBaseResource> theClass, RuntimeResourceDefinition theResourceDef) {
740 
741 		Map<String, RuntimeSearchParam> nameToParam = new HashMap<String, RuntimeSearchParam>();
742 		Map<Field, SearchParamDefinition> compositeFields = new LinkedHashMap<Field, SearchParamDefinition>();
743 
744 		for (Field nextField : theClass.getFields()) {
745 			SearchParamDefinition searchParam = pullAnnotation(theClass, nextField, SearchParamDefinition.class);
746 			if (searchParam != null) {
747 				RestSearchParameterTypeEnum paramType = RestSearchParameterTypeEnum.valueOf(searchParam.type().toUpperCase());
748 				if (paramType == null) {
749 					throw new ConfigurationException("Search param " + searchParam.name() + " has an invalid type: " + searchParam.type());
750 				}
751 				if (paramType == RestSearchParameterTypeEnum.COMPOSITE) {
752 					compositeFields.put(nextField, searchParam);
753 					continue;
754 				}
755 				RuntimeSearchParam param = new RuntimeSearchParam(searchParam.name(), searchParam.description(), searchParam.path(), paramType);
756 				theResourceDef.addSearchParam(param);
757 				nameToParam.put(param.getName(), param);
758 			}
759 		}
760 
761 		for (Entry<Field, SearchParamDefinition> nextEntry : compositeFields.entrySet()) {
762 			SearchParamDefinition searchParam = nextEntry.getValue();
763 
764 			List<RuntimeSearchParam> compositeOf = new ArrayList<RuntimeSearchParam>();
765 			for (String nextName : searchParam.compositeOf()) {
766 				RuntimeSearchParam param = nameToParam.get(nextName);
767 				if (param == null) {
768 					ourLog.warn("Search parameter {}.{} declares that it is a composite with compositeOf value '{}' but that is not a valid parametr name itself. Valid values are: {}", new Object[] { theResourceDef.getName(), searchParam.name(), nextName, nameToParam.keySet() });
769 					continue;
770 				}
771 				compositeOf.add(param);
772 			}
773 
774 			RuntimeSearchParam param = new RuntimeSearchParam(searchParam.name(), searchParam.description(), searchParam.path(), RestSearchParameterTypeEnum.COMPOSITE, compositeOf);
775 			theResourceDef.addSearchParam(param);
776 		}
777 	}

779 	private static Class<?> getGenericCollectionTypeOfCodedField(Field next) {
780 		Class<?> type;
781 		ParameterizedType collectionType = (ParameterizedType) next.getGenericType();
782 		Type firstArg = collectionType.getActualTypeArguments()[0];
783 		if (ParameterizedType.class.isAssignableFrom(firstArg.getClass())) {
784 			ParameterizedType pt = ((ParameterizedType) firstArg);
785 			firstArg = pt.getActualTypeArguments()[0];
786 			type = (Class<?>) firstArg;
787 		} else {
788 			type = (Class<?>) firstArg;
789 		}
790 		return type;
791 	}

793 	static Set<Class<? extends IBase>> scanVersionPropertyFile(Set<Class<? extends IBase>> theDatatypes, Map<String, Class<? extends IBaseResource>> theResourceTypes, FhirVersionEnum version) {
794 		Set<Class<? extends IBase>> retVal = new HashSet<Class<? extends IBase>>();
795 
796 		InputStream str = version.getVersionImplementation().getFhirVersionPropertiesFile();
797 		Properties prop = new Properties();
798 		try {
799 			prop.load(str);
800 			for (Entry<Object, Object> nextEntry : prop.entrySet()) {
801 				String nextKey = nextEntry.getKey().toString();
802 				String nextValue = nextEntry.getValue().toString();
803 
804 				if (nextKey.startsWith("datatype.")) {
805 					if (theDatatypes != null) {
806 						try {
807 							// Datatypes
808 
809 							@SuppressWarnings("unchecked")
810 							Class<? extends IBase> dtType = (Class<? extends IBase>) Class.forName(nextValue);
811 							retVal.add(dtType);
812 
813 							if (IElement.class.isAssignableFrom(dtType)) {
814 								@SuppressWarnings("unchecked")
815 								Class<? extends IElement> nextClass = (Class<? extends IElement>) dtType;
816 								theDatatypes.add(nextClass);
817 							} else if (IBaseDatatype.class.isAssignableFrom(dtType)) {
818 								@SuppressWarnings("unchecked")
819 								Class<? extends IBaseDatatype> nextClass = (Class<? extends IBaseDatatype>) dtType;
820 								theDatatypes.add(nextClass);
821 							} else {
822 								ourLog.warn("Class is not assignable from " + IElement.class.getSimpleName() + " or " + IBaseDatatype.class.getSimpleName() + ": " + nextValue);
823 								continue;
824 							}
825 
826 						} catch (ClassNotFoundException e) {
827 							throw new ConfigurationException("Unknown class[" + nextValue + "] for data type definition: " + nextKey.substring("datatype.".length()), e);
828 						}
829 					}
830 				} else if (nextKey.startsWith("resource.")) {
831 					// Resources
832 					String resName = nextKey.substring("resource.".length()).toLowerCase();
833 					try {
834 						@SuppressWarnings("unchecked")
835 						Class<? extends IBaseResource> nextClass = (Class<? extends IBaseResource>) Class.forName(nextValue);
836 						if (!IBaseResource.class.isAssignableFrom(nextClass)) {
837 							throw new ConfigurationException("Class is not assignable from " + IBaseResource.class.getSimpleName() + ": " + nextValue);
838 						}
839 
840 						theResourceTypes.put(resName, nextClass);
841 					} catch (ClassNotFoundException e) {
842 						throw new ConfigurationException("Unknown class[" + nextValue + "] for resource definition: " + nextKey.substring("resource.".length()), e);
843 					}
844 				} else {
845 					throw new ConfigurationException("Unexpected property in version property file: " + nextKey + "=" + nextValue);
846 				}
847 			}
848 		} catch (IOException e) {
849 			throw new ConfigurationException("Failed to load model property file from classpath: " + "/ca/uhn/fhir/model/dstu/model.properties");
850 		}
851 
852 		return retVal;
853 	}

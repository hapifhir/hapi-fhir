65  /**
66   * The FHIR context is the central starting point for the use of the HAPI FHIR API. It should be created once, and then
67   * used as a factory for various other types of objects (parsers, clients, etc.).
68   * 
69   * <p>
70   * Important usage notes:
71   * </p>
72   * <ul>
73   * <li>
74   * Thread safety: <b>This class is thread safe</b> and may be shared between multiple processing
75   * threads, except for the {@link #registerCustomType} and {@link #registerCustomTypes} methods.
76   * </li>
77   * <li>
78   * Performance: <b>This class is expensive</b> to create, as it scans every resource class it needs to parse or encode
79   * to build up an internal model of those classes. For that reason, you should try to create one FhirContext instance
80   * which remains for the life of your application and reuse that instance. Note that it will not cause problems to
81   * create multiple instances (ie. resources originating from one FhirContext may be passed to parsers originating from
82   * another) but you will incur a performance penalty if a new FhirContext is created for every message you parse/encode.
83   * </li>
84   * </ul>
85   */

177 	private FhirContext(FhirVersionEnum theVersion, Collection<Class<? extends IBaseResource>> theResourceTypes) {
178 		VersionUtil.getVersion();
179 
180 		if (theVersion != null) {
181 			if (!theVersion.isPresentOnClasspath()) {
182 				throw new IllegalStateException(getLocalizer().getMessage(FhirContext.class, "noStructuresForSpecifiedVersion", theVersion.name()));
183 			}
184 			myVersion = theVersion.getVersionImplementation();
185 		} else if (FhirVersionEnum.DSTU1.isPresentOnClasspath()) {
186 			myVersion = FhirVersionEnum.DSTU1.getVersionImplementation();
187 		} else if (FhirVersionEnum.DSTU2.isPresentOnClasspath()) {
188 			myVersion = FhirVersionEnum.DSTU2.getVersionImplementation();
189 		} else if (FhirVersionEnum.DSTU2_HL7ORG.isPresentOnClasspath()) {
190 			myVersion = FhirVersionEnum.DSTU2_HL7ORG.getVersionImplementation();
191 		} else {
192 			throw new IllegalStateException(getLocalizer().getMessage(FhirContext.class, "noStructures"));
193 		}
194 
195 		if (theVersion == null) {
196 			ourLog.info("Creating new FhirContext with auto-detected version [{}]. It is recommended to explicitly select a version for future compatibility by invoking FhirContext.forDstuX()", myVersion.getVersion().name());
197 		} else {
198 			ourLog.info("Creating new FHIR context for FHIR version [{}]", myVersion.getVersion().name());
199 		}
200 		
201 		myResourceTypesToScan = theResourceTypes;
202 		
203 		/*
204 		 * Check if we're running in Android mode and configure the context appropriately if so
205 		 */
206 		try {
207 			Class<?> clazz = Class.forName("ca.uhn.fhir.android.AndroidMarker");
208 			ourLog.info("Android mode detected, configuring FhirContext for Android operation");
209 			try {
210 				Method method = clazz.getMethod("configureContext", FhirContext.class);
211 				method.invoke(null, this);
212 			} catch (Throwable e) {
213 				ourLog.warn("Failed to configure context for Android operation", e);
214 			}
215 		} catch (ClassNotFoundException e) {
216 			ourLog.trace("Android mode not detected");
217 		}
218 		
219 		
220 	}

322 	 * Returns the scanned runtime model for the given type. This is an advanced feature which is generally only needed
323 	 * for extending the core library.
324 	 */
325 	public RuntimeResourceDefinition getResourceDefinition(Class<? extends IBaseResource> theResourceType) {
326 		validateInitialized();
327 		if (theResourceType == null) {
328 			throw new NullPointerException("theResourceType can not be null");
329 		}
330 		if (Modifier.isAbstract(theResourceType.getModifiers())) {
331 			throw new IllegalArgumentException("Can not scan abstract or interface class (resource definitions must be concrete classes): " + theResourceType.getName());
332 		}
333 
334 		RuntimeResourceDefinition retVal = (RuntimeResourceDefinition) myClassToElementDefinition.get(theResourceType);
335 		if (retVal == null) {
336 			retVal = scanResourceType(theResourceType);
337 		}
338 		return retVal;
339 	}

341 	public RuntimeResourceDefinition getResourceDefinition(FhirVersionEnum theVersion, String theResourceName) {
342 		Validate.notNull(theVersion, "theVersion can not be null");
343 		validateInitialized();
344 
345 		if (theVersion.equals(myVersion.getVersion())) {
346 			return getResourceDefinition(theResourceName);
347 		}
348 
349 		Map<String, Class<? extends IBaseResource>> nameToType = myVersionToNameToResourceType.get(theVersion);
350 		if (nameToType == null) {
351 			nameToType = new HashMap<String, Class<? extends IBaseResource>>();
352 			Map<Class<? extends IBase>, BaseRuntimeElementDefinition<?>> existing = Collections.emptyMap();
353 			ModelScanner.scanVersionPropertyFile(null, nameToType, theVersion, existing);
354 
355 			Map<FhirVersionEnum, Map<String, Class<? extends IBaseResource>>> newVersionToNameToResourceType = new HashMap<FhirVersionEnum, Map<String, Class<? extends IBaseResource>>>();
356 			newVersionToNameToResourceType.putAll(myVersionToNameToResourceType);
357 			newVersionToNameToResourceType.put(theVersion, nameToType);
358 			myVersionToNameToResourceType = newVersionToNameToResourceType;
359 		}
360 
361 		Class<? extends IBaseResource> resourceType = nameToType.get(theResourceName.toLowerCase());
362 		if (resourceType == null) {
363 			throw new DataFormatException(createUnknownResourceNameError(theResourceName, theVersion));
364 		}
365 
366 		return getResourceDefinition(resourceType);
367 	}

379 	/**
380 	 * Returns the scanned runtime model for the given type. This is an advanced feature which is generally only needed
381 	 * for extending the core library.
382 	 * <p>
383 	 * Note that this method is case insensitive!
384 	 * </p>
385 	 */
386 	public RuntimeResourceDefinition getResourceDefinition(String theResourceName) {
387 		validateInitialized();
388 		Validate.notBlank(theResourceName, "theResourceName must not be blank");
389 
390 		String resourceName = theResourceName.toLowerCase();
391 		RuntimeResourceDefinition retVal = myNameToResourceDefinition.get(resourceName);
392 
393 		if (retVal == null) {
394 			Class<? extends IBaseResource> clazz = myNameToResourceType.get(resourceName.toLowerCase());
395 			if (clazz == null) {
396 				throw new DataFormatException(createUnknownResourceNameError(theResourceName, myVersion.getVersion()));
397 			}
398 			if (IBaseResource.class.isAssignableFrom(clazz)) {
399 				retVal = scanResourceType(clazz);
400 			}
401 		}
402 
403 		return retVal;
404 	}

601 	private BaseRuntimeElementDefinition<?> scanDatatype(Class<? extends IElement> theResourceType) {
602 		ArrayList<Class<? extends IElement>> resourceTypes = new ArrayList<Class<? extends IElement>>();

615 	private synchronized Map<Class<? extends IBase>, BaseRuntimeElementDefinition<?>> scanResourceTypes(Collection<Class<? extends IElement>> theResourceTypes) {
616 		myInitializing = true;
617 		
618 		List<Class<? extends IBase>> typesToScan = new ArrayList<Class<? extends IBase>>();
619 		if (theResourceTypes != null) {
620 			typesToScan.addAll(theResourceTypes);
621 		}
622 		if (myCustomTypes != null) {
623 			typesToScan.addAll(myCustomTypes);
624 			myCustomTypes = null;
625 		}
626 
627 		ModelScanner scanner = new ModelScanner(this, myVersion.getVersion(), myClassToElementDefinition, typesToScan);
628 		if (myRuntimeChildUndeclaredExtensionDefinition == null) {
629 			myRuntimeChildUndeclaredExtensionDefinition = scanner.getRuntimeChildUndeclaredExtensionDefinition();
630 		}
631 
632 		Map<String, BaseRuntimeElementDefinition<?>> nameToElementDefinition = new HashMap<String, BaseRuntimeElementDefinition<?>>();
633 		nameToElementDefinition.putAll(myNameToElementDefinition);
634 		for (Entry<String, BaseRuntimeElementDefinition<?>> next : scanner.getNameToElementDefinitions().entrySet()) {
635 			if (!nameToElementDefinition.containsKey(next.getKey())) {
636 				nameToElementDefinition.put(next.getKey(), next.getValue());
637 			}
638 		}
639 
640 		Map<String, RuntimeResourceDefinition> nameToResourceDefinition = new HashMap<String, RuntimeResourceDefinition>();
641 		nameToResourceDefinition.putAll(myNameToResourceDefinition);
642 		for (Entry<String, RuntimeResourceDefinition> next : scanner.getNameToResourceDefinition().entrySet()) {
643 			if (!nameToResourceDefinition.containsKey(next.getKey())) {
644 				nameToResourceDefinition.put(next.getKey(), next.getValue());
645 			}
646 		}
647 
648 		Map<Class<? extends IBase>, BaseRuntimeElementDefinition<?>> classToElementDefinition = new HashMap<Class<? extends IBase>, BaseRuntimeElementDefinition<?>>();
649 		classToElementDefinition.putAll(myClassToElementDefinition);
650 		classToElementDefinition.putAll(scanner.getClassToElementDefinitions());
651 		for (BaseRuntimeElementDefinition<?> next : classToElementDefinition.values()) {
652 			if (next instanceof RuntimeResourceDefinition) {
653 				if ("Bundle".equals(next.getName())) {
654 					if (!IBaseBundle.class.isAssignableFrom(next.getImplementingClass())) {
655 						throw new ConfigurationException("Resource type declares resource name Bundle but does not implement IBaseBundle");
656 					}
657 				}
658 			}
659 		}
660 
661 		Map<String, RuntimeResourceDefinition> idToElementDefinition = new HashMap<String, RuntimeResourceDefinition>();
662 		idToElementDefinition.putAll(myIdToResourceDefinition);
663 		idToElementDefinition.putAll(scanner.getIdToResourceDefinition());
664 
665 		myNameToElementDefinition = nameToElementDefinition;
666 		myClassToElementDefinition = classToElementDefinition;
667 		myIdToResourceDefinition = idToElementDefinition;
668 		myNameToResourceDefinition = nameToResourceDefinition;
669 
670 		myNameToResourceType = scanner.getNameToResourceType();
671 
672 		myInitialized = true;
673 		return classToElementDefinition;
674 	}

677 	 * When encoding resources, this setting configures the parser to include

691 	 * </p>
692 	 *

701 	/**
702 	 * Sets the default type which will be used when parsing a resource that is found to be
703 	 * of the given profile.
704 	 * <p>
705 	 * For example, this method is invoked with the profile string of
706 	 * <code>"http://example.com/some_patient_profile"</code> and the type of <code>MyPatient.class</code>,
707 	 * if the parser is parsing a resource and finds that it declares that it conforms to that profile,
708 	 * the <code>MyPatient</code> type will be used unless otherwise specified.
709 	 * </p>
710 	 * 
711 	 * @param theProfile
712 	 *           The profile string, e.g. <code>"http://example.com/some_patient_profile"</code>. Must not be
713 	 *           <code>null</code> or empty.
714 	 * @param theClass
715 	 *           The resource type, or <code>null</code> to clear any existing type
716 	 */
717 	public void setDefaultTypeForProfile(String theProfile, Class<? extends IBaseResource> theClass) {
718 		Validate.notBlank(theProfile, "theProfile must not be null or empty");
719 		if (theClass == null) {
720 			myDefaultTypeForProfile.remove(theProfile);
721 		} else {
722 			myDefaultTypeForProfile.put(theProfile, theClass);
723 		}
724 	}

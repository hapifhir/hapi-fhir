65  	private TemplateEngine getTemplateEngine() {
66  		TemplateEngine engine = new TemplateEngine();
67  		ProfileResourceResolver resolver = new ProfileResourceResolver();
68  		engine.setTemplateResolver(resolver);
69  		if (myMessageResolver != null) {
70  			engine.setMessageResolver(myMessageResolver);
71  		}
72  		StandardDialect dialect = new StandardDialect() {
73  			@Override
74  			public Set<IProcessor> getProcessors(String theDialectPrefix) {
75  				Set<IProcessor> retVal = super.getProcessors(theDialectPrefix);
76  				retVal.add(new NarrativeTagProcessor(theDialectPrefix));
77  				retVal.add(new NarrativeAttributeProcessor(theDialectPrefix));
78  				return retVal;
79  			}
80  
81  		};
82  
83  		engine.setDialect(dialect);
84  		return engine;
85  	}
86  
87  	@Override
88  	protected String applyTemplate(INarrativeTemplate theTemplate, IBase theTargetContext) {
89  
90  		Context context = new Context();
91  		context.setVariable("resource", theTargetContext);
92  		context.setVariable("context", theTargetContext);
93  		context.setVariable("fhirVersion", getFhirContext().getVersion().getVersion().name());
94  
95  		String result = getTemplateEngine().process(theTemplate.getTemplateName(), context);
96  
97  		return result;
98  	}

106 	private String applyTemplateWithinTag(ITemplateContext theTemplateContext, String theName, String theElement) {
107 		IEngineConfiguration configuration = theTemplateContext.getConfiguration();
108 		IStandardExpressionParser expressionParser = StandardExpressions.getExpressionParser(configuration);
109 		final IStandardExpression expression = expressionParser.parseExpression(theTemplateContext, theElement);
110 		Object elementValueObj = expression.execute(theTemplateContext);
111 		final IBase/../../org/hl7/fhir/instance/model/api/IBase.html#IBase">IBase elementValue = (IBase) elementValueObj;
112 		if (elementValue == null) {
113 			return "";
114 		}
115 
116 		Optional<INarrativeTemplate> templateOpt;
117 		if (isNotBlank(theName)) {
118 			templateOpt = getManifest().getTemplateByName(getStyle(), theName);
119 			if (!templateOpt.isPresent()) {
120 				throw new InternalErrorException("Unknown template name: " + theName);
121 			}
122 		} else {
123 			templateOpt = getManifest().getTemplateByElement(getStyle(), elementValue);
124 			if (!templateOpt.isPresent()) {
125 				throw new InternalErrorException("No template for type: " + elementValue.getClass());
126 			}
127 		}
128 
129 		return applyTemplate(templateOpt.get(), elementValue);
130 	}

137 	private class ProfileResourceResolver extends DefaultTemplateResolver {
138 		@Override
139 		protected boolean computeResolvable(IEngineConfiguration theConfiguration, String theOwnerTemplate, String theTemplate, Map<String, Object> theTemplateResolutionAttributes) {
140 			return getManifest().getTemplateByName(getStyle(), theTemplate).isPresent();
141 		}
142 
143 		@Override
144 		protected TemplateMode computeTemplateMode(IEngineConfiguration theConfiguration, String theOwnerTemplate, String theTemplate, Map<String, Object> theTemplateResolutionAttributes) {
145 			return TemplateMode.XML;
146 		}
147 
148 		@Override
149 		protected ITemplateResource computeTemplateResource(IEngineConfiguration theConfiguration, String theOwnerTemplate, String theTemplate, Map<String, Object> theTemplateResolutionAttributes) {
150 			return getManifest()
151 				.getTemplateByName(getStyle(), theTemplate)
152 				.map(t -> new StringTemplateResource(t.getTemplateText()))
153 				.orElseThrow(() -> new IllegalArgumentException("Unknown template: " + theTemplate));
154 		}
155 
156 		@Override
157 		protected ICacheEntryValidity computeValidity(IEngineConfiguration theConfiguration, String theOwnerTemplate, String theTemplate, Map<String, Object> theTemplateResolutionAttributes) {
158 			return AlwaysValidCacheEntryValidity.INSTANCE;
159 		}
160 	}

162 	private class NarrativeTagProcessor extends AbstractElementTagProcessor {
163 
164 		public NarrativeTagProcessor(String dialectPrefix) {
165 			super(TemplateMode.XML, dialectPrefix, "narrative", true, null, true, 0);
166 		}
167 
168 		@Override
169 		protected void doProcess(ITemplateContext theTemplateContext, IProcessableElementTag theTag, IElementTagStructureHandler theStructureHandler) {
170 			String name = theTag.getAttributeValue("th:name");
171 			String element = theTag.getAttributeValue("th:element");
172 
173 			String appliedTemplate = applyTemplateWithinTag(theTemplateContext, name, element);
174 			theStructureHandler.replaceWith(appliedTemplate, false);
175 		}
176 	}
177 
178 	/**
179 	 * This is a thymeleaf extension that allows people to do things like
180 	 * <th:block th:narrative="${result}"/>
181 	 */
182 	private class NarrativeAttributeProcessor extends AbstractAttributeTagProcessor {
183 
184 		protected NarrativeAttributeProcessor(String theDialectPrefix) {
185 			super(TemplateMode.XML, theDialectPrefix, null, false, "narrative", true, 0, true);
186 		}
187 
188 		@Override
189 		protected void doProcess(ITemplateContext theContext, IProcessableElementTag theTag, AttributeName theAttributeName, String theAttributeValue, IElementTagStructureHandler theStructureHandler) {
190 			String text = applyTemplateWithinTag(theContext, null, theAttributeValue);
191 			theStructureHandler.setBody(text, false);
192 		}

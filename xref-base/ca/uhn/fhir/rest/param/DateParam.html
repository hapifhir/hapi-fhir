23  import java.util.Collections;
24  import java.util.Date;
25  import java.util.List;
26  
27  import ca.uhn.fhir.model.api.IQueryParameterOr;
28  import ca.uhn.fhir.model.api.IQueryParameterType;
29  import ca.uhn.fhir.model.dstu.valueset.QuantityCompararatorEnum;
30  import ca.uhn.fhir.model.primitive.DateTimeDt;
31  import ca.uhn.fhir.model.primitive.InstantDt;
32  import ca.uhn.fhir.parser.DataFormatException;
33  import ca.uhn.fhir.rest.method.QualifiedParamList;
34  import ca.uhn.fhir.rest.server.exceptions.InvalidRequestException;
35  
36  public class DateParam extends DateTimeDt implements IQueryParameterType, IQueryParameterOr<DateParam> {
37  
38  	private QuantityCompararatorEnum myComparator;
39  	private BaseParam myBase=new BaseParam.ComposableBaseParam();
40  
41  	/**
42  	 * Constructor
43  	 */
44  	public DateParam() {
45  	}
46  
47  	/**
48  	 * Constructor
49  	 */
50  	public DateParam(QuantityCompararatorEnum theComparator, Date theDate) {
51  		myComparator = theComparator;
52  		setValue(theDate);
53  	}
54  
55  	/**
56  	 * Constructor
57  	 */
58  	public DateParam(QuantityCompararatorEnum theComparator, String theDate) {
59  		myComparator = theComparator;
60  		setValueAsString(theDate);
61  	}
62  
63  	/**
64  	 * Constructor
65  	 */
66  	public DateParam(QuantityCompararatorEnum theComparator, DateTimeDt theDate) {
67  		myComparator = theComparator;
68  		setValueAsString(theDate != null ? theDate.getValueAsString() : null);
69  	}
70  
71  	/**
72  	 * Constructor which takes a complete [qualifier]{date} string.
73  	 * 
74  	 * @param theString
75  	 *            The string
76  	 */
77  	public DateParam(String theString) {
78  		setValueAsQueryToken(null, theString);
79  	}
80  
81  	/**
82  	 * Returns the comparator, or <code>null</code> if none has been set
83  	 */
84  	public QuantityCompararatorEnum getComparator() {
85  		return myComparator;
86  	}
87  
88  	@Override
89  	public String getQueryParameterQualifier() {
90  		if (myBase.getMissing()!=null) {
91  			return myBase.getQueryParameterQualifier();
92  		}
93  		return null;
94  	}
95  
96  	@Override
97  	public String getValueAsQueryToken() {
98  		if (myBase.getMissing()!=null) {
99  			return myBase.getValueAsQueryToken();
100 		}
101 		if (myComparator != null && getValue() != null) {
102 			return myComparator.getCode() + getValueAsString();
103 		} else if (myComparator == null && getValue() != null) {
104 			return getValueAsString();
105 		}
106 		return "";
107 	}
108 
109 	@Override
110 	public List<DateParam> getValuesAsQueryTokens() {
111 		return Collections.singletonList(this);
112 	}
113 
114 	/**
115 	 * Returns <code>true</code> if no date/time is specified. Note that this method does not check the comparator, so a
116 	 * QualifiedDateParam with only a comparator and no date/time is considered empty.
117 	 */
118 	@Override
119 	public boolean isEmpty() {
120 		// Just here to provide a javadoc
121 		return super.isEmpty();
122 	}
123 
124 	public void setComparator(QuantityCompararatorEnum theComparator) {
125 		myComparator = theComparator;
126 	}
127 
128 	@Override
129 	public void setValueAsQueryToken(String theQualifier, String theValue) {
130 		myBase.setValueAsQueryToken(theQualifier, theValue);
131 		if (myBase.getMissing()!=null) {
132 			setValue(null);
133 			myComparator=null;
134 			return;
135 		}
136 
137 		if (theValue.length() < 2) {
138 			throw new DataFormatException("Invalid qualified date parameter: " + theValue);
139 		}
140 
141 		char char0 = theValue.charAt(0);
142 		char char1 = theValue.charAt(1);
143 		if (Character.isDigit(char0)) {
144 			setValueAsString(theValue);
145 		} else {
146 			int dateStart = 2;
147 			if (Character.isDigit(char1)) {
148 				dateStart = 1;
149 			}
150 
151 			String comparatorString = theValue.substring(0, dateStart);
152 			QuantityCompararatorEnum comparator = QuantityCompararatorEnum.VALUESET_BINDER.fromCodeString(comparatorString);
153 			if (comparator == null) {
154 				throw new DataFormatException("Invalid date qualifier: " + comparatorString);
155 			}
156 
157 			String dateString = theValue.substring(dateStart);
158 			setValueAsString(dateString);
159 			setComparator(comparator);
160 		}
161 
162 	}
163 
164 	@Override
165 	public void  setValuesAsQueryTokens(QualifiedParamList theParameters) {
166 		myBase.setMissing(null);
167 		myComparator = null;
168 		setValueAsString(null);
169 		
170 		if (theParameters.size() == 1) {
171 			setValueAsString(theParameters.get(0));
172 		} else if (theParameters.size() > 1) {
173 			throw new InvalidRequestException("This server does not support multi-valued dates for this paramater: " + theParameters);
174 		}
175 		
176 	}
177 
178 	@Override
179 	public String toString() {
180 		StringBuilder b = new StringBuilder();
181 		b.append(getClass().getSimpleName());
182 		b.append("[");
183 		if (myComparator!=null) {
184 			b.append(myComparator.getCode());
185 		}
186 		b.append(getValueAsString());
187 		if (myBase.getMissing()!=null) {
188 			b.append(" missing=").append(myBase.getMissing());
189 		}
190 		b.append("]");
191 		return b.toString();
192 	}
193 
194 	public InstantDt getValueAsInstantDt() {
195 		return new InstantDt(getValue());
196 	}
197 
198 	public DateTimeDt getValueAsDateTimeDt() {
199 		return new DateTimeDt(getValueAsString());
200 	}
201 
202 	@Override
203 	public Boolean getMissing() {
204 		return myBase.getMissing();
205 	}
206 
207 	@Override
208 	public void setMissing(Boolean theMissing) {
209 		myBase.setMissing(theMissing);
210 	}
211 
212 }

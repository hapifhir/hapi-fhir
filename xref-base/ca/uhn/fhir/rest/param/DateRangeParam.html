39  	/**
40  	 * Basic constructor. Values must be supplied by calling {@link #setLowerBound(DateParam)} and

43  	public DateRangeParam() {
44  		// nothing

47  	/**
48  	 * Constructor which takes two Dates representing the lower and upper bounds of the range (inclusive on both ends)
49  	 * 
50  	 * @param theLowerBound
51  	 *            A qualified date param representing the lower date bound (optionally may include time), e.g.
52  	 *            "2011-02-22" or "2011-02-22T13:12:00Z". Will be treated inclusively. Either theLowerBound or theUpperBound may both be populated, or one may be null, but it is not valid for both to be null.
53  	 * @param theUpperBound
54  	 *            A qualified date param representing the upper date bound (optionally may include time), e.g.
55  	 *            "2011-02-22" or "2011-02-22T13:12:00Z". Will be treated inclusively. Either theLowerBound or theUpperBound may both be populated, or one may be null, but it is not valid for both to be null.
56  	 */
57  	public DateRangeParam(Date theLowerBound, Date theUpperBound) {
58  		setRangeFromDatesInclusive(theLowerBound, theUpperBound);
59  	}

61  	/**
62  	 * Sets the range from a single date param. If theDateParam has no qualifier, treats it as the lower and upper bound
63  	 * (e.g. 2011-01-02 would match any time on that day). If theDateParam has a qualifier, treats it as either the
64  	 * lower or upper bound, with no opposite bound.
65  	 */
66  	public DateRangeParam(DateParam theDateParam) {
67  		if (theDateParam == null) {
68  			throw new NullPointerException("theDateParam can not be null");
69  		}
70  		if (theDateParam.isEmpty()) {
71  			throw new IllegalArgumentException("theDateParam can not be empty");
72  		}
73  		if (theDateParam.getComparator() == null) {
74  			setRangeFromDatesInclusive(theDateParam.getValueAsString(), theDateParam.getValueAsString());
75  		} else {
76  			switch (theDateParam.getComparator()) {
77  			case GREATERTHAN:
78  			case GREATERTHAN_OR_EQUALS:
79  				myLowerBound = theDateParam;
80  				myUpperBound = null;
81  				break;
82  			case LESSTHAN:
83  			case LESSTHAN_OR_EQUALS:
84  				myLowerBound = null;
85  				myUpperBound = theDateParam;
86  				break;
87  			default:
88  				// Should not happen
89  				throw new IllegalStateException("Unknown comparator:" + theDateParam.getComparator() + ". This is a bug.");
90  			}
91  		}
92  		validateAndThrowDataFormatExceptionIfInvalid();
93  	}

123 	private void addParam(DateParam theParsed) throws InvalidRequestException {
124 		if (theParsed.getComparator() == null) {
125 			if (myLowerBound != null || myUpperBound != null) {
126 				throw new InvalidRequestException("Can not have multiple date range parameters for the same param without a qualifier");
127 			}
128 
129 			myLowerBound = new DateParam(QuantityCompararatorEnum.GREATERTHAN_OR_EQUALS, theParsed.getValueAsString());
130 			myUpperBound = new DateParam(QuantityCompararatorEnum.LESSTHAN_OR_EQUALS, theParsed.getValueAsString());
131 			
132 		} else {
133 
134 			switch (theParsed.getComparator()) {
135 			case GREATERTHAN:
136 			case GREATERTHAN_OR_EQUALS:
137 				if (myLowerBound != null) {
138 					throw new InvalidRequestException("Can not have multiple date range parameters for the same param that specify a lower bound");
139 				}
140 				myLowerBound = theParsed;
141 				break;
142 			case LESSTHAN:
143 			case LESSTHAN_OR_EQUALS:
144 				if (myUpperBound != null) {
145 					throw new InvalidRequestException("Can not have multiple date range parameters for the same param that specify an upper bound");
146 				}
147 				myUpperBound = theParsed;
148 				break;
149 			default:
150 				throw new InvalidRequestException("Unknown comparator: " + theParsed.getComparator());
151 			}
152 
153 		}
154 	}

160 	public Date getLowerBoundAsInstant() {
161 		if (myLowerBound == null) {
162 			return null;
163 		}
164 		Date retVal = myLowerBound.getValue();
165 		if (myLowerBound.getComparator() != null) {
166 			switch (myLowerBound.getComparator()) {
167 			case GREATERTHAN:
168 				retVal = myLowerBound.getPrecision().add(retVal, 1);
169 				break;
170 			case GREATERTHAN_OR_EQUALS:
171 				break;
172 			case LESSTHAN:
173 			case LESSTHAN_OR_EQUALS:
174 				throw new IllegalStateException("Unvalid lower bound comparator: " + myLowerBound.getComparator());
175 			}
176 		}
177 		return retVal;
178 	}

180 	public DateParam getUpperBound() {
181 		return myUpperBound;
182 	}
183 
184 	public Date getUpperBoundAsInstant() {
185 		if (myUpperBound == null) {
186 			return null;
187 		}
188 		Date retVal = myUpperBound.getValue();
189 		if (myUpperBound.getComparator() != null) {
190 			switch (myUpperBound.getComparator()) {
191 			case LESSTHAN:
192 				retVal = new Date(retVal.getTime() - 1L);
193 				break;
194 			case LESSTHAN_OR_EQUALS:
195 				retVal = myUpperBound.getPrecision().add(retVal, 1);
196 				retVal = new Date(retVal.getTime() - 1L);
197 				break;
198 			case GREATERTHAN_OR_EQUALS:
199 			case GREATERTHAN:
200 				throw new IllegalStateException("Unvalid upper bound comparator: " + myUpperBound.getComparator());
201 			}
202 		}
203 		return retVal;
204 	}

208 		ArrayList<DateParam> retVal = new ArrayList<DateParam>();
209 		if (myLowerBound != null && !myLowerBound.isEmpty()) {

211 		}
212 		if (myUpperBound != null && !myUpperBound.isEmpty()) {
213 			retVal.add((myUpperBound));
214 		}
215 		return retVal;
216 	}

236 	 * Sets the range from a pair of dates, inclusive on both ends
237 	 * 
238 	 * @param theLowerBound
239 	 *            A qualified date param representing the lower date bound (optionally may include time), e.g.
240 	 *            "2011-02-22" or "2011-02-22T13:12:00Z". Will be treated inclusively. Either theLowerBound or theUpperBound may both be populated, or one may be null, but it is not valid for both to be null.
241 	 * @param theUpperBound
242 	 *            A qualified date param representing the upper date bound (optionally may include time), e.g.
243 	 *            "2011-02-22" or "2011-02-22T13:12:00Z". Will be treated inclusively. Either theLowerBound or theUpperBound may both be populated, or one may be null, but it is not valid for both to be null.
244 	 */
245 	public void setRangeFromDatesInclusive(Date theLowerBound, Date theUpperBound) {
246 		myLowerBound = theLowerBound != null ? new DateParam(QuantityCompararatorEnum.GREATERTHAN_OR_EQUALS, theLowerBound) : null;
247 		myUpperBound = theUpperBound != null ? new DateParam(QuantityCompararatorEnum.LESSTHAN_OR_EQUALS, theUpperBound) : null;
248 		validateAndThrowDataFormatExceptionIfInvalid();
249 	}

252 	 * Sets the range from a pair of dates, inclusive on both ends
253 	 * 
254 	 * @param theLowerBound
255 	 *            A qualified date param representing the lower date bound (optionally may include time), e.g.
256 	 *            "2011-02-22" or "2011-02-22T13:12:00Z". Will be treated inclusively. Either theLowerBound or theUpperBound may both be populated, or one may be null, but it is not valid for both to be null.
257 	 * @param theUpperBound
258 	 *            A qualified date param representing the upper date bound (optionally may include time), e.g.
259 	 *            "2011-02-22" or "2011-02-22T13:12:00Z". Will be treated inclusively. Either theLowerBound or theUpperBound may both be populated, or one may be null, but it is not valid for both to be null.
260 	 */
261 	public void setRangeFromDatesInclusive(DateTimeDt theLowerBound, DateTimeDt theUpperBound) {
262 		if (theLowerBound instanceof DateParam) {
263 			myLowerBound = (DateParam) theLowerBound;
264 		} else {
265 			myLowerBound = theLowerBound != null ? new DateParam(QuantityCompararatorEnum.GREATERTHAN_OR_EQUALS, theLowerBound) : null;
266 		} 
267 		if (theUpperBound instanceof DateParam) {
268 			myUpperBound = (DateParam) theUpperBound;
269 		} else {
270 			myUpperBound = theUpperBound != null ? new DateParam(QuantityCompararatorEnum.LESSTHAN_OR_EQUALS, theUpperBound) : null;
271 		}
272 		validateAndThrowDataFormatExceptionIfInvalid();
273 	}

296 	@Override
297 	public void setValuesAsQueryTokens(List<QualifiedParamList> theParameters) throws InvalidRequestException {
298 		
299 		boolean haveHadUnqualifiedParameter = false;
300 		for (QualifiedParamList paramList : theParameters) {
301 			if (paramList.size() == 0) {
302 				continue;
303 			}
304 			if (paramList.size() > 1) {
305 				throw new InvalidRequestException("DateRange parameter does not suppport OR queries");
306 			}
307 			String param = paramList.get(0);
308 			DateParam parsed = new DateParam();
309 			parsed.setValueAsQueryToken(paramList.getQualifier(), param);
310 			addParam(parsed);
311 			
312 			if (parsed.getComparator() == null) {
313 				if (haveHadUnqualifiedParameter) {
314 					throw new InvalidRequestException("Multiple date parameters with the same name and no qualifier (>, <, etc.) is not supported");					
315 				}
316 				haveHadUnqualifiedParameter=true;
317 			}
318 			
319 		}
320 		
321 	}

323 	@Override
324 	public String toString() {
325 		StringBuilder b = new StringBuilder();
326 		b.append(getClass().getSimpleName());
327 		b.append("[");
328 		if (haveLowerBound()) {
329 			if (myLowerBound.getComparator() != null) {
330 				b.append(myLowerBound.getComparator().getCode());
331 			}
332 			b.append(myLowerBound.getValueAsString());
333 		}
334 		if (haveUpperBound()) {
335 			if(haveLowerBound()) {
336 				b.append(" ");
337 			}
338 			if (myUpperBound.getComparator() != null) {
339 				b.append(myUpperBound.getComparator().getCode());
340 			}
341 			b.append(myUpperBound.getValueAsString());
342 		} else {
343 			if (!haveLowerBound()) {
344 				b.append("empty");
345 			}
346 		}
347 		b.append("]");
348 		return b.toString();
349 	}

351 	private void validateAndThrowDataFormatExceptionIfInvalid() {
352 		boolean haveLowerBound = haveLowerBound();
353 		boolean haveUpperBound = haveUpperBound();
354 		if (haveLowerBound && haveUpperBound) {
355 			if (myLowerBound.getValue().getTime() > myUpperBound.getValue().getTime()) {
356 				StringBuilder b = new StringBuilder();
357 				b.append("Lower bound of ");
358 				b.append(myLowerBound.getValueAsString());
359 				b.append(" is after upper bound of ");
360 				b.append(myUpperBound.getValueAsString());
361 				throw new DataFormatException(b.toString());
362 			}
363 		}
364 
365 		if (haveLowerBound) {
366 			if (myLowerBound.getComparator() == null) {
367 				myLowerBound.setComparator(QuantityCompararatorEnum.GREATERTHAN_OR_EQUALS);
368 			}
369 			switch (myLowerBound.getComparator()) {
370 			case GREATERTHAN:
371 			case GREATERTHAN_OR_EQUALS:
372 			default:
373 				break;
374 			case LESSTHAN:
375 			case LESSTHAN_OR_EQUALS:
376 				throw new DataFormatException("Lower bound comparator must be > or >=, can not be " + myLowerBound.getComparator().getCode());
377 			}
378 		}
379 
380 		if (haveUpperBound) {
381 			if (myUpperBound.getComparator() == null) {
382 				myUpperBound.setComparator(QuantityCompararatorEnum.LESSTHAN_OR_EQUALS);
383 			}
384 			switch (myUpperBound.getComparator()) {
385 			case LESSTHAN:
386 			case LESSTHAN_OR_EQUALS:
387 			default:
388 				break;
389 			case GREATERTHAN:
390 			case GREATERTHAN_OR_EQUALS:
391 				throw new DataFormatException("Upper bound comparator must be < or <=, can not be " + myUpperBound.getComparator().getCode());
392 			}
393 		}

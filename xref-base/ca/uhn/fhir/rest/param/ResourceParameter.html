63  	public ResourceParameter(Class<? extends IResource> theParameterType, Object theProvider, Mode theMode) {
64  		Validate.notNull(theParameterType, "theParameterType can not be null");
65  		Validate.notNull(theMode, "theMode can not be null");
66  
67  		myResourceType = theParameterType;
68  		myMode = theMode;
69  
70  		Class<? extends IBaseResource> providerResourceType = null;
71  		if (theProvider instanceof IResourceProvider) {
72  			providerResourceType = ((IResourceProvider) theProvider).getResourceType();
73  		}
74  
75  		if (Modifier.isAbstract(myResourceType.getModifiers()) && providerResourceType != null) {
76  			myResourceType = providerResourceType;
77  		}
78  
79  	}

100 	@Override
101 	public Object translateQueryParametersIntoServerArgument(RequestDetails theRequest, BaseMethodBinding<?> theMethodBinding) throws InternalErrorException, InvalidRequestException {
102 		switch (myMode) {
103 		case BODY:
104 			try {
105 				return IOUtils.toString(createRequestReader(theRequest));
106 			} catch (IOException e) {
107 				// Shouldn't happen since we're reading from a byte array
108 				throw new InternalErrorException("Failed to load request", e);
109 			}
110 		case BODY_BYTE_ARRAY:
111 			return theRequest.loadRequestContents();
112 		case ENCODING:
113 			return RestfulServerUtils.determineRequestEncoding(theRequest);
114 		case RESOURCE:
115 		default:
116 			return parseResourceFromRequest(theRequest, theMethodBinding, myResourceType);
117 		}
118 		// }
119 	}

121 	public static Reader createRequestReader(RequestDetails theRequest, Charset charset) {
122 		Reader requestReader = new InputStreamReader(new ByteArrayInputStream(theRequest.loadRequestContents()), charset);
123 		return requestReader;
124 	}

126 	public static Reader createRequestReader(RequestDetails theRequest) {
127 		return createRequestReader(theRequest, determineRequestCharset(theRequest));
128 	}

130 	public static Charset determineRequestCharset(RequestDetails theRequest) {
131 		Charset charset =  theRequest.getCharset();
132 		if (charset == null) {
133 			charset = Charset.forName("UTF-8");
134 		}
135 		return charset;
136 	}

138 	@SuppressWarnings("unchecked")
139 	public static <T extends IBaseResource> T loadResourceFromRequest(RequestDetails theRequest, BaseMethodBinding<?> theMethodBinding, Class<T> theResourceType) {
140 		FhirContext ctx = theRequest.getServer().getFhirContext();
141 
142 		final Charset charset = determineRequestCharset(theRequest);
143 		Reader requestReader = createRequestReader(theRequest, charset);
144 
145 		RestOperationTypeEnum restOperationType = theMethodBinding != null ? theMethodBinding.getRestOperationType() : null;
146 
147 		EncodingEnum encoding = RestfulServerUtils.determineRequestEncodingNoDefault(theRequest);
148 		if (encoding == null) {
149 			String ctValue = theRequest.getHeader(Constants.HEADER_CONTENT_TYPE);
150 			if (ctValue != null) {
151 				if (ctValue.startsWith("application/x-www-form-urlencoded")) {
152 					String msg = theRequest.getServer().getFhirContext().getLocalizer().getMessage(ResourceParameter.class, "invalidContentTypeInRequest", ctValue, theMethodBinding.getRestOperationType());
153 					throw new InvalidRequestException(msg);
154 				}
155 			}
156 			if (isBlank(ctValue)) {
157 				/*
158 				 * If the client didn't send a content type, try to guess
159 				 */
160 				String body;
161 				try {
162 					body = IOUtils.toString(requestReader);
163 				} catch (IOException e) {
164 					// This shouldn't happen since we're reading from a byte array..
165 					throw new InternalErrorException(e);
166 				}
167 				encoding = MethodUtil.detectEncodingNoDefault(body);
168 				if (encoding == null) {
169 					String msg = ctx.getLocalizer().getMessage(ResourceParameter.class, "noContentTypeInRequest", restOperationType);
170 					throw new InvalidRequestException(msg);
171 				} else {
172 					requestReader = new InputStreamReader(new ByteArrayInputStream(theRequest.loadRequestContents()), charset);
173 				}
174 			} else {
175 				String msg = ctx.getLocalizer().getMessage(ResourceParameter.class, "invalidContentTypeInRequest", ctValue, restOperationType);
176 				throw new InvalidRequestException(msg);
177 			}
178 		}
179 
180 		IParser parser = encoding.newParser(ctx);
181 
182 		T retVal;
183 		if (theResourceType != null) {
184 			retVal = parser.parseResource(theResourceType, requestReader);
185 		} else {
186 			retVal = (T) parser.parseResource(requestReader);
187 		}
188 
189 		if (theRequest.getServer().getFhirContext().getVersion().getVersion().equals(FhirVersionEnum.DSTU1)) {
190 			TagList tagList = new TagList();
191 			for (String nextTagComplete : theRequest.getHeaders(Constants.HEADER_CATEGORY)) {
192 				MethodUtil.parseTagValue(tagList, nextTagComplete);
193 			}
194 			if (tagList.isEmpty() == false) {
195 				((IResource) retVal).getResourceMetadata().put(ResourceMetadataKeyEnum.TAG_LIST, tagList);
196 			}
197 		}
198 		return retVal;
199 	}

201 	public static IBaseResource parseResourceFromRequest(RequestDetails theRequest, BaseMethodBinding<?> theMethodBinding, Class<? extends IBaseResource> theResourceType) {
202 		IBaseResource retVal = null;
203 		
204 		if (IBaseBinary.class.isAssignableFrom(theResourceType)) {
205 			String ct = theRequest.getHeader(Constants.HEADER_CONTENT_TYPE);
206 			if (EncodingEnum.forContentTypeStrict(ct) == null) {
207 			FhirContext ctx = theRequest.getServer().getFhirContext();
208 			IBaseBinary binary = (IBaseBinary) ctx.getResourceDefinition("Binary").newInstance();
209 			binary.setContentType(ct);
210 			binary.setContent(theRequest.loadRequestContents());
211 			retVal = binary;
212 			}
213 		}
214 		
215 		if (retVal == null) {
216 			retVal = loadResourceFromRequest(theRequest, theMethodBinding, theResourceType);
217 		}
218 		return retVal;
219 	}

25  import org.apache.commons.lang3.builder.ToStringBuilder;

30  import ca.uhn.fhir.model.api.IQueryParameterType;

33  public class ReferenceParam extends IdDt implements IQueryParameterType {
34  
35  	private BaseParam myBase=new BaseParam.ComposableBaseParam();
36  	private String myChain;
37  
38  	public ReferenceParam() {
39  	}
40  
41  	public ReferenceParam(String theValue) {
42  		setValueAsQueryToken(null, theValue);
43  	}
44  
45  	public ReferenceParam(String theChain, String theValue) {
46  		setValueAsQueryToken(null, theValue);
47  		setChain(theChain);
48  	}
49  
50  	public ReferenceParam(String theResourceType, String theChain, String theValue) {
51  		if (isNotBlank(theResourceType)) {
52  			setValue(theResourceType + "/" + theValue);
53  		} else {
54  			setValue(theValue);
55  		}
56  		setChain(theChain);
57  	}
58  
59  	public String getChain() {
60  		return myChain;
61  	}
62  
63  	@Override
64  	public Boolean getMissing() {
65  		return myBase.getMissing();
66  	}
67  
68  	@Override
69  	public String getQueryParameterQualifier() {
70  		if (myBase.getMissing()!=null) {
71  			return myBase.getQueryParameterQualifier();
72  		}
73  		
74  		StringBuilder b = new StringBuilder();
75  		if (isNotBlank(getResourceType())) {
76  			b.append(':');
77  			b.append(getResourceType());
78  		}
79  		if (isNotBlank(myChain)) {
80  			b.append('.');
81  			b.append(myChain);
82  		}
83  		if (b.length() != 0) {
84  			return b.toString();
85  		}
86  		return null;
87  	}
88  
89  	public Class<? extends IBaseResource> getResourceType(FhirContext theCtx) {
90  		if (isBlank(getResourceType())) {
91  			return null;
92  		}
93  		return theCtx.getResourceDefinition(getResourceType()).getImplementingClass();
94  	}
95  
96  	@Override
97  	public String getValueAsQueryToken() {
98  		if (myBase.getMissing()!=null) {
99  			return myBase.getValueAsQueryToken();
100 		}
101 		if (isBlank(getResourceType())) {
102 			return getValue(); // e.g. urn:asdjd or 123 or cid:wieiuru or #1
103 		} else {
104 			return getIdPart();
105 		}
106 	}
107 
108 	public void setChain(String theChain) {
109 		myChain = theChain;
110 	}
111 
112 	@Override
113 	public void setMissing(Boolean theMissing) {
114 		myBase.setMissing(theMissing);
115 	}
116 
117 	@Override
118 	public void setValueAsQueryToken(String theQualifier, String theValue) {
119 		myBase.setValueAsQueryToken(theQualifier, theValue);
120 		if (myBase.getMissing()!=null) {
121 			myChain=null;
122 			setValue(null);
123 			return;
124 		}
125 		
126 		String q = theQualifier;
127 		String resourceType = null;
128 		if (isNotBlank(q)) {
129 			if (q.startsWith(":")) {
130 				int nextIdx = q.indexOf('.');
131 				if (nextIdx != -1) {
132 					resourceType = q.substring(1, nextIdx);
133 					myChain = q.substring(nextIdx + 1);
134 				} else {
135 					resourceType = q.substring(1);
136 				}
137 			} else if (q.startsWith(".")) {
138 				myChain = q.substring(1);
139 			}
140 		}
141 
142 		setValue(theValue);
143 
144 		if (isNotBlank(resourceType) && isBlank(getResourceType())) {
145 			setValue(resourceType + '/' + theValue);
146 		}
147 	}

155 	 * in the HAPI FHIR documentation for an example of how to use this method.
156 	 * </p>
157 	 */
158 	public DateParam toDateParam() {
159 		DateParam retVal = new DateParam();
160 		retVal.setValueAsQueryToken(null, getValueAsQueryToken());
161 		return retVal;
162 	}

194 	@Override
195 	public String toString() {
196 		ToStringBuilder b = new ToStringBuilder(this, ToStringStyle.SHORT_PREFIX_STYLE);
197 		if (isNotBlank(myChain)) {
198 			b.append("chain", myChain);
199 		}
200 		b.append("value", getValue());
201 		return b.build();
202 	}

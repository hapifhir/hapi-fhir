127 	static void addTagsToPostOrPut(FhirContext theContext, IBaseResource resource, BaseHttpClientInvocation retVal) {
128 		if (theContext.getVersion().getVersion().equals(FhirVersionEnum.DSTU1)) {
129 			TagList list = (TagList) ((IResource)resource).getResourceMetadata().get(ResourceMetadataKeyEnum.TAG_LIST);
130 			if (list != null) {
131 				for (Tag tag : list) {
132 					if (StringUtils.isNotBlank(tag.getTerm())) {
133 						retVal.addHeader(Constants.HEADER_CATEGORY, tag.toHeaderValue());
134 					}
135 				}
136 			}
137 		}

159 	public static HttpPostClientInvocation createCreateInvocation(IBaseResource theResource, String theResourceBody, String theId, FhirContext theContext) {
160 		RuntimeResourceDefinition def = theContext.getResourceDefinition(theResource);
161 		String resourceName = def.getName();
162 
163 		StringBuilder urlExtension = new StringBuilder();
164 		urlExtension.append(resourceName);
165 
166 		boolean dstu1 = theContext.getVersion().getVersion().equals(FhirVersionEnum.DSTU1);
167 		if (dstu1) {
168 			/*
169 			 * This was allowable at one point, but as of DSTU2 it isn't.
170 			 */
171 			if (StringUtils.isNotBlank(theId)) {
172 				urlExtension.append('/');
173 				urlExtension.append(theId);
174 			}
175 		}
176 		
177 		HttpPostClientInvocation retVal;
178 		if (StringUtils.isBlank(theResourceBody)) {
179 			retVal = new HttpPostClientInvocation(theContext, theResource, urlExtension.toString());
180 		} else {
181 			retVal = new HttpPostClientInvocation(theContext, theResourceBody, false, urlExtension.toString());
182 		}
183 		addTagsToPostOrPut(theContext, theResource, retVal);
184 
185 		if (!dstu1) {
186 			retVal.setOmitResourceId(true);
187 		}
188 		// addContentTypeHeaderBasedOnDetectedType(retVal, theResourceBody);
189 
190 		return retVal;
191 	}

205 	public static HttpPutClientInvocation createUpdateInvocation(FhirContext theContext, IBaseResource theResource, String theResourceBody, Map<String, List<String>> theMatchParams) {
206 		StringBuilder b = new StringBuilder();
207 
208 		String resourceType = theContext.getResourceDefinition(theResource).getName();
209 		b.append(resourceType);
210 
211 		boolean haveQuestionMark = false;
212 		for (Entry<String, List<String>> nextEntry : theMatchParams.entrySet()) {
213 			for (String nextValue : nextEntry.getValue()) {
214 				b.append(haveQuestionMark ? '&' : '?');
215 				haveQuestionMark = true;
216 				b.append(UrlUtil.escape(nextEntry.getKey()));
217 				b.append('=');
218 				b.append(UrlUtil.escape(nextValue));
219 			}
220 		}
221 
222 		HttpPutClientInvocation retVal;
223 		if (StringUtils.isBlank(theResourceBody)) {
224 			retVal = new HttpPutClientInvocation(theContext, theResource, b.toString());
225 		} else {
226 			retVal = new HttpPutClientInvocation(theContext, theResourceBody, false, b.toString());
227 		}
228 
229 		addTagsToPostOrPut(theContext, theResource, retVal);
230 
231 		return retVal;
232 	}

247 	public static HttpPutClientInvocation createUpdateInvocation(IBaseResource theResource, String theResourceBody, IIdType theId, FhirContext theContext) {
248 		String resourceName = theContext.getResourceDefinition(theResource).getName();
249 		StringBuilder urlBuilder = new StringBuilder();
250 		urlBuilder.append(resourceName);
251 		urlBuilder.append('/');
252 		urlBuilder.append(theId.getIdPart());
253 		String urlExtension = urlBuilder.toString();
254 
255 		HttpPutClientInvocation retVal;
256 		if (StringUtils.isBlank(theResourceBody)) {
257 			retVal = new HttpPutClientInvocation(theContext, theResource, urlExtension);
258 		} else {
259 			retVal = new HttpPutClientInvocation(theContext, theResourceBody, false, urlExtension);
260 		}
261 		
262 		retVal.setForceResourceId(theId);
263 
264 		if (theId.hasVersionIdPart()) {
265 			if (theContext.getVersion().getVersion().isNewerThan(FhirVersionEnum.DSTU1)) {
266 				retVal.addHeader(Constants.HEADER_IF_MATCH, '"' + theId.getVersionIdPart() + '"');
267 			} else {
268 				String versionId = theId.getVersionIdPart();
269 				if (StringUtils.isNotBlank(versionId)) {
270 					urlBuilder.append('/');
271 					urlBuilder.append(Constants.PARAM_HISTORY);
272 					urlBuilder.append('/');
273 					urlBuilder.append(versionId);
274 					retVal.addHeader(Constants.HEADER_CONTENT_LOCATION, urlBuilder.toString());
275 				}
276 			}
277 		}
278 
279 		addTagsToPostOrPut(theContext, theResource, retVal);
280 		// addContentTypeHeaderBasedOnDetectedType(retVal, theResourceBody);
281 
282 		return retVal;
283 	}

291 	public static EncodingEnum detectEncodingNoDefault(String theBody) {
292 		EncodingEnum retVal = null;
293 		for (int i = 0; i < theBody.length() && retVal == null; i++) {
294 			switch (theBody.charAt(i)) {
295 				case '<':
296 					retVal = EncodingEnum.XML;
297 					break;
298 				case '{':
299 					retVal = EncodingEnum.JSON;
300 					break;
301 			}
302 		}
303 		return retVal;
304 	}

306 	public static void extractDescription(SearchParameter theParameter, Annotation[] theAnnotations) {
307 		for (Annotation annotation : theAnnotations) {
308 			if (annotation instanceof Description) {
309 				Description desc = (Description) annotation;
310 				if (isNotBlank(desc.formalDefinition())) {
311 					theParameter.setDescription(desc.formalDefinition());
312 				} else {
313 					theParameter.setDescription(desc.shortDefinition());
314 				}
315 			}
316 		}
317 	}

319 	public static Integer findIdParameterIndex(Method theMethod, FhirContext theContext) {
320 		Integer index = MethodUtil.findParamAnnotationIndex(theMethod, IdParam.class);
321 		if (index != null) {
322 			Class<?> paramType = theMethod.getParameterTypes()[index];
323 			if (IIdType.class.equals(paramType)) {
324 				return index;
325 			}
326 			boolean isRi = theContext.getVersion().getVersion().isRi();
327 			boolean usesHapiId = IdDt.class.equals(paramType);
328 			if (isRi == usesHapiId) {
329 				throw new ConfigurationException("Method uses the wrong Id datatype (IdDt / IdType) for the given context FHIR version: " + theMethod.toString());
330 			}
331 		}
332 		return index;
333 	}

335 	public static Integer findParamAnnotationIndex(Method theMethod, Class<?> toFind) {
336 		int paramIndex = 0;
337 		for (Annotation[] annotations : theMethod.getParameterAnnotations()) {
338 			for (int annotationIndex = 0; annotationIndex < annotations.length; annotationIndex++) {
339 				Annotation nextAnnotation = annotations[annotationIndex];
340 				Class<? extends Annotation> class1 = nextAnnotation.getClass();
341 				if (toFind.isAssignableFrom(class1)) {
342 					return paramIndex;
343 				}
344 			}
345 			paramIndex++;
346 		}
347 		return null;
348 	}

358 	@SuppressWarnings("unchecked")
359 	public static List<IParameter> getResourceParameters(final FhirContext theContext, Method theMethod, Object theProvider, RestOperationTypeEnum theRestfulOperationTypeEnum) {
360 		List<IParameter> parameters = new ArrayList<IParameter>();
361 
362 		Class<?>[] parameterTypes = theMethod.getParameterTypes();
363 		int paramIndex = 0;
364 		for (Annotation[] annotations : theMethod.getParameterAnnotations()) {
365 
366 			IParameter param = null;
367 			Class<?> parameterType = parameterTypes[paramIndex];
368 			Class<? extends java.util.Collection<?>> outerCollectionType = null;
369 			Class<? extends java.util.Collection<?>> innerCollectionType = null;
370 			if (SearchParameterMap.class.equals(parameterType)) {
371 				if (theProvider instanceof IDynamicSearchResourceProvider) {
372 					Search searchAnnotation = theMethod.getAnnotation(Search.class);
373 					if (searchAnnotation != null && searchAnnotation.dynamic()) {
374 						param = new DynamicSearchParameter((IDynamicSearchResourceProvider) theProvider);
375 					}
376 				}
377 			} else if (TagList.class.isAssignableFrom(parameterType)) {
378 				// TagList is handled directly within the method bindings
379 				param = new NullParameter();
380 			} else {
381 				if (Collection.class.isAssignableFrom(parameterType)) {
382 					innerCollectionType = (Class<? extends java.util.Collection<?>>) parameterType;
383 					parameterType = ReflectionUtil.getGenericCollectionTypeOfMethodParameter(theMethod, paramIndex);
384 				}
385 				if (Collection.class.isAssignableFrom(parameterType)) {
386 					outerCollectionType = innerCollectionType;
387 					innerCollectionType = (Class<? extends java.util.Collection<?>>) parameterType;
388 					parameterType = ReflectionUtil.getGenericCollectionTypeOfMethodParameter(theMethod, paramIndex);
389 				}
390 				if (Collection.class.isAssignableFrom(parameterType)) {
391 					throw new ConfigurationException("Argument #" + paramIndex + " of Method '" + theMethod.getName() + "' in type '" + theMethod.getDeclaringClass().getCanonicalName() + "' is of an invalid generic type (can not be a collection of a collection of a collection)");
392 				}
393 			}
394 			if (parameterType.equals(HttpServletRequest.class) || parameterType.equals(ServletRequest.class)) {
395 				param = new ServletRequestParameter();
396 			} else if (parameterType.equals(RequestDetails.class)) {
397 				param = new RequestDetailsParameter();
398 			} else if (parameterType.equals(IRequestOperationCallback.class)) {
399 				param = new RequestOperationCallbackParameter();
400 			} else if (parameterType.equals(SummaryEnum.class)) {
401 				param = new SummaryEnumParameter();
402 			} else if (parameterType.equals(HttpServletResponse.class) || parameterType.equals(ServletResponse.class)) {
403 				param = new ServletResponseParameter();
404 			} else {
405 				for (int i = 0; i < annotations.length && param == null; i++) {
406 					Annotation nextAnnotation = annotations[i];
407 
408 					if (nextAnnotation instanceof RequiredParam) {
409 						SearchParameter parameter = new SearchParameter();
410 						parameter.setName(((RequiredParam) nextAnnotation).name());
411 						parameter.setRequired(true);
412 						parameter.setDeclaredTypes(((RequiredParam) nextAnnotation).targetTypes());
413 						parameter.setCompositeTypes(((RequiredParam) nextAnnotation).compositeTypes());
414 						parameter.setChainlists(((RequiredParam) nextAnnotation).chainWhitelist(), ((RequiredParam) nextAnnotation).chainBlacklist());
415 						parameter.setType(theContext, parameterType, innerCollectionType, outerCollectionType);
416 						MethodUtil.extractDescription(parameter, annotations);
417 						param = parameter;
418 					} else if (nextAnnotation instanceof OptionalParam) {
419 						SearchParameter parameter = new SearchParameter();
420 						parameter.setName(((OptionalParam) nextAnnotation).name());
421 						parameter.setRequired(false);
422 						parameter.setDeclaredTypes(((OptionalParam) nextAnnotation).targetTypes());
423 						parameter.setCompositeTypes(((OptionalParam) nextAnnotation).compositeTypes());
424 						parameter.setChainlists(((OptionalParam) nextAnnotation).chainWhitelist(), ((OptionalParam) nextAnnotation).chainBlacklist());
425 						parameter.setType(theContext, parameterType, innerCollectionType, outerCollectionType);
426 						MethodUtil.extractDescription(parameter, annotations);
427 						param = parameter;
428 					} else if (nextAnnotation instanceof IncludeParam) {
429 						Class<? extends Collection<Include>> instantiableCollectionType;
430 						Class<?> specType;
431 
432 						if (parameterType == String.class) {
433 							instantiableCollectionType = null;
434 							specType = String.class;
435 						} else if ((parameterType != Include.class) || innerCollectionType == null || outerCollectionType != null) {
436 							throw new ConfigurationException("Method '" + theMethod.getName() + "' is annotated with @" + IncludeParam.class.getSimpleName() + " but has a type other than Collection<" + Include.class.getSimpleName() + ">");
437 						} else {
438 							instantiableCollectionType = (Class<? extends Collection<Include>>) CollectionBinder.getInstantiableCollectionType(innerCollectionType, "Method '" + theMethod.getName() + "'");
439 							specType = parameterType;
440 						}
441 
442 						param = new IncludeParameter((IncludeParam) nextAnnotation, instantiableCollectionType, specType);
443 					} else if (nextAnnotation instanceof ResourceParam) {
444 						Mode mode;
445 						if (IBaseResource.class.isAssignableFrom(parameterType)) {
446 							mode = Mode.RESOURCE;
447 						} else if (String.class.equals(parameterType)) {
448 							mode = ResourceParameter.Mode.BODY;
449 						} else if (byte[].class.equals(parameterType)) {
450 							mode = ResourceParameter.Mode.BODY_BYTE_ARRAY;
451 						} else if (EncodingEnum.class.equals(parameterType)) {
452 							mode = Mode.ENCODING;
453 						} else {
454 							StringBuilder b = new StringBuilder();
455 							b.append("Method '");
456 							b.append(theMethod.getName());
457 							b.append("' is annotated with @");
458 							b.append(ResourceParam.class.getSimpleName());
459 							b.append(" but has a type that is not an implemtation of ");
460 							b.append(IBaseResource.class.getCanonicalName());
461 							b.append(" or String or byte[]");
462 							throw new ConfigurationException(b.toString());
463 						}
464 						param = new ResourceParameter((Class<? extends IResource>) parameterType, theProvider, mode);
465 					} else if (nextAnnotation instanceof IdParam || nextAnnotation instanceof VersionIdParam) {
466 						param = new NullParameter();
467 					} else if (nextAnnotation instanceof ServerBase) {
468 						param = new ServerBaseParamBinder();
469 					} else if (nextAnnotation instanceof Elements) {
470 						param = new ElementsParameter();
471 					} else if (nextAnnotation instanceof Since) {
472 						param = new SinceParameter();
473 						((SinceParameter)param).setType(theContext, parameterType, innerCollectionType, outerCollectionType);
474 					} else if (nextAnnotation instanceof At) {
475 						param = new AtParameter();
476 						((AtParameter)param).setType(theContext, parameterType, innerCollectionType, outerCollectionType);
477 					} else if (nextAnnotation instanceof Count) {
478 						param = new CountParameter();
479 					} else if (nextAnnotation instanceof Sort) {
480 						param = new SortParameter(theContext);
481 					} else if (nextAnnotation instanceof TransactionParam) {
482 						param = new TransactionParameter(theContext);
483 					} else if (nextAnnotation instanceof ConditionalUrlParam) {
484 						param = new ConditionalParamBinder(theRestfulOperationTypeEnum, ((ConditionalUrlParam)nextAnnotation).supportsMultiple());
485 					} else if (nextAnnotation instanceof OperationParam) {
486 						Operation op = theMethod.getAnnotation(Operation.class);
487 						param = new OperationParameter(theContext, op.name(), ((OperationParam) nextAnnotation));
488 					} else if (nextAnnotation instanceof Validate.Mode) {
489 						if (parameterType.equals(ValidationModeEnum.class) == false) {
490 							throw new ConfigurationException("Parameter annotated with @" + Validate.class.getSimpleName() + "." + Validate.Mode.class.getSimpleName() + " must be of type " + ValidationModeEnum.class.getName());
491 						}
492 						param = new OperationParameter(theContext, Constants.EXTOP_VALIDATE, Constants.EXTOP_VALIDATE_MODE, 0, 1).setConverter(new IOperationParamConverter() {
493 							@Override
494 							public Object incomingServer(Object theObject) {
495 								if (isNotBlank(theObject.toString())) {
496 									ValidationModeEnum retVal = ValidationModeEnum.forCode(theObject.toString());
497 									if (retVal == null) {
498 										OperationParameter.throwInvalidMode(theObject.toString());
499 									}
500 									return retVal;
501 								} else {
502 									return null;
503 								}
504 							}
505 							
506 							@Override
507 							public Object outgoingClient(Object theObject) {
508 								return ParametersUtil.createString(theContext, ((ValidationModeEnum)theObject).getCode());
509 							}
510 						});
511 					} else if (nextAnnotation instanceof Validate.Profile) {
512 						if (parameterType.equals(String.class) == false) {
513 							throw new ConfigurationException("Parameter annotated with @" + Validate.class.getSimpleName() + "." + Validate.Profile.class.getSimpleName() + " must be of type " + String.class.getName());
514 						}
515 						param = new OperationParameter(theContext, Constants.EXTOP_VALIDATE, Constants.EXTOP_VALIDATE_PROFILE, 0, 1).setConverter(new IOperationParamConverter() {
516 							@Override
517 							public Object incomingServer(Object theObject) {
518 								return theObject.toString();
519 							}
520 							
521 							@Override
522 							public Object outgoingClient(Object theObject) {
523 								return ParametersUtil.createString(theContext, theObject.toString());
524 							}
525 						});
526 					} else {
527 						continue;
528 					}
529 
530 				}
531 
532 			}
533 
534 			if (param == null) {
535 				throw new ConfigurationException("Parameter #" + ((paramIndex + 1)) + "/" + (parameterTypes.length) + " of method '" + theMethod.getName() + "' on type '" + theMethod.getDeclaringClass().getCanonicalName()
536 						+ "' has no recognized FHIR interface parameter annotations. Don't know how to handle this parameter");
537 			}

543 		}

547 	public static void parseClientRequestResourceHeaders(IIdType theRequestedId, Map<String, List<String>> theHeaders, IBaseResource resource) {
548 		List<String> lmHeaders = theHeaders.get(Constants.HEADER_LAST_MODIFIED_LOWERCASE);
549 		if (lmHeaders != null && lmHeaders.size() > 0 && StringUtils.isNotBlank(lmHeaders.get(0))) {
550 			String headerValue = lmHeaders.get(0);
551 			Date headerDateValue;
552 			try {
553 				headerDateValue = DateUtils.parseDate(headerValue);
554 				if (resource instanceof IResource) {
555 					IResource iResource = (IResource) resource;
556 					InstantDt existing = ResourceMetadataKeyEnum.UPDATED.get(iResource);
557 					if (existing == null || existing.isEmpty()) {
558 						InstantDt lmValue = new InstantDt(headerDateValue);
559 						iResource.getResourceMetadata().put(ResourceMetadataKeyEnum.UPDATED, lmValue);
560 					}
561 				} else if (resource instanceof IAnyResource) {
562 					IAnyResource anyResource = (IAnyResource) resource;
563 					if (anyResource.getMeta().getLastUpdated() == null) {
564 						anyResource.getMeta().setLastUpdated(headerDateValue);
565 					}
566 				}
567 			} catch (Exception e) {
568 				ourLog.warn("Unable to parse date string '{}'. Error is: {}", headerValue, e.toString());
569 			}
570 		}
571 
572 		List<String> clHeaders = theHeaders.get(Constants.HEADER_CONTENT_LOCATION_LC);
573 		if (clHeaders != null && clHeaders.size() > 0 && StringUtils.isNotBlank(clHeaders.get(0))) {
574 			String headerValue = clHeaders.get(0);
575 			if (isNotBlank(headerValue)) {
576 				new IdDt(headerValue).applyTo(resource);
577 			}
578 		}
579 
580 		IdDt existing = IdDt.of(resource);
581 
582 		List<String> eTagHeaders = theHeaders.get(Constants.HEADER_ETAG_LC);
583 		String eTagVersion = null;
584 		if (eTagHeaders != null && eTagHeaders.size() > 0) {
585 			eTagVersion = parseETagValue(eTagHeaders.get(0));
586 		}
587 		if (isNotBlank(eTagVersion)) {
588 			if (existing == null || existing.isEmpty()) {
589 				if (theRequestedId != null) {
590 					theRequestedId.withVersion(eTagVersion).applyTo(resource);
591 				}
592 			} else if (existing.hasVersionIdPart() == false) {
593 				existing.withVersion(eTagVersion).applyTo(resource);
594 			}
595 		} else if (existing == null || existing.isEmpty()) {
596 			if (theRequestedId != null) {
597 				theRequestedId.applyTo(resource);
598 			}
599 		}
600 
601 		List<String> categoryHeaders = theHeaders.get(Constants.HEADER_CATEGORY_LC);
602 		if (categoryHeaders != null && categoryHeaders.size() > 0 && StringUtils.isNotBlank(categoryHeaders.get(0))) {
603 			TagList tagList = new TagList();
604 			for (String header : categoryHeaders) {
605 				parseTagValue(tagList, header);
606 			}
607 			if (resource instanceof IResource) {
608 				ResourceMetadataKeyEnum.TAG_LIST.put((IResource) resource, tagList);
609 			} else if (resource instanceof IAnyResource) {
610 				IBaseMetaType meta = ((IAnyResource) resource).getMeta();
611 				for (Tag next : tagList) {
612 					meta.addTag().setSystem(next.getScheme()).setCode(next.getTerm()).setDisplay(next.getLabel());
613 				}
614 			}
615 		}
616 	}

618 	public static String parseETagValue(String value) {
619 		String eTagVersion;
620 		value = value.trim();
621 		if (value.length() > 1) {
622 			if (value.charAt(value.length() - 1) == '"') {
623 				if (value.charAt(0) == '"') {
624 					eTagVersion = value.substring(1, value.length() - 1);
625 				} else if (value.length() > 3 && value.charAt(0) == 'W' && value.charAt(1) == '/' && value.charAt(2) == '"') {
626 					eTagVersion = value.substring(3, value.length() - 1);
627 				} else {
628 					eTagVersion = value;
629 				}
630 			} else {
631 				eTagVersion = value;
632 			}
633 		} else {
634 			eTagVersion = value;
635 		}
636 		return eTagVersion;
637 	}

678 	private static void parseTagValue(TagList theTagList, String theCompleteHeaderValue, StringBuilder theBuffer) {
679 		int firstSemicolon = theBuffer.indexOf(";");
680 		int deleteTo;
681 		if (firstSemicolon == -1) {
682 			firstSemicolon = theBuffer.indexOf(",");
683 			if (firstSemicolon == -1) {
684 				firstSemicolon = theBuffer.length();
685 				deleteTo = theBuffer.length();
686 			} else {
687 				deleteTo = firstSemicolon;
688 			}
689 		} else {
690 			deleteTo = firstSemicolon + 1;
691 		}
692 
693 		String term = theBuffer.substring(0, firstSemicolon);
694 		String scheme = null;
695 		String label = null;
696 		if (isBlank(term)) {
697 			return;
698 		}
699 
700 		theBuffer.delete(0, deleteTo);
701 		while (theBuffer.length() > 0 && theBuffer.charAt(0) == ' ') {
702 			theBuffer.deleteCharAt(0);
703 		}
704 
705 		while (theBuffer.length() > 0) {
706 			boolean foundSomething = false;
707 			if (theBuffer.length() > SCHEME.length() && theBuffer.substring(0, SCHEME.length()).equals(SCHEME)) {
708 				int closeIdx = theBuffer.indexOf("\"", SCHEME.length());
709 				scheme = theBuffer.substring(SCHEME.length(), closeIdx);
710 				theBuffer.delete(0, closeIdx + 1);
711 				foundSomething = true;
712 			}
713 			if (theBuffer.length() > LABEL.length() && theBuffer.substring(0, LABEL.length()).equals(LABEL)) {
714 				int closeIdx = theBuffer.indexOf("\"", LABEL.length());
715 				label = theBuffer.substring(LABEL.length(), closeIdx);
716 				theBuffer.delete(0, closeIdx + 1);
717 				foundSomething = true;
718 			}
719 			// TODO: support enc2231-string as described in
720 			// http://tools.ietf.org/html/draft-johnston-http-category-header-02
721 			// TODO: support multiple tags in one header as described in
722 			// http://hl7.org/implement/standards/fhir/http.html#tags
723 
724 			while (theBuffer.length() > 0 && (theBuffer.charAt(0) == ' ' || theBuffer.charAt(0) == ';')) {
725 				theBuffer.deleteCharAt(0);
726 			}
727 
728 			if (!foundSomething) {
729 				break;
730 			}
731 		}
732 
733 		if (theBuffer.length() > 0 && theBuffer.charAt(0) == ',') {
734 			theBuffer.deleteCharAt(0);
735 			while (theBuffer.length() > 0 && theBuffer.charAt(0) == ' ') {
736 				theBuffer.deleteCharAt(0);
737 			}
738 			theTagList.add(new Tag(scheme, term, label));
739 			parseTagValue(theTagList, theCompleteHeaderValue, theBuffer);
740 		} else {
741 			theTagList.add(new Tag(scheme, term, label));
742 		}
743 
744 		if (theBuffer.length() > 0) {
745 			ourLog.warn("Ignoring extra text at the end of " + Constants.HEADER_CATEGORY + " tag '" + theBuffer.toString() + "' - Complete tag value was: " + theCompleteHeaderValue);
746 		}
747 
748 	}

750 	public static MethodOutcome process2xxResponse(FhirContext theContext, int theResponseStatusCode, String theResponseMimeType, Reader theResponseReader, Map<String, List<String>> theHeaders) {
751 		List<String> locationHeaders = new ArrayList<String>();
752 		List<String> lh = theHeaders.get(Constants.HEADER_LOCATION_LC);
753 		if (lh != null) {
754 			locationHeaders.addAll(lh);
755 		}
756 		List<String> clh = theHeaders.get(Constants.HEADER_CONTENT_LOCATION_LC);
757 		if (clh != null) {
758 			locationHeaders.addAll(clh);
759 		}
760 
761 		MethodOutcome retVal = new MethodOutcome();
762 		if (locationHeaders != null && locationHeaders.size() > 0) {
763 			String locationHeader = locationHeaders.get(0);
764 			BaseOutcomeReturningMethodBinding.parseContentLocation(theContext, retVal, locationHeader);
765 		}
766 		if (theResponseStatusCode != Constants.STATUS_HTTP_204_NO_CONTENT) {
767 			EncodingEnum ct = EncodingEnum.forContentType(theResponseMimeType);
768 			if (ct != null) {
769 				PushbackReader reader = new PushbackReader(theResponseReader);
770 
771 				try {
772 					int firstByte = reader.read();
773 					if (firstByte == -1) {
774 						BaseOutcomeReturningMethodBinding.ourLog.debug("No content in response, not going to read");
775 						reader = null;
776 					} else {
777 						reader.unread(firstByte);
778 					}
779 				} catch (IOException e) {
780 					BaseOutcomeReturningMethodBinding.ourLog.debug("No content in response, not going to read", e);
781 					reader = null;
782 				}
783 
784 				if (reader != null) {
785 					IParser parser = ct.newParser(theContext);
786 					IBaseResource outcome = parser.parseResource(reader);
787 					if (outcome instanceof IBaseOperationOutcome) {
788 						retVal.setOperationOutcome((IBaseOperationOutcome) outcome);
789 					} else {
790 						retVal.setResource(outcome);
791 					}
792 				}
793 
794 			} else {
795 				BaseOutcomeReturningMethodBinding.ourLog.debug("Ignoring response content of type: {}", theResponseMimeType);
796 			}
797 		}
798 		return retVal;
799 	}

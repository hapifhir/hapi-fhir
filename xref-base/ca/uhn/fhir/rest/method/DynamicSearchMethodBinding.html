44  public class DynamicSearchMethodBinding extends BaseResourceReturningMethodBinding {
45  
46  	private IDynamicSearchResourceProvider myProvider;
47  	private List<RuntimeSearchParam> mySearchParameters;
48  	private HashSet<String> myParamNames;
49  	private Integer myIdParamIndex;
50  
51  	public DynamicSearchMethodBinding(Class<? extends IBaseResource> theReturnResourceType, Method theMethod, FhirContext theConetxt, IDynamicSearchResourceProvider theProvider) {
52  		super(theReturnResourceType, theMethod, theConetxt, theProvider);
53  
54  		myProvider = theProvider;
55  		mySearchParameters = myProvider.getSearchParameters();
56  
57  		myParamNames = new HashSet<String>();
58  		for (RuntimeSearchParam next : mySearchParameters) {
59  			myParamNames.add(next.getName());
60  		}
61  
62  		myIdParamIndex = MethodUtil.findIdParameterIndex(theMethod);
63  
64  	}
65  
66  	@Override
67  	protected BundleTypeEnum getResponseBundleType() {
68  		return BundleTypeEnum.SEARCHSET;
69  	}
70  
71  
72  	@Override
73  	public List<IParameter> getParameters() {
74  		List<IParameter> retVal = new ArrayList<IParameter>(super.getParameters());
75  		
76  		for (RuntimeSearchParam next : mySearchParameters) {
77  			
78  		}
79  		
80  		return retVal;
81  	}
82  
83  	@Override
84  	public ReturnTypeEnum getReturnType() {
85  		return ReturnTypeEnum.BUNDLE;
86  	}

88  	@Override
89  	public IBundleProvider invokeServer(RequestDetails theRequest, Object[] theMethodParams) throws InvalidRequestException, InternalErrorException {
90  		if (myIdParamIndex != null) {
91  			theMethodParams[myIdParamIndex] = theRequest.getId();
92  		}
93  
94  		Object response = invokeServerMethod(theMethodParams);
95  		return toResourceList(response);
96  	}

108 	private static final org.slf4j.Logger ourLog = org.slf4j.LoggerFactory.getLogger(DynamicSearchMethodBinding.class);
109 
110 	@Override
111 	public boolean incomingServerRequestMatchesMethod(Request theRequest) {
112 		if (!theRequest.getResourceName().equals(getResourceName())) {
113 			ourLog.trace("Method {} doesn't match because resource name {} != {}", getMethod().getName(), theRequest.getResourceName(), getResourceName());
114 			return false;
115 		}
116 		if (theRequest.getId() != null && myIdParamIndex == null) {
117 			ourLog.trace("Method {} doesn't match because ID is not null: {}", theRequest.getId());
118 			return false;
119 		}
120 		if (theRequest.getRequestType() == RequestTypeEnum.GET && theRequest.getOperation() != null && !Constants.PARAM_SEARCH.equals(theRequest.getOperation())) {
121 			ourLog.trace("Method {} doesn't match because request type is GET but operation is not null: {}", theRequest.getId(), theRequest.getOperation());
122 			return false;
123 		}
124 		if (theRequest.getRequestType() == RequestTypeEnum.POST && !Constants.PARAM_SEARCH.equals(theRequest.getOperation())) {
125 			ourLog.trace("Method {} doesn't match because request type is POST but operation is not _search: {}", theRequest.getId(), theRequest.getOperation());
126 			return false;
127 		}
128 		if (theRequest.getRequestType() != RequestTypeEnum.GET && theRequest.getRequestType() != RequestTypeEnum.POST) {
129 			ourLog.trace("Method {} doesn't match because request type is {}", getMethod());
130 			return false;
131 		}
132 		if (theRequest.getCompartmentName() != null) {
133 			ourLog.trace("Method {} doesn't match because it is for compartment {}", new Object[] { getMethod(), theRequest.getCompartmentName() });
134 			return false;
135 		}
136 
137 		for (String next : theRequest.getParameters().keySet()) {
138 			if (next.charAt(0) == '_') {
139 				continue;
140 			}
141 			String nextQualified = next;
142 			int colonIndex = next.indexOf(':');
143 			int dotIndex = next.indexOf('.');
144 			if (colonIndex != -1 || dotIndex != -1) {
145 				int index;
146 				if (colonIndex != -1 && dotIndex != -1) {
147 					index = Math.min(colonIndex, dotIndex);
148 				} else {
149 					index = (colonIndex != -1) ? colonIndex : dotIndex;
150 				}
151 				next = next.substring(0, index);
152 			}
153 			if (!myParamNames.contains(next)) {
154 				ourLog.trace("Method {} doesn't match because has parameter {}", new Object[] { getMethod(), nextQualified });
155 				return false;
156 			}
157 		}
158 
159 		return true;
160 	}

164 		// there should be no way to call this....
165 		throw new UnsupportedOperationException("Dynamic search methods are only used for server implementations");
166 	}

169 		return mySearchParameters;

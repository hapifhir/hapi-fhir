23  import java.util.ArrayList;
24  import java.util.Collection;
25  import java.util.Collections;
26  import java.util.HashSet;
27  import java.util.List;
28  import java.util.Set;
29  import java.util.TreeSet;
30  
31  import ca.uhn.fhir.context.ConfigurationException;
32  import ca.uhn.fhir.context.FhirContext;
33  import ca.uhn.fhir.model.api.Include;
34  import ca.uhn.fhir.model.api.PathSpecification;
35  import ca.uhn.fhir.rest.annotation.IncludeParam;
36  import ca.uhn.fhir.rest.param.BaseQueryParameter;
37  import ca.uhn.fhir.rest.server.exceptions.InternalErrorException;
38  import ca.uhn.fhir.rest.server.exceptions.InvalidRequestException;
39  
40  class IncludeParameter extends BaseQueryParameter {
41  
42  	private Set<String> myAllow;
43  	private Class<? extends Collection<Include>> myInstantiableCollectionType;
44  	private Class<?> mySpecType;
45  	private boolean myReverse;
46  
47  	public IncludeParameter(IncludeParam theAnnotation, Class<? extends Collection<Include>> theInstantiableCollectionType, Class<?> theSpecType) {
48  		myInstantiableCollectionType = theInstantiableCollectionType;
49  		myReverse = theAnnotation.reverse();
50  		if (theAnnotation.allow().length > 0) {
51  			myAllow = new HashSet<String>();
52  			for (String next : theAnnotation.allow()) {
53  				if (next != null) {
54  					myAllow.add(next);
55  				}
56  			}
57  		} else {
58  			myAllow = Collections.emptySet();
59  		}
60  
61  		mySpecType = theSpecType;
62  		if (mySpecType != Include.class && mySpecType != PathSpecification.class && mySpecType != String.class) {
63  			throw new ConfigurationException("Invalid @" + IncludeParam.class.getSimpleName() + " parameter type: " + mySpecType);
64  		}
65  
66  	}
67  
68  	@SuppressWarnings("unchecked")
69  	@Override
70  	public List<QualifiedParamList> encode(FhirContext theContext, Object theObject) throws InternalErrorException {
71  		ArrayList<QualifiedParamList> retVal = new ArrayList<QualifiedParamList>();
72  
73  		if (myInstantiableCollectionType == null) {
74  			if (mySpecType == Include.class) {
75  				retVal.add(QualifiedParamList.singleton(((Include) theObject).getValue()));
76  			} else if (mySpecType == PathSpecification.class) {
77  				retVal.add(QualifiedParamList.singleton(((PathSpecification) theObject).getValue()));
78  			} else {
79  				retVal.add(QualifiedParamList.singleton(((String) theObject)));
80  			}
81  		} else {
82  			Collection<Include> val = (Collection<Include>) theObject;
83  			for (Include pathSpec : val) {
84  				retVal.add(QualifiedParamList.singleton(pathSpec.getValue()));
85  			}
86  		}
87  
88  		return retVal;
89  	}
90  
91  	public Set<String> getAllow() {
92  		return myAllow;
93  	}
94  
95  	@Override
96  	public String getName() {
97  		return myReverse ? "_revinclude" : "_include";
98  	}
99  
100 	@Override
101 	public RestSearchParameterTypeEnum getParamType() {
102 		return null;
103 	}
104 
105 	@Override
106 	public boolean handlesMissing() {
107 		return true;
108 	}
109 
110 	@Override
111 	public boolean isRequired() {
112 		return false;
113 	}
114 
115 	@Override
116 	public Object parse(FhirContext theContext, List<QualifiedParamList> theString) throws InternalErrorException, InvalidRequestException {
117 		Collection<Include> retValCollection = null;
118 		if (myInstantiableCollectionType != null) {
119 			try {
120 				retValCollection = myInstantiableCollectionType.newInstance();
121 			} catch (Exception e) {
122 				throw new InternalErrorException("Failed to instantiate " + myInstantiableCollectionType.getName(), e);
123 			}
124 		}
125 
126 		for (List<String> nextParamList : theString) {
127 			if (nextParamList.isEmpty()) {
128 				continue;
129 			}
130 			if (nextParamList.size() > 1) {
131 				throw new InvalidRequestException(theContext.getLocalizer().getMessage(IncludeParameter.class, "orIncludeInRequest"));
132 			}
133 
134 			String value = nextParamList.get(0);
135 			if (myAllow != null && !myAllow.isEmpty()) {
136 				if (!myAllow.contains(value)) {
137 					if (!myAllow.contains("*")) {
138 						String msg = theContext.getLocalizer().getMessage(IncludeParameter.class, "invalidIncludeNameInRequest", value, new TreeSet<String>(myAllow).toString(), getName());
139 						throw new InvalidRequestException(msg);
140 					}
141 				}
142 			}
143 			if (retValCollection == null) {
144 				if (mySpecType == String.class) {
145 					return value;
146 				} else if (mySpecType == PathSpecification.class) {
147 					return new PathSpecification(value);
148 				} else {
149 					return new Include(value);
150 				}
151 			} else {
152 				if (mySpecType == PathSpecification.class) {
153 					retValCollection.add(new PathSpecification(value));
154 				} else {
155 					retValCollection.add(new Include(value));
156 				}
157 			}
158 		}
159 
160 		return retValCollection;
161 	}
162 
163 }

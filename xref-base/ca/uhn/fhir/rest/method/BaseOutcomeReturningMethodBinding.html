57  abstract class BaseOutcomeReturningMethodBinding extends BaseMethodBinding<MethodOutcome> {
58  	static final org.slf4j.Logger ourLog = org.slf4j.LoggerFactory.getLogger(BaseOutcomeReturningMethodBinding.class);

60  	private boolean myReturnVoid;
61  
62  	public BaseOutcomeReturningMethodBinding(Method theMethod, FhirContext theContext, Class<?> theMethodAnnotation, Object theProvider) {
63  		super(theMethod, theContext, theProvider);
64  
65  		if (!theMethod.getReturnType().equals(MethodOutcome.class)) {
66  			if (!allowVoidReturnType()) {
67  				throw new ConfigurationException("Method " + theMethod.getName() + " in type " + theMethod.getDeclaringClass().getCanonicalName() + " is a @" + theMethodAnnotation.getSimpleName() + " method but it does not return " + MethodOutcome.class);
68  			} else if (theMethod.getReturnType() == void.class) {
69  				myReturnVoid = true;
70  			}
71  		}
72  	}

74  	private void addLocationHeader(Request theRequest, HttpServletResponse theResponse, MethodOutcome response, String headerLocation) {
75  		StringBuilder b = new StringBuilder();
76  		b.append(theRequest.getFhirServerBase());
77  		b.append('/');
78  		b.append(getResourceName());
79  		b.append('/');
80  		b.append(response.getId().getIdPart());
81  		if (response.getId().hasVersionIdPart()) {
82  			b.append("/" + Constants.PARAM_HISTORY + "/");
83  			b.append(response.getId().getVersionIdPart());
84  		} else if (response.getVersionId() != null && response.getVersionId().isEmpty() == false) {
85  			b.append("/" + Constants.PARAM_HISTORY + "/");
86  			b.append(response.getVersionId().getValue());
87  		}
88  		theResponse.addHeader(headerLocation, b.toString());
89  	}

108 	@Override
109 	public boolean incomingServerRequestMatchesMethod(Request theRequest) {
110 		Set<RequestType> allowableRequestTypes = provideAllowableRequestTypes();
111 		RequestType requestType = theRequest.getRequestType();
112 		if (!allowableRequestTypes.contains(requestType)) {
113 			return false;
114 		}
115 		if (!getResourceName().equals(theRequest.getResourceName())) {
116 			return false;
117 		}
118 		if (getMatchingOperation() == null && StringUtils.isNotBlank(theRequest.getOperation())) {
119 			return false;
120 		}
121 		if (getMatchingOperation() != null && !getMatchingOperation().equals(theRequest.getOperation())) {
122 			return false;
123 		}
124 		return true;
125 	}

127 	@Override
128 	public MethodOutcome invokeClient(String theResponseMimeType, Reader theResponseReader, int theResponseStatusCode, Map<String, List<String>> theHeaders) throws IOException, BaseServerResponseException {
129 		switch (theResponseStatusCode) {
130 		case Constants.STATUS_HTTP_200_OK:
131 		case Constants.STATUS_HTTP_201_CREATED:
132 		case Constants.STATUS_HTTP_204_NO_CONTENT:
133 			if (myReturnVoid) {
134 				return null;
135 			}
136 			MethodOutcome retVal = MethodUtil.process2xxResponse(getContext(), getResourceName(), theResponseStatusCode, theResponseMimeType, theResponseReader, theHeaders);
137 			return retVal;
138 		default:
139 			throw processNon2xxResponseAndReturnExceptionToThrow(theResponseStatusCode, theResponseMimeType, theResponseReader);
140 		}
141 
142 	}

144 	@Override
145 	public void invokeServer(RestfulServer theServer, Request theRequest) throws BaseServerResponseException, IOException {
146 		IResource resource;
147 		if (requestContainsResource()) {
148 			resource = parseIncomingServerResource(theRequest);
149 			TagList tagList = new TagList();
150 			for (Enumeration<String> enumeration = theRequest.getServletRequest().getHeaders(Constants.HEADER_CATEGORY); enumeration.hasMoreElements();) {
151 				String nextTagComplete = enumeration.nextElement();
152 				MethodUtil.parseTagValue(tagList, nextTagComplete);
153 			}
154 			if (tagList.isEmpty() == false) {
155 				resource.getResourceMetadata().put(ResourceMetadataKeyEnum.TAG_LIST, tagList);
156 			}
157 		} else {
158 			resource = null;
159 		}
160 
161 		Object[] params = createParametersForServerRequest(theRequest, resource);
162 		addParametersForServerRequest(theRequest, params);
163 
164 		HttpServletResponse servletResponse = theRequest.getServletResponse();
165 		MethodOutcome response;
166 		try {
167 			response = (MethodOutcome) invokeServerMethod(params);
168 		} catch (InternalErrorException e) {
169 			ourLog.error("Internal error during method invocation", e);
170 			EncodingEnum encodingNotNull = RestfulServerUtils.determineResponseEncodingWithDefault(theServer, theRequest.getServletRequest());
171 			streamOperationOutcome(e, theServer, encodingNotNull, servletResponse, theRequest);
172 			return;
173 		} catch (BaseServerResponseException e) {
174 			ourLog.info("Exception during method invocation: " + e.getMessage());
175 			EncodingEnum encodingNotNull = RestfulServerUtils.determineResponseEncodingWithDefault(theServer, theRequest.getServletRequest());
176 			streamOperationOutcome(e, theServer, encodingNotNull, servletResponse, theRequest);
177 			return;
178 		}
179 
180 		if (response != null && response.getId() != null && response.getId().hasResourceType()) {
181 			if (getContext().getResourceDefinition(response.getId().getResourceType()) == null) {
182 				throw new InternalErrorException("Server method returned invalid resource ID: " + response.getId().getValue());
183 			}
184 		}
185 
186 		BaseOperationOutcome outcome = response != null ? response.getOperationOutcome() : null;
187 		for (int i = theServer.getInterceptors().size() - 1; i >= 0; i--) {
188 			IServerInterceptor next = theServer.getInterceptors().get(i);
189 			boolean continueProcessing = next.outgoingResponse(theRequest, outcome, theRequest.getServletRequest(), theRequest.getServletResponse());
190 			if (!continueProcessing) {
191 				return;
192 			}
193 		}
194 
195 		switch (getResourceOperationType()) {
196 		case CREATE:
197 			if (response == null) {
198 				throw new InternalErrorException("Method " + getMethod().getName() + " in type " + getMethod().getDeclaringClass().getCanonicalName() + " returned null, which is not allowed for create operation");
199 			}
200 			if (response.getCreated() == null || Boolean.TRUE.equals(response.getCreated())) {
201 				servletResponse.setStatus(Constants.STATUS_HTTP_201_CREATED);
202 			} else {
203 				servletResponse.setStatus(Constants.STATUS_HTTP_200_OK);
204 			}
205 			addContentLocationHeaders(theRequest, servletResponse, response);
206 			break;
207 
208 		case UPDATE:
209 			if (response == null || response.getCreated() == null || Boolean.FALSE.equals(response.getCreated())) {
210 				servletResponse.setStatus(Constants.STATUS_HTTP_200_OK);
211 			} else {
212 				servletResponse.setStatus(Constants.STATUS_HTTP_201_CREATED);
213 			}
214 			addContentLocationHeaders(theRequest, servletResponse, response);
215 			break;
216 
217 		case VALIDATE:
218 		case DELETE:
219 		default:
220 			if (response == null) {
221 				if (isReturnVoid() == false) {
222 					throw new InternalErrorException("Method " + getMethod().getName() + " in type " + getMethod().getDeclaringClass().getCanonicalName() + " returned null");
223 				}
224 				servletResponse.setStatus(Constants.STATUS_HTTP_204_NO_CONTENT);
225 			} else {
226 				if (response.getOperationOutcome() == null) {
227 					servletResponse.setStatus(Constants.STATUS_HTTP_204_NO_CONTENT);
228 				} else {
229 					servletResponse.setStatus(Constants.STATUS_HTTP_200_OK);
230 				}
231 			}
232 
233 		}
234 
235 		theServer.addHeadersToResponse(servletResponse);
236 
237 		if (outcome != null) {
238 			EncodingEnum encoding = RestfulServerUtils.determineResponseEncodingWithDefault(theServer, theRequest.getServletRequest());
239 			servletResponse.setContentType(encoding.getResourceContentType());
240 			Writer writer = servletResponse.getWriter();
241 			IParser parser = encoding.newParser(getContext());
242 			parser.setPrettyPrint(RestfulServerUtils.prettyPrintResponse(theServer, theRequest));
243 			try {
244 				parser.encodeResourceToWriter(response.getOperationOutcome(), writer);
245 			} finally {
246 				writer.close();
247 			}
248 		} else {
249 			servletResponse.setContentType(Constants.CT_TEXT_WITH_UTF8);
250 			Writer writer = servletResponse.getWriter();
251 			writer.close();
252 		}
253 
254 		// getMethod().in
255 	}

257 	private void addContentLocationHeaders(Request theRequest, HttpServletResponse servletResponse, MethodOutcome response) {
258 		if (response != null && response.getId() != null) {
259 			addLocationHeader(theRequest, servletResponse, response, Constants.HEADER_LOCATION);
260 			addLocationHeader(theRequest, servletResponse, response, Constants.HEADER_CONTENT_LOCATION);
261 		}

268 	/**
269 	 * @throws IOException
270 	 */
271 	protected IResource parseIncomingServerResource(Request theRequest) throws IOException {
272 		EncodingEnum encoding = RestfulServerUtils.determineRequestEncoding(theRequest);
273 		IParser parser = encoding.newParser(getContext());
274 		BufferedReader requestReader = theRequest.getServletRequest().getReader();
275 		
276 		Class<? extends IBaseResource> wantedResourceType = requestContainsResourceType();
277 		IResource retVal;
278 		if (wantedResourceType != null) {
279 			retVal = (IResource) parser.parseResource(wantedResourceType, requestReader);
280 		} else {
281 			retVal = parser.parseResource(requestReader);
282 		}
283 		
284 		retVal.setId(theRequest.getId());
285 		
286 		return retVal;
287 	}

306 	protected void streamOperationOutcome(BaseServerResponseException theE, RestfulServer theServer, EncodingEnum theEncodingNotNull, HttpServletResponse theResponse, Request theRequest) throws IOException {
307 		theResponse.setStatus(theE.getStatusCode());
308 
309 		theServer.addHeadersToResponse(theResponse);
310 		
311 		if (theE.getOperationOutcome() != null) {
312 			theResponse.setContentType(theEncodingNotNull.getResourceContentType());
313 			IParser parser = theEncodingNotNull.newParser(theServer.getFhirContext());
314 			parser.setPrettyPrint(RestfulServerUtils.prettyPrintResponse(theServer, theRequest));
315 			Writer writer = theResponse.getWriter();
316 			try {
317 				parser.encodeResourceToWriter(theE.getOperationOutcome(), writer);
318 			} finally {
319 				writer.close();
320 			}
321 		} else {
322 			theResponse.setContentType(Constants.CT_TEXT);
323 			Writer writer = theResponse.getWriter();
324 			try {
325 				writer.append(theE.getMessage());
326 			} finally {
327 				writer.close();
328 			}
329 		}
330 	}

332 	protected static void parseContentLocation(MethodOutcome theOutcomeToPopulate, String theResourceName, String theLocationHeader) {
333 		if (StringUtils.isBlank(theLocationHeader)) {
334 			return;
335 		}
336 
337 		theOutcomeToPopulate.setId(new IdDt(theLocationHeader));
338 
339 		String resourceNamePart = "/" + theResourceName + "/";
340 		int resourceIndex = theLocationHeader.lastIndexOf(resourceNamePart);
341 		if (resourceIndex > -1) {
342 			int idIndexStart = resourceIndex + resourceNamePart.length();
343 			int idIndexEnd = theLocationHeader.indexOf('/', idIndexStart);
344 			if (idIndexEnd == -1) {
345 				// nothing
346 			} else {
347 				String versionIdPart = "/_history/";
348 				int historyIdStart = theLocationHeader.indexOf(versionIdPart, idIndexEnd);
349 				if (historyIdStart != -1) {
350 					theOutcomeToPopulate.setVersionId(new IdDt(theLocationHeader.substring(historyIdStart + versionIdPart.length())));
351 				}
352 			}
353 		}

60  abstract class BaseOutcomeReturningMethodBinding extends BaseMethodBinding<MethodOutcome> {
61  	static final org.slf4j.Logger ourLog = org.slf4j.LoggerFactory.getLogger(BaseOutcomeReturningMethodBinding.class);
62  
63  	private static EnumSet<RestOperationTypeEnum> ourOperationsWhichAllowPreferHeader = EnumSet.of(RestOperationTypeEnum.CREATE, RestOperationTypeEnum.UPDATE);
64  
65  	private boolean myReturnVoid;
66  
67  	public BaseOutcomeReturningMethodBinding(Method theMethod, FhirContext theContext, Class<?> theMethodAnnotation, Object theProvider) {
68  		super(theMethod, theContext, theProvider);
69  
70  		if (!theMethod.getReturnType().equals(MethodOutcome.class)) {
71  			if (!allowVoidReturnType()) {
72  				throw new ConfigurationException("Method " + theMethod.getName() + " in type " + theMethod.getDeclaringClass().getCanonicalName() + " is a @" + theMethodAnnotation.getSimpleName() + " method but it does not return " + MethodOutcome.class);
73  			} else if (theMethod.getReturnType() == void.class) {
74  				myReturnVoid = true;
75  			}
76  		}
77  	}

82  	 * Subclasses may override to allow a void method return type, which is allowable for some methods (e.g. delete)

91  	 * For servers, this method will match only incoming requests that match the given operation, or which have no

96  	private int getOperationStatus(MethodOutcome response) {
97  		switch (getRestOperationType()) {
98  		case CREATE:
99  			if (response == null) {
100 				throw new InternalErrorException("Method " + getMethod().getName() + " in type " + getMethod().getDeclaringClass().getCanonicalName() + " returned null, which is not allowed for create operation");
101 			}
102 			if (response.getCreated() == null || Boolean.TRUE.equals(response.getCreated())) {
103 				return Constants.STATUS_HTTP_201_CREATED;
104 			} else {
105 				return Constants.STATUS_HTTP_200_OK;
106 			}
107 
108 		case UPDATE:
109 			if (response == null || response.getCreated() == null || Boolean.FALSE.equals(response.getCreated())) {
110 				return Constants.STATUS_HTTP_200_OK;
111 			} else {
112 				return Constants.STATUS_HTTP_201_CREATED;
113 			}
114 
115 		case VALIDATE:
116 		case DELETE:
117 		default:
118 			if (response == null) {
119 				if (isReturnVoid() == false) {
120 					throw new InternalErrorException("Method " + getMethod().getName() + " in type " + getMethod().getDeclaringClass().getCanonicalName() + " returned null");
121 				}
122 				return Constants.STATUS_HTTP_204_NO_CONTENT;
123 			} else {
124 				if (response.getOperationOutcome() == null) {
125 					return Constants.STATUS_HTTP_204_NO_CONTENT;
126 				} else {
127 					return Constants.STATUS_HTTP_200_OK;
128 				}
129 			}
130 		}
131 	}

133 	@Override
134 	public boolean incomingServerRequestMatchesMethod(RequestDetails theRequest) {
135 		Set<RequestTypeEnum> allowableRequestTypes = provideAllowableRequestTypes();
136 		RequestTypeEnum requestType = theRequest.getRequestType();
137 		if (!allowableRequestTypes.contains(requestType)) {
138 			return false;
139 		}
140 		if (!getResourceName().equals(theRequest.getResourceName())) {
141 			return false;
142 		}
143 		if (getMatchingOperation() == null && StringUtils.isNotBlank(theRequest.getOperation())) {
144 			return false;
145 		}
146 		if (getMatchingOperation() != null && !getMatchingOperation().equals(theRequest.getOperation())) {
147 			return false;
148 		}
149 		
150 		/*
151 		 * Note: Technically this will match an update (PUT) method even if
152 		 * there is no ID in the URL - We allow this here because there is no
153 		 * better match for that, and this allows the update/PUT method to give
154 		 * a helpful error if the client has forgotten to include the 
155 		 * ID in the URL.
156 		 * 
157 		 * It's also needed for conditional update..
158 		 */
159 		
160 		return true;
161 	}

163 	@Override
164 	public MethodOutcome invokeClient(String theResponseMimeType, Reader theResponseReader, int theResponseStatusCode, Map<String, List<String>> theHeaders) throws BaseServerResponseException {
165 		if (theResponseStatusCode >= 200 && theResponseStatusCode < 300) {
166 			if (myReturnVoid) {
167 				return null;
168 			}
169 			MethodOutcome retVal = MethodUtil.process2xxResponse(getContext(), theResponseStatusCode, theResponseMimeType, theResponseReader, theHeaders);
170 			return retVal;
171 		} else {
172 			throw processNon2xxResponseAndReturnExceptionToThrow(theResponseStatusCode, theResponseMimeType, theResponseReader);
173 		}
174 	}

176 	@Override
177 	public Object invokeServer(IRestfulServer<?> theServer, RequestDetails theRequest) throws BaseServerResponseException, IOException {
178 
179 		Object[] params = createParametersForServerRequest(theRequest);
180 		addParametersForServerRequest(theRequest, params);
181 
182 		/*
183 		 * No need to catch and handle exceptions here, we already handle them one level up including invoking interceptors
184 		 * on them
185 		 */
186 		MethodOutcome response;
187 		Object methodReturn = invokeServerMethod(theServer, theRequest, params);
188 		
189 		if (methodReturn instanceof IBaseOperationOutcome) {
190 			response = new MethodOutcome();
191 			response.setOperationOutcome((IBaseOperationOutcome) methodReturn);
192 		} else {
193 			response = (MethodOutcome) methodReturn;
194 		}
195 		
196 		if (response != null && response.getId() != null && response.getId().hasResourceType()) {
197 			if (getContext().getResourceDefinition(response.getId().getResourceType()) == null) {
198 				throw new InternalErrorException("Server method returned invalid resource ID: " + response.getId().getValue());
199 			}
200 		}
201 
202 		IBaseOperationOutcome outcome = response != null ? response.getOperationOutcome() : null;
203 		IBaseResource resource = response != null ? response.getResource() : null;
204 
205 		return returnResponse(theServer, theRequest, response, outcome, resource);
206 	}
207 	public boolean isReturnVoid() {
208 		return myReturnVoid;
209 	}

213 	private Object returnResponse(IRestfulServer<?> theServer, RequestDetails theRequest, MethodOutcome response, IBaseResource originalOutcome, IBaseResource resource) throws IOException {
214 		boolean allowPrefer = false;
215 		int operationStatus = getOperationStatus(response);
216 		IBaseResource outcome = originalOutcome;
217 
218 		if (ourOperationsWhichAllowPreferHeader.contains(getRestOperationType())) {
219 			allowPrefer = true;
220 		}
221 
222 		if (resource != null && allowPrefer) {
223 			String prefer = theRequest.getHeader(Constants.HEADER_PREFER);
224 			PreferReturnEnum preferReturn = RestfulServerUtils.parsePreferHeader(prefer);
225 			if (preferReturn != null) {
226 				if (preferReturn == PreferReturnEnum.REPRESENTATION) {
227 					outcome = resource;
228 				}
229 			}
230 		}
231 
232 		for (int i = theServer.getInterceptors().size() - 1; i >= 0; i--) {
233 			IServerInterceptor next = theServer.getInterceptors().get(i);
234 			boolean continueProcessing = next.outgoingResponse(theRequest, outcome);
235 			if (!continueProcessing) {
236 				return null;
237 			}
238 		}
239 		
240 		IRestfulResponse restfulResponse = theRequest.getResponse();
241 		
242 		if (response != null) {
243 			if (response.getResource() != null) {
244 				restfulResponse.setOperationResourceLastUpdated(RestfulServerUtils.extractLastUpdatedFromResource(response.getResource()));
245 			}
246 			
247 			IIdType responseId = response.getId();
248 			if (responseId != null && responseId.getResourceType() == null && responseId.hasIdPart()) {
249 				responseId = responseId.withResourceType(getResourceName());
250 			}
251 			
252 			if (responseId != null) {
253 				String serverBase = theRequest.getFhirServerBase();
254 				responseId = RestfulServerUtils.fullyQualifyResourceIdOrReturnNull(theServer, resource, serverBase, responseId);
255 				restfulResponse.setOperationResourceId(responseId);
256 			}
257 		}
258 
259 		boolean prettyPrint = RestfulServerUtils.prettyPrintResponse(theServer, theRequest);
260 		Set<SummaryEnum> summaryMode = Collections.emptySet();
261 
262 		return restfulResponse.streamResponseAsResource(outcome, prettyPrint, summaryMode, operationStatus, null, theRequest.isRespondGzip(), true);
263 //		return theRequest.getResponse().returnResponse(ParseAction.create(outcome), operationStatus, allowPrefer, response, getResourceName());
264 	}

266 	protected void streamOperationOutcome(BaseServerResponseException theE, RestfulServer theServer, EncodingEnum theEncodingNotNull, HttpServletResponse theResponse, RequestDetails theRequest) throws IOException {
267 		theResponse.setStatus(theE.getStatusCode());
268 
269 		theServer.addHeadersToResponse(theResponse);
270 
271 		if (theE.getOperationOutcome() != null) {
272 			theResponse.setContentType(theEncodingNotNull.getResourceContentType());
273 			IParser parser = theEncodingNotNull.newParser(theServer.getFhirContext());
274 			parser.setPrettyPrint(RestfulServerUtils.prettyPrintResponse(theServer, theRequest));
275 			Writer writer = theResponse.getWriter();
276 			try {
277 				parser.encodeResourceToWriter(theE.getOperationOutcome(), writer);
278 			} finally {
279 				writer.close();
280 			}
281 		} else {
282 			theResponse.setContentType(Constants.CT_TEXT);
283 			Writer writer = theResponse.getWriter();
284 			try {
285 				writer.append(theE.getMessage());
286 			} finally {
287 				writer.close();
288 			}
289 		}
290 	}

292 	protected static void parseContentLocation(FhirContext theContext, MethodOutcome theOutcomeToPopulate, String theLocationHeader) {
293 		if (StringUtils.isBlank(theLocationHeader)) {
294 			return;
295 		}
296 
297 		IIdType id = theContext.getVersion().newIdType();
298 		id.setValue(theLocationHeader);
299 		theOutcomeToPopulate.setId(id);
300 	}

211 		try {
212 			inputStream = getClass().getResourceAsStream("/META-INF/MANIFEST.MF");
213 			if (inputStream != null) {
214 				Manifest manifest = new Manifest(inputStream);
215 				result.setConformanceDate(manifest.getMainAttributes().getValue("Build-Time"));
216 			}
217 		} catch (IOException e) {
218 			// fall through
219 		} finally {
220 			if (inputStream != null) {
221 				IOUtils.closeQuietly(inputStream);
222 			}
223 		}
224 		return result;
225 	}

244 	public BaseMethodBinding<?> determineResourceMethod(RequestDetails requestDetails, String requestPath) {
245 		RequestTypeEnum requestType = requestDetails.getRequestType();
246 
247 		ResourceBinding resourceBinding = null;
248 		BaseMethodBinding<?> resourceMethod = null;
249 		String resourceName = requestDetails.getResourceName();
250 		if (myServerConformanceMethod.incomingServerRequestMatchesMethod(requestDetails)) {
251 			resourceMethod = myServerConformanceMethod;
252 		} else if (resourceName == null) {
253 			resourceBinding = myServerBinding;
254 		} else {
255 			resourceBinding = myResourceNameToBinding.get(resourceName);
256 			if (resourceBinding == null) {
257 				throw new ResourceNotFoundException("Unknown resource type '" + resourceName + "' - Server knows how to handle: " + myResourceNameToBinding.keySet());
258 			}
259 		}
260 
261 		if (resourceMethod == null) {
262 			if (resourceBinding != null) {
263 				resourceMethod = resourceBinding.getMethod(requestDetails);
264 			}
265 		}
266 		if (resourceMethod == null) {
267 			if (isBlank(requestPath)) {
268 				throw new InvalidRequestException(myFhirContext.getLocalizer().getMessage(RestfulServer.class, "rootRequest"));
269 			} else {
270 				throw new InvalidRequestException(myFhirContext.getLocalizer().getMessage(RestfulServer.class, "unknownMethod", requestType.name(), requestPath, requestDetails.getParameters().keySet()));
271 			}
272 		}
273 		return resourceMethod;
274 	}

277 	 * Count length of URL string, but treating unescaped sequences (e.g. ' ') as their unescaped equivalent (%20)
278 	 */
279 	protected int escapedLength(String theServletPath) {
280 		int delta = 0;
281 		for (int i = 0; i < theServletPath.length(); i++) {
282 			char next = theServletPath.charAt(i);
283 			if (next == ' ') {
284 				delta = delta + 2;
285 			}
286 		}
287 		return theServletPath.length() + delta;
288 	}

290 	private void findResourceMethods(Object theProvider) {
291 
292 		ourLog.info("Scanning type for RESTful methods: {}", theProvider.getClass());
293 		int count = 0;
294 
295 		Class<?> clazz = theProvider.getClass();
296 		Class<?> supertype = clazz.getSuperclass();
297 		while (!Object.class.equals(supertype)) {
298 			count += findResourceMethods(theProvider, supertype);
299 			supertype = supertype.getSuperclass();
300 		}
301 
302 		count += findResourceMethods(theProvider, clazz);
303 
304 		if (count == 0) {
305 			throw new ConfigurationException("Did not find any annotated RESTful methods on provider class " + theProvider.getClass().getCanonicalName());
306 		}
307 	}

309 	private int findResourceMethods(Object theProvider, Class<?> clazz) throws ConfigurationException {
310 		int count = 0;
311 
312 		for (Method m : ReflectionUtil.getDeclaredMethods(clazz)) {
313 			BaseMethodBinding<?> foundMethodBinding = BaseMethodBinding.bindMethod(m, getFhirContext(), theProvider);
314 			if (foundMethodBinding == null) {
315 				continue;
316 			}
317 
318 			count++;
319 
320 			if (foundMethodBinding instanceof ConformanceMethodBinding) {
321 				myServerConformanceMethod = foundMethodBinding;
322 				continue;
323 			}
324 
325 			if (!Modifier.isPublic(m.getModifiers())) {
326 				throw new ConfigurationException("Method '" + m.getName() + "' is not public, FHIR RESTful methods must be public");
327 			} else {
328 				if (Modifier.isStatic(m.getModifiers())) {
329 					throw new ConfigurationException("Method '" + m.getName() + "' is static, FHIR RESTful methods must not be static");
330 				} else {
331 					ourLog.debug("Scanning public method: {}#{}", theProvider.getClass(), m.getName());
332 
333 					String resourceName = foundMethodBinding.getResourceName();
334 					ResourceBinding resourceBinding;
335 					if (resourceName == null) {
336 						resourceBinding = myServerBinding;
337 					} else {
338 						RuntimeResourceDefinition definition = getFhirContext().getResourceDefinition(resourceName);
339 						if (myResourceNameToBinding.containsKey(definition.getName())) {
340 							resourceBinding = myResourceNameToBinding.get(definition.getName());
341 						} else {
342 							resourceBinding = new ResourceBinding();
343 							resourceBinding.setResourceName(resourceName);
344 							myResourceNameToBinding.put(resourceName, resourceBinding);
345 						}
346 					}
347 
348 					List<Class<?>> allowableParams = foundMethodBinding.getAllowableParamAnnotations();
349 					if (allowableParams != null) {
350 						for (Annotation[] nextParamAnnotations : m.getParameterAnnotations()) {
351 							for (Annotation annotation : nextParamAnnotations) {
352 								Package pack = annotation.annotationType().getPackage();
353 								if (pack.equals(IdParam.class.getPackage())) {
354 									if (!allowableParams.contains(annotation.annotationType())) {
355 										throw new ConfigurationException("Method[" + m.toString() + "] is not allowed to have a parameter annotated with " + annotation);
356 									}
357 								}
358 							}
359 						}
360 					}
361 
362 					resourceBinding.addMethod(foundMethodBinding);
363 					ourLog.debug(" * Method: {}#{} is a handler", theProvider.getClass(), m.getName());
364 				}
365 			}
366 		}
367 
368 		return count;
369 	}

531 	protected void handleRequest(RequestTypeEnum theRequestType, HttpServletRequest theRequest, HttpServletResponse theResponse) throws ServletException, IOException {
532 		String fhirServerBase = null;
533 		ServletRequestDetails requestDetails = new ServletRequestDetails();
534 		requestDetails.setServer(this);
535 		requestDetails.setRequestType(theRequestType);
536 		requestDetails.setServletRequest(theRequest);
537 		requestDetails.setServletResponse(theResponse);
538 
539 		theRequest.setAttribute(SERVLET_CONTEXT_ATTRIBUTE, getServletContext());
540 
541 		try {
542 
543 			/* ***********************************
544 			 * Parse out the request parameters
545 			 * ***********************************/
546 			
547 			String requestFullPath = StringUtils.defaultString(theRequest.getRequestURI());
548 			String servletPath = StringUtils.defaultString(theRequest.getServletPath());
549 			StringBuffer requestUrl = theRequest.getRequestURL();
550 			String servletContextPath = IncomingRequestAddressStrategy.determineServletContextPath(theRequest, this);
551 
552 			/*
553 			 * Just for debugging..
554 			 */
555 			if (ourLog.isTraceEnabled()) {
556 				ourLog.trace("Request FullPath: {}", requestFullPath);
557 				ourLog.trace("Servlet Path: {}", servletPath);
558 				ourLog.trace("Request Url: {}", requestUrl);
559 				ourLog.trace("Context Path: {}", servletContextPath);
560 			}
561 
562 			String completeUrl;
563 			Map<String, String[]> params = null;
564 			if (StringUtils.isNotBlank(theRequest.getQueryString())) {
565 				completeUrl = requestUrl + "?" + theRequest.getQueryString();
566 				/*
567 				 * By default, we manually parse the request params (the URL params, or the body for
568 				 * POST form queries) since Java containers can't be trusted to use UTF-8 encoding
569 				 * when parsing. Specifically Tomcat 7 and Glassfish 4.0 use 8859-1 for some dumb
570 				 * reason.... grr.....
571 				 */
572 				if (isIgnoreServerParsedRequestParameters()) {
573 					String contentType = theRequest.getHeader(Constants.HEADER_CONTENT_TYPE);
574 					if (theRequestType == RequestTypeEnum.POST && isNotBlank(contentType) && contentType.startsWith(Constants.CT_X_FORM_URLENCODED)) {
575 						String requestBody = new String(requestDetails.loadRequestContents(), Constants.CHARSET_UTF8);
576 						params = UrlUtil.parseQueryStrings(theRequest.getQueryString(), requestBody);
577 					} else if (theRequestType == RequestTypeEnum.GET) {
578 						params = UrlUtil.parseQueryString(theRequest.getQueryString());
579 					}
580 				}
581 			} else {
582 				completeUrl = requestUrl.toString();
583 			}
584 
585 			if (params == null) {
586 				params = new HashMap<String, String[]>(theRequest.getParameterMap());
587 			}
588 
589 			requestDetails.setParameters(params);
590 
591 			/* *************************
592 			 * Notify interceptors about the incoming request
593 			 * *************************/
594 			
595 			for (IServerInterceptor next : myInterceptors) {
596 				boolean continueProcessing = next.incomingRequestPreProcessed(theRequest, theResponse);
597 				if (!continueProcessing) {

671 			 * is an object that wraps a specific implementing (user-supplied) method, but
672 			 * handles its input and provides its output back to the client.

675 			 * method binding replies to the client and closes the response.

746 					ourLog.debug("Interceptor {} returned false, not continuing processing");
747 					return;
748 				}
749 			}
750 
751 			/*
752 			 * If we're handling an exception, no summary mode should be applied
753 			 */
754 			requestDetails.getParameters().remove(Constants.PARAM_SUMMARY);
755 			requestDetails.getParameters().remove(Constants.PARAM_ELEMENTS);
756 
757 			/*
758 			 * If nobody handles it, default behaviour is to stream back the OperationOutcome to the client.
759 			 */
760 			DEFAULT_EXCEPTION_HANDLER.handleException(requestDetails, exception, theRequest, theResponse);

949 	public void populateRequestDetailsFromRequestPath(RequestDetails theRequestDetails, String theRequestPath) {
950 		StringTokenizer tok = new UrlPathTokenizer(theRequestPath);
951 		String resourceName = null;
952 
953 		IIdType id = null;
954 		String operation = null;
955 		String compartment = null;
956 		if (tok.hasMoreTokens()) {
957 			resourceName = tok.nextToken();
958 			if (partIsOperation(resourceName)) {
959 				operation = resourceName;
960 				resourceName = null;
961 			}
962 		}
963 		theRequestDetails.setResourceName(resourceName);
964 
965 		if (tok.hasMoreTokens()) {
966 			String nextString = tok.nextToken();
967 			if (partIsOperation(nextString)) {
968 				operation = nextString;
969 			} else {
970 				id = myFhirContext.getVersion().newIdType();
971 				id.setParts(null, resourceName, UrlUtil.unescape(nextString), null);
972 			}
973 		}
974 
975 		if (tok.hasMoreTokens()) {
976 			String nextString = tok.nextToken();
977 			if (nextString.equals(Constants.PARAM_HISTORY)) {
978 				if (tok.hasMoreTokens()) {
979 					String versionString = tok.nextToken();
980 					if (id == null) {
981 						throw new InvalidRequestException("Don't know how to handle request path: " + theRequestPath);
982 					}
983 					id.setParts(null, resourceName, id.getIdPart(), UrlUtil.unescape(versionString));
984 				} else {
985 					operation = Constants.PARAM_HISTORY;
986 				}
987 			} else if (partIsOperation(nextString)) {
988 				if (operation != null) {
989 					throw new InvalidRequestException("URL Path contains two operations: " + theRequestPath);
990 				}
991 				operation = nextString;
992 			} else {
993 				compartment = nextString;
994 			}
995 		}
996 
997 		// Secondary is for things like ..../_tags/_delete
998 		String secondaryOperation = null;
999 
1000 		while (tok.hasMoreTokens()) {
1001 			String nextString = tok.nextToken();
1002 			if (operation == null) {
1003 				operation = nextString;
1004 			} else if (secondaryOperation == null) {
1005 				secondaryOperation = nextString;
1006 			} else {
1007 				throw new InvalidRequestException("URL path has unexpected token '" + nextString + "' at the end: " + theRequestPath);
1008 			}
1009 		}
1010 
1011 		theRequestDetails.setId(id);
1012 		theRequestDetails.setOperation(operation);
1013 		theRequestDetails.setSecondaryOperation(secondaryOperation);
1014 		theRequestDetails.setCompartmentName(compartment);
1015 	}

1063 	protected void registerProviders(Collection<? extends Object> providers, boolean inInit) {
1064 		List<IResourceProvider> newResourceProviders = new ArrayList<IResourceProvider>();
1065 		List<Object> newPlainProviders = new ArrayList<Object>();

1075 					}

1080 					}
1081 					if (!inInit) {
1082 						myResourceProviders.add(rsrcProvider);
1083 					}
1084 					myTypeToProvider.put(resourceName, rsrcProvider);
1085 					providedResourceScanner.scanForProvidedResources(rsrcProvider);
1086 					newResourceProviders.add(rsrcProvider);
1087 				} else {
1088 					if (!inInit) {
1089 						myPlainProviders.add(provider);
1090 					}
1091 					newPlainProviders.add(provider);
1092 				}
1093 
1094 			}
1095 			if (!newResourceProviders.isEmpty()) {
1096 				ourLog.info("Added {} resource provider(s). Total {}", newResourceProviders.size(), myTypeToProvider.size());
1097 				for (IResourceProvider provider : newResourceProviders) {
1098 					assertProviderIsValid(provider);
1099 					findResourceMethods(provider);
1100 				}
1101 			}
1102 			if (!newPlainProviders.isEmpty()) {
1103 				ourLog.info("Added {} plain provider(s). Total {}", newPlainProviders.size());
1104 				for (Object provider : newPlainProviders) {
1105 					assertProviderIsValid(provider);
1106 					findResourceMethods(provider);
1107 				}
1108 			}
1109 			if (!inInit) {
1110 				ourLog.trace("Invoking provider initialize methods");
1111 				if (!newResourceProviders.isEmpty()) {
1112 					for (IResourceProvider provider : newResourceProviders) {
1113 						invokeInitialize(provider);
1114 					}
1115 				}
1116 				if (!newPlainProviders.isEmpty()) {
1117 					for (Object provider : newPlainProviders) {
1118 						invokeInitialize(provider);
1119 					}
1120 				}
1121 			}
1122 		}

43  /**
44   * Provides methods to intercept requests and responses. Note that implementations of this interface may wish to use {@link InterceptorAdapter} in order to not need to implement every method.
45   * <p>
46   * <b>See:</b> See the <a href="http://jamesagnew.github.io/hapi-fhir/doc_rest_server_interceptor.html">server interceptor documentation</a> for more information on how to use this class.
47   * </p>
48   */
49  public interface IServerInterceptor {
50  
51  	/**
52  	 * This method is called upon any exception being thrown within the server's request processing code. This includes any exceptions thrown within resource provider methods (e.g. {@link Search} and
53  	 * {@link Read} methods) as well as any runtime exceptions thrown by the server itself. This also includes any {@link AuthenticationException}s thrown.
54  	 * <p>
55  	 * Implementations of this method may choose to ignore/log/count/etc exceptions, and return <code>true</code>. In this case, processing will continue, and the server will automatically generate an
56  	 * {@link BaseOperationOutcome OperationOutcome}. Implementations may also choose to provide their own response to the client. In this case, they should return <code>false</code>, to indicate that
57  	 * they have handled the request and processing should stop.
58  	 * </p>
59  	 * 
60  	 *
61  	 * @param theRequestDetails
62  	 *           A bean containing details about the request that is about to be processed, including details such as the resource type and logical ID (if any) and other FHIR-specific aspects of the
63  	 *           request which have been pulled out of the {@link javax.servlet.http.HttpServletRequest servlet request}. Note that the bean properties are not all guaranteed to be populated, depending
64  	 *           on how early during processing the exception occurred.
65  	 * @param theServletRequest
66  	 *           The incoming request
67  	 * @param theServletResponse
68  	 *           The response. Note that interceptors may choose to provide a response (i.e. by calling {@link javax.servlet.http.HttpServletResponse#getWriter()}) but in that case it is important to
69  	 *           return <code>false</code> to indicate that the server itself should not also provide a response.
70  	 * @return Return <code>true</code> if processing should continue normally. This is generally the right thing to do. If your interceptor is providing a response rather than letting HAPI handle the
71  	 *         response normally, you must return <code>false</code>. In this case, no further processing will occur and no further interceptors will be called.
72  	 * @throws ServletException
73  	 *            If this exception is thrown, it will be re-thrown up to the container for handling.
74  	 * @throws IOException
75  	 *            If this exception is thrown, it will be re-thrown up to the container for handling.
76  	 */
77  	boolean handleException(RequestDetails theRequestDetails, BaseServerResponseException theException, HttpServletRequest theServletRequest, HttpServletResponse theServletResponse)
78  			throws ServletException, IOException;
79  
80  	/**
81  	 * This method is called just before the actual implementing server method is invoked.
82  	 * <p>
83  	 * Note about exceptions:
84  	 * </p>
85  	 * 
86  	 * @param theRequestDetails
87  	 *           A bean containing details about the request that is about to be processed, including details such as the resource type and logical ID (if any) and other FHIR-specific aspects of the
88  	 *           request which have been pulled out of the {@link HttpServletRequest servlet request}.
89  	 * @param theRequest
90  	 *           The incoming request
91  	 * @param theResponse
92  	 *           The response. Note that interceptors may choose to provide a response (i.e. by calling {@link HttpServletResponse#getWriter()}) but in that case it is important to return
93  	 *           <code>false</code> to indicate that the server itself should not also provide a response.
94  	 * @return Return <code>true</code> if processing should continue normally. This is generally the right thing to do. If your interceptor is providing a response rather than letting HAPI handle the
95  	 *         response normally, you must return <code>false</code>. In this case, no further processing will occur and no further interceptors will be called.
96  	 * @throws AuthenticationException
97  	 *            This exception may be thrown to indicate that the interceptor has detected an unauthorized access attempt. If thrown, processing will stop and an HTTP 401 will be returned to the
98  	 *            client.
99  	 */
100 	boolean incomingRequestPostProcessed(RequestDetails theRequestDetails, HttpServletRequest theRequest, HttpServletResponse theResponse) throws AuthenticationException;
101 
102 	/**
103 	 * Invoked before an incoming request is processed
104 	 * 
105 	 * @param theServletRequest
106 	 *           The incoming servlet request as provided by the servlet container
107 	 * @param theOperation
108 	 *           The type of operation that the FHIR server has determined that the client is trying to invoke
109 	 * @param theProcessedRequest
110 	 *           An object which will be populated with the details which were extracted from the raw request by the server,
111 	 *           e.g. the FHIR operation type and the parsed resource body (if any).
112 	 */
113 	void incomingRequestPreHandled(RestOperationTypeEnum theOperation, ActionRequestDetails theProcessedRequest);
114 
115 	/**
116 	 * This method is called before any other processing takes place for each incoming request. It may be used to provide alternate handling for some requests, or to screen requests before they are
117 	 * handled, etc.
118 	 * <p>
119 	 * Note that any exceptions thrown by this method will not be trapped by HAPI (they will be passed up to the server)
120 	 * </p>
121 	 * 
122 	 * @param theRequest
123 	 *           The incoming request
124 	 * @param theResponse
125 	 *           The response. Note that interceptors may choose to provide a response (i.e. by calling {@link HttpServletResponse#getWriter()}) but in that case it is important to return
126 	 *           <code>false</code> to indicate that the server itself should not also provide a response.
127 	 * @return Return <code>true</code> if processing should continue normally. This is generally the right thing to do. If your interceptor is providing a response rather than letting HAPI handle the
128 	 *         response normally, you must return <code>false</code>. In this case, no further processing will occur and no further interceptors will be called.
129 	 */
130 	boolean incomingRequestPreProcessed(HttpServletRequest theRequest, HttpServletResponse theResponse);
131 
132 	/**
133 	 * This method is called after the server implementation method has been called, but before any attempt to stream the response back to the client
134 	 * 
135 	 * @param theRequestDetails
136 	 *           A bean containing details about the request that is about to be processed, including
137 	 * @param theResponseObject
138 	 *           The actual object which is being streamed to the client as a response
139 	 * @param theServletRequest
140 	 *           The incoming request
141 	 * @param theServletResponse
142 	 *           The response. Note that interceptors may choose to provide a response (i.e. by calling {@link HttpServletResponse#getWriter()}) but in that case it is important to return
143 	 *           <code>false</code> to indicate that the server itself should not also provide a response.
144 	 * @return Return <code>true</code> if processing should continue normally. This is generally the right thing to do. If your interceptor is providing a response rather than letting HAPI handle the
145 	 *         response normally, you must return <code>false</code>. In this case, no further processing will occur and no further interceptors will be called.
146 	 * @throws AuthenticationException
147 	 *            This exception may be thrown to indicate that the interceptor has detected an unauthorized access attempt. If thrown, processing will stop and an HTTP 401 will be returned to the
148 	 *            client.
149 	 */
150 	boolean outgoingResponse(RequestDetails theRequestDetails, Bundle theResponseObject, HttpServletRequest theServletRequest, HttpServletResponse theServletResponse) throws AuthenticationException;

194 	 * This method is called after the server implementation method has been called, but before any attempt to stream the response back to the client
195 	 * 
196 	 * @param theRequestDetails

214 	/**
215 	 * This method is called upon any exception being thrown within the server's request processing code. This includes any exceptions thrown within resource provider methods (e.g. {@link Search} and
216 	 * {@link Read} methods) as well as any runtime exceptions thrown by the server itself. This method is invoked for each interceptor (until one of them returns a non-<code>null</code> response or
217 	 * the end of the list is reached), after which {@link #handleException(RequestDetails, BaseServerResponseException, HttpServletRequest, HttpServletResponse)} is called for each interceptor.
218 	 * <p>
219 	 * This may be used to add an OperationOutcome to a response, or to convert between exception types for any reason.
220 	 * </p>
221 	 * <p>
222 	 * Implementations of this method may choose to ignore/log/count/etc exceptions, and return <code>null</code>. In this case, processing will continue, and the server will automatically generate an
223 	 * {@link BaseOperationOutcome OperationOutcome}. Implementations may also choose to provide their own response to the client. In this case, they should return a non-<code>null</code>, to indicate
224 	 * that they have handled the request and processing should stop.
225 	 * </p>
226 	 * 
227 	 * @return Returns the new exception to use for processing, or <code>null</code> if this interceptor is not trying to modify the exception. For example, if this interceptor has nothing to do with
228 	 *         exception processing, it should always return <code>null</code>. If this interceptor adds an OperationOutcome to the exception, it should return an exception.
229 	 */
230 	BaseServerResponseException preProcessOutgoingException(RequestDetails theRequestDetails, Throwable theException, HttpServletRequest theServletRequest) throws ServletException;

232 	public static class ActionRequestDetails {
233 		private final IIdType myId;
234 		private boolean myLocked = false;
235 		private IBaseResource myResource;
236 		private final String myResourceType;
237 
238 		public ActionRequestDetails(IIdType theId, String theResourceType) {
239 			myId = theId;
240 			myResourceType = theResourceType;
241 		}
242 
243 		public ActionRequestDetails(RequestDetails theRequestDetails) {
244 			myId = theRequestDetails.getId();
245 			myResourceType = theRequestDetails.getResourceName();
246 		}
247 
248 		public ActionRequestDetails(IIdType theId, String theResourceType, IBaseResource theResource) {
249 			this(theId, theResourceType);
250 			myResource = theResource;
251 		}
252 
253 		/**
254 		 * Returns the ID of the incoming request (typically this is from the request URL)
255 		 */
256 		public IIdType getId() {
257 			return myId;
258 		}
259 
260 		/**
261 		 * For requests where a resource is passed from the client to the server (e.g. create, update, etc.) this method will return the resource which was provided by the client. Otherwise, this method
262 		 * will return <code>null</code>.
263 		 * <p>
264 		 * Note that this method is currently only populated if the handling method has a parameter annotated with the {@link ResourceParam} annotation.
265 		 * </p>
266 		 */
267 		public IBaseResource getResource() {
268 			return myResource;
269 		}
270 
271 		/**
272 		 * Returns the resource type this request pertains to, or <code>null</code> if this request is not type specific (e.g. server-history)
273 		 */
274 		public String getResourceType() {
275 			return myResourceType;
276 		}
277 
278 		/**
279 		 * Prevent any further changes to this object
280 		 */
281 		public void lock() {
282 			myLocked = true;
283 		}
284 
285 		/**
286 		 * This method should not be called by client code
287 		 */
288 		public void setResource(IBaseResource theObject) {
289 			validateNotLocked();
290 			myResource = theObject;
291 		}
292 
293 		private void validateNotLocked() {
294 			if (myLocked) {
295 				throw new IllegalStateException("Values on this object may not be changed");
296 			}
297 		}
298 	}

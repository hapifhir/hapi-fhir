45  /**
46   * Provides methods to intercept requests and responses. Note that implementations of this interface may wish to use
47   * {@link InterceptorAdapter} in order to not need to implement every method.
48   * <p>
49   * <b>See:</b> See the <a href="http://jamesagnew.github.io/hapi-fhir/doc_rest_server_interceptor.html">server
50   * interceptor documentation</a> for more information on how to use this class.
51   * </p>
52   */

55  	/**
56  	 * This method is called upon any exception being thrown within the server's request processing code. This includes
57  	 * any exceptions thrown within resource provider methods (e.g. {@link Search} and {@link Read} methods) as well as
58  	 * any runtime exceptions thrown by the server itself. This also includes any {@link AuthenticationException}s
59  	 * thrown.
60  	 * <p>
61  	 * Implementations of this method may choose to ignore/log/count/etc exceptions, and return <code>true</code>. In
62  	 * this case, processing will continue, and the server will automatically generate an {@link BaseOperationOutcome
63  	 * OperationOutcome}. Implementations may also choose to provide their own response to the client. In this case, they
64  	 * should return <code>false</code>, to indicate that they have handled the request and processing should stop.
65  	 * </p>
66  	 * 
67  	 *
68  	 * @param theRequestDetails
69  	 *           A bean containing details about the request that is about to be processed, including details such as the
70  	 *           resource type and logical ID (if any) and other FHIR-specific aspects of the request which have been
71  	 *           pulled out of the {@link javax.servlet.http.HttpServletRequest servlet request}. Note that the bean
72  	 *           properties are not all guaranteed to be populated, depending on how early during processing the
73  	 *           exception occurred.
74  	 * @param theServletRequest
75  	 *           The incoming request
76  	 * @param theServletResponse
77  	 *           The response. Note that interceptors may choose to provide a response (i.e. by calling
78  	 *           {@link javax.servlet.http.HttpServletResponse#getWriter()}) but in that case it is important to return
79  	 *           <code>false</code> to indicate that the server itself should not also provide a response.
80  	 * @return Return <code>true</code> if processing should continue normally. This is generally the right thing to do.
81  	 *         If your interceptor is providing a response rather than letting HAPI handle the response normally, you
82  	 *         must return <code>false</code>. In this case, no further processing will occur and no further interceptors
83  	 *         will be called.
84  	 * @throws ServletException
85  	 *            If this exception is thrown, it will be re-thrown up to the container for handling.
86  	 * @throws IOException
87  	 *            If this exception is thrown, it will be re-thrown up to the container for handling.
88  	 */
89  	boolean handleException(RequestDetails theRequestDetails, BaseServerResponseException theException, HttpServletRequest theServletRequest, HttpServletResponse theServletResponse) throws ServletException, IOException;

100 	 *           pulled out of the {@link HttpServletRequest servlet request}.

105 	 *           {@link HttpServletResponse#getWriter()}) but in that case it is important to return <code>false</code>

117 	/**
118 	 * Invoked before an incoming request is processed. Note that this method is called
119 	 * after the server has begin preparing the response to the incoming client request. 
120 	 * As such, it is not able to supply a response to the incoming request in the way that
121 	 * {@link #incomingRequestPreHandled(RestOperationTypeEnum, ActionRequestDetails)} and
122 	 * {@link #incomingRequestPostProcessed(RequestDetails, HttpServletRequest, HttpServletResponse)} 
123 	 * are.
124 	 * <p>
125 	 * This method may however throw a subclass of {@link BaseServerResponseException}, and processing
126 	 * will be aborted with an appropriate error returned to the client.
127 	 * </p>
128 	 * 
129 	 * @param theServletRequest
130 	 *           The incoming servlet request as provided by the servlet container
131 	 * @param theOperation
132 	 *           The type of operation that the FHIR server has determined that the client is trying to invoke
133 	 * @param theProcessedRequest
134 	 *           An object which will be populated with the details which were extracted from the raw request by the
135 	 *           server, e.g. the FHIR operation type and the parsed resource body (if any).
136 	 */
137 	void incomingRequestPreHandled(RestOperationTypeEnum theOperation, ActionRequestDetails theProcessedRequest);

341 	/**
342 	 * This method is called upon any exception being thrown within the server's request processing code. This includes
343 	 * any exceptions thrown within resource provider methods (e.g. {@link Search} and {@link Read} methods) as well as
344 	 * any runtime exceptions thrown by the server itself. This method is invoked for each interceptor (until one of them
345 	 * returns a non-<code>null</code> response or the end of the list is reached), after which
346 	 * {@link #handleException(RequestDetails, BaseServerResponseException, HttpServletRequest, HttpServletResponse)} is
347 	 * called for each interceptor.
348 	 * <p>
349 	 * This may be used to add an OperationOutcome to a response, or to convert between exception types for any reason.
350 	 * </p>
351 	 * <p>
352 	 * Implementations of this method may choose to ignore/log/count/etc exceptions, and return <code>null</code>. In
353 	 * this case, processing will continue, and the server will automatically generate an {@link BaseOperationOutcome
354 	 * OperationOutcome}. Implementations may also choose to provide their own response to the client. In this case, they
355 	 * should return a non-<code>null</code>, to indicate that they have handled the request and processing should stop.
356 	 * </p>
357 	 * 
358 	 * @return Returns the new exception to use for processing, or <code>null</code> if this interceptor is not trying to
359 	 *         modify the exception. For example, if this interceptor has nothing to do with exception processing, it
360 	 *         should always return <code>null</code>. If this interceptor adds an OperationOutcome to the exception, it
361 	 *         should return an exception.
362 	 */
363 	BaseServerResponseException preProcessOutgoingException(RequestDetails theRequestDetails, Throwable theException, HttpServletRequest theServletRequest) throws ServletException;

365 	public static class ActionRequestDetails {
366 		private final FhirContext myContext;
367 		private final IIdType myId;
368 		private RequestDetails myRequestDetails;
369 		private IBaseResource myResource;
370 		private final String myResourceType;
371 
372 		public ActionRequestDetails(IIdType theId, String theResourceType, FhirContext theContext, RequestDetails theRequestDetails) {
373 			myId = theId;
374 			myResourceType = theResourceType;
375 			myContext = theContext;
376 			myRequestDetails = theRequestDetails;
377 		}
378 
379 		public ActionRequestDetails(IIdType theId, String theResourceType, IBaseResource theResource, FhirContext theContext, RequestDetails theRequestDetails) {
380 			this(theId, theResourceType, theContext, theRequestDetails);
381 			myResource = theResource;
382 		}
383 
384 		public ActionRequestDetails(RequestDetails theRequestDetails) {
385 			myId = theRequestDetails.getId();
386 			myResourceType = theRequestDetails.getResourceName();
387 			myContext = theRequestDetails.getServer().getFhirContext();
388 			myRequestDetails = theRequestDetails;
389 		}
390 
391 		public FhirContext getContext() {
392 			return myContext;
393 		}
394 
395 		/**
396 		 * Returns the ID of the incoming request (typically this is from the request URL)
397 		 */
398 		public IIdType getId() {
399 			return myId;
400 		}
401 
402 		/**
403 		 * For requests where a resource is passed from the client to the server (e.g. create, update, etc.) this method
404 		 * will return the resource which was provided by the client. Otherwise, this method will return <code>null</code>
405 		 * .
406 		 * <p>
407 		 * Note that this method is currently only populated if the handling method has a parameter annotated with the
408 		 * {@link ResourceParam} annotation.
409 		 * </p>
410 		 */
411 		public IBaseResource getResource() {
412 			return myResource;
413 		}
414 
415 		/**
416 		 * Returns the resource type this request pertains to, or <code>null</code> if this request is not type specific
417 		 * (e.g. server-history)
418 		 */
419 		public String getResourceType() {
420 			return myResourceType;
421 		}
422 
423 		/**
424 		 * Returns the same map which was 
425 		 */
426 		public Map<Object, Object> getUserData() {
427 			return myRequestDetails.getUserData();
428 		}
429 
430 		/**
431 		 * This method should not be called by client code
432 		 */
433 		public void setResource(IBaseResource theObject) {
434 			myResource = theObject;
435 		}

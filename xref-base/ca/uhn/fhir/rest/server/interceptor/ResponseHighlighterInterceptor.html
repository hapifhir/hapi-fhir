48  /**
49   * This interceptor detects when a request is coming from a browser, and automatically returns a response with syntax
50   * highlighted (coloured) HTML for the response instead of just returning raw XML/JSON.
51   * 
52   * @since 1.0
53   */
54  public class ResponseHighlighterInterceptor extends InterceptorAdapter {
55  
56  	private static final org.slf4j.Logger ourLog = org.slf4j.LoggerFactory.getLogger(ResponseHighlighterInterceptor.class);
57  	private static final String[] PARAM_FORMAT_VALUE_JSON = new String[] { Constants.FORMAT_JSON };
58  	private static final String[] PARAM_FORMAT_VALUE_XML = new String[] { Constants.FORMAT_XML };
59  
60  	/**
61  	 * TODO: As of HAPI 1.6 (2016-06-10) this parameter has been replaced with simply
62  	 * requesting _format=json or xml so eventually this parameter should be removed
63  	 */
64  	public static final String PARAM_RAW = "_raw";

66  	public static final String PARAM_RAW_TRUE = "true";

69  	private String format(String theResultBody, EncodingEnum theEncodingEnum) {
70  		String str = StringEscapeUtils.escapeHtml4(theResultBody);
71  		if (str == null || theEncodingEnum == null) {
72  			return str;
73  		}
74  
75  		StringBuilder b = new StringBuilder();
76  
77  		if (theEncodingEnum == EncodingEnum.JSON) {
78  
79  			boolean inValue = false;
80  			boolean inQuote = false;
81  			for (int i = 0; i < str.length(); i++) {
82  				char prevChar = (i > 0) ? str.charAt(i - 1) : ' ';
83  				char nextChar = str.charAt(i);
84  				char nextChar2 = (i + 1) < str.length() ? str.charAt(i + 1) : ' ';
85  				char nextChar3 = (i + 2) < str.length() ? str.charAt(i + 2) : ' ';
86  				char nextChar4 = (i + 3) < str.length() ? str.charAt(i + 3) : ' ';
87  				char nextChar5 = (i + 4) < str.length() ? str.charAt(i + 4) : ' ';
88  				char nextChar6 = (i + 5) < str.length() ? str.charAt(i + 5) : ' ';
89  				if (inQuote) {
90  					b.append(nextChar);
91  					if (prevChar != '\\' && nextChar == '&' && nextChar2 == 'q' && nextChar3 == 'u' && nextChar4 == 'o' && nextChar5 == 't' && nextChar6 == ';') {
92  						b.append("quot;</span>");
93  						i += 5;
94  						inQuote = false;
95  					} else if (nextChar == '\\' && nextChar2 == '"') {
96  						b.append("quot;</span>");
97  						i += 5;
98  						inQuote = false;
99  					}
100 				} else {
101 					if (nextChar == ':') {
102 						inValue = true;
103 						b.append(nextChar);
104 					} else if (nextChar == '[' || nextChar == '{') {
105 						b.append("<span class='hlControl'>");
106 						b.append(nextChar);
107 						b.append("</span>");
108 						inValue = false;
109 					} else if (nextChar == '}' || nextChar == '}' || nextChar == ',') {
110 						b.append("<span class='hlControl'>");
111 						b.append(nextChar);
112 						b.append("</span>");
113 						inValue = false;
114 					} else if (nextChar == '&' && nextChar2 == 'q' && nextChar3 == 'u' && nextChar4 == 'o' && nextChar5 == 't' && nextChar6 == ';') {
115 						if (inValue) {
116 							b.append("<span class='hlQuot'>&quot;");
117 						} else {
118 							b.append("<span class='hlTagName'>&quot;");
119 						}
120 						inQuote = true;
121 						i += 5;
122 					} else if (nextChar == ':') {
123 						b.append("<span class='hlControl'>");
124 						b.append(nextChar);
125 						b.append("</span>");
126 						inValue = true;
127 					} else {
128 						b.append(nextChar);
129 					}
130 				}
131 			}
132 
133 		} else {
134 			boolean inQuote = false;
135 			boolean inTag = false;
136 			for (int i = 0; i < str.length(); i++) {
137 				char nextChar = str.charAt(i);
138 				char nextChar2 = (i + 1) < str.length() ? str.charAt(i + 1) : ' ';
139 				char nextChar3 = (i + 2) < str.length() ? str.charAt(i + 2) : ' ';
140 				char nextChar4 = (i + 3) < str.length() ? str.charAt(i + 3) : ' ';
141 				char nextChar5 = (i + 4) < str.length() ? str.charAt(i + 4) : ' ';
142 				char nextChar6 = (i + 5) < str.length() ? str.charAt(i + 5) : ' ';
143 				if (inQuote) {
144 					b.append(nextChar);
145 					if (nextChar == '&' && nextChar2 == 'q' && nextChar3 == 'u' && nextChar4 == 'o' && nextChar5 == 't' && nextChar6 == ';') {
146 						b.append("quot;</span>");
147 						i += 5;
148 						inQuote = false;
149 					}
150 				} else if (inTag) {
151 					if (nextChar == '&' && nextChar2 == 'g' && nextChar3 == 't' && nextChar4 == ';') {
152 						b.append("</span><span class='hlControl'>&gt;</span>");
153 						inTag = false;
154 						i += 3;
155 					} else if (nextChar == ' ') {
156 						b.append("</span><span class='hlAttr'>");
157 						b.append(nextChar);
158 					} else if (nextChar == '&' && nextChar2 == 'q' && nextChar3 == 'u' && nextChar4 == 'o' && nextChar5 == 't' && nextChar6 == ';') {
159 						b.append("<span class='hlQuot'>&quot;");
160 						inQuote = true;
161 						i += 5;
162 					} else {
163 						b.append(nextChar);
164 					}
165 				} else {
166 					if (nextChar == '&' && nextChar2 == 'l' && nextChar3 == 't' && nextChar4 == ';') {
167 						b.append("<span class='hlControl'>&lt;</span><span class='hlTagName'>");
168 						inTag = true;
169 						i += 3;
170 					} else {
171 						b.append(nextChar);
172 					}
173 				}
174 			}
175 		}
176 
177 		return b.toString();
178 	}

180 	@Override
181 	public boolean handleException(RequestDetails theRequestDetails, BaseServerResponseException theException, HttpServletRequest theServletRequest, HttpServletResponse theServletResponse) throws ServletException, IOException {
182 		/*
183 		 * It's not a browser...
184 		 */
185 		Set<String> accept = RestfulServerUtils.parseAcceptHeaderAndReturnHighestRankedOptions(theServletRequest);
186 		if (!accept.contains(Constants.CT_HTML)) {
187 			return super.handleException(theRequestDetails, theException, theServletRequest, theServletResponse);
188 		}
189 
190 		/*
191 		 * It's an AJAX request, so no HTML
192 		 */
193 		String requestedWith = theServletRequest.getHeader("X-Requested-With");
194 		if (requestedWith != null) {
195 			return super.handleException(theRequestDetails, theException, theServletRequest, theServletResponse);
196 		}
197 
198 		/*
199 		 * Not a GET
200 		 */
201 		if (theRequestDetails.getRequestType() != RequestTypeEnum.GET) {
202 			return super.handleException(theRequestDetails, theException, theServletRequest, theServletResponse);
203 		}
204 
205 		if (theException.getOperationOutcome() == null) {
206 			return super.handleException(theRequestDetails, theException, theServletRequest, theServletResponse);
207 		}
208 
209 		streamResponse(theRequestDetails, theServletResponse, theException.getOperationOutcome());
210 
211 		return false;
212 	}

214 	@Override
215 	public boolean outgoingResponse(RequestDetails theRequestDetails, IBaseResource theResponseObject, HttpServletRequest theServletRequest, HttpServletResponse theServletResponse) throws AuthenticationException {
216 
217 		/*
218 		 * Request for _raw
219 		 */
220 		String[] rawParamValues = theRequestDetails.getParameters().get(PARAM_RAW);
221 		if (rawParamValues != null && rawParamValues.length > 0 && rawParamValues[0].equals(PARAM_RAW_TRUE)) {
222 			ourLog.warn("Client is using non-standard/legacy  _raw parameter - Use _format=json or _format=xml instead, as this parmameter will be removed at some point");
223 			return super.outgoingResponse(theRequestDetails, theResponseObject, theServletRequest, theServletResponse);
224 		}
225 		
226 		boolean force = false;
227 		String[] formatParams = theRequestDetails.getParameters().get(Constants.PARAM_FORMAT);
228 		if (formatParams != null && formatParams.length > 0) {
229 			String formatParam = formatParams[0];
230 			if (Constants.FORMATS_HTML.contains(formatParam)) { // this is a set
231 				force = true;
232 			} else if (Constants.FORMATS_HTML_XML.equals(formatParam)) {
233 				force = true;
234 				theRequestDetails.getParameters().put(Constants.PARAM_FORMAT, PARAM_FORMAT_VALUE_XML);
235 			} else if (Constants.FORMATS_HTML_JSON.equals(formatParam)) {
236 				force = true;
237 				theRequestDetails.getParameters().put(Constants.PARAM_FORMAT, PARAM_FORMAT_VALUE_JSON);
238 			} else {
239 				return super.outgoingResponse(theRequestDetails, theResponseObject, theServletRequest, theServletResponse);
240 			}
241 		}
242 		
243 		/*
244 		 * It's not a browser...
245 		 */
246 		Set<String> highestRankedAcceptValues = RestfulServerUtils.parseAcceptHeaderAndReturnHighestRankedOptions(theServletRequest);
247 		if (!force && highestRankedAcceptValues.contains(Constants.CT_HTML) == false) {
248 			return super.outgoingResponse(theRequestDetails, theResponseObject, theServletRequest, theServletResponse);
249 		}
250 
251 		/*
252 		 * It's an AJAX request, so no HTML
253 		 */
254 		if (!force && isNotBlank(theServletRequest.getHeader("X-Requested-With"))) {
255 			return super.outgoingResponse(theRequestDetails, theResponseObject, theServletRequest, theServletResponse);
256 		}
257 		/*
258 		 * If the request has an Origin header, it is probably an AJAX request
259 		 */
260 		if (!force && isNotBlank(theServletRequest.getHeader(Constants.HEADER_ORIGIN))) {
261 			return super.outgoingResponse(theRequestDetails, theResponseObject, theServletRequest, theServletResponse);
262 		}
263 		
264 
265 		/*
266 		 * Not a GET
267 		 */
268 		if (!force && theRequestDetails.getRequestType() != RequestTypeEnum.GET) {
269 			return super.outgoingResponse(theRequestDetails, theResponseObject, theServletRequest, theServletResponse);
270 		}
271 
272 		/*
273 		 * Not binary
274 		 */
275 		if (!force && "Binary".equals(theRequestDetails.getResourceName())) {
276 			return super.outgoingResponse(theRequestDetails, theResponseObject, theServletRequest, theServletResponse);
277 		}
278 
279 		streamResponse(theRequestDetails, theServletResponse, theResponseObject);
280 
281 		return false;
282 	}

284 	private void streamResponse(RequestDetails theRequestDetails, HttpServletResponse theServletResponse, IBaseResource resource) {
285 		IParser p;
286 		Map<String, String[]> parameters = theRequestDetails.getParameters();
287 		if (parameters.containsKey(Constants.PARAM_FORMAT)) {
288 			p = RestfulServerUtils.getNewParser(theRequestDetails.getServer().getFhirContext(), theRequestDetails);
289 		} else {
290 			EncodingEnum defaultResponseEncoding = theRequestDetails.getServer().getDefaultResponseEncoding();
291 			p = defaultResponseEncoding.newParser(theRequestDetails.getServer().getFhirContext());
292 			RestfulServerUtils.configureResponseParser(theRequestDetails, p);
293 		}
294 
295 		EncodingEnum encoding = p.getEncoding();
296 		String encoded = p.encodeResourceToString(resource);
297 
298 		theServletResponse.setContentType(Constants.CT_HTML_WITH_UTF8);
299 
300 		StringBuilder b = new StringBuilder();
301 		b.append("<html lang=\"en\">\n");
302 		b.append("	<head>\n");
303 		b.append("		<meta charset=\"utf-8\" />\n");
304 		b.append("       <style>\n");
305 		b.append(".hlQuot {\n");
306 		b.append("  color: #88F;\n");
307 		b.append("}\n");
308 		b.append(".hlAttr {\n");
309 		b.append("  color: #888;\n");
310 		b.append("}\n");
311 		b.append(".hlTagName {\n");
312 		b.append("  color: #006699;\n");
313 		b.append("}\n");
314 		b.append(".hlControl {\n");
315 		b.append("  color: #660000;\n");
316 		b.append("}\n");
317 		b.append(".hlText {\n");
318 		b.append("  color: #000000;\n");
319 		b.append("}\n");
320 		b.append(".hlUrlBase {\n");
321 		b.append("}");
322 		b.append(".headersDiv {\n");
323 		b.append("  background: #EEE;");
324 		b.append("}");
325 		b.append(".headerName {\n");
326 		b.append("  color: #888;\n");
327 		b.append("  font-family: monospace;\n");
328 		b.append("}");
329 		b.append(".headerValue {\n");
330 		b.append("  color: #88F;\n");
331 		b.append("  font-family: monospace;\n");
332 		b.append("}");
333 		b.append("BODY {\n");
334 		b.append("  font-family: Arial;\n");
335 		b.append("}");
336 		b.append("       </style>\n");
337 		b.append("	</head>\n");
338 		b.append("\n");
339 		b.append("	<body>");
340 		
341 		b.append("<p>");
342 		b.append("This result is being rendered in HTML for easy viewing. ");
343 		b.append("You may access this content as ");
344 		
345 		b.append("<a href=\"");
346 		b.append(createLinkHref(parameters, Constants.FORMAT_JSON));
347 		b.append("\">Raw JSON</a> or ");
348 
349 		b.append("<a href=\"");
350 		b.append(createLinkHref(parameters, Constants.FORMAT_XML));
351 		b.append("\">Raw XML</a>, ");
352 
353 		b.append(" or view this content in ");
354 		
355 		b.append("<a href=\"");
356 		b.append(createLinkHref(parameters, Constants.FORMATS_HTML_JSON));
357 		b.append("\">HTML JSON</a> ");
358 
359 		b.append("or ");
360 		b.append("<a href=\"");
361 		b.append(createLinkHref(parameters, Constants.FORMATS_HTML_XML));
362 		b.append("\">HTML XML</a>.");
363 		b.append("</p>");
364 		
365 		b.append("\n");
366 		
367 		//		if (isEncodeHeaders()) {
368 		//			b.append("<h1>Request Headers</h1>");
369 		//			b.append("<div class=\"headersDiv\">");
370 		//			for (int next : theRequestDetails.get)
371 		//			b.append("</div>");
372 		//			b.append("<h1>Response Headers</h1>");
373 		//			b.append("<div class=\"headersDiv\">");
374 		//			b.append("</div>");
375 		//			b.append("<h1>Response Body</h1>");
376 		//		}
377 		b.append("<pre>");
378 		b.append(format(encoded, encoding));
379 		b.append("</pre>");
380 		b.append("   </body>");
381 		b.append("</html>");
382 		//@formatter:off
383 		String out = b.toString();
384 		//@formatter:on
385 
386 		try {
387 			theServletResponse.getWriter().append(out);
388 			theServletResponse.getWriter().close();
389 		} catch (IOException e) {
390 			throw new InternalErrorException(e);
391 		}
392 	}

394 	private String createLinkHref(Map<String, String[]> parameters, String formatValue) {
395 		StringBuilder rawB = new StringBuilder();
396 		for (String next : parameters.keySet()) {
397 			if (Constants.PARAM_FORMAT.equals(next)) {
398 				continue;
399 			}
400 			for (String nextValue : parameters.get(next)) {
401 				if (isBlank(nextValue)) {
402 					continue;
403 				}
404 				if (rawB.length() == 0) {
405 					rawB.append('?');
406 				} else {
407 					rawB.append('&');
408 				}
409 				rawB.append(UrlUtil.escape(next));
410 				rawB.append('=');
411 				rawB.append(UrlUtil.escape(nextValue));
412 			}
413 		}
414 		if (rawB.length() == 0) {
415 			rawB.append('?');
416 		} else {
417 			rawB.append('&');
418 		}
419 		rawB.append(Constants.PARAM_FORMAT).append('=').append(formatValue);
420 
421 		String link = rawB.toString();
422 		return link;
423 	}

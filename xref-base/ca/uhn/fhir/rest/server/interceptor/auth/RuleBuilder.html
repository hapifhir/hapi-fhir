231 			private class RuleBuilderRuleOpClassifier implements IAuthRuleBuilderRuleOpClassifier {
232 
233 				private ClassifierTypeEnum myClassifierType;
234 				private String myInCompartmentName;
235 				private Collection<? extends IIdType> myInCompartmentOwners;
236 
237 				private IAuthRuleBuilderRuleOpClassifierFinished finished() {
238 
239 					RuleImplOp rule = new RuleImplOp(myRuleName);
240 					rule.setMode(myRuleMode);
241 					rule.setOp(myRuleOp);
242 					rule.setAppliesTo(myAppliesTo);
243 					rule.setAppliesToTypes(myAppliesToTypes);
244 					rule.setClassifierType(myClassifierType);
245 					rule.setClassifierCompartmentName(myInCompartmentName);
246 					rule.setClassifierCompartmentOwners(myInCompartmentOwners);
247 					myRules.add(rule);
248 
249 					return new RuleBuilderFinished();
250 				}
251 
252 				@Override
253 				public IAuthRuleBuilderRuleOpClassifierFinished inCompartment(String theCompartmentName, Collection<? extends IIdType> theOwners) {
254 					Validate.notBlank(theCompartmentName, "theCompartmentName must not be null");
255 					Validate.notNull(theOwners, "theOwners must not be null");
256 					Validate.noNullElements(theOwners, "theOwners must not contain any null elements");
257 					for (IIdType next : theOwners) {
258 						validateOwner(next);
259 					}
260 					myClassifierType = ClassifierTypeEnum.IN_COMPARTMENT;
261 					return finished();
262 				}
263 
264 				@Override
265 				public IAuthRuleBuilderRuleOpClassifierFinished inCompartment(String theCompartmentName, IIdType theOwner) {
266 					Validate.notBlank(theCompartmentName, "theCompartmentName must not be null");
267 					Validate.notNull(theOwner, "theOwner must not be null");
268 					validateOwner(theOwner);
269 					myInCompartmentName = theCompartmentName;
270 					myInCompartmentOwners = Collections.singletonList(theOwner);
271 					myClassifierType = ClassifierTypeEnum.IN_COMPARTMENT;
272 					return finished();
273 				}
274 
275 				private void validateOwner(IIdType theOwner) {
276 					Validate.notBlank(theOwner.getIdPart(), "owner.getIdPart() must not be null or empty");
277 					Validate.notBlank(theOwner.getIdPart(), "owner.getResourceType() must not be null or empty");
278 				}
279 
280 				@Override
281 				public IAuthRuleBuilderRuleOpClassifierFinished withAnyId() {
282 					myClassifierType = ClassifierTypeEnum.ANY_ID;
283 					return finished();
284 				}
285 
286 			}

292 			@Override
293 			public IAuthRuleBuilderOperationNamed named(String theOperationName) {
294 				Validate.notBlank(theOperationName, "theOperationName must not be null or empty");
295 				return new RuleBuilderRuleOperationNamed(theOperationName);
296 			}
297 
298 			@Override
299 			public IAuthRuleBuilderOperationNamed withAnyName() {
300 				return new RuleBuilderRuleOperationNamed(null);
301 			}
302 
303 			private class RuleBuilderRuleOperationNamed implements IAuthRuleBuilderOperationNamed {
304 
305 				private String myOperationName;
306 
307 				public RuleBuilderRuleOperationNamed(String theOperationName) {
308 					if (theOperationName != null && !theOperationName.startsWith("$")) {
309 						myOperationName = '$' + theOperationName;
310 					} else {
311 						myOperationName = theOperationName;
312 					}
313 				}
314 
315 				private OperationRule createRule() {
316 					OperationRule rule = new OperationRule(myRuleName);
317 					rule.setOperationName(myOperationName);
318 					rule.setMode(myRuleMode);
319 					return rule;
320 				}
321 
322 				@Override
323 				public IAuthRuleBuilderRuleOpClassifierFinished onInstance(IIdType theInstanceId) {
324 					Validate.notNull(theInstanceId, "theInstanceId must not be null");
325 					Validate.notBlank(theInstanceId.getResourceType(), "theInstanceId does not have a resource type");
326 					Validate.notBlank(theInstanceId.getIdPart(), "theInstanceId does not have an ID part");
327 
328 					OperationRule rule = createRule();
329 					ArrayList<IIdType> ids = new ArrayList<IIdType>();
330 					ids.add(theInstanceId);
331 					rule.appliesToInstances(ids);
332 					myRules.add(rule);
333 					return new RuleBuilderFinished();
334 				}
335 
336 				@Override
337 				public IAuthRuleBuilderRuleOpClassifierFinished onServer() {
338 					OperationRule rule = createRule();
339 					rule.appliesToServer();
340 					myRules.add(rule);
341 					return new RuleBuilderFinished();
342 				}
343 
344 				@Override
345 				public IAuthRuleBuilderRuleOpClassifierFinished onType(Class<? extends IBaseResource> theType) {
346 					Validate.notNull(theType, "theType must not be null");
347 
348 					OperationRule rule = createRule();
349 					HashSet<Class<? extends IBaseResource>> appliesToTypes = new HashSet<Class<? extends IBaseResource>>();
350 					appliesToTypes.add(theType);
351 					rule.appliesToTypes(appliesToTypes);
352 					myRules.add(rule);
353 					return new RuleBuilderFinished();
354 				}
355 
356 			}
357

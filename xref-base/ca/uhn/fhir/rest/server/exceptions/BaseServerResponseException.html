34  /**
35   * Base class for RESTful client and server exceptions. RESTful client methods will only throw exceptions which are subclasses of this exception type, and RESTful server methods should also only call
36   * subclasses of this exception type.
37   * <p>
38   * HAPI provides a number of subclasses of BaseServerResponseException, and each one corresponds to a specific
39   * HTTP status code. For example, if a {@link IResourceProvider resource provider} method throws 
40   * {@link ResourceNotFoundException}, this is a signal to the server that an <code>HTTP 404</code> should
41   * be returned to the client.
42   * </p>
43   * <p>
44   * <b>See:</b> A complete list of available exceptions is in the <a href="./package-summary.html">package summary</a>.
45   * If an exception doesn't exist for a condition you want to represent, let us know by filing an
46   * <a href="https://github.com/jamesagnew/hapi-fhir/issues">issue in our tracker</a>. You may also
47   * use {@link UnclassifiedServerFailureException} to represent any error code you want.
48   * </p>
49   */
50  public abstract class BaseServerResponseException extends RuntimeException {

99  		super(theMessages != null && theMessages.length > 0 ? theMessages[0] : null);

265 	public static BaseServerResponseException newInstance(int theStatusCode, String theMessage) {
266 		if (ourStatusCodeToExceptionType.containsKey(theStatusCode)) {
267 			try {
268 				return ourStatusCodeToExceptionType.get(theStatusCode).getConstructor(new Class[] { String.class }).newInstance(theMessage);
269 			} catch (InstantiationException e) {
270 				throw new InternalErrorException(e);
271 			} catch (IllegalAccessException e) {
272 				throw new InternalErrorException(e);
273 			} catch (IllegalArgumentException e) {
274 				throw new InternalErrorException(e);
275 			} catch (InvocationTargetException e) {
276 				throw new InternalErrorException(e);
277 			} catch (NoSuchMethodException e) {
278 				throw new InternalErrorException(e);
279 			} catch (SecurityException e) {
280 				throw new InternalErrorException(e);
281 			}
282 		} else {
283 			return new UnclassifiedServerFailureException(theStatusCode, theMessage);
284 		}
285 	}
286 
287 	static void registerExceptionType(int theStatusCode, Class<? extends BaseServerResponseException> theType) {
288 		if (ourStatusCodeToExceptionType.containsKey(theStatusCode)) {
289 			throw new Error("Can not register " + theType + " to status code " + theStatusCode + " because " + ourStatusCodeToExceptionType.get(theStatusCode) + " already registers that code");
290 		}
291 		ourStatusCodeToExceptionType.put(theStatusCode, theType);
292 	}

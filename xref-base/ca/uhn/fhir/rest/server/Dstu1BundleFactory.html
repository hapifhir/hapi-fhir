65  	@Override
66  	public void addResourcesToBundle(List<IBaseResource> theResult, BundleTypeEnum theBundleType, String theServerBase, BundleInclusionRule theBundleInclusionRule, Set<Include> theIncludes) {
67  		if (myBundle == null) {
68  			myBundle = new Bundle();
69  		}
70  		
71  		List<IResource> includedResources = new ArrayList<IResource>();
72  		Set<IdDt> addedResourceIds = new HashSet<IdDt>();
73  	
74  		for (IBaseResource next : theResult) {
75  			if (next.getIdElement().isEmpty() == false) {
76  				addedResourceIds.add((IdDt) next.getIdElement());
77  			}
78  		}
79  	
80  		for (IBaseResource nextBaseRes : theResult) {
81  			IResource next = (IResource)nextBaseRes;
82  			
83  			Set<String> containedIds = new HashSet<String>();
84  			for (IResource nextContained : next.getContained().getContainedResources()) {
85  				if (nextContained.getId().isEmpty() == false) {
86  					containedIds.add(nextContained.getId().getValue());
87  				}
88  			}
89  	
90  			if (myContext.getNarrativeGenerator() != null) {
91  				String title = myContext.getNarrativeGenerator().generateTitle(next);
92  				ourLog.trace("Narrative generator created title: {}", title);
93  				if (StringUtils.isNotBlank(title)) {
94  					ResourceMetadataKeyEnum.TITLE.put(next, title);
95  				}
96  			} else {
97  				ourLog.trace("No narrative generator specified");
98  			}
99  
100             List<ResourceReferenceInfo> references = myContext.newTerser().getAllResourceReferences(next);
101 			do {
102 				List<IResource> addedResourcesThisPass = new ArrayList<IResource>();
103 
104                 for (ResourceReferenceInfo nextRefInfo : references) {
105                     if (!theBundleInclusionRule.shouldIncludeReferencedResource(nextRefInfo, theIncludes))
106                         continue;
107 
108 					IResource nextRes = (IResource) nextRefInfo.getResourceReference().getResource();
109 					if (nextRes != null) {
110 						if (nextRes.getId().hasIdPart()) {
111 							if (containedIds.contains(nextRes.getId().getValue())) {
112 								// Don't add contained IDs as top level resources
113 								continue;
114 							}
115 	
116 							IdDt id = nextRes.getId();
117 							if (id.hasResourceType() == false) {
118 								String resName = myContext.getResourceDefinition(nextRes).getName();
119 								id = id.withResourceType(resName);
120 							}
121 	
122 							if (!addedResourceIds.contains(id)) {
123 								addedResourceIds.add(id);
124 								addedResourcesThisPass.add(nextRes);
125 							}
126 	
127 						}
128 					}
129 				}
130 
131                 includedResources.addAll(addedResourcesThisPass);
132 
133 				// Linked resources may themselves have linked resources
134 				references = new ArrayList<ResourceReferenceInfo>();
135 				for (IResource iResource : addedResourcesThisPass) {
136 					List<ResourceReferenceInfo> newReferences = myContext.newTerser().getAllResourceReferences(iResource);
137 					references.addAll(newReferences);
138 				}
139 			} while (references.isEmpty() == false);
140 	
141 			myBundle.addResource(next, myContext, theServerBase);
142 		}
143 	
144 		/*
145 		 * Actually add the resources to the bundle
146 		 */
147 		for (IResource next : includedResources) {
148 			BundleEntry entry = myBundle.addResource(next, myContext, theServerBase);
149 			if (myContext.getVersion().getVersion().isNewerThan(FhirVersionEnum.DSTU1)) {
150 				if (entry.getSearchMode().isEmpty()) {
151 					entry.getSearchMode().setValueAsEnum(BundleEntrySearchModeEnum.INCLUDE);
152 				}
153 			}
154 		}
155 	
156 	}

158     @Override
159 	public void addRootPropertiesToBundle(String theAuthor, String theServerBase, String theCompleteUrl, Integer theTotalResults, BundleTypeEnum theBundleType, IPrimitiveType<Date> theLastUpdated) {
160 		if (myBundle.getAuthorName().isEmpty()) {
161 			myBundle.getAuthorName().setValue(theAuthor);
162 		}
163 
164 		if (myBundle.getUpdated().isEmpty() && isNotBlank(theLastUpdated.getValueAsString())) {
165 			myBundle.getUpdated().setValueAsString(theLastUpdated.getValueAsString());
166 		}
167 		
168 		if (myBundle.getBundleId().isEmpty()) {
169 			myBundle.getBundleId().setValue(UUID.randomUUID().toString());
170 		}
171 		
172 		if (myBundle.getLinkBase().isEmpty()) {
173 			myBundle.getLinkBase().setValue(theServerBase);
174 		}
175 		
176 		if (myBundle.getLinkSelf().isEmpty()) {
177 			myBundle.getLinkSelf().setValue(theCompleteUrl);
178 		}
179 		
180 		if (theBundleType != null && myBundle.getType().isEmpty()) {
181 			myBundle.getType().setValueAsString(theBundleType.getCode());
182 		}
183 
184 		if (myBundle.getTotalResults().isEmpty() && theTotalResults != null) {
185 			myBundle.getTotalResults().setValue(theTotalResults);
186 		}
187 	}

199 	@Override
200 	public void initializeBundleFromBundleProvider(RestfulServer theServer, IBundleProvider theResult, EncodingEnum theResponseEncoding, String theServerBase, String theCompleteUrl, boolean thePrettyPrint, int theOffset, Integer theLimit, String theSearchId, BundleTypeEnum theBundleType, Set<Include> theIncludes) {
201 		int numToReturn;
202 		String searchId = null;
203 		List<IBaseResource> resourceList;
204 		if (theServer.getPagingProvider() == null) {
205 			numToReturn = theResult.size();
206 			resourceList = theResult.getResources(0, numToReturn);
207 			RestfulServerUtils.validateResourceListNotNull(resourceList);
208 	
209 		} else {
210 			IPagingProvider pagingProvider = theServer.getPagingProvider();
211 			if (theLimit == null) {
212 				numToReturn = pagingProvider.getDefaultPageSize();
213 			} else {
214 				numToReturn = Math.min(pagingProvider.getMaximumPageSize(), theLimit);
215 			}
216 
217 			numToReturn = Math.min(numToReturn, theResult.size() - theOffset);
218 			resourceList = theResult.getResources(theOffset, numToReturn + theOffset);
219 			RestfulServerUtils.validateResourceListNotNull(resourceList);
220 	
221 			if (theSearchId != null) {
222 				searchId = theSearchId;
223 			} else {
224 				if (theResult.size() > numToReturn) {
225 					searchId = pagingProvider.storeResultList(theResult);
226 					Validate.notNull(searchId, "Paging provider returned null searchId");
227 				}
228 			}
229 		}
230 	
231 		for (IBaseResource next : resourceList) {
232 			if (next.getIdElement() == null || next.getIdElement().isEmpty()) {
233 				if (!(next instanceof BaseOperationOutcome)) {
234 					throw new InternalErrorException("Server method returned resource of type[" + next.getClass().getSimpleName() + "] with no ID specified (IResource#setId(IdDt) must be called)");
235 				}
236 			}
237 		}
238 	
239 		if (theServer.getAddProfileTag() != AddProfileTagEnum.NEVER) {
240 			for (IBaseResource nextRes : resourceList) {
241 				RuntimeResourceDefinition def = theServer.getFhirContext().getResourceDefinition(nextRes);
242 				if (theServer.getAddProfileTag() == AddProfileTagEnum.ALWAYS || !def.isStandardProfile()) {
243 					RestfulServerUtils.addProfileToBundleEntry(theServer.getFhirContext(), nextRes, theServerBase);
244 				}
245 			}
246 		}
247 	
248 		addResourcesToBundle(new ArrayList<IBaseResource>(resourceList), theBundleType, theServerBase, theServer.getBundleInclusionRule(), theIncludes);
249 		addRootPropertiesToBundle(null, theServerBase, theCompleteUrl, theResult.size(), theBundleType, theResult.getPublished());
250 
251 		if (theServer.getPagingProvider() != null) {
252 			int limit;
253 			limit = theLimit != null ? theLimit : theServer.getPagingProvider().getDefaultPageSize();
254 			limit = Math.min(limit, theServer.getPagingProvider().getMaximumPageSize());
255 	
256 			if (searchId != null) {
257 				if (theOffset + numToReturn < theResult.size()) {
258 					myBundle.getLinkNext().setValue(RestfulServerUtils.createPagingLink(theIncludes, theServerBase, searchId, theOffset + numToReturn, numToReturn, theResponseEncoding, thePrettyPrint));
259 				}
260 				if (theOffset > 0) {
261 					int start = Math.max(0, theOffset - limit);
262 					myBundle.getLinkPrevious().setValue(RestfulServerUtils.createPagingLink(theIncludes, theServerBase, searchId, start, limit, theResponseEncoding, thePrettyPrint));
263 				}
264 			}
265 		}
266 	}

268 	@Override
269 	public void initializeBundleFromResourceList(String theAuthor, List<? extends IBaseResource> theResult, String theServerBase, String theCompleteUrl, int theTotalResults, BundleTypeEnum theBundleType) {
270 		myBundle = new Bundle();
271 		
272 		myBundle.getAuthorName().setValue(theAuthor);
273 		myBundle.getBundleId().setValue(UUID.randomUUID().toString());
274 		myBundle.getLinkBase().setValue(theServerBase);
275 		myBundle.getLinkSelf().setValue(theCompleteUrl);
276 		myBundle.getType().setValueAsEnum(theBundleType);
277 	
278 		List<IBaseResource> includedResources = new ArrayList<IBaseResource>();
279 		Set<IIdType> addedResourceIds = new HashSet<IIdType>();
280 	
281 		for (IBaseResource next : theResult) {
282 			if (next.getIdElement().isEmpty() == false) {
283 				addedResourceIds.add(next.getIdElement());
284 			}
285 		}
286 	
287 		for (IBaseResource nextRes : theResult) {
288 			IResource next = (IResource) nextRes;
289 			
290 			Set<String> containedIds = new HashSet<String>();
291 			for (IResource nextContained : next.getContained().getContainedResources()) {
292 				if (nextContained.getId().isEmpty() == false) {
293 					containedIds.add(nextContained.getId().getValue());
294 				}
295 			}
296 	
297 			if (myContext.getNarrativeGenerator() != null) {
298 				String title = myContext.getNarrativeGenerator().generateTitle(next);
299 				ourLog.trace("Narrative generator created title: {}", title);
300 				if (StringUtils.isNotBlank(title)) {
301 					ResourceMetadataKeyEnum.TITLE.put(next, title);
302 				}
303 			} else {
304 				ourLog.trace("No narrative generator specified");
305 			}
306 	
307 			List<BaseResourceReferenceDt> references = myContext.newTerser().getAllPopulatedChildElementsOfType(next, BaseResourceReferenceDt.class);
308 			do {
309 				List<IBaseResource> addedResourcesThisPass = new ArrayList<IBaseResource>();
310 	
311 				for (BaseResourceReferenceDt nextRef : references) {
312 					IBaseResource nextRefRes = nextRef.getResource();
313 					if (nextRefRes != null) {
314 						if (nextRefRes.getIdElement().hasIdPart()) {
315 							if (containedIds.contains(nextRefRes.getIdElement().getValue())) {
316 								// Don't add contained IDs as top level resources
317 								continue;
318 							}
319 	
320 							IIdType id = nextRefRes.getIdElement();
321 							if (id.hasResourceType() == false) {
322 								String resName = myContext.getResourceDefinition(nextRefRes).getName();
323 								id = id.withResourceType(resName);
324 							}
325 	
326 							if (!addedResourceIds.contains(id)) {
327 								addedResourceIds.add(id);
328 								addedResourcesThisPass.add(nextRefRes);
329 							}
330 	
331 						}
332 					}
333 				}
334 	
335 				// Linked resources may themselves have linked resources
336 				references = new ArrayList<BaseResourceReferenceDt>();
337 				for (IBaseResource iResource : addedResourcesThisPass) {
338 					List<BaseResourceReferenceDt> newReferences = myContext.newTerser().getAllPopulatedChildElementsOfType(iResource, BaseResourceReferenceDt.class);
339 					references.addAll(newReferences);
340 				}
341 	
342 				includedResources.addAll(addedResourcesThisPass);
343 	
344 			} while (references.isEmpty() == false);
345 	
346 			myBundle.addResource(next, myContext, theServerBase);
347 	
348 		}
349 	
350 		/*
351 		 * Actually add the resources to the bundle
352 		 */
353 		for (IBaseResource next : includedResources) {
354 			BundleEntry entry = myBundle.addResource((IResource) next, myContext, theServerBase);
355 			if (myContext.getVersion().getVersion().isNewerThan(FhirVersionEnum.DSTU1)) {
356 				if (entry.getSearchMode().isEmpty()) {
357 					entry.getSearchMode().setValueAsEnum(BundleEntrySearchModeEnum.INCLUDE);
358 				}
359 			}
360 		}
361 	
362 		myBundle.getTotalResults().setValue(theTotalResults);
363 	}

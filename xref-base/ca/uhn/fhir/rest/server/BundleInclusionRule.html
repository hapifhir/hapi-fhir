23  import java.util.Set;
24  
25  import ca.uhn.fhir.model.api.Include;
26  import ca.uhn.fhir.util.ResourceReferenceInfo;
27  
28  /**
29   * Created by Bill de Beaubien on 3/4/2015.
30   *
31   * Controls how bundles decide whether referenced resources should be included
32   */
33  public enum BundleInclusionRule {
34  	
35  	/**
36  	 * Decision is based on whether the resource's Include is in the IncludeSet (e.g. DiagnosticReport.result). Note that the resource has to be populated to be included.
37  	 *
38  	 * This is the default behavior
39  	 */
40  	BASED_ON_INCLUDES {
41  		@Override
42  		public boolean shouldIncludeReferencedResource(ResourceReferenceInfo theReferenceInfo, Set<Include> theIncludes) {
43  			return theReferenceInfo.matchesIncludeSet(theIncludes);
44  		}
45  	},
46  
47  	/**
48  	 * Decision is based on whether the resource reference is set to a populated resource (in which case its included) or just an id (in which case it's not included)
49  	 *
50  	 * This is the original HAPI behavior
51  	 */
52  	BASED_ON_RESOURCE_PRESENCE {
53  		@Override
54  		public boolean shouldIncludeReferencedResource(ResourceReferenceInfo theReferenceInfo, Set<Include> theIncludes) {
55  			return true;
56  		}
57  	};
58  
59  	public abstract boolean shouldIncludeReferencedResource(ResourceReferenceInfo theReferenceInfo, Set<Include> theIncludes);
60  }

24  import org.apache.http.client.HttpClient;
25  import org.hl7.fhir.instance.model.api.IBaseResource;
26  
27  import ca.uhn.fhir.context.FhirContext;
28  import ca.uhn.fhir.rest.api.SummaryEnum;
29  import ca.uhn.fhir.rest.client.IClientInterceptor;
30  import ca.uhn.fhir.rest.server.EncodingEnum;
31  
32  public interface IRestfulClient {
33  
34  	/**
35  	 * Retrieve the contents at the given URL and parse them as a resource. This
36  	 * method could be used as a low level implementation of a read/vread/search
37  	 * operation.
38  	 * 
39  	 * @param theResourceType The resource type to parse
40  	 * @param theUrl The URL to load
41  	 * @return The parsed resource
42  	 */
43  	<T extends IBaseResource> T fetchResourceFromUrl(Class<T> theResourceType, String theUrl);
44  	
45  	/**
46  	 * Returns the FHIR context associated with this client
47  	 */
48  	FhirContext getFhirContext();
49  	
50  	/**
51  	 * Do not call this method in client code. It is a part of the internal HAPI API and 
52  	 * is subject to change!
53  	 */
54  	HttpClient getHttpClient();
55  	
56  	/**
57  	 * Base URL for the server, with no trailing "/"
58  	 */
59  	String getServerBase();
60  	
61  	/**
62  	 * Register a new interceptor for this client. An interceptor can be used to add additional
63  	 * logging, or add security headers, or pre-process responses, etc. 
64  	 */
65  	void registerInterceptor(IClientInterceptor theInterceptor);
66  
67  	/**
68  	 * Specifies that the client should use the given encoding to do its 
69  	 * queries. This means that the client will append the "_format" param
70  	 * to GET methods (read/search/etc), and will add an appropriate header for
71  	 * write methods. 
72  	 * 
73  	 * @param theEncoding The encoding to use in the request, or <code>null</code> not specify
74  	 * an encoding (which generally implies the use of XML). The default is <code>null</code>.
75  	 */
76  	void setEncoding(EncodingEnum theEncoding);
77  	
78  	/**
79  	 * Specifies that the client should request that the server respond with "pretty printing"
80  	 * enabled. Note that this is a non-standard parameter, not all servers will
81  	 * support it.
82  	 * 
83  	 * @param thePrettyPrint The pretty print flag to use in the request (default is <code>false</code>)
84  	 */
85  	void setPrettyPrint(Boolean thePrettyPrint);
86  
87  	/**
88  	 * If not set to <code>null</code>, specifies a value for the <code>_summary</code> parameter
89  	 * to be applied globally on this client.
90  	 */
91  	void setSummary(SummaryEnum theSummary);
92  
93  	/**
94  	 * Remove an intercaptor that was previously registered using {@link IRestfulClient#registerInterceptor(IClientInterceptor)}
95  	 */
96  	void unregisterInterceptor(IClientInterceptor theInterceptor);
97  	
98  }

35  import org.apache.commons.io.IOUtils;
36  import org.apache.http.Header;
37  import org.apache.http.HttpEntity;

39  import org.apache.http.client.methods.CloseableHttpResponse;
40  import org.apache.http.entity.ContentType;

42  import ca.uhn.fhir.rest.client.api.IHttpResponse;

46  /**
47   * A Http Response based on Apache. This is an adapter around the class
48   * {@link org.apache.http.HttpResponse HttpResponse}
49   * 
50   * @author Peter Van Houte | peter.vanhoute@agfa.com | Agfa Healthcare
51   */
52  public class ApacheHttpResponse implements IHttpResponse {
53  
54  	private static final org.slf4j.Logger ourLog = org.slf4j.LoggerFactory.getLogger(ApacheHttpResponse.class);
55  
56  	private boolean myEntityBuffered = false;
57  	private byte[] myEntityBytes;
58  	private final HttpResponse myResponse;
59  
60  	public ApacheHttpResponse(HttpResponse theResponse) {
61  		this.myResponse = theResponse;
62  	}
63  
64  	@Override
65  	public void bufferEntitity() throws IOException {
66  		bufferEntity();
67  	}
68  
69  	@Override
70  	public void bufferEntity() throws IOException {
71  		if (myEntityBuffered) {
72  			return;
73  		}
74  		InputStream respEntity = readEntity();
75  		if (respEntity != null) {
76  			this.myEntityBuffered = true;
77  			try {
78  				this.myEntityBytes = IOUtils.toByteArray(respEntity);
79  			} catch (IllegalStateException e) {
80  				throw new InternalErrorException(e);
81  			}
82  		}
83  	}
84  
85  	@Override
86  	public void close() {
87  		if (myResponse instanceof CloseableHttpResponse) {
88  			try {
89  				((CloseableHttpResponse) myResponse).close();
90  			} catch (IOException e) {
91  				ourLog.debug("Failed to close response", e);
92  			}
93  		}
94  	}
95  
96  	@Override
97  	public Reader createReader() throws IOException {
98  		HttpEntity entity = myResponse.getEntity();
99  		if (entity == null) {
100 			return new StringReader("");
101 		}
102 		Charset charset = null;
103 		if (entity.getContentType() != null && entity.getContentType().getElements() != null
104 				&& entity.getContentType().getElements().length > 0) {
105 			ContentType ct = ContentType.get(entity);
106 			charset = ct.getCharset();
107 		}
108 		if (charset == null) {
109 			if (Constants.STATUS_HTTP_204_NO_CONTENT != myResponse.getStatusLine().getStatusCode()) {
110 				ourLog.warn("Response did not specify a charset.");
111 			}
112 			charset = Charset.forName("UTF-8");
113 		}
114 
115 		Reader reader = new InputStreamReader(readEntity(), charset);
116 		return reader;
117 	}
118 
119 	@Override
120 	public Map<String, List<String>> getAllHeaders() {
121 		Map<String, List<String>> headers = new HashMap<String, List<String>>();
122 		if (myResponse.getAllHeaders() != null) {
123 			for (Header next : myResponse.getAllHeaders()) {
124 				String name = next.getName().toLowerCase();
125 				List<String> list = headers.get(name);
126 				if (list == null) {
127 					list = new ArrayList<String>();
128 					headers.put(name, list);
129 				}
130 				list.add(next.getValue());
131 			}
132 
133 		}
134 		return headers;
135 	}
136 
137 	@Override
138 	public String getMimeType() {
139 		ContentType ct = ContentType.get(myResponse.getEntity());
140 		return ct != null ? ct.getMimeType() : null;
141 	}
142 
143 	@Override
144 	public HttpResponse getResponse() {
145 		return myResponse;
146 	}
147 
148 	@Override
149 	public int getStatus() {
150 		return myResponse.getStatusLine().getStatusCode();
151 	}
152 
153 	@Override
154 	public String getStatusInfo() {
155 		return myResponse.getStatusLine().getReasonPhrase();
156 	}
157 
158 	@Override
159 	public InputStream readEntity() throws IOException {
160 		if (this.myEntityBuffered) {
161 			return new ByteArrayInputStream(myEntityBytes);
162 		} else if (myResponse.getEntity() != null) {
163 			return myResponse.getEntity().getContent();
164 		} else {
165 			return null;
166 		}
167 	}
168 }

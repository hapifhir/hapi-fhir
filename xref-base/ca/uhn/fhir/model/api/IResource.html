23  import java.util.Arrays;
24  import java.util.HashSet;
25  import java.util.Set;
26  
27  import ca.uhn.fhir.model.api.annotation.ResourceDef;
28  import ca.uhn.fhir.model.base.composite.BaseContainedDt;
29  import ca.uhn.fhir.model.base.composite.BaseNarrativeDt;
30  import ca.uhn.fhir.model.base.resource.ResourceMetadataMap;
31  import ca.uhn.fhir.model.primitive.CodeDt;
32  import ca.uhn.fhir.model.primitive.IdDt;
33  
34  /**
35   * This interface is the parent interface for all FHIR Resource definition
36   * classes. Classes implementing this interface should be annotated
37   * with the {@link ResourceDef @ResourceDef} annotation.
38   * 
39   * <p>
40   * Note that this class is a part of HAPI's model API, used to define
41   * structure classes. Users will often interact with this interface, but
42   * should not need to implement it directly.
43   * </p>
44   */
45  public interface IResource extends ICompositeElement, org.hl7.fhir.instance.model.api.IBaseResource {
46      public static final Include INCLUDE_ALL = new Include("*");
47      public static final Set<Include> WILDCARD_ALL_SET = new HashSet<Include>(Arrays.asList(INCLUDE_ALL));
48  
49      /**
50  	 * Returns the contained resource list for this resource.
51  	 * <p>
52  	 * Usage note: HAPI will generally populate and use the resources from this
53  	 * list automatically (placing inline resources in the contained list when
54  	 * encoding, and copying contained resources from this list to their
55  	 * appropriate references when parsing) so it is generally not neccesary to
56  	 * interact with this list directly. Instead, in a server you can place
57  	 * resource instances in reference fields (such as <code>Patient#setManagingOrganization(ResourceReferenceDt)</code> )
58  	 * and the resource will be automatically contained. In a client, contained resources will
59  	 * be automatically populated into their appropriate fields by the HAPI parser.
60  	 * </p>
61  	 * TODO: document contained resources and link there
62  	 */
63  	BaseContainedDt getContained();
64  
65  	/**
66  	 * Returns the ID of this resource. Note that this identifier is the URL (or a portion
67  	 * of the URL) used to access this resource, and is not the same thing as any business
68  	 * identifiers stored within the resource. For example, a Patient resource might
69  	 * have any number of medical record numbers but these are not stored here.
70  	 * <p>
71  	 * This ID is specified as the "Logical ID" and "Version ID" in the FHIR specification, see 
72  	 * <a href="http://www.hl7.org/implement/standards/fhir/resources.html#metadata">here</a>
73  	 * </p>
74  	 */
75  	IdDt getId();
76  
77  	/**
78  	 * Gets the language of the resource itself - <b>NOTE that this language attribute
79  	 * applies to the resource itself, it is not (for example) the language spoken by
80  	 * a practitioner or patient</b>
81  	 */
82  	CodeDt getLanguage();
83  
84  	/**
85  	 * Returns the metadata map for this object, creating it if neccesary.
86  	 * Metadata entries are used to get/set feed bundle entries, such as the
87  	 * resource version, or the last updated timestamp.
88  	 * <p>
89  	 * Keys in this map are enumerated in the {@link ResourceMetadataKeyEnum},
90  	 * and each key has a specific value type that it must use.
91  	 * </p>
92  	 * 
93  	 * @see ResourceMetadataKeyEnum for a list of allowable keys and the object
94  	 *      types that values of a given key must use.
95  	 */
96  	ResourceMetadataMap getResourceMetadata();
97  
98  	/**
99  	 * Returns the narrative block for this resource
100 	 */
101 	BaseNarrativeDt getText();
102 	
103 	/**
104 	 * Sets the ID of this resource. Note that this identifier is the URL (or a portion
105 	 * of the URL) used to access this resource, and is not the same thing as any business
106 	 * identifiers stored within the resource. For example, a Patient resource might
107 	 * have any number of medical record numbers but these are not stored here.
108 	 * <p>
109 	 * This ID is specified as the "Logical ID" and "Version ID" in the FHIR specification, see 
110 	 * <a href="http://www.hl7.org/implement/standards/fhir/resources.html#metadata">here</a>
111 	 * </p>
112 	 */
113 	void setId(IdDt theId);
114 
115 	/**
116 	 * Sets the language of the resource itself - <b>NOTE that this language attribute
117 	 * applies to the resource itself, it is not (for example) the language spoken by
118 	 * a practitioner or patient</b>
119 	 */
120 	void setLanguage(CodeDt theLanguage);
121 
122 	/**
123 	 * Sets the metadata map for this object. Metadata entries are used to
124 	 * get/set feed bundle entries, such as the resource version, or the last
125 	 * updated timestamp.
126 	 * 
127 	 * @throws NullPointerException
128 	 *             The map must not be null
129 	 */
130 	void setResourceMetadata(ResourceMetadataMap theMap);
131 
132 	/**
133 	 * Returns a String representing the name of this Resource. This return
134 	 * value is not used for anything by HAPI itself, but is provided as a 
135 	 * convenience to developers using the API.
136 	 * 
137 	 * @return the name of this resource, e.g. "Patient", or "Observation"
138 	 */
139 	String getResourceName();
140 
141 	/**
142 	 * Returns the FHIR version represented by this structure
143 	 */
144     public ca.uhn.fhir.context.FhirVersionEnum getStructureFhirVersionEnum();
145 
146 	
147 }

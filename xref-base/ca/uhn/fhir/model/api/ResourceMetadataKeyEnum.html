42  /**
43   * Keys in this map refer to <b>resource metadata keys</b>, which are keys used to access information about specific resource instances that live outside of the resource body. Typically, these are
44   * data elements which are sent/receieved in HTTP Headers along with read/create resource requests, or properties which can be found in bundle entries.
45   * <p>
46   * To access or set resource metadata values, every resource has a metadata map, and this class provides convenient getters/setters for interacting with that map. For example, to get a resource's
47   * {@link #UPDATED} value, which is the "last updated" time for that resource, use the following code:
48   * </p>
49   * <p>
50   * <code>InstantDt updated = ResourceMetadataKeyEnum.UPDATED.get(resource);</code>
51   * <p>
52   * <p>
53   * To set this value, use the following:
54   * </p>
55   * <p>
56   * <code>InstantDt update = new InstantDt("2011-01-02T11:22:33.0000Z"); // populate with the actual time<br>
57   * ResourceMetadataKeyEnum.UPDATED.put(resource, update);</code>
58   * </p>
59   * <p>
60   * Note that this class is not a Java Enum, and can therefore be extended (this is why it is not actually an Enum). Users of HAPI-FHIR are able to create their own classes extending this class to
61   * define their own keys for storage in resource metadata if needed.
62   * </p>
63   */
64  public abstract class ResourceMetadataKeyEnum<T> {

438 	private static DecimalDt getDecimalFromMetadataOrNullIfNone(Map<ResourceMetadataKeyEnum<?>, Object> theResourceMetadata, ResourceMetadataKeyEnum<DecimalDt> theKey) {
439 		Object retValObj = theResourceMetadata.get(theKey);
440 		if (retValObj == null) {
441 			return null;
442 		} else if (retValObj instanceof DecimalDt) {
443 			if (((DecimalDt) retValObj).isEmpty()) {
444 				return null;
445 			} else {
446 				return (DecimalDt) retValObj;
447 			}
448 		} else if (retValObj instanceof String) {
449 			if (StringUtils.isBlank((String) retValObj)) {
450 				return null;
451 			}
452 			return new DecimalDt((String) retValObj);
453 		} else if (retValObj instanceof Double) {
454 			return new DecimalDt((Double) retValObj);
455 		}
456 		throw new InternalErrorException("Found an object of type '" + retValObj.getClass().getCanonicalName() + "' in resource metadata for key " + theKey.name() + " - Expected "
457 				+ InstantDt.class.getCanonicalName());
458 	}

463 		Object retValObj = theResourceMetadata.get(theKey);

468 		} else if (retValObj instanceof String) {
469 			return theBinder.fromCodeString((String) retValObj);

472 				+ InstantDt.class.getCanonicalName());
473 	}

479 	private static List<IdDt> getIdListFromMetadataOrNullIfNone(Map<ResourceMetadataKeyEnum<?>, Object> theResourceMetadata, ResourceMetadataKeyEnum<?> theKey) {
480 		Object retValObj = theResourceMetadata.get(theKey);
481 		if (retValObj == null) {
482 			return null;
483 		} else if (retValObj instanceof List) {
484 			List<?> retValList = (List<?>) retValObj;
485 			for (Object next : retValList) {
486 				if (!(next instanceof IdDt)) {
487 					List<IdDt> retVal = new ArrayList<IdDt>();
488 					for (Object nextVal : retValList) {
489 						retVal.add(toId(theKey, nextVal));
490 					}
491 					return Collections.unmodifiableList(retVal);
492 				}
493 			}
494 			@SuppressWarnings("unchecked")
495 			List<IdDt> retVal = (List<IdDt>) retValList;
496 			return Collections.unmodifiableList(retVal);
497 		} else {
498 			return Collections.singletonList(toId(theKey, retValObj));
499 		}
500 	}

550 		throw new InternalErrorException("Found an object of type '" + retValObj.getClass().getCanonicalName() + "' in resource metadata for key " + theKey.name() + " - Expected "
551 				+ String.class.getCanonicalName());
552 	}

554 	private static IdDt toId(ResourceMetadataKeyEnum<?> theKey, Object retValObj) {
555 		if (retValObj == null) {
556 			return null;
557 		} else if (retValObj instanceof String) {
558 			if (isNotBlank((String) retValObj)) {
559 				return new IdDt((String) retValObj);
560 			} else {
561 				return null;
562 			}
563 		} else if (retValObj instanceof IdDt) {
564 			if (((IdDt) retValObj).isEmpty()) {
565 				return null;
566 			} else {
567 				return (IdDt) retValObj;
568 			}
569 		} else if (retValObj instanceof Number) {
570 			return new IdDt(((Number) retValObj).toString());
571 		}
572 		throw new InternalErrorException("Found an object of type '" + retValObj.getClass().getCanonicalName() + "' in resource metadata for key " + theKey.name() + " - Expected "
573 				+ IdDt.class.getCanonicalName());
574 	}

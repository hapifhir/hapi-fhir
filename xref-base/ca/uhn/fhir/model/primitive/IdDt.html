45  /**
46   * Represents the FHIR ID type. This is the actual resource ID, meaning the ID that will be used in RESTful URLs, Resource References, etc. to represent a specific instance of a resource.
47   * 
48   * <p>
49   * <b>Description</b>: A whole number in the range 0 to 2^64-1 (optionally represented in hex), a uuid, an oid, or any other combination of lowercase letters, numerals, "-" and ".", with a length
50   * limit of 36 characters.
51   * </p>
52   * <p>
53   * regex: [a-z-Z0-9\-\.]{1,36}
54   * </p>
55   */
56  @DatatypeDef(name = "id", profileOf = StringDt.class)

284 	 * Returns the value of this ID. Note that this value may be a fully qualified URL, a relative/partial URL, or a simple ID. Use {@link #getIdPart()} to get just the ID portion.
285 	 * 
286 	 * @see #getIdPart()
287 	 */
288 	@Override
289 	public String getValue() {
290 		if (super.getValue() == null && myHaveComponentParts) {
291 
292 			if (isLocal() || isUrn()) {
293 				return myUnqualifiedId;
294 			}
295 
296 			StringBuilder b = new StringBuilder();
297 			if (isNotBlank(myBaseUrl)) {
298 				b.append(myBaseUrl);
299 				if (myBaseUrl.charAt(myBaseUrl.length() - 1) != '/') {
300 					b.append('/');
301 				}
302 			}
303 
304 			if (isNotBlank(myResourceType)) {
305 				b.append(myResourceType);
306 			}
307 
308 			if (b.length() > 0) {
309 				b.append('/');
310 			}
311 
312 			b.append(myUnqualifiedId);
313 			if (isNotBlank(myUnqualifiedVersionId)) {
314 				b.append('/');
315 				b.append(Constants.PARAM_HISTORY);
316 				b.append('/');
317 				b.append(myUnqualifiedVersionId);
318 			}
319 			String value = b.toString();
320 			super.setValue(value);
321 		}
322 		return super.getValue();
323 	}

390 	@Override
391 	public boolean isIdPartValid() {
392 		String id = getIdPart();
393 		if (StringUtils.isBlank(id)) {
394 			return false;
395 		}
396 		if (id.length() > 64) {
397 			return false;
398 		}
399 		for (int i = 0; i < id.length(); i++) {
400 			char nextChar = id.charAt(i);
401 			if (nextChar >= 'a' && nextChar <= 'z') {
402 				continue;
403 			}
404 			if (nextChar >= 'A' && nextChar <= 'Z') {
405 				continue;
406 			}
407 			if (nextChar >= '0' && nextChar <= '9') {
408 				continue;
409 			}
410 			if (nextChar == '-' || nextChar == '.') {
411 				continue;
412 			}
413 			return false;
414 		}
415 		return true;
416 	}

470 	/**
471 	 * Set the value
472 	 * 
473 	 * <p>
474 	 * <b>Description</b>: A whole number in the range 0 to 2^64-1 (optionally represented in hex), a uuid, an oid, or any other combination of lowercase letters, numerals, "-" and ".", with a length
475 	 * limit of 36 characters.
476 	 * </p>
477 	 * <p>
478 	 * regex: [a-z0-9\-\.]{1,36}
479 	 * </p>
480 	 */
481 	@Override
482 	public IdDt setValue(String theValue) throws DataFormatException {
483 		// TODO: add validation
484 		super.setValue(theValue);
485 		myHaveComponentParts = false;
486 
487 		if (StringUtils.isBlank(theValue)) {
488 			myBaseUrl = null;
489 			super.setValue(null);
490 			myUnqualifiedId = null;
491 			myUnqualifiedVersionId = null;
492 			myResourceType = null;
493 		} else if (theValue.charAt(0) == '#' && theValue.length() > 1) {
494 			super.setValue(theValue);
495 			myBaseUrl = null;
496 			myUnqualifiedId = theValue;
497 			myUnqualifiedVersionId = null;
498 			myResourceType = null;
499 			myHaveComponentParts = true;
500 		} else if (theValue.startsWith("urn:")) {
501 			myBaseUrl = null;
502 			myUnqualifiedId = theValue;
503 			myUnqualifiedVersionId = null;
504 			myResourceType = null;
505 			myHaveComponentParts = true;
506 		} else {
507 			int vidIndex = theValue.indexOf("/_history/");
508 			int idIndex;
509 			if (vidIndex != -1) {
510 				myUnqualifiedVersionId = theValue.substring(vidIndex + "/_history/".length());
511 				idIndex = theValue.lastIndexOf('/', vidIndex - 1);
512 				myUnqualifiedId = theValue.substring(idIndex + 1, vidIndex);
513 			} else {
514 				idIndex = theValue.lastIndexOf('/');
515 				myUnqualifiedId = theValue.substring(idIndex + 1);
516 				myUnqualifiedVersionId = null;
517 			}
518 
519 			myBaseUrl = null;
520 			if (idIndex <= 0) {
521 				myResourceType = null;
522 			} else {
523 				int typeIndex = theValue.lastIndexOf('/', idIndex - 1);
524 				if (typeIndex == -1) {
525 					myResourceType = theValue.substring(0, idIndex);
526 				} else {
527 					myResourceType = theValue.substring(typeIndex + 1, idIndex);
528 
529 					if (typeIndex > 4) {
530 						myBaseUrl = theValue.substring(0, typeIndex);
531 					}
532 
533 				}
534 			}
535 
536 		}
537 		return this;
538 	}

620 	 * @return A new instance of IdDt which is identical, but refers to the specific version of this resource ID noted by theVersion.

622 	@Override
623 	public IdDt withVersion(String theVersion) {
624 		Validate.notBlank(theVersion, "Version may not be null or empty");
625 
626 		if (isLocal() || isUrn()) {
627 			return new IdDt(getValueAsString());
628 		}
629 
630 		String existingValue = getValue();
631 
632 		int i = existingValue.indexOf(Constants.PARAM_HISTORY);
633 		String value;
634 		if (i > 1) {
635 			value = existingValue.substring(0, i - 1);
636 		} else {
637 			value = existingValue;
638 		}
639 
640 		return new IdDt(value + '/' + Constants.PARAM_HISTORY + '/' + theVersion);
641 	}

643 	private static boolean isValidLong(String id) {
644 		if (StringUtils.isBlank(id)) {
645 			return false;
646 		}
647 		for (int i = 0; i < id.length(); i++) {
648 			if (Character.isDigit(id.charAt(i)) == false) {
649 				return false;
650 			}
651 		}
652 		return true;
653 	}

666 	public static IdDt of(IBaseResource theResouce) {
667 		if (theResouce == null) {
668 			throw new NullPointerException("theResource can not be null");
669 		} else {
670 			IIdType retVal = theResouce.getIdElement();
671 			if (retVal == null) {
672 				return null;
673 			} else if (retVal instanceof IdDt) {
674 				return (IdDt) retVal;
675 			} else {
676 				return new IdDt(retVal.getValue());
677 			}
678 		}
679 	}

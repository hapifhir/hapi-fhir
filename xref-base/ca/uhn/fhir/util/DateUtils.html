23  import java.lang.ref.SoftReference;
24  import java.text.ParsePosition;
25  import java.text.SimpleDateFormat;
26  import java.util.*;
27  
28  /**
29   * A utility class for parsing and formatting HTTP dates as used in cookies and
30   * other headers.  This class handles dates as defined by RFC 2616 section
31   * 3.3.1 as well as some other common non-standard formats.
32   * <p>
33   * This class is basically intended to be a high-performance workaround
34   * for the fact that Java SimpleDateFormat is kind of expensive to
35   * create and yet isn't thread safe.
36   * </p>
37   * <p>
38   * This class was adapted from the class with the same name from the Jetty
39   * project, licensed under the terms of the Apache Software License 2.0.
40   * </p>
41   */
42  public final class DateUtils {
43  
44  	public static final TimeZone GMT = TimeZone.getTimeZone("GMT");
45  	/**
46  	 * Date format pattern used to parse HTTP date headers in RFC 1123 format.
47  	 */
48  	private static final String PATTERN_RFC1123 = "EEE, dd MMM yyyy HH:mm:ss zzz";
49  	/**
50  	 * Date format pattern used to parse HTTP date headers in RFC 1036 format.
51  	 */
52  	private static final String PATTERN_RFC1036 = "EEE, dd-MMM-yy HH:mm:ss zzz";
53  	/**
54  	 * Date format pattern used to parse HTTP date headers in ANSI C
55  	 * {@code asctime()} format.
56  	 */
57  	private static final String PATTERN_ASCTIME = "EEE MMM d HH:mm:ss yyyy";
58  	private static final String[] DEFAULT_PATTERNS = new String[]{
59  		PATTERN_RFC1123,
60  		PATTERN_RFC1036,
61  		PATTERN_ASCTIME
62  	};
63  	private static final Date DEFAULT_TWO_DIGIT_YEAR_START;
64  
65  	static {
66  		final Calendar calendar = Calendar.getInstance();
67  		calendar.setTimeZone(GMT);
68  		calendar.set(2000, Calendar.JANUARY, 1, 0, 0, 0);
69  		calendar.set(Calendar.MILLISECOND, 0);
70  		DEFAULT_TWO_DIGIT_YEAR_START = calendar.getTime();
71  	}
72  
73  	/**
74  	 * This class should not be instantiated.
75  	 */
76  	private DateUtils() {
77  	}
78  
79  	/**
80  	 * A factory for {@link SimpleDateFormat}s. The instances are stored in a
81  	 * threadlocal way because SimpleDateFormat is not thread safe as noted in
82  	 * {@link SimpleDateFormat its javadoc}.
83  	 */
84  	final static class DateFormatHolder {
85  
86  		private static final ThreadLocal<SoftReference<Map<String, SimpleDateFormat>>> THREADLOCAL_FORMATS = ThreadLocal.withInitial(() -> new SoftReference<>(new HashMap<>()));
87  
88  		/**
89  		 * creates a {@link SimpleDateFormat} for the requested format string.
90  		 *
91  		 * @param pattern a non-{@code null} format String according to
92  		 *                {@link SimpleDateFormat}. The format is not checked against
93  		 *                {@code null} since all paths go through
94  		 *                {@link DateUtils}.
95  		 * @return the requested format. This simple DateFormat should not be used
96  		 * to {@link SimpleDateFormat#applyPattern(String) apply} to a
97  		 * different pattern.
98  		 */
99  		static SimpleDateFormat formatFor(final String pattern) {
100 			final SoftReference<Map<String, SimpleDateFormat>> ref = THREADLOCAL_FORMATS.get();
101 			Map<String, SimpleDateFormat> formats = ref.get();
102 			if (formats == null) {
103 				formats = new HashMap<>();
104 				THREADLOCAL_FORMATS.set(
105 					new SoftReference<>(formats));
106 			}
107 
108 			SimpleDateFormat format = formats.get(pattern);
109 			if (format == null) {
110 				format = new SimpleDateFormat(pattern, Locale.US);
111 				format.setTimeZone(TimeZone.getTimeZone("GMT"));
112 				formats.put(pattern, format);
113 			}
114 
115 			return format;
116 		}
117 
118 	}
119 
120 	/**
121 	 * Parses a date value.  The formats used for parsing the date value are retrieved from
122 	 * the default http params.
123 	 *
124 	 * @param theDateValue the date value to parse
125 	 * @return the parsed date or null if input could not be parsed
126 	 */
127 	public static Date parseDate(final String theDateValue) {
128 		notNull(theDateValue, "Date value");
129 		String v = theDateValue;
130 		if (v.length() > 1 && v.startsWith("'") && v.endsWith("'")) {
131 			v = v.substring(1, v.length() - 1);
132 		}
133 
134 		for (final String dateFormat : DEFAULT_PATTERNS) {
135 			final SimpleDateFormat dateParser = DateFormatHolder.formatFor(dateFormat);
136 			dateParser.set2DigitYearStart(DEFAULT_TWO_DIGIT_YEAR_START);
137 			final ParsePosition pos = new ParsePosition(0);
138 			final Date result = dateParser.parse(v, pos);
139 			if (pos.getIndex() != 0) {
140 				return result;
141 			}
142 		}
143 		return null;
144 	}
145 
146 	/**
147 	 * Formats the given date according to the RFC 1123 pattern.
148 	 *
149 	 * @param date The date to format.
150 	 * @return An RFC 1123 formatted date string.
151 	 * @see #PATTERN_RFC1123
152 	 */
153 	public static String formatDate(final Date date) {
154 		notNull(date, "Date");
155 		notNull(PATTERN_RFC1123, "Pattern");
156 		final SimpleDateFormat formatter = DateFormatHolder.formatFor(PATTERN_RFC1123);
157 		return formatter.format(date);
158 	}
159 
160 	public static <T> T notNull(final T argument, final String name) {
161 		if (argument == null) {
162 			throw new IllegalArgumentException(name + " may not be null");
163 		}
164 		return argument;
165 	}
166 
167 }

117 	 * Clones all values from a source object into the equivalent fields in a target object
118 	 * @param theSource The source object (must not be null)
119 	 * @param theTarget The target object to copy values into (must not be null)
120 	 * @param theIgnoreMissingFields The ignore fields in the target which do not exist (if false, an exception will be thrown if the target is unable to accept a value from the source)
121 	 */
122 	public void cloneInto(IBase theSource, IBase theTarget, boolean theIgnoreMissingFields) {
123 		Validate.notNull(theSource, "theSource must not be null");
124 		Validate.notNull(theTarget, "theTarget must not be null");
125 		
126 		if (theSource instanceof IPrimitiveType<?>) {
127 			if (theTarget instanceof IPrimitiveType<?>) {
128 				((IPrimitiveType<?>)theTarget).setValueAsString(((IPrimitiveType<?>)theSource).getValueAsString());
129 				return;
130 			} else {
131 				if (theIgnoreMissingFields) {
132 					return;
133 				} else {
134 					throw new DataFormatException("Can not copy value from primitive of type " + theSource.getClass().getName() + " into type " + theTarget.getClass().getName());
135 				}
136 			}
137 		}
138 		
139 		BaseRuntimeElementCompositeDefinition<?> sourceDef = (BaseRuntimeElementCompositeDefinition<?>) myContext.getElementDefinition(theSource.getClass()); 
140 		BaseRuntimeElementCompositeDefinition<?> targetDef = (BaseRuntimeElementCompositeDefinition<?>) myContext.getElementDefinition(theTarget.getClass());
141 		
142 		List<BaseRuntimeChildDefinition> children = sourceDef.getChildren();
143 		if (sourceDef instanceof RuntimeExtensionDtDefinition) {
144 			children = ((RuntimeExtensionDtDefinition)sourceDef).getChildrenIncludingUrl();
145 		}
146 		
147 		for (BaseRuntimeChildDefinition nextChild : children) {
148 			for (IBase nextValue : nextChild.getAccessor().getValues(theSource)) {
149 				String elementName = nextChild.getChildNameByDatatype(nextValue.getClass());
150 				BaseRuntimeChildDefinition targetChild = targetDef.getChildByName(elementName);
151 				if (targetChild == null) {
152 					if (theIgnoreMissingFields) {
153 						continue;
154 					} else {
155 						throw new DataFormatException("Type " + theTarget.getClass().getName() + " does not have a child with name " + elementName);
156 					}
157 				}
158 				
159 				BaseRuntimeElementDefinition<?> childDef = targetChild.getChildByName(elementName);
160 				IBase target = childDef.newInstance();
161 				targetChild.getMutator().addValue(theTarget, target);
162 				cloneInto(nextValue, target, theIgnoreMissingFields);
163 			}
164 		}
165 		
166 	}

169 	 * Returns a list containing all child elements (including the resource itself) which are <b>non-empty</b> and are either of the exact type specified, or are a subclass of that type.
170 	 * <p>
171 	 * For example, specifying a type of {@link StringDt} would return all non-empty string instances within the message. Specifying a type of {@link IResource} would return the resource itself, as
172 	 * well as any contained resources.
173 	 * </p>
174 	 * <p>
175 	 * Note on scope: This method will descend into any contained resources ({@link IResource#getContained()}) as well, but will not descend into linked resources (e.g.
176 	 * {@link BaseResourceReferenceDt#getResource()}) or embedded resources (e.g. Bundle.entry.resource)
177 	 * </p>
178 	 * 
179 	 * @param theResource
180 	 *           The resource instance to search. Must not be null.
181 	 * @param theType
182 	 *           The type to search for. Must not be null.
183 	 * @return Returns a list of all matching elements
184 	 */
185 	public <T extends IBase> List<T> getAllPopulatedChildElementsOfType(IBaseResource theResource, final Class<T> theType) {
186 		final ArrayList<T> retVal = new ArrayList<T>();
187 		BaseRuntimeElementCompositeDefinition<?> def = myContext.getResourceDefinition(theResource);
188 		visit(new IdentityHashMap<Object, Object>(), theResource, null, null, def, new IModelVisitor() {
189 			@SuppressWarnings("unchecked")
190 			@Override
191 			public void acceptElement(IBase theElement, List<String> thePathToElement, BaseRuntimeChildDefinition theChildDefinition, BaseRuntimeElementDefinition<?> theDefinition) {
192 				if (theElement == null || theElement.isEmpty()) {
193 					return;
194 				}
195 
196 				if (theType.isAssignableFrom(theElement.getClass())) {
197 					retVal.add((T) theElement);
198 				}
199 			}
200 
201 			@Override
202 			public void acceptUndeclaredExtension(ISupportsUndeclaredExtensions theContainingElement, List<String> thePathToElement, BaseRuntimeChildDefinition theChildDefinition,
203 					BaseRuntimeElementDefinition<?> theDefinition, ExtensionDt theNextExt) {
204 				// nothing
205 			}
206 		});
207 		return retVal;
208 	}

234 		BaseRuntimeChildDefinition nextDef = theCurrentDef.getChildByNameOrThrowDataFormatException(theSubList.get(0));

237 			return nextDef;

240 			return getDefinition(cmp, theSubList.subList(1, theSubList.size()));

264 	public <T> T getSingleValueOrNull(IBase theTarget, String thePath, Class<T> theWantedType) {
265 		Validate.notNull(theTarget, "theTarget must not be null");
266 		Validate.notBlank(thePath, "thePath must not be empty");
267 
268 		BaseRuntimeElementDefinition<?> def = myContext.getElementDefinition(theTarget.getClass());
269 		if (!(def instanceof BaseRuntimeElementCompositeDefinition)) {
270 			throw new IllegalArgumentException("Target is not a composite type: " + theTarget.getClass().getName());
271 		}
272 
273 		BaseRuntimeElementCompositeDefinition<?> currentDef = (BaseRuntimeElementCompositeDefinition<?>) def;
274 		Object currentObj = theTarget;
275 
276 		List<String> parts = Arrays.asList(thePath.split("\\."));
277 		List<T> retVal = getValues(currentDef, currentObj, parts, theWantedType);
278 		if (retVal.isEmpty()) {
279 			return null;
280 		} else {
281 			return retVal.get(0);
282 		}
283 	}

285 	@SuppressWarnings("unchecked")
286 	private <T> List<T> getValues(BaseRuntimeElementCompositeDefinition<?> theCurrentDef, Object theCurrentObj, List<String> theSubList, Class<T> theWantedClass) {
287 		String name = theSubList.get(0);
288 				
289 		BaseRuntimeChildDefinition nextDef = theCurrentDef.getChildByNameOrThrowDataFormatException(name);
290 		List<? extends IBase> values = nextDef.getAccessor().getValues(theCurrentObj);
291 		List<T> retVal = new ArrayList<T>();
292 
293 		if (theSubList.size() == 1) {
294 			if (nextDef instanceof RuntimeChildChoiceDefinition) {
295 				for (IBase next : values) {
296 					if (next != null) {
297 						if (name.endsWith("[x]")) {
298 							if (theWantedClass == null || theWantedClass.isAssignableFrom(next.getClass())) {
299 								retVal.add((T) next);
300 							}
301 						} else {
302 							String childName = nextDef.getChildNameByDatatype(next.getClass());
303 							if (theSubList.get(0).equals(childName)) {
304 								if (theWantedClass == null || theWantedClass.isAssignableFrom(next.getClass())) {
305 									retVal.add((T) next);
306 								}
307 							}
308 						}
309 					}
310 				}
311 			} else {
312 				for (IBase next : values) {
313 					if (next != null) {
314 						if (theWantedClass == null || theWantedClass.isAssignableFrom(next.getClass())) {
315 							retVal.add((T) next);
316 						}
317 					}
318 				}
319 			}
320 		} else {
321 			for (IBase nextElement : values) {
322 				BaseRuntimeElementCompositeDefinition<?> nextChildDef = (BaseRuntimeElementCompositeDefinition<?>) myContext.getElementDefinition(nextElement.getClass());
323 				List<T> foundValues = getValues(nextChildDef, nextElement, theSubList.subList(1, theSubList.size()), theWantedClass);
324 				retVal.addAll(foundValues);
325 			}
326 		}
327 		return retVal;
328 	}

352 	 * Returns <code>true</code> if <code>theSource</code> is in the compartment named <code>theCompartmentName</code>
353 	 * belonging to resource <code>theTarget</code>
354 	 * 
355 	 * @param theCompartmentName The name of the compartment
356 	 * @param theSource The potential member of the compartment
357 	 * @param theTarget The owner of the compartment. Note that both the resource type and ID must be filled in on this IIdType or the method will throw an {@link IllegalArgumentException}
358 	 * @return <code>true</code> if <code>theSource</code> is in the compartment
359 	 * @throws IllegalArgumentException If theTarget does not contain both a resource type and ID
360 	 */
361 	public boolean isSourceInCompartmentForTarget(String theCompartmentName, IBaseResource theSource, IIdType theTarget) {
362 		Validate.notBlank(theCompartmentName, "theCompartmentName must not be null or blank");
363 		Validate.notNull(theSource, "theSource must not be null");
364 		Validate.notNull(theTarget, "theTarget must not be null");
365 		Validate.notBlank(defaultString(theTarget.getResourceType()), "theTarget must have a populated resource type (theTarget.getResourceType() does not return a value)");
366 		Validate.notBlank(defaultString(theTarget.getIdPart()), "theTarget must have a populated ID (theTarget.getIdPart() does not return a value)");
367 		
368 		String wantRef = theTarget.toUnqualifiedVersionless().getValue();
369 		
370 		RuntimeResourceDefinition sourceDef = myContext.getResourceDefinition(theSource);
371 		if (theSource.getIdElement().hasIdPart()) {
372 			if (wantRef.equals(sourceDef.getName() + '/' + theSource.getIdElement().getIdPart())) {
373 				return true;
374 			}
375 		}
376 		
377 		List<RuntimeSearchParam> params = sourceDef.getSearchParamsForCompartmentName(theCompartmentName);
378 		for (RuntimeSearchParam nextParam : params) {
379 			for (String nextPath : nextParam.getPathsSplit()) {
380 				for (IBaseReference nextValue : getValues(theSource, nextPath, IBaseReference.class)) {
381 					String nextRef = nextValue.getReferenceElement().toUnqualifiedVersionless().getValue();
382 					if (wantRef.equals(nextRef)) {
383 						return true;
384 					}
385 				}
386 			}
387 		}
388 		
389 		return false;
390 	}

392 	private void visit(IBase theElement, BaseRuntimeChildDefinition theChildDefinition, BaseRuntimeElementDefinition<?> theDefinition, IModelVisitor2 theCallback, List<IBase> theContainingElementPath,
393 			List<BaseRuntimeChildDefinition> theChildDefinitionPath, List<BaseRuntimeElementDefinition<?>> theElementDefinitionPath) {
394 		if (theChildDefinition != null) {
395 			theChildDefinitionPath.add(theChildDefinition);
396 		}
397 		theContainingElementPath.add(theElement);
398 		theElementDefinitionPath.add(theDefinition);
399 
400 		theCallback.acceptElement(theElement, Collections.unmodifiableList(theContainingElementPath), Collections.unmodifiableList(theChildDefinitionPath),
401 				Collections.unmodifiableList(theElementDefinitionPath));
402 
403 		/*
404 		 * Visit undeclared extensions
405 		 */
406 		if (theElement instanceof ISupportsUndeclaredExtensions) {
407 			ISupportsUndeclaredExtensions containingElement = (ISupportsUndeclaredExtensions) theElement;
408 			for (ExtensionDt nextExt : containingElement.getUndeclaredExtensions()) {
409 				theContainingElementPath.add(nextExt);
410 				theCallback.acceptUndeclaredExtension(nextExt, theContainingElementPath, theChildDefinitionPath, theElementDefinitionPath);
411 				theContainingElementPath.remove(theContainingElementPath.size() - 1);
412 			}
413 		}
414 
415 		/*
416 		 * Now visit the children of the given element
417 		 */
418 		switch (theDefinition.getChildType()) {
419 		case ID_DATATYPE:
420 		case PRIMITIVE_XHTML_HL7ORG:
421 		case PRIMITIVE_XHTML:
422 		case PRIMITIVE_DATATYPE:
423 			// These are primitive types, so we don't need to visit their children
424 			break;
425 		case RESOURCE:
426 		case RESOURCE_BLOCK:
427 		case COMPOSITE_DATATYPE: {
428 			BaseRuntimeElementCompositeDefinition<?> childDef = (BaseRuntimeElementCompositeDefinition<?>) theDefinition;
429 			for (BaseRuntimeChildDefinition nextChild : childDef.getChildrenAndExtension()) {
430 				List<? extends IBase> values = nextChild.getAccessor().getValues(theElement);
431 				if (values != null) {
432 					for (IBase nextValue : values) {
433 						if (nextValue == null) {
434 							continue;
435 						}
436 						if (nextValue.isEmpty()) {
437 							continue;
438 						}
439 						BaseRuntimeElementDefinition<?> childElementDef;
440 						childElementDef = nextChild.getChildElementDefinitionByDatatype(nextValue.getClass());
441 
442 						if (childElementDef == null) {
443 							StringBuilder b = new StringBuilder();
444 							b.append("Found value of type[");
445 							b.append(nextValue.getClass().getSimpleName());
446 							b.append("] which is not valid for field[");
447 							b.append(nextChild.getElementName());
448 							b.append("] in ");
449 							b.append(childDef.getName());
450 							b.append(" - Valid types: ");
451 							for (Iterator<String> iter = new TreeSet<String>(nextChild.getValidChildNames()).iterator(); iter.hasNext();) {
452 								BaseRuntimeElementDefinition<?> childByName = nextChild.getChildByName(iter.next());
453 								b.append(childByName.getImplementingClass().getSimpleName());
454 								if (iter.hasNext()) {
455 									b.append(", ");
456 								}
457 							}
458 							throw new DataFormatException(b.toString());
459 						}
460 
461 						if (nextChild instanceof RuntimeChildDirectResource) {
462 							// Don't descend into embedded resources
463 							theContainingElementPath.add(nextValue);
464 							theChildDefinitionPath.add(nextChild);
465 							theElementDefinitionPath.add(myContext.getElementDefinition(nextValue.getClass()));
466 							theCallback.acceptElement(nextValue, Collections.unmodifiableList(theContainingElementPath), Collections.unmodifiableList(theChildDefinitionPath),
467 									Collections.unmodifiableList(theElementDefinitionPath));
468 							theChildDefinitionPath.remove(theChildDefinitionPath.size() - 1);
469 							theContainingElementPath.remove(theContainingElementPath.size() - 1);
470 							theElementDefinitionPath.remove(theElementDefinitionPath.size() - 1);
471 						} else {
472 							visit(nextValue, nextChild, childElementDef, theCallback, theContainingElementPath, theChildDefinitionPath, theElementDefinitionPath);
473 						}
474 					}
475 				}
476 			}
477 			break;
478 		}
479 		case CONTAINED_RESOURCES: {
480 			BaseContainedDt value = (BaseContainedDt) theElement;
481 			for (IResource next : value.getContainedResources()) {
482 				BaseRuntimeElementCompositeDefinition<?> def = myContext.getResourceDefinition(next);
483 				visit(next, null, def, theCallback, theContainingElementPath, theChildDefinitionPath, theElementDefinitionPath);
484 			}
485 			break;
486 		}
487 		case EXTENSION_DECLARED:
488 		case UNDECL_EXT: {
489 			throw new IllegalStateException("state should not happen: " + theDefinition.getChildType());
490 		}
491 		case CONTAINED_RESOURCE_LIST: {
492 			if (theElement != null) {
493 				BaseRuntimeElementDefinition<?> def = myContext.getElementDefinition(theElement.getClass());
494 				visit(theElement, null, def, theCallback, theContainingElementPath, theChildDefinitionPath, theElementDefinitionPath);
495 			}
496 			break;
497 		}
498 		}
499 
500 		if (theChildDefinition != null) {
501 			theChildDefinitionPath.remove(theChildDefinitionPath.size() - 1);
502 		}
503 		theContainingElementPath.remove(theContainingElementPath.size() - 1);
504 		theElementDefinitionPath.remove(theElementDefinitionPath.size() - 1);
505 	}

526 	 * Visit all elements in a given resource
527 	 * 
528 	 * THIS ALTERNATE METHOD IS STILL EXPERIMENTAL
529 	 * 
530 	 * <p>
531 	 * Note on scope: This method will descend into any contained resources ({@link IResource#getContained()}) as well, but will not descend into linked resources (e.g.
532 	 * {@link BaseResourceReferenceDt#getResource()}) or embedded resources (e.g. Bundle.entry.resource)
533 	 * </p>
534 	 * 
535 	 * @param theResource
536 	 *           The resource to visit
537 	 * @param theVisitor
538 	 *           The visitor
539 	 */
540 	void visit(IBaseResource theResource, IModelVisitor2 theVisitor) {
541 		BaseRuntimeElementCompositeDefinition<?> def = myContext.getResourceDefinition(theResource);
542 		visit(theResource, null, def, theVisitor, new ArrayList<IBase>(), new ArrayList<BaseRuntimeChildDefinition>(), new ArrayList<BaseRuntimeElementDefinition<?>>());
543 	}
544 
545 	private void visit(IdentityHashMap<Object, Object> theStack, IBase theElement, List<String> thePathToElement, BaseRuntimeChildDefinition theChildDefinition,
546 			BaseRuntimeElementDefinition<?> theDefinition, IModelVisitor theCallback) {
547 		List<String> pathToElement = addNameToList(thePathToElement, theChildDefinition);
548 
549 		if (theStack.put(theElement, theElement) != null) {
550 			return;
551 		}
552 		
553 		theCallback.acceptElement(theElement, pathToElement, theChildDefinition, theDefinition);
554 		addUndeclaredExtensions(theElement, theDefinition, theChildDefinition, theCallback);
555 
556 		BaseRuntimeElementDefinition<?> def = theDefinition;
557 		if (def.getChildType() == ChildTypeEnum.CONTAINED_RESOURCE_LIST) {
558 			def = myContext.getElementDefinition(theElement.getClass());
559 		}
560 		
561 		switch (def.getChildType()) {
562 		case ID_DATATYPE:
563 		case PRIMITIVE_XHTML_HL7ORG:
564 		case PRIMITIVE_XHTML:
565 		case PRIMITIVE_DATATYPE:
566 			// These are primitive types
567 			break;
568 		case RESOURCE:
569 		case RESOURCE_BLOCK:
570 		case COMPOSITE_DATATYPE: {
571 			BaseRuntimeElementCompositeDefinition<?> childDef = (BaseRuntimeElementCompositeDefinition<?>) def;
572 			for (BaseRuntimeChildDefinition nextChild : childDef.getChildrenAndExtension()) {
573 				
574 				List<?> values = nextChild.getAccessor().getValues(theElement);
575 				if (values != null) {
576 					for (Object nextValueObject : values) {
577 						IBase nextValue;
578 						try {
579 							nextValue = (IBase) nextValueObject;
580 						} catch (ClassCastException e) {
581 							String s = "Found instance of " + nextValueObject.getClass() + " - Did you set a field value to the incorrect type? Expected " + IBase.class.getName();
582 							throw new ClassCastException(s);
583 						}
584 						if (nextValue == null) {
585 							continue;
586 						}
587 						if (nextValue.isEmpty()) {
588 							continue;
589 						}
590 						BaseRuntimeElementDefinition<?> childElementDef;
591 						childElementDef = nextChild.getChildElementDefinitionByDatatype(nextValue.getClass());
592 
593 						if (childElementDef == null) {
594 							childElementDef = myContext.getElementDefinition(nextValue.getClass());
595 						}
596 
597 						if (nextChild instanceof RuntimeChildDirectResource) {
598 							// Don't descend into embedded resources
599 							theCallback.acceptElement(nextValue, null, nextChild, childElementDef);
600 						} else {
601 							visit(theStack, nextValue, pathToElement, nextChild, childElementDef, theCallback);
602 						}
603 					}
604 				}
605 			}
606 			break;
607 		}
608 		case CONTAINED_RESOURCES: {
609 			BaseContainedDt value = (BaseContainedDt) theElement;
610 			for (IResource next : value.getContainedResources()) {
611 				def = myContext.getResourceDefinition(next);
612 				visit(theStack, next, pathToElement, null, def, theCallback);
613 			}
614 			break;
615 		}
616 		case CONTAINED_RESOURCE_LIST:
617 		case EXTENSION_DECLARED:
618 		case UNDECL_EXT: {
619 			throw new IllegalStateException("state should not happen: " + def.getChildType());
620 		}
621 		}
622 		
623 		theStack.remove(theElement);
624 		
625 	}

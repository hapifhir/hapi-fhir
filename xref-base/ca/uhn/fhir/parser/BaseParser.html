98  	protected Iterable<CompositeChildElement> compositeChildIterator(final List<? extends BaseRuntimeChildDefinition> theChildren, final boolean theContainedResource, final CompositeChildElement theParent) {
99  		return new Iterable<BaseParser.CompositeChildElement>() {
100 			@Override
101 			public Iterator<CompositeChildElement> iterator() {
102 
103 				return new Iterator<CompositeChildElement>() {
104 					private Iterator<? extends BaseRuntimeChildDefinition> myChildrenIter;
105 					private Boolean myHasNext = null;
106 					private CompositeChildElement myNext;
107 
108 					/**
109 					 * Constructor
110 					 */
111 					{
112 						myChildrenIter = theChildren.iterator();
113 					}
114 
115 					@Override
116 					public boolean hasNext() {
117 						if (myHasNext != null) {
118 							return myHasNext;
119 						}
120 
121 						myNext = null;
122 						do {
123 							if (myChildrenIter.hasNext() == false) {
124 								myHasNext = Boolean.FALSE;
125 								return false;
126 							}
127 
128 							myNext = new CompositeChildElement(theParent, myChildrenIter.next());
129 
130 							/*
131 							 * There are lots of reasons we might skip encoding a particular child
132 							 */
133 							if (myNext.getDef().getElementName().equals("extension") || myNext.getDef().getElementName().equals("modifierExtension")) {
134 								myNext = null;
135 							} else if (myNext.getDef().getElementName().equals("id")) {
136 								myNext = null;
137 							} else if (!myNext.shouldBeEncoded()) {
138 								myNext = null;
139 							} else if (isSummaryMode() && !myNext.getDef().isSummary()) {
140 								myNext = null;
141 							} else if (myNext.getDef() instanceof RuntimeChildNarrativeDefinition) {
142 								if (isSuppressNarratives() || isSummaryMode()) {
143 									myNext = null;
144 								} else if (theContainedResource) {
145 									myNext = null;
146 								}
147 							} else if (myNext.getDef() instanceof RuntimeChildContainedResources) {
148 								if (theContainedResource) {
149 									myNext = null;
150 								}
151 							}
152 
153 						} while (myNext == null);
154 
155 						myHasNext = true;
156 						return true;
157 					}
158 
159 					@Override
160 					public CompositeChildElement next() {
161 						if (myHasNext == null) {
162 							if (!hasNext()) {
163 								throw new IllegalStateException();
164 							}
165 						}
166 						CompositeChildElement retVal = myNext;
167 						myNext = null;
168 						myHasNext = null;
169 						return retVal;
170 					}
171 
172 					@Override
173 					public void remove() {
174 						throw new UnsupportedOperationException();
175 					}
176 				};
177 			}
178 		};
179 	}

181 	private void containResourcesForEncoding(ContainedResources theContained, IBaseResource theResource, IBaseResource theTarget) {
182 		Set<String> allIds = new HashSet<String>();
183 		Map<String, IBaseResource> existingIdToContainedResource = null;
184 
185 		if (theTarget instanceof IResource) {
186 			List<? extends IResource> containedResources = ((IResource) theTarget).getContained().getContainedResources();
187 			for (IResource next : containedResources) {
188 				String nextId = next.getId().getValue();
189 				if (StringUtils.isNotBlank(nextId)) {
190 					if (!nextId.startsWith("#")) {
191 						nextId = '#' + nextId;
192 					}
193 					allIds.add(nextId);
194 					if (existingIdToContainedResource == null) {
195 						existingIdToContainedResource = new HashMap<String, IBaseResource>();
196 					}
197 					existingIdToContainedResource.put(nextId, next);
198 				}
199 			}
200 		} else if (theTarget instanceof IDomainResource) {
201 			List<? extends IAnyResource> containedResources = ((IDomainResource) theTarget).getContained();
202 			for (IAnyResource next : containedResources) {
203 				String nextId = next.getIdElement().getValue();
204 				if (StringUtils.isNotBlank(nextId)) {
205 					if (!nextId.startsWith("#")) {
206 						nextId = '#' + nextId;
207 					}
208 					allIds.add(nextId);
209 					if (existingIdToContainedResource == null) {
210 						existingIdToContainedResource = new HashMap<String, IBaseResource>();
211 					}
212 					existingIdToContainedResource.put(nextId, next);
213 				}
214 			}
215 		} else {
216 			// no resources to contain
217 		}
218 
219 		{
220 			List<IBaseReference> allElements = myContext.newTerser().getAllPopulatedChildElementsOfType(theResource, IBaseReference.class);
221 			for (IBaseReference next : allElements) {
222 				IBaseResource resource = next.getResource();
223 				if (resource != null) {
224 					if (resource.getIdElement().isEmpty() || resource.getIdElement().isLocal()) {
225 						if (theContained.getResourceId(resource) != null) {
226 							// Prevent infinite recursion if there are circular loops in the contained resources
227 							continue;
228 						}
229 						theContained.addContained(resource);
230 					} else {
231 						continue;
232 					}
233 
234 					containResourcesForEncoding(theContained, resource, theTarget);
235 				} else if (next.getReferenceElement().isLocal()) {
236 					if (existingIdToContainedResource != null) {
237 						IBaseResource potentialTarget = existingIdToContainedResource.remove(next.getReferenceElement().getValue());
238 						if (potentialTarget != null) {
239 							theContained.addContained(next.getReferenceElement(), potentialTarget);
240 							containResourcesForEncoding(theContained, potentialTarget, theTarget);
241 						}
242 					}
243 				}
244 			}
245 		}
246 
247 	}

252 		myContainedResources = contained;

255 	protected String determineReferenceText(IBaseReference theRef) {
256 		IIdType ref = theRef.getReferenceElement();
257 		if (isBlank(ref.getIdPart())) {
258 			String reference = ref.getValue();
259 			if (theRef.getResource() != null) {
260 				IIdType containedId = getContainedResources().getResourceId(theRef.getResource());
261 				if (containedId != null && !containedId.isEmpty()) {
262 					if (containedId.isLocal()) {
263 						reference = containedId.getValue();
264 					} else {
265 						reference = "#" + containedId.getValue();
266 					}
267 				} else {
268 					IIdType refId = theRef.getResource().getIdElement();
269 					if (refId != null) {
270 						if (refId.hasIdPart()) {
271 							if (!refId.hasResourceType()) {
272 								refId = refId.withResourceType(myContext.getResourceDefinition(theRef.getResource()).getName());
273 							}
274 							if (isStripVersionsFromReferences()) {
275 								reference = refId.toVersionless().getValue();
276 							} else {
277 								reference = refId.getValue();
278 							}
279 						}
280 					}
281 				}
282 			}
283 			return reference;
284 		} else {
285 			if (!ref.hasResourceType() && !ref.isLocal() && theRef.getResource() != null) {
286 				ref = ref.withResourceType(myContext.getResourceDefinition(theRef.getResource()).getName());
287 			}
288 			if (isNotBlank(myServerBaseUrl) && StringUtils.equals(myServerBaseUrl, ref.getBaseUrl())) {
289 				if (isStripVersionsFromReferences()) {
290 					return ref.toUnqualifiedVersionless().getValue();
291 				} else {
292 					return ref.toUnqualified().getValue();
293 				}
294 			} else {
295 				if (isStripVersionsFromReferences()) {
296 					return ref.toVersionless().getValue();
297 				} else {
298 					return ref.getValue();
299 				}
300 			}
301 		}
302 	}

304 	protected String determineResourceBaseUrl(String bundleBaseUrl, BundleEntry theEntry) {
305 		IResource resource = theEntry.getResource();
306 		if (resource == null) {
307 			return null;
308 		}
309 
310 		String resourceBaseUrl = null;
311 		if (resource.getId() != null && resource.getId().hasBaseUrl()) {
312 			if (!resource.getId().getBaseUrl().equals(bundleBaseUrl)) {
313 				resourceBaseUrl = resource.getId().getBaseUrl();
314 			}
315 		}
316 		return resourceBaseUrl;

472 	@Override
473 	public <T extends IBaseResource> T parseResource(Class<T> theResourceType, Reader theReader) throws DataFormatException {
474 		T retVal = doParseResource(theResourceType, theReader);
475 
476 		RuntimeResourceDefinition def = myContext.getResourceDefinition(retVal);
477 		if ("Bundle".equals(def.getName())) {
478 
479 			BaseRuntimeChildDefinition entryChild = def.getChildByName("entry");
480 			BaseRuntimeElementCompositeDefinition<?> entryDef = (BaseRuntimeElementCompositeDefinition<?>) entryChild.getChildByName("entry");
481 			List<IBase> entries = entryChild.getAccessor().getValues(retVal);
482 			if (entries != null) {
483 				for (IBase nextEntry : entries) {
484 
485 					/**
486 					 * If Bundle.entry.fullUrl is populated, set the resource ID to that
487 					 */
488 					// TODO: should emit a warning and maybe notify the error handler if the resource ID doesn't match the
489 					// fullUrl idPart
490 					BaseRuntimeChildDefinition fullUrlChild = entryDef.getChildByName("fullUrl");
491 					if (fullUrlChild == null) {
492 						continue; // TODO: remove this once the data model in tinder plugin catches up to 1.2
493 					}
494 					List<IBase> fullUrl = fullUrlChild.getAccessor().getValues(nextEntry);
495 					if (fullUrl != null && !fullUrl.isEmpty()) {
496 						IPrimitiveType<?> value = (IPrimitiveType<?>) fullUrl.get(0);
497 						if (value.isEmpty() == false) {
498 							List<IBase> entryResources = entryDef.getChildByName("resource").getAccessor().getValues(nextEntry);
499 							if (entryResources != null && entryResources.size() > 0) {
500 								IBaseResource res = (IBaseResource) entryResources.get(0);
501 								String versionId = res.getIdElement().getVersionIdPart();
502 								res.setId(value.getValueAsString());
503 								if (isNotBlank(versionId) && res.getIdElement().hasVersionIdPart() == false) {
504 									res.setId(res.getIdElement().withVersion(versionId));
505 								}
506 							}
507 						}
508 					}
509 
510 				}
511 			}
512 
513 		}
514 
515 		return retVal;
516 	}

540 	@SuppressWarnings("cast")
541 	protected List<? extends IBase> preProcessValues(BaseRuntimeChildDefinition metaChildUncast, List<? extends IBase> theValues) {
542 		if (myContext.getVersion().getVersion().equals(FhirVersionEnum.DSTU2_HL7ORG)) {
543 			if (shouldAddSubsettedTag() && metaChildUncast.getValidChildNames().contains("meta")) {
544 				BaseRuntimeElementDefinition<?> childByName = metaChildUncast.getChildByName("meta");
545 				if (childByName instanceof BaseRuntimeElementCompositeDefinition<?>) {
546 					BaseRuntimeElementCompositeDefinition<?> metaChildUncast1 = (BaseRuntimeElementCompositeDefinition<?>) childByName;
547 					if (metaChildUncast1 != null) {
548 						if (IBaseMetaType.class.isAssignableFrom(metaChildUncast1.getImplementingClass())) {
549 							IBaseMetaType metaValue;
550 							if (theValues != null && theValues.size() >= 1) {
551 								metaValue = (IBaseMetaType) theValues.iterator().next();
552 								try {
553 									metaValue = (IBaseMetaType) metaValue.getClass().getMethod("copy").invoke(metaValue);
554 								} catch (Exception e) {
555 									throw new InternalErrorException("Failed to duplicate meta", e);
556 								}
557 							} else {
558 								metaValue = (IBaseMetaType) metaChildUncast1.newInstance();
559 							}
560 
561 							ArrayList<IBase> retVal = new ArrayList<IBase>();
562 							retVal.add(metaValue);
563 
564 							BaseRuntimeChildDefinition tagChild = metaChildUncast1.getChildByName("tag");
565 							BaseRuntimeElementCompositeDefinition<?> codingDef = (BaseRuntimeElementCompositeDefinition<?>) ((BaseRuntimeElementCompositeDefinition<?>) tagChild.getChildByName("tag"));
566 							IBase coding = codingDef.newInstance();
567 							tagChild.getMutator().addValue(metaValue, coding);
568 
569 							BaseRuntimeChildDefinition systemChild = codingDef.getChildByName("system");
570 							IPrimitiveType<?> system = (IPrimitiveType<?>) myContext.getElementDefinition("uri").newInstance();
571 							system.setValueAsString(Constants.TAG_SUBSETTED_SYSTEM);
572 							systemChild.getMutator().addValue(coding, system);
573 
574 							BaseRuntimeChildDefinition codeChild = codingDef.getChildByName("code");
575 							IPrimitiveType<?> code = (IPrimitiveType<?>) myContext.getElementDefinition("code").newInstance();
576 							code.setValueAsString(Constants.TAG_SUBSETTED_CODE);
577 							codeChild.getMutator().addValue(coding, code);
578 
579 							BaseRuntimeChildDefinition displayChild = codingDef.getChildByName("display");
580 							IPrimitiveType<?> display = (IPrimitiveType<?>) myContext.getElementDefinition("string").newInstance();
581 							display.setValueAsString("Resource encoded in summary mode");
582 							displayChild.getMutator().addValue(coding, display);
583 
584 							return retVal;
585 						}
586 					}
587 				}
588 			}
589 		}
590 
591 		return theValues;
592 	}

594 	@Override
595 	public void setEncodeElements(Set<String> theEncodeElements) {
596 		myEncodeElementsIncludesStars = false;
597 		if (theEncodeElements == null || theEncodeElements.isEmpty()) {
598 			myEncodeElements = null;
599 		} else {
600 			myEncodeElements = theEncodeElements;
601 			for (String next : theEncodeElements) {

606 		}

660 		if (nextChild instanceof BaseRuntimeDeclaredChildDefinition) {

662 			b.append(((BaseRuntimeDeclaredChildDefinition) nextChild).getElementName());
663 			b.append(" has type ");
664 			b.append(theType.getName());
665 			b.append(" but this is not a valid type for this element");
666 			if (nextChild instanceof RuntimeChildChoiceDefinition) {

672 		throw new DataFormatException(nextChild + " has no child of type " + theType);

683 	protected class CompositeChildElement {
684 		private final BaseRuntimeChildDefinition myDef;
685 		private final CompositeChildElement myParent;
686 		private final RuntimeResourceDefinition myResDef;
687 
688 		public CompositeChildElement(CompositeChildElement theParent, BaseRuntimeChildDefinition theDef) {
689 			myDef = theDef;
690 			myParent = theParent;
691 			myResDef = null;
692 
693 			if (ourLog.isTraceEnabled()) {
694 				if (theParent != null) {
695 					StringBuilder path = theParent.buildPath();
696 					if (path != null) {
697 						path.append('.');
698 						path.append(myDef.getElementName());
699 						ourLog.trace(" * Next path: {}", path.toString());
700 					}
701 				}
702 			}
703 
704 		}
705 
706 
707 		public CompositeChildElement(RuntimeResourceDefinition theResDef) {
708 			myResDef = theResDef;
709 			myDef = null;
710 			myParent = null;
711 		}
712 
713 		private StringBuilder buildPath() {
714 			if (myResDef != null) {
715 				StringBuilder b = new StringBuilder();
716 				b.append(myResDef.getName());
717 				return b;
718 			} else {
719 				StringBuilder b = myParent.buildPath();
720 				if (b != null && myDef != null) {
721 					b.append('.');
722 					b.append(myDef.getElementName());
723 				}
724 				return b;
725 			}
726 		}
727 
728 		private boolean checkIfParentShouldBeEncodedAndBuildPath(StringBuilder theB, boolean theStarPass) {
729 			if (myResDef != null) {
730 				if (myEncodeElementsAppliesToResourceTypes != null) {
731 					if (!myEncodeElementsAppliesToResourceTypes.contains(myResDef.getName())) {
732 						return true;
733 					}
734 				}
735 				if (theStarPass) {
736 					theB.append('*');
737 				} else {
738 					theB.append(myResDef.getName());
739 				}
740 				if (myEncodeElements.contains(theB.toString())) {
741 					return true;
742 				} else {
743 					return false;
744 				}
745 			} else if (myParent != null) {
746 				if (myParent.checkIfParentShouldBeEncodedAndBuildPath(theB, theStarPass)) {
747 					return true;
748 				}
749 
750 				if (myDef != null) {
751 					theB.append('.');
752 					theB.append(myDef.getElementName());
753 					return myEncodeElements.contains(theB.toString());
754 				}
755 			}
756 
757 			return true;
758 		}
759 
760 		public BaseRuntimeChildDefinition getDef() {
761 			return myDef;
762 		}
763 
764 		public CompositeChildElement getParent() {
765 			return myParent;
766 		}
767 
768 		public RuntimeResourceDefinition getResDef() {
769 			return myResDef;
770 		}
771 
772 		public boolean shouldBeEncoded() {
773 			if (myEncodeElements == null) {
774 				return true;
775 			}
776 			boolean retVal = checkIfParentShouldBeEncodedAndBuildPath(new StringBuilder(), false);
777 			if (retVal == false && myEncodeElementsIncludesStars) {
778 				retVal = checkIfParentShouldBeEncodedAndBuildPath(new StringBuilder(), true);
779 			}
780 			return retVal;
781 		}

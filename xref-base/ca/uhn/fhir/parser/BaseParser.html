107 	protected Iterable<CompositeChildElement> compositeChildIterator(IBase theCompositeElement, final boolean theContainedResource, final CompositeChildElement theParent) {
108 		
109 		BaseRuntimeElementCompositeDefinition<?> elementDef = (BaseRuntimeElementCompositeDefinition<?>) myContext.getElementDefinition(theCompositeElement.getClass());
110 		final List<BaseRuntimeChildDefinition> children = elementDef.getChildrenAndExtension();
111 		
112 		return new Iterable<BaseParser.CompositeChildElement>() {
113 			@Override
114 			public Iterator<CompositeChildElement> iterator() {
115 
116 				return new Iterator<CompositeChildElement>() {
117 					private Iterator<? extends BaseRuntimeChildDefinition> myChildrenIter;
118 					private Boolean myHasNext = null;
119 					private CompositeChildElement myNext;
120 
121 					/**
122 					 * Constructor
123 					 */
124 					{
125 						myChildrenIter = children.iterator();
126 					}
127 
128 					@Override
129 					public boolean hasNext() {
130 						if (myHasNext != null) {
131 							return myHasNext;
132 						}
133 
134 						myNext = null;
135 						do {
136 							if (myChildrenIter.hasNext() == false) {
137 								myHasNext = Boolean.FALSE;
138 								return false;
139 							}
140 
141 							myNext = new CompositeChildElement(theParent, myChildrenIter.next());
142 
143 							/*
144 							 * There are lots of reasons we might skip encoding a particular child
145 							 */
146 //							if (myNext.getDef().getElementName().equals("extension") || myNext.getDef().getElementName().equals("modifierExtension")) {
147 //								myNext = null;
148 //							} else 
149 							if (myNext.getDef().getElementName().equals("id")) {
150 								myNext = null;
151 							} else if (!myNext.shouldBeEncoded()) {
152 								myNext = null;
153 							} else if (isSummaryMode() && !myNext.getDef().isSummary()) {
154 								myNext = null;
155 							} else if (myNext.getDef() instanceof RuntimeChildNarrativeDefinition) {
156 								if (isSuppressNarratives() || isSummaryMode()) {
157 									myNext = null;
158 								} else if (theContainedResource) {
159 									myNext = null;
160 								}
161 							} else if (myNext.getDef() instanceof RuntimeChildContainedResources) {
162 								if (theContainedResource) {
163 									myNext = null;
164 								}
165 							}
166 
167 						} while (myNext == null);
168 
169 						myHasNext = true;
170 						return true;
171 					}
172 
173 					@Override
174 					public CompositeChildElement next() {
175 						if (myHasNext == null) {
176 							if (!hasNext()) {
177 								throw new IllegalStateException();
178 							}
179 						}
180 						CompositeChildElement retVal = myNext;
181 						myNext = null;
182 						myHasNext = null;
183 						return retVal;
184 					}
185 
186 					@Override
187 					public void remove() {
188 						throw new UnsupportedOperationException();
189 					}
190 				};
191 			}
192 		};
193 	}

195 	private void containResourcesForEncoding(ContainedResources theContained, IBaseResource theResource, IBaseResource theTarget) {
196 		Set<String> allIds = new HashSet<String>();
197 		Map<String, IBaseResource> existingIdToContainedResource = null;
198 
199 		if (theTarget instanceof IResource) {
200 			List<? extends IResource> containedResources = ((IResource) theTarget).getContained().getContainedResources();
201 			for (IResource next : containedResources) {
202 				String nextId = next.getId().getValue();
203 				if (StringUtils.isNotBlank(nextId)) {
204 					if (!nextId.startsWith("#")) {
205 						nextId = '#' + nextId;
206 					}
207 					allIds.add(nextId);
208 					if (existingIdToContainedResource == null) {
209 						existingIdToContainedResource = new HashMap<String, IBaseResource>();
210 					}
211 					existingIdToContainedResource.put(nextId, next);
212 				}
213 			}
214 		} else if (theTarget instanceof IDomainResource) {
215 			List<? extends IAnyResource> containedResources = ((IDomainResource) theTarget).getContained();
216 			for (IAnyResource next : containedResources) {
217 				String nextId = next.getIdElement().getValue();
218 				if (StringUtils.isNotBlank(nextId)) {
219 					if (!nextId.startsWith("#")) {
220 						nextId = '#' + nextId;
221 					}
222 					allIds.add(nextId);
223 					if (existingIdToContainedResource == null) {
224 						existingIdToContainedResource = new HashMap<String, IBaseResource>();
225 					}
226 					existingIdToContainedResource.put(nextId, next);
227 				}
228 			}
229 		} else {
230 			// no resources to contain
231 		}
232 
233 		{
234 			List<IBaseReference> allElements = myContext.newTerser().getAllPopulatedChildElementsOfType(theResource, IBaseReference.class);
235 			for (IBaseReference next : allElements) {
236 				IBaseResource resource = next.getResource();
237 				if (resource != null) {
238 					if (resource.getIdElement().isEmpty() || resource.getIdElement().isLocal()) {
239 						if (theContained.getResourceId(resource) != null) {
240 							// Prevent infinite recursion if there are circular loops in the contained resources
241 							continue;
242 						}
243 						theContained.addContained(resource);
244 					} else {
245 						continue;
246 					}
247 
248 					containResourcesForEncoding(theContained, resource, theTarget);
249 				} else if (next.getReferenceElement().isLocal()) {
250 					if (existingIdToContainedResource != null) {
251 						IBaseResource potentialTarget = existingIdToContainedResource.remove(next.getReferenceElement().getValue());
252 						if (potentialTarget != null) {
253 							theContained.addContained(next.getReferenceElement(), potentialTarget);
254 							containResourcesForEncoding(theContained, potentialTarget, theTarget);
255 						}
256 					}
257 				}
258 			}
259 		}
260 
261 	}

269 	private String determineReferenceText(IBaseReference theRef) {
270 		IIdType ref = theRef.getReferenceElement();
271 		if (isBlank(ref.getIdPart())) {
272 			String reference = ref.getValue();
273 			if (theRef.getResource() != null) {
274 				IIdType containedId = getContainedResources().getResourceId(theRef.getResource());
275 				if (containedId != null && !containedId.isEmpty()) {
276 					if (containedId.isLocal()) {
277 						reference = containedId.getValue();
278 					} else {
279 						reference = "#" + containedId.getValue();
280 					}
281 				} else {
282 					IIdType refId = theRef.getResource().getIdElement();
283 					if (refId != null) {
284 						if (refId.hasIdPart()) {
285 							if (!refId.hasResourceType()) {
286 								refId = refId.withResourceType(myContext.getResourceDefinition(theRef.getResource()).getName());
287 							}
288 							if (isStripVersionsFromReferences()) {
289 								reference = refId.toVersionless().getValue();
290 							} else {
291 								reference = refId.getValue();
292 							}
293 						}
294 					}
295 				}
296 			}
297 			return reference;
298 		} else {
299 			if (!ref.hasResourceType() && !ref.isLocal() && theRef.getResource() != null) {
300 				ref = ref.withResourceType(myContext.getResourceDefinition(theRef.getResource()).getName());
301 			}
302 			if (isNotBlank(myServerBaseUrl) && StringUtils.equals(myServerBaseUrl, ref.getBaseUrl())) {
303 				if (isStripVersionsFromReferences()) {
304 					return ref.toUnqualifiedVersionless().getValue();
305 				} else {
306 					return ref.toUnqualified().getValue();
307 				}
308 			} else {
309 				if (isStripVersionsFromReferences()) {
310 					return ref.toVersionless().getValue();
311 				} else {
312 					return ref.getValue();
313 				}
314 			}
315 		}
316 	}

318 	protected String determineResourceBaseUrl(String bundleBaseUrl, BundleEntry theEntry) {
319 		IResource resource = theEntry.getResource();
320 		if (resource == null) {
321 			return null;
322 		}
323 
324 		String resourceBaseUrl = null;
325 		if (resource.getId() != null && resource.getId().hasBaseUrl()) {
326 			if (!resource.getId().getBaseUrl().equals(bundleBaseUrl)) {
327 				resourceBaseUrl = resource.getId().getBaseUrl();
328 			}
329 		}
330 		return resourceBaseUrl;

396 		for (int i = 0; i < tagList.size(); i++) {
397 			if (isBlank(tagList.get(i).getCode()) && isBlank(tagList.get(i).getSystem())) {

458 	@SuppressWarnings("deprecation")
459 	protected <T extends IPrimitiveType<String>> List<T> getProfileTagsForEncoding(IBaseResource theResource, List<T> theProfiles) {
460 		switch (myContext.getAddProfileTagWhenEncoding()) {
461 		case NEVER:
462 			return theProfiles;
463 		case ONLY_FOR_CUSTOM:
464 			RuntimeResourceDefinition resDef = myContext.getResourceDefinition(theResource);
465 			if (resDef.isStandardType()) {
466 				return theProfiles;
467 			}
468 			break;
469 		case ALWAYS:
470 			break;
471 		}
472 
473 		if (myContext.getVersion().getVersion().isNewerThan(FhirVersionEnum.DSTU1)) {
474 			RuntimeResourceDefinition nextDef = myContext.getResourceDefinition(theResource);
475 			String profile = nextDef.getResourceProfile(myServerBaseUrl);
476 			if (isNotBlank(profile)) {
477 				for (T next : theProfiles) {
478 					if (profile.equals(next.getValue())) {
479 						return theProfiles;
480 					}
481 				}
482 
483 				List<T> newList = new ArrayList<T>();
484 				newList.addAll(theProfiles);
485 
486 				BaseRuntimeElementDefinition<?> idElement = myContext.getElementDefinition("id");
487 				@SuppressWarnings("unchecked")
488 				T newId = (T) idElement.newInstance();
489 				newId.setValue(profile);
490 
491 				newList.add(newId);
492 				return newList;
493 			}
494 		}
495 
496 		return theProfiles;
497 	}

553 	@Override
554 	public <T extends IBaseResource> T parseResource(Class<T> theResourceType, Reader theReader) throws DataFormatException {
555 		
556 		if (theResourceType != null) {
557 			RuntimeResourceDefinition def = myContext.getResourceDefinition(theResourceType);
558 			if (def.getStructureVersion() != myContext.getVersion().getVersion()) {
559 				throw new IllegalArgumentException("This parser is for FHIR version " + myContext.getVersion().getVersion() + " - Can not parse a structure for version " + def.getStructureVersion());
560 			}
561 		}
562 		
563 		T retVal = doParseResource(theResourceType, theReader);
564 
565 		RuntimeResourceDefinition def = myContext.getResourceDefinition(retVal);
566 		if ("Bundle".equals(def.getName())) {
567 
568 			BaseRuntimeChildDefinition entryChild = def.getChildByName("entry");
569 			BaseRuntimeElementCompositeDefinition<?> entryDef = (BaseRuntimeElementCompositeDefinition<?>) entryChild.getChildByName("entry");
570 			List<IBase> entries = entryChild.getAccessor().getValues(retVal);
571 			if (entries != null) {
572 				for (IBase nextEntry : entries) {
573 
574 					/**
575 					 * If Bundle.entry.fullUrl is populated, set the resource ID to that
576 					 */
577 					// TODO: should emit a warning and maybe notify the error handler if the resource ID doesn't match the
578 					// fullUrl idPart
579 					BaseRuntimeChildDefinition fullUrlChild = entryDef.getChildByName("fullUrl");
580 					if (fullUrlChild == null) {
581 						continue; // TODO: remove this once the data model in tinder plugin catches up to 1.2
582 					}
583 					List<IBase> fullUrl = fullUrlChild.getAccessor().getValues(nextEntry);
584 					if (fullUrl != null && !fullUrl.isEmpty()) {
585 						IPrimitiveType<?> value = (IPrimitiveType<?>) fullUrl.get(0);
586 						if (value.isEmpty() == false) {
587 							List<IBase> entryResources = entryDef.getChildByName("resource").getAccessor().getValues(nextEntry);
588 							if (entryResources != null && entryResources.size() > 0) {
589 								IBaseResource res = (IBaseResource) entryResources.get(0);
590 								String versionId = res.getIdElement().getVersionIdPart();
591 								res.setId(value.getValueAsString());
592 								if (isNotBlank(versionId) && res.getIdElement().hasVersionIdPart() == false) {
593 									res.setId(res.getIdElement().withVersion(versionId));
594 								}
595 							}
596 						}
597 					}
598 
599 				}
600 			}
601 
602 		}
603 
604 		return retVal;
605 	}

629 	protected List<? extends IBase> preProcessValues(BaseRuntimeChildDefinition metaChildUncast, IBaseResource theResource, List<? extends IBase> theValues) {
630 		if (myContext.getVersion().getVersion().isRi()) {
631 
632 			/*
633 			 * If we're encoding the meta tag, we do some massaging of the meta values before
634 			 * encoding. Buf if there is no meta element at all, we create one since we're possibly going to be
635 			 * adding things to it
636 			 */
637 			if (theValues.isEmpty() && metaChildUncast.getElementName().equals("meta")) {
638 				BaseRuntimeElementDefinition<?> metaChild = metaChildUncast.getChildByName("meta");
639 				if (IBaseMetaType.class.isAssignableFrom(metaChild.getImplementingClass())) {
640 					IBaseMetaType newType = (IBaseMetaType) metaChild.newInstance();
641 					theValues = Collections.singletonList(newType);
642 				}
643 			}
644 
645 			if (theValues.size() == 1 && theValues.get(0) instanceof IBaseMetaType) {
646 
647 				IBaseMetaType metaValue = (IBaseMetaType) theValues.get(0);
648 				try {
649 					metaValue = (IBaseMetaType) metaValue.getClass().getMethod("copy").invoke(metaValue);
650 				} catch (Exception e) {
651 					throw new InternalErrorException("Failed to duplicate meta", e);
652 				}
653 
654 				if (isBlank(metaValue.getVersionId())) {
655 					if (theResource.getIdElement().hasVersionIdPart()) {
656 						metaValue.setVersionId(theResource.getIdElement().getVersionIdPart());
657 					}
658 				}
659 
660 				filterCodingsWithNoCodeOrSystem(metaValue.getTag());
661 				filterCodingsWithNoCodeOrSystem(metaValue.getSecurity());
662 
663 				List<? extends IPrimitiveType<String>> newProfileList = getProfileTagsForEncoding(theResource, metaValue.getProfile());
664 				List<? extends IPrimitiveType<String>> oldProfileList = metaValue.getProfile();
665 				if (oldProfileList != newProfileList) {
666 					oldProfileList.clear();
667 					for (IPrimitiveType<String> next : newProfileList) {
668 						if (isNotBlank(next.getValue())) {
669 							metaValue.addProfile(next.getValue());
670 						}
671 					}
672 				}
673 
674 				if (shouldAddSubsettedTag()) {
675 					IBaseCoding coding = metaValue.addTag();
676 					coding.setCode(Constants.TAG_SUBSETTED_CODE);
677 					coding.setSystem(Constants.TAG_SUBSETTED_SYSTEM);
678 					coding.setDisplay(subsetDescription());
679 				}
680 
681 				return Collections.singletonList(metaValue);
682 			}
683 		}
684 
685 		@SuppressWarnings("unchecked")
686 		List<IBase> retVal = (List<IBase>) theValues;
687 		
688 		for (int i = 0; i < retVal.size(); i++) {
689 			IBase next = retVal.get(i);
690 			
691 			/*
692 			 * If we have automatically contained any resources via
693 			 * their references, this ensures that we output the new
694 			 * local reference
695 			 */
696 			if (next instanceof IBaseReference) {
697 				IBaseReference nextRef = (IBaseReference)next;
698 				String refText = determineReferenceText(nextRef);
699 				if (!StringUtils.equals(refText, nextRef.getReferenceElement().getValue())) {
700 					
701 					if (retVal == theValues) {
702 						retVal = new ArrayList<IBase>(theValues);
703 					}
704 					IBaseReference newRef = (IBaseReference) myContext.getElementDefinition(nextRef.getClass()).newInstance();
705 					myContext.newTerser().cloneInto(nextRef, newRef, true);
706 					newRef.setReference(refText);
707 					retVal.set(i, newRef);
708 					
709 				}
710 			}
711 		}
712 		
713 		return retVal;
714 	}

867 	static boolean hasExtensions(IBase theElement) {
868 		if (theElement instanceof ISupportsUndeclaredExtensions) {
869 			ISupportsUndeclaredExtensions res = (ISupportsUndeclaredExtensions) theElement;
870 			if (res.getUndeclaredExtensions().size() > 0 || res.getUndeclaredModifierExtensions().size() > 0) {
871 				return true;
872 			}
873 		}
874 		if (theElement instanceof IBaseHasExtensions) {
875 			IBaseHasExtensions res = (IBaseHasExtensions) theElement;
876 			if (res.hasExtension()) {
877 				return true;
878 			}
879 		}
880 		if (theElement instanceof IBaseHasModifierExtensions) {
881 			IBaseHasModifierExtensions res = (IBaseHasModifierExtensions) theElement;
882 			if (res.hasModifierExtension()) {
883 				return true;
884 			}
885 		}
886 		return false;
887 	}

918 		private StringBuilder buildPath() {
919 			if (myResDef != null) {
920 				StringBuilder b = new StringBuilder();
921 				b.append(myResDef.getName());
922 				return b;
923 			} else {
924 				StringBuilder b = myParent.buildPath();
925 				if (b != null && myDef != null) {
926 					b.append('.');
927 					b.append(myDef.getElementName());
928 				}
929 				return b;
930 			}
931 		}

933 		private boolean checkIfParentShouldBeEncodedAndBuildPath(StringBuilder thePathBuilder, boolean theStarPass) {
934 			return checkIfPathMatches(thePathBuilder, theStarPass, myEncodeElementsAppliesToResourceTypes, myEncodeElements, true);
935 		}
936 
937 		private boolean checkIfParentShouldNotBeEncodedAndBuildPath(StringBuilder thePathBuilder, boolean theStarPass) {
938 			return checkIfPathMatches(thePathBuilder, theStarPass, null, myDontEncodeElements, false);
939 		}

941 		private boolean checkIfPathMatches(StringBuilder thePathBuilder, boolean theStarPass, Set<String> theResourceTypes, Set<String> theElements, boolean theCheckingForWhitelist) {
942 			if (myResDef != null) {
943 				if (theResourceTypes != null) {
944 					if (!theResourceTypes.contains(myResDef.getName())) {
945 						return true;
946 					}
947 				}
948 				if (theStarPass) {
949 					thePathBuilder.append('*');
950 				} else {
951 					thePathBuilder.append(myResDef.getName());
952 				}
953 				if (theElements.contains(thePathBuilder.toString())) {
954 					return true;
955 				} else {
956 					return false;
957 				}
958 			} else if (myParent != null) {
959 				boolean parentCheck;
960 				if (theCheckingForWhitelist) {
961 					parentCheck = myParent.checkIfParentShouldBeEncodedAndBuildPath(thePathBuilder, theStarPass);
962 				} else {
963 					parentCheck = myParent.checkIfParentShouldNotBeEncodedAndBuildPath(thePathBuilder, theStarPass);
964 				}
965 				if (parentCheck) {
966 					return true;
967 				}
968 
969 				if (myDef != null) {
970 					thePathBuilder.append('.');
971 					thePathBuilder.append(myDef.getElementName());
972 					return theElements.contains(thePathBuilder.toString());
973 				}
974 			}
975 
976 			return true;
977 		}

991 		public boolean shouldBeEncoded() {
992 			boolean retVal = true;
993 			if (myEncodeElements != null) {
994 				retVal = checkIfParentShouldBeEncodedAndBuildPath(new StringBuilder(), false);
995 				if (retVal == false && myEncodeElementsIncludesStars) {
996 					retVal = checkIfParentShouldBeEncodedAndBuildPath(new StringBuilder(), true);
997 				}
998 			}
999 			if (retVal && myDontEncodeElements != null) {
1000 				retVal = !checkIfParentShouldNotBeEncodedAndBuildPath(new StringBuilder(), false);
1001 				if (retVal && myDontEncodeElementsIncludesStars) {
1002 					retVal = !checkIfParentShouldNotBeEncodedAndBuildPath(new StringBuilder(), true);
1003 				}
1004 			}
1005 			return retVal;

171 	private void putPlacerResourceInDeletedEntry(BundleEntry entry) {
172 		IdDt id = null;
173 		if (entry.getLinkSelf() != null && entry.getLinkSelf().isEmpty() == false) {
174 			id = new IdDt(entry.getLinkSelf().getValue());
175 		} else {
176 			id = entry.getId();
177 		}
178 
179 		IResource resource = entry.getResource();
180 		if (resource == null && id != null && isNotBlank(id.getResourceType())) {
181 			String resourceType = id.getResourceType();
182 			RuntimeResourceDefinition def = myContext.getResourceDefinition(resourceType);
183 			if (def == null) {
184 				throw new DataFormatException("Entry references unknown resource type: " + resourceType);
185 			}
186 			resource = (IResource) def.newInstance();
187 			resource.setId(id);
188 			entry.setResource(resource);
189 		}
190 
191 		if (resource != null) {
192 			resource.getResourceMetadata().put(ResourceMetadataKeyEnum.DELETED_AT, entry.getDeletedAt());
193 			resource.getResourceMetadata().put(ResourceMetadataKeyEnum.VERSION_ID, id);
194 		}
195 	}

283 
284 		private static final int STATE_LABEL = 2;
285 		private static final int STATE_NONE = 0;
286 		private static final int STATE_SCHEME = 3;
287 		private static final int STATE_TERM = 1;
288 
289 		private int myCatState = STATE_NONE;
290 		private String myLabel;
291 		private String myScheme;
292 		private TagList myTagList;
293 		private String myTerm;
294 
295 		public AtomCategoryState(TagList theTagList) {
296 			super(null);
297 			myTagList = theTagList;
298 		}
299 
300 		@Override
301 		public void attributeValue(String theName, String theValue) throws DataFormatException {
302 			if ("term".equals(theName)) {
303 				myTerm = theValue;
304 			} else if ("label".equals(theName)) {
305 				myLabel = theValue;
306 			} else if ("scheme".equals(theName)) {
307 				myScheme = theValue;
308 			} else if ("value".equals(theName)) {
309 				/*
310 				 * This handles XML parsing, which is odd for this quasi-resource type, since the tag has three values instead of one like everything else.
311 				 */
312 				switch (myCatState) {
313 				case STATE_LABEL:
314 					myLabel = theValue;
315 					break;
316 				case STATE_TERM:
317 					myTerm = theValue;
318 					break;
319 				case STATE_SCHEME:
320 					myScheme = theValue;
321 					break;
322 				default:
323 					super.string(theValue);
324 					break;
325 				}
326 
327 			}
328 		}
329 
330 		@Override
331 		public void endingElement() throws DataFormatException {
332 			if (myCatState != STATE_NONE) {
333 				myCatState = STATE_NONE;
334 			} else {
335 				if (isNotEmpty(myScheme) || isNotBlank(myTerm) || isNotBlank(myLabel)) {
336 					myTagList.addTag(myScheme, myTerm, myLabel);
337 				}
338 				pop();
339 			}
340 		}
341 
342 		@Override
343 		public void enteringNewElement(String theNamespaceUri, String theName) throws DataFormatException {
344 			if (myCatState != STATE_NONE) {
345 				throw new DataFormatException("Unexpected element in entry: " + theName);
346 			}
347 			if ("term".equals(theName)) {
348 				myCatState = STATE_TERM;
349 			} else if ("label".equals(theName)) {
350 				myCatState = STATE_LABEL;
351 			} else if ("scheme".equals(theName)) {
352 				myCatState = STATE_SCHEME;
353 			}
354 		}

478 			if ("title".equals(theLocalPart)) {
479 				push(new AtomPrimitiveState(myEntry.getTitle()));
480 			} else if ("id".equals(theLocalPart)) {
481 				push(new AtomPrimitiveState(myEntry.getId()));
482 			} else if ("link".equals(theLocalPart)) {
483 				push(new AtomLinkState(myEntry));
484 			} else if ("updated".equals(theLocalPart)) {
485 				push(new AtomPrimitiveState(myEntry.getUpdated()));
486 			} else if ("published".equals(theLocalPart)) {
487 				push(new AtomPrimitiveState(myEntry.getPublished()));
488 			} else if ("author".equals(theLocalPart)) {
489 				push(new AtomAuthorState(myEntry));
490 			} else if ("content".equals(theLocalPart)) {
491 				push(new PreResourceStateHapi(myEntry, myResourceType).setRequireResourceType(false));
492 			} else if ("summary".equals(theLocalPart)) {
493 				push(new XhtmlState(getPreResourceState(), myEntry.getSummary(), false));
494 			} else if ("category".equals(theLocalPart)) {
495 				push(new AtomCategoryState(myEntry.getCategories()));
496 			} else if ("deleted".equals(theLocalPart) && myJsonMode) {
497 				// JSON and XML deleted entries are completely different for some reason
498 				myDeleted = true;
499 				push(new AtomDeletedJsonWhenState(myEntry.getDeletedAt()));
500 			} else {
501 				throw new DataFormatException("Unexpected element in entry: " + theLocalPart);
502 			}
503 
504 			// TODO: handle category
505 		}

511 		@SuppressWarnings("deprecation")
512 		private void populateResourceMetadata() {
513 			if (myEntry.getResource() == null) {
514 				return;
515 			}
516 
517 			IdDt id = myEntry.getId();
518 			if (id != null && id.isEmpty() == false) {
519 				myEntry.getResource().setId(id);
520 			}
521 
522 			Map<ResourceMetadataKeyEnum<?>, Object> metadata = myEntry.getResource().getResourceMetadata();
523 			if (myEntry.getPublished().isEmpty() == false) {
524 				ResourceMetadataKeyEnum.PUBLISHED.put(myEntry.getResource(), myEntry.getPublished());
525 			}
526 			if (myEntry.getUpdated().isEmpty() == false) {
527 				ResourceMetadataKeyEnum.UPDATED.put(myEntry.getResource(), myEntry.getUpdated());
528 			}
529 
530 			ResourceMetadataKeyEnum.TITLE.put(myEntry.getResource(), myEntry.getTitle().getValue());
531 
532 			if (myEntry.getCategories().isEmpty() == false) {
533 				TagList tagList = new TagList();
534 				for (Tag next : myEntry.getCategories()) {
535 					tagList.add(next);
536 				}
537 				ResourceMetadataKeyEnum.TAG_LIST.put(myEntry.getResource(), tagList);
538 			}
539 			if (!myEntry.getLinkSelf().isEmpty()) {
540 				String linkSelfValue = myEntry.getLinkSelf().getValue();
541 				IdDt linkSelf = new IdDt(linkSelfValue);
542 				myEntry.getResource().setId(linkSelf);
543 				if (isNotBlank(linkSelf.getVersionIdPart())) {
544 					metadata.put(ResourceMetadataKeyEnum.VERSION_ID, linkSelf);
545 				}
546 			}
547 			if (!myEntry.getLinkAlternate().isEmpty()) {
548 				ResourceMetadataKeyEnum.LINK_ALTERNATE.put(myEntry.getResource(), myEntry.getLinkAlternate().getValue());
549 			}
550 			if (!myEntry.getLinkSearch().isEmpty()) {
551 				ResourceMetadataKeyEnum.LINK_SEARCH.put(myEntry.getResource(), myEntry.getLinkSearch().getValue());
552 			}
553 
554 		}

585 		public void endingElement() throws DataFormatException {
586 			if (myInstance != null) {
587 				if ("self".equals(myRel)) {
588 					myInstance.getLinkSelf().setValueAsString(myHref);
589 				} else if ("first".equals(myRel)) {
590 					myInstance.getLinkFirst().setValueAsString(myHref);
591 				} else if ("previous".equals(myRel)) {
592 					myInstance.getLinkPrevious().setValueAsString(myHref);
593 				} else if ("next".equals(myRel)) {
594 					myInstance.getLinkNext().setValueAsString(myHref);
595 				} else if ("last".equals(myRel)) {
596 					myInstance.getLinkLast().setValueAsString(myHref);
597 				} else if ("fhir-base".equals(myRel)) {
598 					myInstance.getLinkBase().setValueAsString(myHref);
599 				}
600 			} else {
601 				if ("self".equals(myRel)) {
602 					myEntry.getLinkSelf().setValueAsString(myHref);

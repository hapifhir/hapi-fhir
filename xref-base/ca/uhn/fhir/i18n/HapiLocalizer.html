23  import java.text.MessageFormat;
24  import java.util.Map;
25  import java.util.ResourceBundle;
26  import java.util.concurrent.ConcurrentHashMap;
27  
28  /**
29   * This feature is not yet in its final state and should be considered an internal part of HAPI for now - use with caution
30   */
31  public class HapiLocalizer {
32  
33  	private ResourceBundle myBundle;
34  	private final Map<String, MessageFormat> myKeyToMessageFormat = new ConcurrentHashMap<String, MessageFormat>();
35  
36  	public HapiLocalizer() {
37  		myBundle = ResourceBundle.getBundle(HapiLocalizer.class.getPackage().getName() + ".hapi-messages");
38  	}
39  
40  	public String getMessage(Class<?> theType, String theKey, Object... theParameters) {
41  		return getMessage(theType.getName() + '.' + theKey, theParameters);
42  	}
43  	
44  	public String getMessage(String theQualifiedKey, Object... theParameters) {
45  		if (theParameters != null && theParameters.length > 0) {
46  			MessageFormat format = myKeyToMessageFormat.get(theQualifiedKey);
47  			if (format != null) {
48  				return format.format(theParameters).toString();
49  			}
50  			
51  			String formatString = myBundle.getString(theQualifiedKey);
52  			if (formatString== null) {
53  				formatString = "!MESSAGE!";
54  			}
55  			
56  			format = new MessageFormat(formatString);
57  			myKeyToMessageFormat.put(theQualifiedKey, format);
58  			return format.format(theParameters).toString();
59  		} else {
60  			String retVal = myBundle.getString(theQualifiedKey);
61  			if (retVal == null) {
62  				retVal = "!MESSAGE!";
63  			}
64  			return retVal;
65  		}
66  	}
67  	
68  	
69  }

310 	private static Map<String, ModalityEnum> CODE_TO_ENUM = new HashMap<String, ModalityEnum>();
311 	private static Map<String, Map<String, ModalityEnum>> SYSTEM_TO_CODE_TO_ENUM = new HashMap<String, Map<String, ModalityEnum>>();
312 	
313 	private final String myCode;
314 	private final String mySystem;
315 	
316 	static {
317 		for (ModalityEnum next : ModalityEnum.values()) {
318 			CODE_TO_ENUM.put(next.getCode(), next);
319 			
320 			if (!SYSTEM_TO_CODE_TO_ENUM.containsKey(next.getSystem())) {
321 				SYSTEM_TO_CODE_TO_ENUM.put(next.getSystem(), new HashMap<String, ModalityEnum>());
322 			}
323 			SYSTEM_TO_CODE_TO_ENUM.get(next.getSystem()).put(next.getCode(), next);			
324 		}
325 	}

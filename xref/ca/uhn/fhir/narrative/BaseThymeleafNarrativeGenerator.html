103 	@Override
104 	public NarrativeDt generateNarrative(String theProfile, IResource theResource) {
105 		if (!myInitialized) {
106 			initialize();
107 		}
108 
109 		String name = null;
110 		if (StringUtils.isNotBlank(theProfile)) {
111 			name = myProfileToName.get(theProfile);
112 		}
113 		if (name == null) {
114 			name = myClassToName.get(theResource.getClass());
115 		}
116 
117 		if (name == null) {
118 			if (myIgnoreMissingTemplates) {
119 				ourLog.debug("No narrative template available for profile: {}", theProfile);
120 				return new NarrativeDt(new XhtmlDt("<div>No narrative template available for resource profile: " + theProfile + "</div>"), NarrativeStatusEnum.EMPTY);
121 			} else {
122 				throw new DataFormatException("No narrative template for class " + theResource.getClass().getCanonicalName());
123 			}
124 		}
125 
126 		try {
127 			Context context = new Context();
128 			context.setVariable("resource", theResource);
129 
130 			String result = myProfileTemplateEngine.process(name, context);
131 
132 			if (myCleanWhitespace) {
133 				ourLog.trace("Pre-whitespace cleaning: ", result);
134 				result = cleanWhitespace(result);
135 				ourLog.trace("Post-whitespace cleaning: ", result);
136 			}
137 
138 			XhtmlDt div = new XhtmlDt(result);
139 			return new NarrativeDt(div, NarrativeStatusEnum.GENERATED);
140 		} catch (Exception e) {
141 			if (myIgnoreFailures) {
142 				ourLog.error("Failed to generate narrative", e);
143 				return new NarrativeDt(new XhtmlDt("<div>No narrative available - Error: " + e.getMessage() + "</div>"), NarrativeStatusEnum.EMPTY);
144 			} else {
145 				throw new DataFormatException(e);
146 			}
147 		}
148 	}

155 	@Override
156 	public String generateTitle(String theProfile, IResource theResource) {
157 		if (!myInitialized) {
158 			initialize();
159 		}
160 
161 		ourLog.trace("Generating resource title {}", theResource);
162 
163 		String name = null;
164 		if (StringUtils.isNotBlank(theProfile)) {
165 			name = myProfileToName.get(theProfile);
166 		}
167 		if (name == null) {
168 			name = myClassToName.get(theResource.getClass());
169 		}
170 
171 		ourLog.trace("Template name is {}", name);
172 
173 		if (name == null) {
174 			if (myIgnoreMissingTemplates) {
175 				ourLog.debug("No title template available for profile: {}", theProfile);
176 				return null;
177 			} else {
178 				throw new DataFormatException("No title template for class " + theResource.getClass().getCanonicalName());
179 			}
180 		}
181 
182 		try {
183 			Context context = new Context();
184 			context.setVariable("resource", theResource);
185 
186 			String result = myTitleTemplateEngine.process(name, context);
187 
188 			ourLog.trace("Produced {}", result);
189 
190 			StringBuilder b = new StringBuilder();
191 			boolean inTag = false;
192 			for (int i = 0; i < result.length(); i++) {
193 				char nextChar = result.charAt(i);
194 				char prevChar = i > 0 ? result.charAt(i - 1) : '\n';
195 				if (nextChar == '<') {
196 					inTag = true;
197 					continue;
198 				} else if (inTag) {
199 					if (nextChar == '>') {
200 						inTag = false;
201 					}
202 					continue;
203 				} else if (nextChar <= ' ') {
204 					if (prevChar <= ' ' || prevChar == '>') {
205 						continue;
206 					} else {
207 						b.append(' ');
208 					}
209 				} else {
210 					b.append(nextChar);
211 				}
212 			}
213 
214 			while (b.length() > 0 && b.charAt(b.length() - 1) == ' ') {
215 				b.setLength(b.length() - 1);
216 			}
217 
218 			result = b.toString();
219 			if (result.startsWith("<") && result.contains(">")) {
220 				result = result.substring(result.indexOf('>') + 1);
221 			}
222 			if (result.endsWith(">") && result.contains("<")) {
223 				result = result.substring(0, result.lastIndexOf('<'));
224 			}
225 
226 			result = result.replace("&gt;", ">").replace("&lt;", "<").replace("&amp;", "&");
227 
228 			return result;
229 		} catch (Exception e) {
230 			if (myIgnoreFailures) {
231 				ourLog.error("Failed to generate narrative", e);
232 				return "No title available - Error: " + e.getMessage();
233 			} else {
234 				throw new DataFormatException(e);
235 			}
236 		}
237 	}

250 	private synchronized void initialize() {
251 		if (myInitialized) {
252 			return;
253 		}
254 
255 		ourLog.info("Initializing narrative generator");
256 
257 		myProfileToName = new HashMap<String, String>();
258 		myClassToName = new HashMap<Class<?>, String>();
259 		myNameToNarrativeTemplate = new HashMap<String, String>();
260 		myNameToTitleTemplate = new HashMap<String, String>();
261 
262 		List<String> propFileName = getPropertyFile();
263 
264 		try {
265 			if (myApplyDefaultDatatypeTemplates) {
266 				loadProperties(DefaultThymeleafNarrativeGenerator.NARRATIVES_PROPERTIES);
267 			}
268 			for (String next : propFileName) {
269 				loadProperties(next);
270 			}
271 		} catch (IOException e) {
272 			throw new ConfigurationException("Can not load property file " + propFileName, e);
273 		}
274 
275 		{
276 			myProfileTemplateEngine = new TemplateEngine();
277 			TemplateResolver resolver = new TemplateResolver();
278 			resolver.setResourceResolver(new ProfileResourceResolver());
279 			myProfileTemplateEngine.setTemplateResolver(resolver);
280 			StandardDialect dialect = new StandardDialect();
281 			HashSet<IProcessor> additionalProcessors = new HashSet<IProcessor>();
282 			additionalProcessors.add(new NarrativeAttributeProcessor());
283 			dialect.setAdditionalProcessors(additionalProcessors);
284 			myProfileTemplateEngine.setDialect(dialect);
285 			myProfileTemplateEngine.initialize();
286 		}
287 		{
288 			myTitleTemplateEngine = new TemplateEngine();
289 			TemplateResolver resolver = new TemplateResolver();
290 			resolver.setResourceResolver(new TitleResourceResolver());
291 			myTitleTemplateEngine.setTemplateResolver(resolver);
292 			StandardDialect dialect = new StandardDialect();
293 			HashSet<IProcessor> additionalProcessors = new HashSet<IProcessor>();
294 			additionalProcessors.add(new NarrativeAttributeProcessor());
295 			dialect.setAdditionalProcessors(additionalProcessors);
296 			myTitleTemplateEngine.setDialect(dialect);
297 			myTitleTemplateEngine.initialize();
298 		}
299 
300 		myInitialized = true;
301 	}

331 	private void loadProperties(String propFileName) throws IOException {
332 		ourLog.debug("Loading narrative properties file: {}", propFileName);
333 
334 		Properties file = new Properties();
335 
336 		InputStream resource = loadResource(propFileName);
337 		file.load(resource);
338 		for (Object nextKeyObj : file.keySet()) {
339 			String nextKey = (String) nextKeyObj;
340 			if (nextKey.endsWith(".profile")) {
341 				String name = nextKey.substring(0, nextKey.indexOf(".profile"));
342 				if (isBlank(name)) {
343 					continue;
344 				}
345 
346 				String narrativePropName = name + ".narrative";
347 				String narrativeName = file.getProperty(narrativePropName);
348 				String titlePropName = name + ".title";
349 				String titleName = file.getProperty(titlePropName);
350 				if (isBlank(narrativeName) && isBlank(titleName)) {
351 					throw new ConfigurationException("Found property '" + nextKey + "' but no corresponding property '" + narrativePropName + "' or '" + titlePropName + "' in file " + propFileName);
352 				}
353 
354 				myProfileToName.put(file.getProperty(nextKey), name);
355 
356 				if (StringUtils.isNotBlank(narrativeName)) {
357 					String narrative = IOUtils.toString(loadResource(narrativeName));
358 					myNameToNarrativeTemplate.put(name, narrative);
359 				}
360 				if (StringUtils.isNotBlank(titleName)) {
361 					String title = IOUtils.toString(loadResource(titleName));
362 					myNameToTitleTemplate.put(name, title);
363 				}
364 
365 			} else if (nextKey.endsWith(".class")) {
366 
367 				String name = nextKey.substring(0, nextKey.indexOf(".class"));
368 				if (isBlank(name)) {
369 					continue;
370 				}
371 
372 				String className = file.getProperty(nextKey);
373 
374 				Class<?> clazz;
375 				try {
376 					clazz = Class.forName(className);
377 				} catch (ClassNotFoundException e) {
378 					ourLog.warn("Unknown datatype class '{}' identified in narrative file {}", name, propFileName);
379 					continue;
380 				}
381 
382 				String narrativePropName = name + ".narrative";
383 				String narrativeName = file.getProperty(narrativePropName);
384 				String titlePropName = name + ".title";
385 				String titleName = file.getProperty(titlePropName);
386 				if (isBlank(narrativeName) && isBlank(titleName)) {
387 					throw new ConfigurationException("Found property '" + nextKey + "' but no corresponding property '" + narrativePropName + "' or '" + titlePropName + "' in file " + propFileName);
388 				}
389 
390 				myClassToName.put(clazz, name);
391 
392 				if (StringUtils.isNotBlank(narrativeName)) {
393 					String narrative = IOUtils.toString(loadResource(narrativeName));
394 					myNameToNarrativeTemplate.put(name, narrative);
395 				}
396 				if (StringUtils.isNotBlank(titleName)) {
397 					String title = IOUtils.toString(loadResource(titleName));
398 					myNameToTitleTemplate.put(name, title);
399 				}
400 
401 			} else if (nextKey.endsWith(".narrative")) {
402 				continue;
403 			} else if (nextKey.endsWith(".title")) {
404 				continue;
405 			} else {
406 				throw new ConfigurationException("Invalid property name: " + nextKey);
407 			}
408 
409 		}
410 	}

412 	private InputStream loadResource(String name) throws IOException {
413 		if (name.startsWith("classpath:")) {
414 			String cpName = name.substring("classpath:".length());
415 			InputStream resource = DefaultThymeleafNarrativeGenerator.class.getResourceAsStream(cpName);
416 			if (resource == null) {
417 				resource = DefaultThymeleafNarrativeGenerator.class.getResourceAsStream("/" + cpName);
418 				if (resource == null) {
419 					throw new IOException("Can not find '" + cpName + "' on classpath");
420 				}
421 			}
422 			return resource;
423 		} else if (name.startsWith("file:")) {
424 			File file = new File(name.substring("file:".length()));
425 			if (file.exists() == false) {
426 				throw new IOException("File not found: " + file.getAbsolutePath());
427 			}
428 			return new FileInputStream(file);
429 		} else {
430 			throw new IOException("Invalid resource name: '" + name + "' (must start with classpath: or file: )");
431 		}
432 	}

462 	static String cleanWhitespace(String theResult) {
463 		StringBuilder b = new StringBuilder();
464 		boolean inWhitespace = false;
465 		boolean betweenTags = false;
466 		boolean lastNonWhitespaceCharWasTagEnd = false;
467 		boolean inPre = false;
468 		for (int i = 0; i < theResult.length(); i++) {
469 			char nextChar = theResult.charAt(i);
470 			if (inPre) {
471 				b.append(nextChar);
472 				continue;
473 			} else if (nextChar == '>') {
474 				b.append(nextChar);
475 				betweenTags = true;
476 				lastNonWhitespaceCharWasTagEnd = true;
477 				continue;
478 			} else if (nextChar == '\n' || nextChar == '\r') {
479 				// if (inWhitespace) {
480 				// b.append(' ');
481 				// inWhitespace = false;
482 				// }
483 				continue;
484 			}
485 
486 			if (betweenTags) {
487 				if (Character.isWhitespace(nextChar)) {
488 					inWhitespace = true;
489 				} else if (nextChar == '<') {
490 					if (inWhitespace && !lastNonWhitespaceCharWasTagEnd) {
491 						b.append(' ');
492 					}
493 					inWhitespace = false;
494 					b.append(nextChar);
495 					inWhitespace = false;
496 					betweenTags = false;
497 					lastNonWhitespaceCharWasTagEnd = false;
498 					if (i + 3 < theResult.length()) {
499 						char char1 = Character.toLowerCase(theResult.charAt(i + 1));
500 						char char2 = Character.toLowerCase(theResult.charAt(i + 2));
501 						char char3 = Character.toLowerCase(theResult.charAt(i + 3));
502 						char char4 = Character.toLowerCase((i + 4 < theResult.length()) ? theResult.charAt(i + 4) : ' ');
503 						if (char1 == 'p' && char2 == 'r' && char3 == 'e') {
504 							inPre = true;
505 						} else if (char1 == '/' && char2 == 'p' && char3 == 'r'&&char4=='e') {
506 							inPre = false;
507 						}
508 					}
509 				} else {
510 					lastNonWhitespaceCharWasTagEnd = false;
511 					if (inWhitespace) {
512 						b.append(' ');
513 						inWhitespace = false;
514 					}
515 					b.append(nextChar);
516 				}
517 			} else {
518 				b.append(nextChar);
519 			}
520 		}
521 		return b.toString();
522 	}

524 	public class NarrativeAttributeProcessor extends AbstractAttrProcessor {
525 
526 		protected NarrativeAttributeProcessor() {
527 			super("narrative");
528 		}
529 
530 		@Override
531 		public int getPrecedence() {
532 			return 0;
533 		}
534 
535 		@Override
536 		protected ProcessorResult processAttribute(Arguments theArguments, Element theElement, String theAttributeName) {
537 			final String attributeValue = theElement.getAttributeValue(theAttributeName);
538 
539 			final Configuration configuration = theArguments.getConfiguration();
540 			final IStandardExpressionParser expressionParser = StandardExpressions.getExpressionParser(configuration);
541 
542 			final IStandardExpression expression = expressionParser.parseExpression(configuration, theArguments, attributeValue);
543 			final Object value = expression.execute(configuration, theArguments);
544 
545 			theElement.removeAttribute(theAttributeName);
546 			theElement.clearChildren();
547 
548 			if (value == null) {
549 				return ProcessorResult.ok();
550 			}
551 
552 			Context context = new Context();
553 			context.setVariable("resource", value);
554 
555 			String name = null;
556 			if (value != null) {
557 				Class<? extends Object> nextClass = value.getClass();
558 				do {
559 					name = myClassToName.get(nextClass);
560 					nextClass = nextClass.getSuperclass();
561 				} while (name == null && nextClass.equals(Object.class) == false);
562 			}
563 
564 			if (name == null) {
565 				if (myIgnoreMissingTemplates) {
566 					ourLog.debug("No narrative template available for type: {}", value.getClass());
567 					return ProcessorResult.ok();
568 				} else {
569 					throw new DataFormatException("No narrative template for class " + value.getClass());
570 				}
571 			}
572 
573 			String result = myProfileTemplateEngine.process(name, context);
574 			String trim = result.trim();
575 			Document dom = getXhtmlDOMFor(new StringReader(trim));
576 
577 			Element firstChild = (Element) dom.getFirstChild();
578 			for (int i = 0; i < firstChild.getChildren().size(); i++) {
579 				Node next = firstChild.getChildren().get(i);
580 				if (i == 0 && firstChild.getChildren().size() == 1) {
581 					if (next instanceof org.thymeleaf.dom.Text) {
582 						org.thymeleaf.dom.Text nextText = (org.thymeleaf.dom.Text) next;
583 						nextText.setContent(nextText.getContent().trim());
584 					}
585 				}
586 				theElement.addChild(next);
587 			}
588 
589 			return ProcessorResult.ok();
590 		}

613 		@Override
614 		public InputStream getResourceAsStream(TemplateProcessingParameters theTemplateProcessingParameters, String theName) {
615 			String template = myNameToNarrativeTemplate.get(theName);

24  import org.apache.http.client.HttpClient;
25  
26  import ca.uhn.fhir.context.FhirContext;
27  import ca.uhn.fhir.rest.server.EncodingEnum;
28  
29  public interface IRestfulClient {
30  
31  	/**
32  	 * Returns the FHIR context associated with this client
33  	 */
34  	FhirContext getFhirContext();
35  	
36  	/**
37  	 * Do not call this method in client code. It is a part of the internal HAPI API and 
38  	 * is subject to change!
39  	 */
40  	HttpClient getHttpClient();
41  	
42  	/**
43  	 * Specifies that the client should use the given encoding to do its 
44  	 * queries. This means that the client will append the "_format" param
45  	 * to GET methods (read/search/etc), and will add an appropriate header for
46  	 * write methods. 
47  	 * 
48  	 * @param theEncoding The encoding to use in the request, or <code>null</code> not specify
49  	 * an encoding (which generally implies the use of XML). The default is <code>null</code>.
50  	 */
51  	void setEncoding(EncodingEnum theEncoding);
52  	
53  	/**
54  	 * Specifies that the client should request that the server respond with "pretty printing"
55  	 * enabled. Note that this is a non-standard parameter, so it may only 
56  	 * work against HAPI based servers.
57  	 * 
58  	 * @param thePrettyPrint The pretty print flag to use in the request (default is <code>false</code>)
59  	 */
60  	void setPrettyPrint(boolean thePrettyPrint);
61  	
62  	/**
63  	 * Base URL for the server, with no trailing "/"
64  	 */
65  	String getServerBase();
66  	
67  	
68  
69  }

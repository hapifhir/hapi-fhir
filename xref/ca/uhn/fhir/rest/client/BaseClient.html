112 	<T> T invokeClient(IClientResponseHandler<T> binding, BaseHttpClientInvocation clientInvocation, boolean theLogRequestAndResponse) {
113 		// TODO: handle non 2xx status codes by throwing the correct exception,
114 		// and ensure it's passed upwards
115 		HttpRequestBase httpRequest;
116 		HttpResponse response;
117 		try {
118 			httpRequest = clientInvocation.asHttpRequest(myUrlBase, createExtraParams(), getEncoding());
119 
120 			if (theLogRequestAndResponse) {
121 				ourLog.info("Client invoking: {}", httpRequest);
122 				if (httpRequest instanceof HttpEntityEnclosingRequest) {
123 					HttpEntity entity = ((HttpEntityEnclosingRequest) httpRequest).getEntity();
124 					if (entity.isRepeatable()) {
125 						String content = IOUtils.toString(entity.getContent());
126 						ourLog.info("Client request body: {}", content);
127 					}
128 				}
129 			}
130 
131 			response = myClient.execute(httpRequest);
132 		} catch (DataFormatException e) {
133 			throw new FhirClientConnectionException(e);
134 		} catch (IOException e) {
135 			throw new FhirClientConnectionException(e);
136 		}
137 
138 		try {
139 			ContentType ct = ContentType.get(response.getEntity());
140 			String mimeType = ct != null ? ct.getMimeType() : null;
141 
142 			Map<String, List<String>> headers = new HashMap<String, List<String>>();
143 			if (response.getAllHeaders() != null) {
144 				for (Header next : response.getAllHeaders()) {
145 					String name = next.getName().toLowerCase();
146 					List<String> list = headers.get(name);
147 					if (list == null) {
148 						list = new ArrayList<String>();
149 						headers.put(name, list);
150 					}
151 					list.add(next.getValue());
152 				}
153 			}
154 
155 			if (response.getStatusLine().getStatusCode() < 200 || response.getStatusLine().getStatusCode() > 299) {
156 				String body=null;
157 				Reader reader=null;
158 				try {
159 					reader = createReaderFromResponse(response);
160 					body = IOUtils.toString(reader);
161 				} catch (Exception e) {
162 					ourLog.debug("Failed to read input stream", e);
163 				} finally {
164 					IOUtils.closeQuietly(reader);
165 				}
166 				
167 				String message = "HTTP " + response.getStatusLine().getStatusCode()+" " +response.getStatusLine().getReasonPhrase();
168 				if (Constants.CT_TEXT.equals(mimeType)) {
169 					message = message+": " + body;
170 				}
171 				
172 				keepResponseAndLogIt(theLogRequestAndResponse, response, body);
173 
174 				BaseServerResponseException exception = BaseServerResponseException.newInstance(response.getStatusLine().getStatusCode(), message);
175 
176 				if(body!=null) {
177 					exception.setResponseBody(body);
178 				}
179 
180 				throw exception;
181 			}
182 			if (binding instanceof IClientResponseHandlerHandlesBinary) {
183 				IClientResponseHandlerHandlesBinary<T> handlesBinary = (IClientResponseHandlerHandlesBinary<T>) binding;
184 				if (handlesBinary.isBinary()) {
185 					InputStream reader = response.getEntity().getContent();
186 					try {
187 					
188 					if (ourLog.isTraceEnabled() || myKeepResponses || theLogRequestAndResponse) {
189 						byte[] responseBytes = IOUtils.toByteArray(reader);
190 						if (myKeepResponses) {
191 							myLastResponse = response;
192 							myLastResponseBody = null;
193 						}
194 						String message = "HTTP " + response.getStatusLine().getStatusCode()+" " +response.getStatusLine().getReasonPhrase();
195 						if (theLogRequestAndResponse) {
196 							ourLog.info("Client response: {} - {} bytes", message, responseBytes.length);
197 						}else {
198 							ourLog.trace("Client response: {} - {} bytes", message, responseBytes.length);
199 						}
200 						reader = new ByteArrayInputStream(responseBytes);
201 					}
202 
203 						return handlesBinary.invokeClient(mimeType, reader, response.getStatusLine().getStatusCode(), headers);
204 					} finally {
205 						IOUtils.closeQuietly(reader);
206 					}
207 				}
208 			}
209 			
210 			Reader reader = createReaderFromResponse(response);
211 
212 			if (ourLog.isTraceEnabled() || myKeepResponses || theLogRequestAndResponse) {
213 				String responseString = IOUtils.toString(reader);
214 				keepResponseAndLogIt(theLogRequestAndResponse, response, responseString);
215 				reader = new StringReader(responseString);
216 			}
217 
218 			try {
219 				return binding.invokeClient(mimeType, reader, response.getStatusLine().getStatusCode(), headers);
220 			} finally {
221 				IOUtils.closeQuietly(reader);
222 			}
223 
224 		} catch (IllegalStateException e) {
225 			throw new FhirClientConnectionException(e);
226 		} catch (IOException e) {
227 			throw new FhirClientConnectionException(e);
228 		} finally {
229 			if (response instanceof CloseableHttpResponse) {
230 				try {
231 					((CloseableHttpResponse) response).close();
232 				} catch (IOException e) {
233 					ourLog.debug("Failed to close response", e);
234 				}
235 			}
236 		}
237 	}

239 	private void keepResponseAndLogIt(boolean theLogRequestAndResponse, HttpResponse response, String responseString) {
240 		if (myKeepResponses) {
241 			myLastResponse = response;
242 			myLastResponseBody = responseString;
243 		}
244 		if (theLogRequestAndResponse) {
245 			String message = "HTTP " + response.getStatusLine().getStatusCode()+" " +response.getStatusLine().getReasonPhrase();
246 			if (StringUtils.isNotBlank(responseString)) {
247 				ourLog.info("Client response: {}\n{}", message, responseString);
248 			}else {
249 				ourLog.info("Client response: {}", message, responseString);
250 			}
251 		}else {
252 			ourLog.trace("FHIR response:\n{}\n{}", response, responseString);
253 		}
254 	}

256 	protected Map<String, List<String>> createExtraParams() {
257 		HashMap<String, List<String>> retVal = new LinkedHashMap<String, List<String>>();
258 
259 		if (getEncoding() == EncodingEnum.XML) {
260 			retVal.put(Constants.PARAM_FORMAT, Collections.singletonList("xml"));
261 		} else if (getEncoding() == EncodingEnum.JSON) {
262 			retVal.put(Constants.PARAM_FORMAT, Collections.singletonList("json"));
263 		}
264 
265 		if (isPrettyPrint()) {
266 			retVal.put(Constants.PARAM_PRETTY, Collections.singletonList(Constants.PARAM_PRETTY_VALUE_TRUE));
267 		}
268 
269 		return retVal;
270 	}

300 	public static Reader createReaderFromResponse(HttpResponse theResponse) throws IllegalStateException, IOException {
301 		HttpEntity entity = theResponse.getEntity();
302 		if (entity == null) {
303 			return new StringReader("");
304 		}
305 		Charset charset = null;
306 		if (entity.getContentType() != null && entity.getContentType().getElements() != null && entity.getContentType().getElements().length > 0) {
307 			ContentType ct = ContentType.get(entity);
308 			charset = ct.getCharset();
309 		}
310 		if (charset == null) {
311 			ourLog.warn("Response did not specify a charset.");
312 			charset = Charset.forName("UTF-8");
313 		}
314 
315 		Reader reader = new InputStreamReader(theResponse.getEntity().getContent(), charset);
316 		return reader;
317 	}

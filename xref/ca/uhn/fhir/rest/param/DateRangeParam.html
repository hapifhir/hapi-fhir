36  public class DateRangeParam implements IQueryParameterAnd {
37  
38  	private QualifiedDateParam myLowerBound;
39  	private QualifiedDateParam myUpperBound;
40  
41  	/**
42  	 * Basic constructor. Values must be supplied by calling {@link #setLowerBound(QualifiedDateParam)} and
43  	 * {@link #setUpperBound(QualifiedDateParam)}
44  	 */
45  	public DateRangeParam() {
46  		// nothing
47  	}
48  
49  	/**
50  	 * Constructor which takes two Dates representing the lower and upper bounds of the range (inclusive on both ends)
51  	 * 
52  	 * @param theLowerBound
53  	 *            A qualified date param representing the lower date bound (optionally may include time), e.g.
54  	 *            "2011-02-22" or "2011-02-22T13:12:00". Will be treated inclusively.
55  	 * @param theUpperBound
56  	 *            A qualified date param representing the upper date bound (optionally may include time), e.g.
57  	 *            "2011-02-22" or "2011-02-22T13:12:00". Will be treated inclusively.
58  	 */
59  	public DateRangeParam(Date theLowerBound, Date theUpperBound) {
60  		setRangeFromDatesInclusive(theLowerBound, theUpperBound);
61  	}
62  
63  	/**
64  	 * Sets the range from a single date param. If theDateParam has no qualifier, treats it as the lower and upper bound
65  	 * (e.g. 2011-01-02 would match any time on that day). If theDateParam has a qualifier, treats it as either the
66  	 * lower or upper bound, with no opposite bound.
67  	 */
68  	public DateRangeParam(QualifiedDateParam theDateParam) {
69  		if (theDateParam == null) {
70  			throw new NullPointerException("theDateParam can not be null");
71  		}
72  		if (theDateParam.isEmpty()) {
73  			throw new IllegalArgumentException("theDateParam can not be empty");
74  		}
75  		if (theDateParam.getComparator() == null) {
76  			setRangeFromDatesInclusive(theDateParam.getValueAsString(), theDateParam.getValueAsString());
77  		} else {
78  			switch (theDateParam.getComparator()) {
79  			case GREATERTHAN:
80  			case GREATERTHAN_OR_EQUALS:
81  				myLowerBound = theDateParam;
82  				myUpperBound = null;
83  				break;
84  			case LESSTHAN:
85  			case LESSTHAN_OR_EQUALS:
86  				myLowerBound = null;
87  				myUpperBound = theDateParam;
88  				break;
89  			default:
90  				// Should not happen
91  				throw new IllegalStateException("Unknown comparator:" + theDateParam.getComparator() + ". This is a bug.");
92  			}
93  		}
94  		validateAndThrowDataFormatExceptionIfInvalid();
95  	}
96  
97  	/**
98  	 * Constructor which takes two strings representing the lower and upper bounds of the range (inclusive on both ends)
99  	 * 
100 	 * @param theLowerBound
101 	 *            A qualified date param representing the lower date bound (optionally may include time), e.g.
102 	 *            "2011-02-22" or "2011-02-22T13:12:00"
103 	 * @param theUpperBound
104 	 *            A qualified date param representing the upper date bound (optionally may include time), e.g.
105 	 *            "2011-02-22" or "2011-02-22T13:12:00"
106 	 */
107 	public DateRangeParam(String theLowerBound, String theUpperBound) {
108 		setRangeFromDatesInclusive(theLowerBound, theUpperBound);
109 	}
110 
111 	public QualifiedDateParam getLowerBound() {
112 		return myLowerBound;
113 	}
114 
115 	public Date getLowerBoundAsInstant() {
116 		if (myLowerBound == null) {
117 			return null;
118 		}
119 		Date retVal = myLowerBound.getValue();
120 		if (myLowerBound.getComparator() != null) {
121 			switch (myLowerBound.getComparator()) {
122 			case GREATERTHAN:
123 				retVal = myLowerBound.getPrecision().add(retVal, 1);
124 				break;
125 			case GREATERTHAN_OR_EQUALS:
126 				break;
127 			case LESSTHAN:
128 			case LESSTHAN_OR_EQUALS:
129 				throw new IllegalStateException("Unvalid lower bound comparator: " + myLowerBound.getComparator());
130 			}
131 		}
132 		return retVal;
133 	}
134 
135 	public QualifiedDateParam getUpperBound() {
136 		return myUpperBound;
137 	}
138 
139 	public Date getUpperBoundAsInstant() {
140 		if (myUpperBound == null) {
141 			return null;
142 		}
143 		Date retVal = myUpperBound.getValue();
144 		if (myUpperBound.getComparator() != null) {
145 			switch (myUpperBound.getComparator()) {
146 			case LESSTHAN:
147 				retVal = new Date(retVal.getTime() - 1L);
148 				break;
149 			case LESSTHAN_OR_EQUALS:
150 				retVal = myUpperBound.getPrecision().add(retVal, 1);
151 				retVal = new Date(retVal.getTime() - 1L);
152 				break;
153 			case GREATERTHAN_OR_EQUALS:
154 			case GREATERTHAN:
155 				throw new IllegalStateException("Unvalid upper bound comparator: " + myUpperBound.getComparator());
156 			}
157 		}
158 		return retVal;
159 	}

161 	@Override
162 	public List<IQueryParameterOr> getValuesAsQueryTokens() {
163 		ArrayList<IQueryParameterOr> retVal = new ArrayList<IQueryParameterOr>();
164 		if (myLowerBound != null) {
165 			retVal.add(ParameterUtil.singleton(myLowerBound));
166 		}
167 		if (myUpperBound != null) {
168 			retVal.add(ParameterUtil.singleton(myUpperBound));
169 		}
170 		return retVal;
171 	}

173 	public void setLowerBound(QualifiedDateParam theLowerBound) {
174 		myLowerBound = theLowerBound;
175 		validateAndThrowDataFormatExceptionIfInvalid();
176 	}
177 
178 	/**
179 	 * Sets the range from a pair of dates, inclusive on both ends
180 	 * 
181 	 * @param theLowerBound
182 	 *            A qualified date param representing the lower date bound (optionally may include time), e.g.
183 	 *            "2011-02-22" or "2011-02-22T13:12:00". Will be treated inclusively.
184 	 * @param theUpperBound
185 	 *            A qualified date param representing the upper date bound (optionally may include time), e.g.
186 	 *            "2011-02-22" or "2011-02-22T13:12:00". Will be treated inclusively.
187 	 */
188 	public void setRangeFromDatesInclusive(Date theLowerBound, Date theUpperBound) {
189 		if (theLowerBound != null) {
190 			myLowerBound = new QualifiedDateParam(QuantityCompararatorEnum.GREATERTHAN_OR_EQUALS, theLowerBound);
191 		}
192 		if (theUpperBound != null) {
193 			myUpperBound = new QualifiedDateParam(QuantityCompararatorEnum.LESSTHAN_OR_EQUALS, theUpperBound);
194 		}
195 		validateAndThrowDataFormatExceptionIfInvalid();
196 	}
197 
198 	/**
199 	 * Sets the range from a pair of dates, inclusive on both ends
200 	 * 
201 	 * @param theLowerBound
202 	 *            A qualified date param representing the lower date bound (optionally may include time), e.g.
203 	 *            "2011-02-22" or "2011-02-22T13:12:00". Will be treated inclusively.
204 	 * @param theUpperBound
205 	 *            A qualified date param representing the upper date bound (optionally may include time), e.g.
206 	 *            "2011-02-22" or "2011-02-22T13:12:00". Will be treated inclusively.
207 	 */
208 	public void setRangeFromDatesInclusive(String theLowerBound, String theUpperBound) {
209 		if (StringUtils.isNotBlank(theLowerBound)) {
210 			myLowerBound = new QualifiedDateParam(QuantityCompararatorEnum.GREATERTHAN_OR_EQUALS, theLowerBound);
211 		}
212 		if (StringUtils.isNotBlank(theUpperBound)) {
213 			myUpperBound = new QualifiedDateParam(QuantityCompararatorEnum.LESSTHAN_OR_EQUALS, theUpperBound);
214 		}
215 		validateAndThrowDataFormatExceptionIfInvalid();
216 	}
217 
218 	public void setUpperBound(QualifiedDateParam theUpperBound) {
219 		myUpperBound = theUpperBound;
220 		validateAndThrowDataFormatExceptionIfInvalid();
221 	}

223 	@Override
224 	public void setValuesAsQueryTokens(List<QualifiedParamList> theParameters) throws InvalidRequestException {
225 		for (List<String> paramList : theParameters) {
226 			if (paramList.size() == 0) {
227 				continue;
228 			}
229 			if (paramList.size() > 1) {
230 				throw new InvalidRequestException("DateRange parameter does not suppport OR queries");
231 			}
232 			String param = paramList.get(0);
233 			QualifiedDateParam parsed = new QualifiedDateParam();
234 			parsed.setValueAsQueryToken(null, param);
235 			addParam(parsed);
236 		}
237 	}
238 
239 	private void addParam(QualifiedDateParam theParsed) throws InvalidRequestException {
240 		if (theParsed.getComparator() == null) {
241 			if (myLowerBound != null || myUpperBound != null) {
242 				throw new InvalidRequestException("Can not have multiple date range parameters for the same param without a qualifier");
243 			}
244 
245 			myLowerBound = theParsed;
246 			myUpperBound = theParsed;
247 			// TODO: in this case, should set lower and upper to exact moments
248 			// using specified precision
249 		} else {
250 
251 			switch (theParsed.getComparator()) {
252 			case GREATERTHAN:
253 			case GREATERTHAN_OR_EQUALS:
254 				if (myLowerBound != null) {
255 					throw new InvalidRequestException("Can not have multiple date range parameters for the same param that specify a lower bound");
256 				}
257 				myLowerBound = theParsed;
258 				break;
259 			case LESSTHAN:
260 			case LESSTHAN_OR_EQUALS:
261 				if (myUpperBound != null) {
262 					throw new InvalidRequestException("Can not have multiple date range parameters for the same param that specify an upper bound");
263 				}
264 				myUpperBound = theParsed;
265 				break;
266 			default:
267 				throw new InvalidRequestException("Unknown comparator: " + theParsed.getComparator());
268 			}
269 
270 		}
271 	}
272 
273 	private void validateAndThrowDataFormatExceptionIfInvalid() {
274 		boolean haveLowerBound = myLowerBound != null && myLowerBound.isEmpty() == false;
275 		boolean haveUpperBound = myUpperBound != null && myUpperBound.isEmpty() == false;
276 		if (haveLowerBound && haveUpperBound) {
277 			if (myLowerBound.getValue().after(myUpperBound.getValue())) {
278 				throw new DataFormatException("Lower bound of " + myLowerBound.getValueAsString() + " is after upper bound of " + myUpperBound.getValueAsString());
279 			}
280 		}
281 
282 		if (haveLowerBound) {
283 			if (myLowerBound.getComparator() == null) {
284 				myLowerBound.setComparator(QuantityCompararatorEnum.GREATERTHAN_OR_EQUALS);
285 			}
286 			switch (myLowerBound.getComparator()) {
287 			case GREATERTHAN:
288 			case GREATERTHAN_OR_EQUALS:
289 			default:
290 				break;
291 			case LESSTHAN:
292 			case LESSTHAN_OR_EQUALS:
293 				throw new DataFormatException("Lower bound comparator must be > or >=, can not be " + myLowerBound.getComparator().getCode());
294 			}
295 		}
296 
297 		if (haveUpperBound) {
298 			if (myUpperBound.getComparator() == null) {
299 				myUpperBound.setComparator(QuantityCompararatorEnum.LESSTHAN_OR_EQUALS);
300 			}
301 			switch (myUpperBound.getComparator()) {
302 			case LESSTHAN:
303 			case LESSTHAN_OR_EQUALS:
304 			default:
305 				break;
306 			case GREATERTHAN:
307 			case GREATERTHAN_OR_EQUALS:
308 				throw new DataFormatException("Upper bound comparator must be < or <=, can not be " + myUpperBound.getComparator().getCode());
309 			}
310 		}
311 
312 	}

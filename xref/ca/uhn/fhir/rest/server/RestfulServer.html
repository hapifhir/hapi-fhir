156 	private void findResourceMethods(Object theProvider) throws Exception {
157 
158 		ourLog.info("Scanning type for RESTful methods: {}", theProvider.getClass());
159 		int count = 0;
160 
161 		Class<?> clazz = theProvider.getClass();
162 		Class<?> supertype = clazz.getSuperclass();
163 		while (!Object.class.equals(supertype)) {
164 			count += findResourceMethods(theProvider, supertype);
165 			supertype = supertype.getSuperclass();
166 		}
167 
168 		count += findResourceMethods(theProvider, clazz);
169 
170 		if (count == 0) {
171 			throw new ConfigurationException("Did not find any annotated RESTful methods on provider class " + theProvider.getClass().getCanonicalName());
172 		}
173 	}

175 	private int findResourceMethods(Object theProvider, Class<?> clazz) throws ConfigurationException {
176 		int count = 0;
177 
178 		for (Method m : clazz.getDeclaredMethods()) {
179 			BaseMethodBinding<?> foundMethodBinding = BaseMethodBinding.bindMethod(m, myFhirContext, theProvider);
180 			if (foundMethodBinding == null) {
181 				continue;
182 			}
183 
184 			count++;
185 
186 			if (!Modifier.isPublic(m.getModifiers())) {
187 				throw new ConfigurationException("Method '" + m.getName() + "' is not public, FHIR RESTful methods must be public");
188 			} else {
189 				if (Modifier.isStatic(m.getModifiers())) {
190 					throw new ConfigurationException("Method '" + m.getName() + "' is static, FHIR RESTful methods must not be static");
191 				} else {
192 					ourLog.debug("Scanning public method: {}#{}", theProvider.getClass(), m.getName());
193 
194 					String resourceName = foundMethodBinding.getResourceName();
195 					ResourceBinding resourceBinding;
196 					if (resourceName == null) {
197 						resourceBinding = myNullResourceBinding;
198 					} else {
199 						RuntimeResourceDefinition definition = myFhirContext.getResourceDefinition(resourceName);
200 						if (myResourceNameToProvider.containsKey(definition.getName())) {
201 							resourceBinding = myResourceNameToProvider.get(definition.getName());
202 						} else {
203 							resourceBinding = new ResourceBinding();
204 							resourceBinding.setResourceName(resourceName);
205 							myResourceNameToProvider.put(resourceName, resourceBinding);
206 						}
207 					}
208 
209 					resourceBinding.addMethod(foundMethodBinding);
210 					ourLog.debug(" * Method: {}#{} is a handler", theProvider.getClass(), m.getName());
211 				}
212 			}
213 		}
214 
215 		return count;
216 	}

225 	private void findSystemMethods(Object theSystemProvider, Class<?> clazz) {
226 		Class<?> supertype = clazz.getSuperclass();
227 		if (!Object.class.equals(supertype)) {
228 			findSystemMethods(theSystemProvider, supertype);
229 		}
230 
231 		for (Method m : clazz.getDeclaredMethods()) {
232 			if (Modifier.isPublic(m.getModifiers())) {
233 				ourLog.debug("Scanning public method: {}#{}", theSystemProvider.getClass(), m.getName());
234 
235 				BaseMethodBinding<?> foundMethodBinding = BaseMethodBinding.bindMethod(m, myFhirContext, theSystemProvider);
236 				if (foundMethodBinding != null) {
237 					if (foundMethodBinding instanceof ConformanceMethodBinding) {
238 						myServerConformanceMethod = foundMethodBinding;
239 					}
240 					ourLog.info(" * Method: {}#{} is a handler", theSystemProvider.getClass(), m.getName());
241 				} else {
242 					ourLog.debug(" * Method: {}#{} is not a handler", theSystemProvider.getClass(), m.getName());
243 				}
244 			}
245 		}

328 	private void handlePagingRequest(Request theRequest, HttpServletResponse theResponse, String thePagingAction) throws IOException {
329 		IBundleProvider resultList = getPagingProvider().retrieveResultList(thePagingAction);
330 		if (resultList == null) {
331 			ourLog.info("Client requested unknown paging ID[{}]", thePagingAction);
332 			theResponse.setStatus(Constants.STATUS_HTTP_410_GONE);
333 			addHeadersToResponse(theResponse);
334 			theResponse.setContentType("text/plain");
335 			theResponse.setCharacterEncoding("UTF-8");
336 			theResponse.getWriter().append("Search ID[" + thePagingAction + "] does not exist and may have expired.");
337 			theResponse.getWriter().close();
338 			return;
339 		}
340 
341 		Integer count = extractCountParameter(theRequest.getServletRequest());
342 		if (count == null) {
343 			count = getPagingProvider().getDefaultPageSize();
344 		} else if (count > getPagingProvider().getMaximumPageSize()) {
345 			count = getPagingProvider().getMaximumPageSize();
346 		}
347 
348 		Integer offsetI = tryToExtractNamedParameter(theRequest.getServletRequest(), Constants.PARAM_PAGINGOFFSET);
349 		if (offsetI == null || offsetI < 0) {
350 			offsetI = 0;
351 		}
352 
353 		int start = Math.min(offsetI, resultList.size() - 1);
354 
355 		EncodingEnum responseEncoding = determineRequestEncoding(theRequest);
356 		boolean prettyPrint = prettyPrintResponse(theRequest);
357 		boolean requestIsBrowser = requestIsBrowser(theRequest.getServletRequest());
358 		NarrativeModeEnum narrativeMode = determineNarrativeMode(theRequest);
359 		boolean respondGzip = theRequest.isRespondGzip();
360 		streamResponseAsBundle(this, theResponse, resultList, responseEncoding, theRequest.getFhirServerBase(), theRequest.getCompleteUrl(), prettyPrint, requestIsBrowser, narrativeMode, start,
361 				count, thePagingAction, respondGzip);
362 
363 	}

365 	protected void handleRequest(SearchMethodBinding.RequestType theRequestType, HttpServletRequest theRequest, HttpServletResponse theResponse) throws ServletException, IOException {
366 		try {
367 
368 			if (null != mySecurityManager) {
369 				mySecurityManager.authenticate(theRequest);
370 			}
371 
372 			String resourceName = null;
373 			String requestFullPath = StringUtils.defaultString(theRequest.getRequestURI());
374 			String servletPath = StringUtils.defaultString(theRequest.getServletPath());
375 			StringBuffer requestUrl = theRequest.getRequestURL();
376 			String servletContextPath = "";
377 			if (theRequest.getServletContext() != null) {
378 				servletContextPath = StringUtils.defaultString(theRequest.getServletContext().getContextPath());
379 				// } else {
380 				// servletContextPath = servletPath;
381 			}
382 
383 			if (ourLog.isTraceEnabled()) {
384 				ourLog.trace("Request FullPath: {}", requestFullPath);
385 				ourLog.trace("Servlet Path: {}", servletPath);
386 				ourLog.trace("Request Url: {}", requestUrl);
387 				ourLog.trace("Context Path: {}", servletContextPath);
388 			}
389 
390 			IdDt id = null;
391 			IdDt versionId = null;
392 			String operation = null;
393 
394 			String requestPath = requestFullPath.substring(servletContextPath.length() + servletPath.length());
395 			if (requestPath.length() > 0 && requestPath.charAt(0) == '/') {
396 				requestPath = requestPath.substring(1);
397 			}
398 
399 			String fhirServerBase;
400 			fhirServerBase = myServerAddressStrategy.determineServerBase(theRequest);
401 
402 			if (fhirServerBase.endsWith("/")) {
403 				fhirServerBase = fhirServerBase.substring(0, fhirServerBase.length() - 1);
404 			}
405 
406 			String completeUrl = StringUtils.isNotBlank(theRequest.getQueryString()) ? requestUrl + "?" + theRequest.getQueryString() : requestUrl.toString();
407 
408 			Map<String, String[]> params = new HashMap<String, String[]>(theRequest.getParameterMap());
409 
410 			StringTokenizer tok = new StringTokenizer(requestPath, "/");
411 			if (tok.hasMoreTokens()) {
412 				resourceName = tok.nextToken();
413 				if (resourceName.startsWith("_")) {
414 					operation = resourceName;
415 					resourceName = null;
416 				}
417 			}
418 
419 			ResourceBinding resourceBinding = null;
420 			BaseMethodBinding<?> resourceMethod = null;
421 			if ("metadata".equals(resourceName)) {
422 				resourceMethod = myServerConformanceMethod;
423 			} else if (resourceName == null) {
424 				resourceBinding = myNullResourceBinding;
425 			} else {
426 				resourceBinding = myResourceNameToProvider.get(resourceName);
427 				if (resourceBinding == null) {
428 					throw new InvalidRequestException("Unknown resource type '" + resourceName + "' - Server knows how to handle: " + myResourceNameToProvider.keySet());
429 				}
430 			}
431 
432 			if (tok.hasMoreTokens()) {
433 				String nextString = tok.nextToken();
434 				if (nextString.startsWith("_")) {
435 					operation = nextString;
436 				} else {
437 					id = new IdDt(resourceName, nextString);
438 				}
439 			}
440 
441 			if (tok.hasMoreTokens()) {
442 				String nextString = tok.nextToken();
443 				if (nextString.equals(Constants.PARAM_HISTORY)) {
444 					if (tok.hasMoreTokens()) {
445 						String versionString = tok.nextToken();
446 						if (id == null) {
447 							throw new InvalidRequestException("Don't know how to handle request path: " + requestPath);
448 						}
449 						id = new IdDt(resourceName + "/" + id.getIdPart() + "/_history/" + versionString);
450 						versionId = id;
451 					} else {
452 						operation = Constants.PARAM_HISTORY;
453 					}
454 				} else if (nextString.startsWith("_")) {
455 					if (operation != null) {
456 						throw new InvalidRequestException("URL Path contains two operations (part beginning with _): " + requestPath);
457 					}
458 					operation = nextString;
459 				}
460 			}
461 
462 			// Secondary is for things like ..../_tags/_delete
463 			String secondaryOperation = null;
464 
465 			while (tok.hasMoreTokens()) {
466 				String nextString = tok.nextToken();
467 				if (operation == null) {
468 					operation = nextString;
469 				} else if (secondaryOperation == null) {
470 					secondaryOperation = nextString;
471 				} else {
472 					throw new InvalidRequestException("URL path has unexpected token '" + nextString + "' at the end: " + requestPath);
473 				}
474 			}
475 
476 			if (theRequestType == RequestType.PUT && versionId == null) {
477 				String contentLocation = theRequest.getHeader("Content-Location");
478 				if (contentLocation != null) {
479 					versionId = new IdDt(contentLocation);
480 				}
481 			}
482 
483 			// TODO: look for more tokens for version, compartments, etc...
484 
485 			String acceptEncoding = theRequest.getHeader(Constants.HEADER_ACCEPT_ENCODING);
486 			boolean respondGzip = false;
487 			if (acceptEncoding != null) {
488 				String[] parts = acceptEncoding.trim().split("\\s*,\\s*");
489 				for (String string : parts) {
490 					if (string.equals("gzip")) {
491 						respondGzip = true;
492 					}
493 				}
494 			}
495 
496 			Request r = new Request();
497 			r.setResourceName(resourceName);
498 			r.setId(id);
499 			r.setVersion(versionId);
500 			r.setOperation(operation);
501 			r.setSecondaryOperation(secondaryOperation);
502 			r.setParameters(params);
503 			r.setRequestType(theRequestType);
504 			r.setFhirServerBase(fhirServerBase);
505 			r.setCompleteUrl(completeUrl);
506 			r.setServletRequest(theRequest);
507 			r.setServletResponse(theResponse);
508 			r.setRespondGzip(respondGzip);
509 
510 			String pagingAction = theRequest.getParameter(Constants.PARAM_PAGINGACTION);
511 			if (getPagingProvider() != null && isNotBlank(pagingAction)) {
512 				handlePagingRequest(r, theResponse, pagingAction);
513 				return;
514 			}
515 
516 			if (resourceMethod == null && resourceBinding != null) {
517 				resourceMethod = resourceBinding.getMethod(r);
518 			}
519 			if (null == resourceMethod) {
520 				StringBuilder b = new StringBuilder();
521 				b.append("No resource method available for ");
522 				b.append(theRequestType.name());
523 				b.append(" operation[");
524 				b.append(requestPath);
525 				b.append("]");
526 				b.append(" with parameters ");
527 				b.append(params.keySet());
528 				throw new InvalidRequestException(b.toString());
529 			}
530 
531 			resourceMethod.invokeServer(this, r, theResponse);
532 
533 		} catch (AuthenticationException e) {
534 			if (requestIsBrowser(theRequest)) {
535 				// if request is coming from a browser, prompt the user to enter login credentials
536 				theResponse.setHeader("WWW-Authenticate", "BASIC realm=\"FHIR\"");
537 			}
538 			theResponse.setStatus(e.getStatusCode());
539 			addHeadersToResponse(theResponse);
540 			theResponse.setContentType("text/plain");
541 			theResponse.setCharacterEncoding("UTF-8");
542 			theResponse.getWriter().write(e.getMessage());
543 
544 		} catch (Throwable e) {
545 
546 			OperationOutcome oo = new OperationOutcome();
547 			Issue issue = oo.addIssue();
548 			issue.getSeverity().setValueAsEnum(IssueSeverityEnum.ERROR);
549 
550 			int statusCode = 500;
551 			if (e instanceof InternalErrorException) {
552 				ourLog.error("Failure during REST processing", e);
553 				issue.getDetails().setValue(e.toString() + "\n\n" + ExceptionUtils.getStackTrace(e));
554 			} else if (e instanceof BaseServerResponseException) {
555 				ourLog.warn("Failure during REST processing: {}", e.toString());
556 				statusCode = ((BaseServerResponseException) e).getStatusCode();
557 				issue.getDetails().setValue(e.getMessage());
558 			} else {
559 				ourLog.error("Failure during REST processing", e);
560 				issue.getDetails().setValue(e.toString() + "\n\n" + ExceptionUtils.getStackTrace(e));
561 			}
562 
563 			streamResponseAsResource(this, theResponse, oo, determineResponseEncoding(theRequest), true, false, NarrativeModeEnum.NORMAL, statusCode, false);
564 
565 			theResponse.setStatus(statusCode);
566 			addHeadersToResponse(theResponse);
567 			theResponse.setContentType("text/plain");
568 			theResponse.setCharacterEncoding("UTF-8");
569 			theResponse.getWriter().append(e.getMessage());
570 			theResponse.getWriter().close();
571 
572 		}
573 
574 	}

576 	/**
577 	 * Initializes the server. Note that this method is final to avoid accidentally introducing bugs in implementations, but subclasses may put initialization code in {@link #initialize()}, which is
578 	 * called immediately before beginning initialization of the restful server's internal init.
579 	 */
580 	@Override
581 	public final void init() throws ServletException {
582 		initialize();
583 		try {
584 			ourLog.info("Initializing HAPI FHIR restful server");
585 
586 			mySecurityManager = getSecurityManager();
587 			if (null == mySecurityManager) {
588 				ourLog.trace("No security manager has been provided");
589 			}
590 
591 			Collection<IResourceProvider> resourceProvider = getResourceProviders();
592 			if (resourceProvider != null) {
593 				Map<Class<? extends IResource>, IResourceProvider> typeToProvider = new HashMap<Class<? extends IResource>, IResourceProvider>();
594 				for (IResourceProvider nextProvider : resourceProvider) {
595 					Class<? extends IResource> resourceType = nextProvider.getResourceType();
596 					if (resourceType == null) {
597 						throw new NullPointerException("getResourceType() on class '" + nextProvider.getClass().getCanonicalName() + "' returned null");
598 					}
599 					if (typeToProvider.containsKey(resourceType)) {
600 						throw new ServletException("Multiple providers for type: " + resourceType.getCanonicalName());
601 					}
602 					typeToProvider.put(resourceType, nextProvider);
603 				}
604 				ourLog.info("Got {} resource providers", typeToProvider.size());
605 				for (IResourceProvider provider : typeToProvider.values()) {
606 					assertProviderIsValid(provider);
607 					findResourceMethods(provider);
608 				}
609 			}
610 
611 			Collection<Object> providers = getPlainProviders();
612 			if (providers != null) {
613 				for (Object next : providers) {
614 					assertProviderIsValid(next);
615 					findResourceMethods(next);
616 				}
617 			}
618 
619 			findResourceMethods(getServerProfilesProvider());
620 			findSystemMethods(getServerConformanceProvider());
621 
622 		} catch (Exception ex) {
623 			ourLog.error("An error occurred while loading request handlers!", ex);
624 			throw new ServletException("Failed to initialize FHIR Restful server", ex);
625 		}
626 
627 		myStarted = true;
628 		ourLog.info("A FHIR has been lit on this server");
629 	}

849 	public static EncodingEnum determineRequestEncoding(Request theReq) {
850 		Enumeration<String> acceptValues = theReq.getServletRequest().getHeaders(Constants.HEADER_CONTENT_TYPE);
851 		if (acceptValues != null) {
852 			while (acceptValues.hasMoreElements()) {
853 				String nextAcceptHeaderValue = acceptValues.nextElement();
854 				if (nextAcceptHeaderValue != null && isNotBlank(nextAcceptHeaderValue)) {
855 					for (String nextPart : nextAcceptHeaderValue.split(",")) {
856 						int scIdx = nextPart.indexOf(';');
857 						if (scIdx == 0) {
858 							continue;
859 						}
860 						if (scIdx != -1) {
861 							nextPart = nextPart.substring(0, scIdx);
862 						}
863 						nextPart = nextPart.trim();
864 						EncodingEnum retVal = Constants.FORMAT_VAL_TO_ENCODING.get(nextPart);
865 						if (retVal != null) {
866 							return retVal;
867 						}
868 					}
869 				}
870 			}
871 		}
872 		return EncodingEnum.XML;
873 	}

966 	public static void streamResponseAsBundle(RestfulServer theServer, HttpServletResponse theHttpResponse, IBundleProvider theResult, EncodingEnum theResponseEncoding, String theServerBase,
967 			String theCompleteUrl, boolean thePrettyPrint, boolean theRequestIsBrowser, NarrativeModeEnum theNarrativeMode, int theOffset, Integer theLimit, String theSearchId, boolean theRespondGzip)
968 			throws IOException {
969 		assert !theServerBase.endsWith("/");
970 
971 		theHttpResponse.setStatus(200);
972 
973 		if (theRequestIsBrowser && theServer.isUseBrowserFriendlyContentTypes()) {
974 			theHttpResponse.setContentType(theResponseEncoding.getBrowserFriendlyBundleContentType());
975 		} else if (theNarrativeMode == NarrativeModeEnum.ONLY) {
976 			theHttpResponse.setContentType(Constants.CT_HTML);
977 		} else {
978 			theHttpResponse.setContentType(theResponseEncoding.getBundleContentType());
979 		}
980 
981 		theHttpResponse.setCharacterEncoding(Constants.CHARSET_UTF_8);
982 
983 		theServer.addHeadersToResponse(theHttpResponse);
984 
985 		int numToReturn;
986 		String searchId = null;
987 		List<IResource> resourceList;
988 		if (theServer.getPagingProvider() == null) {
989 			numToReturn = theResult.size();
990 			resourceList = theResult.getResources(0, numToReturn);
991 		} else {
992 			IPagingProvider pagingProvider = theServer.getPagingProvider();
993 			if (theLimit == null) {
994 				numToReturn = pagingProvider.getDefaultPageSize();
995 			} else {
996 				numToReturn = Math.min(pagingProvider.getMaximumPageSize(), theLimit);
997 			}
998 
999 			numToReturn = Math.min(numToReturn, theResult.size() - theOffset);
1000 			resourceList = theResult.getResources(theOffset, numToReturn + theOffset);
1001 
1002 			if (theSearchId != null) {
1003 				searchId = theSearchId;
1004 			} else {
1005 				if (theResult.size() > numToReturn) {
1006 					searchId = pagingProvider.storeResultList(theResult);
1007 					Validate.notNull(searchId, "Paging provider returned null searchId");
1008 				}
1009 			}
1010 		}
1011 
1012 		for (IResource next : resourceList) {
1013 			if (next.getId() == null || next.getId().isEmpty()) {
1014 				throw new InternalErrorException("Server method returned resource of type[" + next.getClass().getSimpleName() + "] with no ID specified (IResource#setId(IdDt) must be called)");
1015 			}
1016 		}
1017 
1018 		Bundle bundle = createBundleFromResourceList(theServer.getFhirContext(), theServer.getServerName(), resourceList, theServerBase, theCompleteUrl, theResult.size());
1019 
1020 		bundle.setPublished(theResult.getPublished());
1021 
1022 		if (theServer.getPagingProvider() != null) {
1023 			int limit;
1024 			limit = theLimit != null ? theLimit : theServer.getPagingProvider().getDefaultPageSize();
1025 			limit = Math.min(limit, theServer.getPagingProvider().getMaximumPageSize());
1026 
1027 			if (searchId != null) {
1028 				if (theOffset + numToReturn < theResult.size()) {
1029 					bundle.getLinkNext().setValue(createPagingLink(theServerBase, searchId, theOffset + numToReturn, numToReturn, theResponseEncoding, thePrettyPrint));
1030 				}
1031 				if (theOffset > 0) {
1032 					int start = Math.max(0, theOffset - limit);
1033 					bundle.getLinkPrevious().setValue(createPagingLink(theServerBase, searchId, start, limit, theResponseEncoding, thePrettyPrint));
1034 				}
1035 			}
1036 		}
1037 
1038 		Writer writer = getWriter(theHttpResponse, theRespondGzip);
1039 		try {
1040 			if (theNarrativeMode == NarrativeModeEnum.ONLY) {
1041 				for (IResource next : resourceList) {
1042 					writer.append(next.getText().getDiv().getValueAsString());
1043 					writer.append("<hr/>");
1044 				}
1045 			} else {
1046 				RestfulServer.getNewParser(theServer.getFhirContext(), theResponseEncoding, thePrettyPrint, theNarrativeMode).encodeBundleToWriter(bundle, writer);
1047 			}
1048 		} finally {
1049 			writer.close();
1050 		}
1051 	}

1059 	private static void streamResponseAsResource(RestfulServer theServer, HttpServletResponse theHttpResponse, IResource theResource, EncodingEnum theResponseEncoding, boolean thePrettyPrint,
1060 			boolean theRequestIsBrowser, NarrativeModeEnum theNarrativeMode, int stausCode, boolean theRespondGzip) throws IOException {
1061 		theHttpResponse.setStatus(stausCode);
1062 
1063 		if (theResource instanceof Binary) {
1064 			Binary bin = (Binary) theResource;
1065 			if (isNotBlank(bin.getContentType())) {
1066 				theHttpResponse.setContentType(bin.getContentType());
1067 			} else {
1068 				theHttpResponse.setContentType(Constants.CT_OCTET_STREAM);
1069 			}
1070 			if (bin.getContent() == null || bin.getContent().length == 0) {
1071 				return;
1072 			}
1073 			theHttpResponse.setContentLength(bin.getContent().length);
1074 			ServletOutputStream oos = theHttpResponse.getOutputStream();
1075 			oos.write(bin.getContent());
1076 			oos.close();
1077 			return;
1078 		}
1079 
1080 		if (theRequestIsBrowser && theServer.isUseBrowserFriendlyContentTypes()) {
1081 			theHttpResponse.setContentType(theResponseEncoding.getBrowserFriendlyBundleContentType());
1082 		} else if (theNarrativeMode == NarrativeModeEnum.ONLY) {
1083 			theHttpResponse.setContentType(Constants.CT_HTML);
1084 		} else {
1085 			theHttpResponse.setContentType(theResponseEncoding.getResourceContentType());
1086 		}
1087 		theHttpResponse.setCharacterEncoding(Constants.CHARSET_UTF_8);
1088 
1089 		theServer.addHeadersToResponse(theHttpResponse);
1090 
1091 		InstantDt lastUpdated = ResourceMetadataKeyEnum.UPDATED.get(theResource);
1092 		if (lastUpdated != null) {
1093 			theHttpResponse.addHeader(Constants.HEADER_LAST_MODIFIED, lastUpdated.getValueAsString());
1094 		}
1095 
1096 		TagList list = (TagList) theResource.getResourceMetadata().get(ResourceMetadataKeyEnum.TAG_LIST);
1097 		if (list != null) {
1098 			for (Tag tag : list) {
1099 				if (StringUtils.isNotBlank(tag.getTerm())) {
1100 					theHttpResponse.addHeader(Constants.HEADER_CATEGORY, tag.toHeaderValue());
1101 				}
1102 			}
1103 		}
1104 
1105 		Writer writer = getWriter(theHttpResponse, theRespondGzip);
1106 		try {
1107 			if (theNarrativeMode == NarrativeModeEnum.ONLY) {
1108 				writer.append(theResource.getText().getDiv().getValueAsString());
1109 			} else {
1110 				RestfulServer.getNewParser(theServer.getFhirContext(), theResponseEncoding, thePrettyPrint, theNarrativeMode).encodeResourceToWriter(theResource, writer);
1111 			}
1112 		} finally {
1113 			writer.close();
1114 		}
1115 	}

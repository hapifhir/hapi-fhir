39  import ca.uhn.fhir.model.dstu.valueset.RestfulConformanceModeEnum;

56  public class ServerConformanceProvider {
57  
58  	private volatile Conformance myConformance;
59  	private final RestfulServer myRestfulServer;
60  	private boolean myCache = true;
61  
62  	public ServerConformanceProvider(RestfulServer theRestfulServer) {
63  		myRestfulServer = theRestfulServer;
64  	}
65  
66  	@Metadata
67  	public Conformance getServerConformance() {
68  		if (myConformance != null && myCache) {
69  			return myConformance;
70  		}
71  
72  		Conformance retVal = new Conformance();
73  		retVal.getSoftware().setName(myRestfulServer.getServerName());
74  		retVal.getSoftware().setVersion(myRestfulServer.getServerVersion());
75  		retVal.addFormat(Constants.CT_FHIR_XML);
76  		retVal.addFormat(Constants.CT_FHIR_JSON);
77  
78  		Rest rest = retVal.addRest();
79  		rest.setMode(RestfulConformanceModeEnum.SERVER);
80  
81  		Set<RestfulOperationSystemEnum> systemOps = new HashSet<RestfulOperationSystemEnum>();
82  
83  		List<ResourceBinding> bindings = new ArrayList<ResourceBinding>(myRestfulServer.getResourceBindings());
84  		Collections.sort(bindings, new Comparator<ResourceBinding>() {
85  			@Override
86  			public int compare(ResourceBinding theArg0, ResourceBinding theArg1) {
87  				return theArg0.getResourceName().compareToIgnoreCase(theArg1.getResourceName());
88  			}
89  		});
90  
91  		for (ResourceBinding next : bindings) {
92  
93  			Set<RestfulOperationTypeEnum> resourceOps = new HashSet<RestfulOperationTypeEnum>();
94  			RestResource resource = rest.addResource();
95  
96  			String resourceName = next.getResourceName();
97  			RuntimeResourceDefinition def = myRestfulServer.getFhirContext().getResourceDefinition(resourceName);
98  			resource.getType().setValue(def.getName());
99  			resource.getProfile().setId(new IdDt(def.getResourceProfile()));
100 
101 			Map<String, Conformance.RestResourceSearchParam> nameToSearchParam = new HashMap<String, Conformance.RestResourceSearchParam>();
102 			for (BaseMethodBinding<?> nextMethodBinding : next.getMethodBindings()) {
103 				RestfulOperationTypeEnum resOp = nextMethodBinding.getResourceOperationType();
104 				if (resOp != null) {
105 					if (resourceOps.contains(resOp) == false) {
106 						resourceOps.add(resOp);
107 						resource.addOperation().setCode(resOp);
108 					}
109 				}
110 
111 				RestfulOperationSystemEnum sysOp = nextMethodBinding.getSystemOperationType();
112 				if (sysOp != null) {
113 					if (systemOps.contains(sysOp) == false) {
114 						systemOps.add(sysOp);
115 						rest.addOperation().setCode(sysOp);
116 					}
117 				}
118 
119 				if (nextMethodBinding instanceof SearchMethodBinding) {
120 					List<IParameter> params = ((SearchMethodBinding) nextMethodBinding).getParameters();
121 					List<SearchParameter> searchParameters = new ArrayList<SearchParameter>();
122 					for (IParameter nextParameter : params) {
123 						if ((nextParameter instanceof SearchParameter)) {
124 							searchParameters.add((SearchParameter) nextParameter);
125 						}
126 					}
127 					Collections.sort(searchParameters, new Comparator<SearchParameter>() {
128 						@Override
129 						public int compare(SearchParameter theO1, SearchParameter theO2) {
130 							if (theO1.isRequired() == theO2.isRequired()) {
131 								return 0;
132 							}
133 							if (theO1.isRequired()) {
134 								return 1;
135 							}
136 							return -1;
137 						}
138 					});
139 					if (searchParameters.isEmpty()) {
140 						continue;
141 					}
142 					boolean allOptional = searchParameters.get(0).isRequired() == false;
143 
144 					RestResourceSearchParam searchParam = null;
145 					ExtensionDt searchParamChain = null;
146 					for (SearchParameter nextParameter : searchParameters) {
147 
148 						if (searchParam == null || allOptional) {
149 							if (!nameToSearchParam.containsKey(nextParameter.getName())) {
150 								RestResourceSearchParam param = resource.addSearchParam();
151 								param.setName(nextParameter.getName());
152 								param.setDocumentation(nextParameter.getDescription());
153 								param.setType(nextParameter.getParamType());
154 								searchParam = param;
155 							} else {
156 								searchParam = nameToSearchParam.get(nextParameter.getName());
157 							}
158 
159 							if (searchParam != null) {
160 								searchParam.setType(nextParameter.getParamType());
161 							}
162 
163 						} else {
164 
165 							searchParamChain = searchParam.addUndeclaredExtension(false, ExtensionConstants.CONF_ADDITIONAL_PARAM);
166 
167 							// if (searchParamChain == null) {
168 							// } else {
169 							// searchParamChain = searchParamChain.addUndeclaredExtension(false,
170 							// ExtensionConstants.CONF_ADDITIONAL_PARAM);
171 							// }
172 
173 							ExtensionDt ext = new ExtensionDt();
174 							ext.setUrl(ExtensionConstants.CONF_ADDITIONAL_PARAM_NAME);
175 							ext.setValue(new StringDt(nextParameter.getName()));
176 							searchParamChain.getUndeclaredExtensions().add(ext);
177 
178 							ext = new ExtensionDt();
179 							ext.setUrl(ExtensionConstants.CONF_ADDITIONAL_PARAM_DESCRIPTION);
180 							ext.setValue(new StringDt(nextParameter.getDescription()));
181 							searchParamChain.getUndeclaredExtensions().add(ext);
182 
183 							ext = new ExtensionDt();
184 							ext.setUrl(ExtensionConstants.CONF_ADDITIONAL_PARAM_TYPE);
185 							if (nextParameter.getParamType() != null) {
186 								ext.setValue(new CodeDt(nextParameter.getParamType().getCode()));
187 							}
188 							searchParamChain.getUndeclaredExtensions().add(ext);
189 
190 							ext = new ExtensionDt();
191 							ext.setUrl(ExtensionConstants.CONF_ADDITIONAL_PARAM_REQUIRED);
192 							ext.setValue(new BooleanDt(nextParameter.isRequired()));
193 							searchParamChain.getUndeclaredExtensions().add(ext);
194 
195 						}
196 
197 					}
198 				}
199 
200 				Collections.sort(resource.getOperation(), new Comparator<RestResourceOperation>() {
201 					@Override
202 					public int compare(RestResourceOperation theO1, RestResourceOperation theO2) {
203 						RestfulOperationTypeEnum o1 = theO1.getCode().getValueAsEnum();
204 						RestfulOperationTypeEnum o2 = theO2.getCode().getValueAsEnum();
205 						if (o1 == null && o2 == null) {
206 							return 0;
207 						}
208 						if (o1 == null) {
209 							return 1;
210 						}
211 						if (o2 == null) {
212 							return -1;
213 						}
214 						return o1.ordinal() - o2.ordinal();
215 					}
216 				});
217 
218 			}
219 
220 		}
221 
222 		myConformance = retVal;
223 		return retVal;
224 	}
225 
226 	/**
227 	 * Sets the cache property (default is true). If set to true, the same response will be returned for each
228 	 * invocation.
229 	 */
230 	public void setCache(boolean theCache) {
231 		myCache = theCache;
232 	}

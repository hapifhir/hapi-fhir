41  abstract class BaseOutcomeReturningMethodBindingWithResourceParam extends BaseOutcomeReturningMethodBinding {
42  	private int myResourceParameterIndex;
43  	private String myResourceName;
44  	private boolean myBinary;
45  
46  	public BaseOutcomeReturningMethodBindingWithResourceParam(Method theMethod, FhirContext theContext, Class<?> theMethodAnnotation, Object theProvider) {
47  		super(theMethod, theContext, theMethodAnnotation, theProvider);
48  
49  		ResourceParameter resourceParameter = null;
50  
51  		int index = 0;
52  		for (IParameter next : getParameters()) {
53  			if (next instanceof ResourceParameter) {
54  				resourceParameter = (ResourceParameter) next;
55  				Class<? extends IResource> resourceType = resourceParameter.getResourceType();
56  
57  				if (theProvider instanceof IResourceProvider) {
58  					resourceType = ((IResourceProvider) theProvider).getResourceType();
59  				}
60  
61  				if (resourceType.isAssignableFrom(Binary.class)) {
62  					myBinary = true;
63  				}
64  
65  				myResourceName = theContext.getResourceDefinition(resourceType).getName();
66  
67  				myResourceParameterIndex = index;
68  			}
69  			index++;
70  		}
71  
72  		if (resourceParameter == null) {
73  			throw new ConfigurationException("Method " + theMethod.getName() + " in type " + theMethod.getDeclaringClass().getCanonicalName() + " does not have a parameter annotated with @" + ResourceParam.class.getSimpleName());
74  		}
75  
76  	}
77  
78  	@Override
79  	protected IResource parseIncomingServerResource(Request theRequest) throws IOException {
80  		if (myBinary) {
81  			String ct = theRequest.getServletRequest().getHeader(Constants.HEADER_CONTENT_TYPE);
82  			byte[] contents = IOUtils.toByteArray(theRequest.getServletRequest().getInputStream());
83  			return new Binary(ct, contents);
84  		} else {
85  			return super.parseIncomingServerResource(theRequest);
86  		}
87  	}
88  
89  	@Override
90  	public RestfulOperationTypeEnum getResourceOperationType() {
91  		// TODO Auto-generated method stub
92  		return null;
93  	}
94  
95  	@Override
96  	public RestfulOperationSystemEnum getSystemOperationType() {
97  		// TODO Auto-generated method stub
98  		return null;
99  	}
100 
101 	/**
102 	 * For subclasses to override
103 	 */
104 	@Override
105 	protected void addParametersForServerRequest(Request theRequest, Object[] theParams) {
106 		// nothing
107 	}
108 
109 	@Override
110 	public String getResourceName() {
111 		return myResourceName;
112 	}
113 
114 	@Override
115 	public BaseHttpClientInvocation invokeClient(Object[] theArgs) throws InternalErrorException {
116 		IResource resource = (IResource) theArgs[myResourceParameterIndex];
117 		if (resource == null) {
118 			throw new NullPointerException("Resource can not be null");
119 		}
120 
121 		BaseHttpClientInvocation retVal = createClientInvocation(theArgs, resource);
122 		return retVal;
123 	}

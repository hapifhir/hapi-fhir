23  import java.lang.annotation.ElementType;
24  import java.lang.annotation.Retention;
25  import java.lang.annotation.RetentionPolicy;
26  import java.lang.annotation.Target;
27  
28  import ca.uhn.fhir.model.api.IResource;
29  import ca.uhn.fhir.model.api.TagList;
30  import ca.uhn.fhir.model.primitive.IdDt;
31  
32  /**
33   * RESTful method annotation to be used for the FHIR <a
34   * href="http://hl7.org/implement/standards/fhir/http.html#tags">Tag
35   * Operations</a> which have to do with deleting tags.
36   * <ul>
37   * <li>
38   * To delete tag(s) <b>to the given resource
39   * instance</b>, this annotation should contain a {@link #type()} attribute
40   * specifying the resource type, and the method should have a parameter of type
41   * {@link IdDt} annotated with the {@link IdParam} annotation, as well as 
42   * a parameter of type {@link TagList} which will contain the list of tags
43   * to be deleted.
44   * Note that for a
45   * server implementation, the {@link #type()} annotation is optional if the
46   * method is defined in a <a href=
47   * "http://hl7api.sourceforge.net/hapi-fhir/doc_rest_server.html#resource_providers"
48   * >resource provider</a>, since the type is implied.</li>
49   * <li>
50   * To delete tag(s) on the server <b>to the given version of the
51   * resource instance</b>, this annotation should contain a {@link #type()}
52   * attribute specifying the resource type, and the method should have a
53   * parameter of type {@link IdDt} annotated with the {@link VersionIdParam}
54   * annotation, <b>and</b> a parameter of type {@link IdDt} annotated with the
55   * {@link IdParam} annotation, as well as 
56   * a parameter of type {@link TagList} which will contain the list of tags
57   * to be deleted.
58   * Note that for a server implementation, the
59   * {@link #type()} annotation is optional if the method is defined in a <a href=
60   * "http://hl7api.sourceforge.net/hapi-fhir/doc_rest_server.html#resource_providers"
61   * >resource provider</a>, since the type is implied.</li>
62   * </ul>
63   */
64  @Target(value= ElementType.METHOD)
65  @Retention(value=RetentionPolicy.RUNTIME)
66  public @interface DeleteTags {
67  
68  	/**
69  	 * If set to a type other than the default (which is {@link IResource.class}
70  	 * , this method is expected to return a TagList containing only tags which
71  	 * are specific to the given resource type.
72  	 */
73  	Class<? extends IResource> type() default IResource.class;
74  
75  }

123 	private void putPlacerResourceInDeletedEntry(BundleEntry entry) {
124 		IdDt id = null;
125 		if (entry.getLinkSelf() != null && entry.getLinkSelf().isEmpty() == false) {
126 			id = new IdDt(entry.getLinkSelf().getValue());
127 		} else {
128 			id = entry.getId();
129 		}
130 
131 		IResource resource = entry.getResource();
132 		if (resource == null && id != null && isNotBlank(id.getResourceType())) {
133 			String resourceType = id.getResourceType();
134 			RuntimeResourceDefinition def = myContext.getResourceDefinition(resourceType);
135 			if (def == null) {
136 				throw new DataFormatException("Entry references unknown resource type: " + resourceType);
137 			}
138 			resource = def.newInstance();
139 			resource.setId(id);
140 			entry.setResource(resource);
141 		}
142 
143 		if (resource != null) {
144 			resource.getResourceMetadata().put(ResourceMetadataKeyEnum.DELETED_AT, entry.getDeletedAt());
145 			resource.getResourceMetadata().put(ResourceMetadataKeyEnum.VERSION_ID, id);
146 		}
147 	}

217 	public class AtomCategoryState extends BaseState {
218 
219 		private static final int STATE_LABEL = 2;
220 		private static final int STATE_NONE = 0;
221 		private static final int STATE_SCHEME = 3;
222 		private static final int STATE_TERM = 1;
223 
224 		private int myCatState = STATE_NONE;
225 		private Tag myInstance;
226 
227 		public AtomCategoryState(Tag theEntry) {
228 			super(null);
229 			myInstance = theEntry;
230 		}
231 
232 		@Override
233 		public void attributeValue(String theName, String theValue) throws DataFormatException {
234 			if ("term".equals(theName)) {
235 				myInstance.setTerm(theValue);
236 			} else if ("label".equals(theName)) {
237 				myInstance.setLabel(theValue);
238 			} else if ("scheme".equals(theName)) {
239 				myInstance.setScheme(theValue);
240 			} else if ("value".equals(theName)) {
241 				/*
242 				 * This handles XML parsing, which is odd for this quasi-resource type, since the tag has three values
243 				 * instead of one like everything else.
244 				 */
245 				switch (myCatState) {
246 				case STATE_LABEL:
247 					myInstance.setLabel(theValue);
248 					break;
249 				case STATE_TERM:
250 					myInstance.setTerm(theValue);
251 					break;
252 				case STATE_SCHEME:
253 					myInstance.setScheme(theValue);
254 					break;
255 				default:
256 					super.string(theValue);
257 					break;
258 				}
259 
260 			}
261 		}
262 
263 		@Override
264 		public void endingElement() throws DataFormatException {
265 			if (myCatState != STATE_NONE) {
266 				myCatState = STATE_NONE;
267 			} else {
268 				pop();
269 			}
270 		}
271 
272 		@Override
273 		public void enteringNewElement(String theNamespaceURI, String theName) throws DataFormatException {
274 			if (myCatState != STATE_NONE) {
275 				throw new DataFormatException("Unexpected element in entry: " + theName);
276 			}
277 			if ("term".equals(theName)) {
278 				myCatState = STATE_TERM;
279 			} else if ("label".equals(theName)) {
280 				myCatState = STATE_LABEL;
281 			} else if ("scheme".equals(theName)) {
282 				myCatState = STATE_SCHEME;
283 			}
284 		}

384 
385 		private boolean myDeleted;
386 		private BundleEntry myEntry;
387 		private Class<? extends IResource> myResourceType;
388 
389 		public AtomEntryState(Bundle theInstance, Class<? extends IResource> theResourceType) {
390 			super(null);
391 			myEntry = new BundleEntry();
392 			myResourceType = theResourceType;
393 			theInstance.getEntries().add(myEntry);
394 		}
395 
396 		@Override
397 		public void endingElement() throws DataFormatException {
398 			populateResourceMetadata();
399 			pop();
400 
401 			if (myDeleted) {
402 				putPlacerResourceInDeletedEntry(myEntry);
403 			}
404 		}
405 
406 		@Override
407 		public void enteringNewElement(String theNamespaceURI, String theLocalPart) throws DataFormatException {
408 			if ("title".equals(theLocalPart)) {
409 				push(new AtomPrimitiveState(myEntry.getTitle()));
410 			} else if ("id".equals(theLocalPart)) {
411 				push(new AtomPrimitiveState(myEntry.getId()));
412 			} else if ("link".equals(theLocalPart)) {
413 				push(new AtomLinkState(myEntry));
414 			} else if ("updated".equals(theLocalPart)) {
415 				push(new AtomPrimitiveState(myEntry.getUpdated()));
416 			} else if ("published".equals(theLocalPart)) {
417 				push(new AtomPrimitiveState(myEntry.getPublished()));
418 			} else if ("author".equals(theLocalPart)) {
419 				push(new AtomAuthorState(myEntry));
420 			} else if ("content".equals(theLocalPart)) {
421 				push(new PreResourceState(myEntry, myResourceType));
422 			} else if ("summary".equals(theLocalPart)) {
423 				push(new XhtmlState(getPreResourceState(), myEntry.getSummary(), false));
424 			} else if ("category".equals(theLocalPart)) {
425 				push(new AtomCategoryState(myEntry.addCategory()));
426 			} else if ("deleted".equals(theLocalPart) && myJsonMode) {
427 				// JSON and XML deleted entries are completely different for some reason
428 				myDeleted = true;
429 				push(new AtomDeletedJsonWhenState(myEntry.getDeletedAt()));
430 			} else {
431 				throw new DataFormatException("Unexpected element in entry: " + theLocalPart);
432 			}
433 
434 			// TODO: handle category
435 		}

438 			return myEntry;

441 		@SuppressWarnings("deprecation")
442 		private void populateResourceMetadata() {
443 			if (myEntry.getResource() == null) {
444 				return;
445 			}
446 
447 			IdDt id = myEntry.getId();
448 			if (id != null && id.isEmpty() == false) {
449 				myEntry.getResource().setId(id);
450 			}
451 
452 			Map<ResourceMetadataKeyEnum<?>, Object> metadata = myEntry.getResource().getResourceMetadata();
453 			if (myEntry.getPublished().isEmpty() == false) {
454 				ResourceMetadataKeyEnum.PUBLISHED.put(myEntry.getResource(), myEntry.getPublished());
455 			}
456 			if (myEntry.getUpdated().isEmpty() == false) {
457 				ResourceMetadataKeyEnum.UPDATED.put(myEntry.getResource(), myEntry.getUpdated());
458 			}
459 
460 			ResourceMetadataKeyEnum.TITLE.put(myEntry.getResource(), myEntry.getTitle().getValue());
461 
462 			if (myEntry.getCategories().isEmpty() == false) {
463 				TagList tagList = new TagList();
464 				for (Tag next : myEntry.getCategories()) {
465 					tagList.add(next);
466 				}
467 				ResourceMetadataKeyEnum.TAG_LIST.put(myEntry.getResource(), tagList);
468 			}
469 			if (!myEntry.getLinkSelf().isEmpty()) {
470 				String linkSelfValue = myEntry.getLinkSelf().getValue();
471 				IdDt linkSelf = new IdDt(linkSelfValue);
472 				myEntry.getResource().setId(linkSelf);
473 				if (isNotBlank(linkSelf.getVersionIdPart())) {
474 					metadata.put(ResourceMetadataKeyEnum.VERSION_ID, linkSelf);
475 				}
476 			}
477 			if (!myEntry.getLinkAlternate().isEmpty()) {
478 				ResourceMetadataKeyEnum.LINK_ALTERNATE.put(myEntry.getResource(), myEntry.getLinkAlternate().getValue());
479 			}
480 			if (!myEntry.getLinkSearch().isEmpty()) {
481 				ResourceMetadataKeyEnum.LINK_SEARCH.put(myEntry.getResource(), myEntry.getLinkSearch().getValue());
482 			}
483 
484 		}

535 				} else if ("alternate".equals(myRel)) {

598 
599 		private Bundle myInstance;
600 		private Class<? extends IResource> myResourceType;
601 
602 		public AtomState(Bundle theInstance, Class<? extends IResource> theResourceType) {
603 			super(null);
604 			myInstance = theInstance;
605 			myResourceType = theResourceType;
606 		}
607 
608 		@Override
609 		public void endingElement() throws DataFormatException {
610 			pop();
611 		}
612 
613 		@Override
614 		public void enteringNewElement(String theNamespaceURI, String theLocalPart) throws DataFormatException {
615 			if ("entry".equals(theLocalPart) && verifyNamespace(XmlParser.ATOM_NS, theNamespaceURI)) {
616 				push(new AtomEntryState(myInstance, myResourceType));
617 			} else if (theLocalPart.equals("published")) {
618 				push(new AtomPrimitiveState(myInstance.getPublished()));
619 			} else if (theLocalPart.equals("title")) {
620 				push(new AtomPrimitiveState(myInstance.getTitle()));
621 			} else if ("id".equals(theLocalPart)) {
622 				push(new AtomPrimitiveState(myInstance.getBundleId()));
623 			} else if ("link".equals(theLocalPart)) {
624 				push(new AtomLinkState(myInstance));
625 			} else if ("totalResults".equals(theLocalPart) && (verifyNamespace(XmlParser.OPENSEARCH_NS, theNamespaceURI) || verifyNamespace(Constants.OPENSEARCH_NS_OLDER, theNamespaceURI))) {
626 				push(new AtomPrimitiveState(myInstance.getTotalResults()));
627 			} else if ("updated".equals(theLocalPart)) {
628 				push(new AtomPrimitiveState(myInstance.getUpdated()));
629 			} else if ("author".equals(theLocalPart)) {
630 				push(new AtomAuthorState(myInstance));

993 				ContainedDt newDt;

997 				} else {
998 					newDt = (ContainedDt) values.get(0);
999 				}

1001 				push(state);
1002 				return;
1003 			}

1005 			case RESOURCE:
1006 			case EXTENSION_DECLARED: {
